48      public static List<ModuleDetails> readAllModulesIncludingThirdPartyIfAny(
49              String... thirdPartyPackages) {
50          final Set<String> standardModuleFileNames =
51                  new Reflections("com.puppycrawl.tools.checkstyle.meta",
52                          new ResourcesScanner()).getResources(Pattern.compile(".*\\.xml"));
53          final Set<String> allMetadataSources = new HashSet<>(standardModuleFileNames);
54          for (String packageName : thirdPartyPackages) {
55              final Set<String> thirdPartyModuleFileNames =
56                      new Reflections(packageName, new ResourcesScanner())
57                              .getResources(Pattern.compile(".*checkstylemeta-.*\\.xml"));
58              allMetadataSources.addAll(thirdPartyModuleFileNames);
59          }
60  
61          final List<ModuleDetails> result = new ArrayList<>();
62          for (String fileName : allMetadataSources) {
63              final ModuleType moduleType;
64              if (fileName.endsWith("FileFilter.xml")) {
65                  moduleType = ModuleType.FILEFILTER;
66              }
67              else if (fileName.endsWith("Filter.xml")) {
68                  moduleType = ModuleType.FILTER;
69              }
70              else {
71                  moduleType = ModuleType.CHECK;
72              }
73              final ModuleDetails moduleDetails;
74              try {
75                  moduleDetails = read(XmlMetaReader.class.getResourceAsStream("/" + fileName),
76                          moduleType);
77              }
78              catch (ParserConfigurationException | IOException | SAXException ex) {
79                  throw new IllegalStateException("Problem to read all modules including third "
80                          + "party if any. Problem detected at file: " + fileName, ex);
81              }
82              result.add(moduleDetails);
83          }
84  
85          return result;
86      }

135         final List<ModulePropertyDetails> result = new ArrayList<>();
136         final NodeList propertyList = properties.getElementsByTagName("property");
137         for (int i = 0; i < propertyList.getLength(); i++) {
138             final ModulePropertyDetailstyDetails.html#ModulePropertyDetails">ModulePropertyDetails propertyDetails = new ModulePropertyDetails();
139             final Element prop = (Element) propertyList.item(i);
140             propertyDetails.setName(getAttributeValue(prop, "name"));
141             propertyDetails.setType(getAttributeValue(prop, "type"));
142             if (prop.hasAttribute("default-value")) {
143                 propertyDetails.setDefaultValue(getAttributeValue(prop, "default-value"));
144             }
145             if (prop.hasAttribute("validation-type")) {
146                 propertyDetails.setValidationType(getAttributeValue(prop, "validation-type"));

155     private static List<String> getListContentByAttribute(Element element, String listParent,
156                                                          String listOption, String attribute) {
157         final List<Element> children = getDirectChildsByTag(element, listParent);
158         List<String> result = null;
159         if (!children.isEmpty()) {
160             final NodeList nodeList = children.get(0).getElementsByTagName(listOption);
161             final List<String> listContent = new ArrayList<>();
162             for (int j = 0; j < nodeList.getLength(); j++) {
163                 listContent.add(getAttributeValue((Element) nodeList.item(j), attribute));
164             }
165             result = listContent;
166         }
167         return result;
168     }

170     private static List<Element> getDirectChildsByTag(Element element, String sTagName) {
171         final NodeList children = element.getElementsByTagName(sTagName);
172         final List<Element> res = new ArrayList<>();
173         for (int i = 0; i < children.getLength(); i++) {
174             if (children.item(i).getParentNode().equals(element)) {
175                 res.add((Element) children.item(i));
176             }
177         }
178         return res;
179     }

181     private static String getAttributeValue(Element element, String attribute) {
182         return element.getAttributes().getNamedItem(attribute).getNodeValue();
183     }
