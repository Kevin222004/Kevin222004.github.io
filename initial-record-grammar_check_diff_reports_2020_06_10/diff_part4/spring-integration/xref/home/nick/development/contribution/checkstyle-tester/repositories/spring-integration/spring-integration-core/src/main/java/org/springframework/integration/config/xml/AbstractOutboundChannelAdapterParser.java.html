26  import org.springframework.beans.factory.support.ManagedList;
27  import org.springframework.beans.factory.xml.ParserContext;
28  import org.springframework.integration.config.ConsumerEndpointFactoryBean;
29  import org.springframework.integration.handler.AbstractReplyProducingMessageHandler;
30  import org.springframework.util.CollectionUtils;
31  import org.springframework.util.StringUtils;

34  /**
35   * Base class for outbound Channel Adapter parsers.
36   *
37   * If this component is defined as the top-level element in the Spring application context it will produce
38   * an {@link org.springframework.integration.endpoint.AbstractEndpoint} depending on the channel type.
39   * If this component is defined as nested element (e.g., inside of the chain) it will produce
40   * a {@link org.springframework.messaging.MessageHandler}.
41   *
42   * @author Mark Fisher
43   * @author Gary Russell
44   * @author Artem Bilan
45   */
46  public abstract class AbstractOutboundChannelAdapterParser extends AbstractChannelAdapterParser {
47  
48  	@Override
49  	protected AbstractBeanDefinition doParse(Element element, ParserContext parserContext, String channelName) {
50  		if (parserContext.isNested()) {
51  			if (channelName != null) {
52  				String elementDescription = IntegrationNamespaceUtils.createElementDescription(element);
53  				parserContext.getReaderContext().error(
54  						"The 'channel' attribute isn't allowed for " +
55  						elementDescription +
56  						" when it is used as a nested element," +
57  								" e.g. inside a <chain/>", element);
58  			}
59  			AbstractBeanDefinition consumerBeanDefinition = this.parseConsumer(element, parserContext);
60  			this.configureRequestHandlerAdviceChain(element, parserContext, consumerBeanDefinition, null);
61  			return consumerBeanDefinition;
62  		}
63  		BeanDefinitionBuilder builder =  BeanDefinitionBuilder.genericBeanDefinition(ConsumerEndpointFactoryBean.class);
64  		Element pollerElement = DomUtils.getChildElementByTagName(element, "poller");
65  		BeanComponentDefinition handlerBeanComponentDefinition = this.doParseAndRegisterConsumer(element, parserContext);
66  		builder.addPropertyReference("handler", handlerBeanComponentDefinition.getBeanName());
67  		IntegrationNamespaceUtils.checkAndConfigureFixedSubscriberChannel(element, parserContext, channelName,
68  				handlerBeanComponentDefinition.getBeanName());
69  		if (pollerElement != null) {
70  			if (!StringUtils.hasText(channelName)) {
71  				parserContext.getReaderContext().error(
72  						"outbound channel adapter with a 'poller' requires a 'channel' to poll", element);
73  			}
74  			IntegrationNamespaceUtils.configurePollerMetadata(pollerElement, builder, parserContext);
75  		}
76  		builder.addPropertyValue("inputChannelName", channelName);
77  
78  		this.configureRequestHandlerAdviceChain(element, parserContext, handlerBeanComponentDefinition.getBeanDefinition(), builder);
79  
80  		return builder.getBeanDefinition();
81  	}
82  
83  	private void configureRequestHandlerAdviceChain(Element element, ParserContext parserContext,
84  							BeanDefinition handlerBeanDefinition, BeanDefinitionBuilder consumerBuilder) {
85  		Element txElement = DomUtils.getChildElementByTagName(element, "transactional");
86  		Element adviceChainElement = DomUtils.getChildElementByTagName(element,
87  				IntegrationNamespaceUtils.REQUEST_HANDLER_ADVICE_CHAIN);
88  		@SuppressWarnings("rawtypes")
89  		ManagedList adviceChain =
90  				IntegrationNamespaceUtils.configureAdviceChain(adviceChainElement, txElement, handlerBeanDefinition,
91  						parserContext);
92  		if (!CollectionUtils.isEmpty(adviceChain)) {
93  			/*
94  			 * For ARPMH, the advice chain is injected so just the handleRequestMessage method is advised.
95  			 * Sometime ARPMHs do double duty as a gateway and a channel adapter. The parser subclass
96  			 * can indicate this by overriding isUsingReplyProducer(), or we can try to determine it from
97  			 * the bean class.
98  			 */
99  			boolean isReplyProducer = this.isUsingReplyProducer();
100 			if (!isReplyProducer) {
101 				Class<?> beanClass = null;
102 				if (handlerBeanDefinition instanceof AbstractBeanDefinition) {
103 					AbstractBeanDefinition abstractBeanDefinition = (AbstractBeanDefinition) handlerBeanDefinition;
104 					if (abstractBeanDefinition.hasBeanClass()) {
105 						beanClass = abstractBeanDefinition.getBeanClass();
106 					}
107 				}
108 				isReplyProducer = beanClass != null && AbstractReplyProducingMessageHandler.class.isAssignableFrom(beanClass);
109 			}
110 
111 			if (isReplyProducer) {
112 				handlerBeanDefinition.getPropertyValues().add("adviceChain", adviceChain);
113 			}
114 			else if (consumerBuilder != null) {
115 				consumerBuilder.addPropertyValue("adviceChain", adviceChain);
116 			}
117 			else {
118 				String elementDescription = IntegrationNamespaceUtils.createElementDescription(element);
119 				parserContext.getReaderContext().error("'request-handler-advice-chain' isn't allowed for " +
120 						elementDescription +
121 						" within a <chain/>, because its Handler " +
122 						"isn't an AbstractReplyProducingMessageHandler", element);
123 			}
124 		}
125 	}
126 
127 	/**
128 	 * Override this method to control the registration process and return the bean name.
129 	 * If parsing a bean definition whose name can be auto-generated, consider using
130 	 * {@link #parseConsumer(Element, ParserContext)} instead.
131 	 *
132 	 * @param element The element.
133 	 * @param parserContext The parser context.
134 	 * @return The bean component definition.
135 	 */
136 	protected BeanComponentDefinition doParseAndRegisterConsumer(Element element, ParserContext parserContext) {
137 		AbstractBeanDefinition definition = this.parseConsumer(element, parserContext);
138 		if (definition == null) {
139 			parserContext.getReaderContext().error("Consumer parsing must return an AbstractBeanDefinition.", element);
140 		}
141 		String order = element.getAttribute(IntegrationNamespaceUtils.ORDER);
142 		if (StringUtils.hasText(order)) {
143 			definition.getPropertyValues().addPropertyValue(IntegrationNamespaceUtils.ORDER, order);
144 		}
145 		String beanName = BeanDefinitionReaderUtils.generateBeanName(definition, parserContext.getRegistry());
146 		String[] handlerAlias = IntegrationNamespaceUtils.generateAlias(element);
147 		BeanComponentDefinition beanComponentDefinition = new BeanComponentDefinition(definition, beanName, handlerAlias);
148 		parserContext.registerBeanComponent(beanComponentDefinition);
149 		return beanComponentDefinition;
150 	}
151 
152 	/**
153 	 * Override this method to return the BeanDefinition for the MessageConsumer. It will
154 	 * be registered with a generated name.
155 	 *
156 	 * @param element The element.
157 	 * @param parserContext The parser context.
158 	 * @return The bean definition.
159 	 */
160 	protected abstract AbstractBeanDefinition parseConsumer(Element element, ParserContext parserContext);
161 
162 	/**
163 	 * Override this to signal that this channel adapter is actually using a AbstractReplyProducingMessageHandler
164 	 * while it is not possible for this parser to determine that because, say, a FactoryBean is being used.
165 	 *
166 	 * @return false, unless overridden.
167 	 */
168 	protected boolean isUsingReplyProducer() {
169 		return false;
170 	}
