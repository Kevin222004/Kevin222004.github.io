17  package org.springframework.integration.config.xml;
18  
19  import java.util.Map;
20  
21  import org.w3c.dom.Element;
22  
23  import org.springframework.beans.factory.support.BeanDefinitionBuilder;
24  import org.springframework.beans.factory.xml.ParserContext;
25  import org.springframework.integration.IntegrationMessageHeaderAccessor;
26  import org.springframework.integration.handler.MethodInvokingMessageProcessor;
27  import org.springframework.messaging.MessageHeaders;
28  import org.springframework.util.StringUtils;
29  
30  /**
31   * Parser for the &lt;header-enricher&gt; element within the core integration
32   * namespace. This is used for setting the <em>standard</em>, out-of-the-box
33   * configurable {@link MessageHeaders}, such as 'reply-channel', 'priority',
34   * and 'correlation-id'. It will also accept custom header values (or bean
35   * references) if provided as 'header' sub-elements.
36   *
37   * @author Mark Fisher
38   * @author Oleg Zhurakousky
39   * @author Artem Bilan
40   */
41  public class StandardHeaderEnricherParser extends HeaderEnricherParserSupport {
42  
43  	public StandardHeaderEnricherParser() {
44  		addElementToHeaderMapping("reply-channel", MessageHeaders.REPLY_CHANNEL);
45  		addElementToHeaderMapping("error-channel", MessageHeaders.ERROR_CHANNEL);
46  		addElementToHeaderMapping("correlation-id", IntegrationMessageHeaderAccessor.CORRELATION_ID);
47  		addElementToHeaderMapping("expiration-date", IntegrationMessageHeaderAccessor.EXPIRATION_DATE,
48  				Long.class.getName());
49  		addElementToHeaderMapping("priority", IntegrationMessageHeaderAccessor.PRIORITY, Integer.class.getName());
50  		addElementToHeaderMapping("routing-slip", IntegrationMessageHeaderAccessor.ROUTING_SLIP, Map.class.getName());
51  	}
52  
53  	@Override
54  	protected void postProcessHeaderEnricher(BeanDefinitionBuilder builder, Element element,
55  			ParserContext parserContext) {
56  		String ref = element.getAttribute("ref");
57  		String method = element.getAttribute("method");
58  		if (StringUtils.hasText(ref) || StringUtils.hasText(method)) {
59  			if (!StringUtils.hasText(ref) || !StringUtils.hasText(method)) {
60  				parserContext.getReaderContext().error(
61  						"If either 'ref' or 'method' is provided, then they are both required.",
62  						parserContext.extractSource(element));
63  				return;
64  			}
65  			BeanDefinitionBuilder processorBuilder =
66  					BeanDefinitionBuilder.genericBeanDefinition(MethodInvokingMessageProcessor.class);
67  			processorBuilder.addConstructorArgReference(ref);
68  			processorBuilder.addConstructorArgValue(method);
69  			builder.addPropertyValue("messageProcessor", processorBuilder.getBeanDefinition());
70  		}
71  	}
72  
73  }
