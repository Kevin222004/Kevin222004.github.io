7   import java.util.HashSet;

10  import org.apache.commons.lang3.mutable.MutableInt;

12  import net.sourceforge.pmd.lang.apex.ast.ASTBooleanExpression;

19  
20  /**
21   * See the doc for the Java metric.
22   *
23   * @author Cl√©ment Fournier
24   */
25  public class CycloMetric extends AbstractApexOperationMetric {
26  
27  
28      @Override
29      public double computeFor(ASTMethod node, MetricOptions options) {
30          return ((MutableInt) node.jjtAccept(new StandardCycloVisitor(), new MutableInt(1))).doubleValue();
31      }
32      
33  
34      /**
35       * Computes the number of control flow paths through that expression, which is the number of {@code ||} and {@code
36       * &&} operators. Used both by Npath and Cyclo.
37       *
38       * @param expression Boolean expression
39       *
40       * @return The complexity of the expression
41       */
42      public static int booleanExpressionComplexity(ASTStandardCondition expression) {
43          Set<ASTBooleanExpression> subs = new HashSet<>(expression.findDescendantsOfType(ASTBooleanExpression.class));
44          int complexity = 0;
45  
46          for (ASTBooleanExpression sub : subs) {
47              BooleanOp op = sub.getOperator();
48              if (op != null && (op == BooleanOp.AND || op == BooleanOp.OR)) {
49                  complexity++;
50              }
51          }
52  
53          return complexity;
54      }
55  }
