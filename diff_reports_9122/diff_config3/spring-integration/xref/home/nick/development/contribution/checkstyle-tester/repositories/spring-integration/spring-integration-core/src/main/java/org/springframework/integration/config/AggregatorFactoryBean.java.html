189 	@Override
190 	protected AggregatingMessageHandler createHandler() {
191 		MessageGroupProcessor outputProcessor;
192 		if (this.processorBean instanceof MessageGroupProcessor) {
193 			outputProcessor = (MessageGroupProcessor) this.processorBean;
194 		}
195 		else {
196 			if (!StringUtils.hasText(this.methodName)) {
197 				outputProcessor = new MethodInvokingMessageGroupProcessor(this.processorBean);
198 			}
199 			else {
200 				outputProcessor = new MethodInvokingMessageGroupProcessor(this.processorBean, this.methodName);
201 			}
202 		}
203 
204 		if (this.headersFunction != null) {
205 			if (outputProcessor instanceof AbstractAggregatingMessageGroupProcessor) {
206 				((AbstractAggregatingMessageGroupProcessor) outputProcessor).setHeadersFunction(this.headersFunction);
207 			}
208 			else {
209 				outputProcessor = new DelegatingMessageGroupProcessor(outputProcessor, this.headersFunction);
210 			}
211 		}
212 
213 		AggregatingMessageHandler aggregator = new AggregatingMessageHandler(outputProcessor);
214 		JavaUtils.INSTANCE
215 				.acceptIfNotNull(this.expireGroupsUponCompletion, aggregator::setExpireGroupsUponCompletion)
216 				.acceptIfNotNull(this.sendTimeout, aggregator::setSendTimeout)
217 				.acceptIfNotNull(this.outputChannelName, aggregator::setOutputChannelName)
218 				.acceptIfNotNull(this.lockRegistry, aggregator::setLockRegistry)
219 				.acceptIfNotNull(this.messageStore, aggregator::setMessageStore)
220 				.acceptIfNotNull(this.correlationStrategy, aggregator::setCorrelationStrategy)
221 				.acceptIfNotNull(this.releaseStrategy, aggregator::setReleaseStrategy)
222 				.acceptIfNotNull(this.groupTimeoutExpression, aggregator::setGroupTimeoutExpression)
223 				.acceptIfNotNull(this.forceReleaseAdviceChain, aggregator::setForceReleaseAdviceChain)
224 				.acceptIfNotNull(this.taskScheduler, aggregator::setTaskScheduler)
225 				.acceptIfNotNull(this.discardChannel, aggregator::setDiscardChannel)
226 				.acceptIfNotNull(this.discardChannelName, aggregator::setDiscardChannelName)
227 				.acceptIfNotNull(this.sendPartialResultOnExpiry, aggregator::setSendPartialResultOnExpiry)
228 				.acceptIfNotNull(this.minimumTimeoutForEmptyGroups, aggregator::setMinimumTimeoutForEmptyGroups)
229 				.acceptIfNotNull(this.expireGroupsUponTimeout, aggregator::setExpireGroupsUponTimeout)
230 				.acceptIfNotNull(this.popSequence, aggregator::setPopSequence)
231 				.acceptIfNotNull(this.releaseLockBeforeSend, aggregator::setReleaseLockBeforeSend)
232 				.acceptIfNotNull(this.expireDuration,
233 						(duration) -> aggregator.setExpireDuration(Duration.ofMillis(duration)))
234 				.acceptIfNotNull(this.expireTimeout, aggregator::setExpireTimeout);
235 
236 		return aggregator;
237 	}
