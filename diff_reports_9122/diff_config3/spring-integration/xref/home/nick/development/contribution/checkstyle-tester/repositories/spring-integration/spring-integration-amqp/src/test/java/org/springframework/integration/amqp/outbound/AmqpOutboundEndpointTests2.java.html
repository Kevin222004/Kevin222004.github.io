58  @SpringJUnitConfig
59  @RabbitAvailable(queues = "testConfirmOk")
60  @DirtiesContext
61  public class AmqpOutboundEndpointTests2 {
62  
63  	@Test
64  	void testConfirmOk(@Autowired IntegrationFlow flow, @Autowired RabbitTemplate template) {
65  		flow.getInputChannel().send(new GenericMessage<>("test", Collections.singletonMap("rk", "testConfirmOk")));
66  		assertThat(template.receive("testConfirmOk")).isNotNull();
67  	}
68  
69  	@Test
70  	void testWithReturn(@Autowired IntegrationFlow flow) {
71  		assertThatThrownBy(() -> flow.getInputChannel()
72  				.send(new GenericMessage<>("test", Collections.singletonMap("rk", "junkjunk"))))
73  				.isInstanceOf(MessageHandlingException.class)
74  				.hasCauseInstanceOf(AmqpException.class)
75  				.extracting(Throwable::getCause)
76  				.extracting(Throwable::getMessage)
77  				.isEqualTo("Message was returned by the broker");
78  	}
79  
80  	@Test
81  	void testReturnConfirmNoChannels(@Autowired IntegrationFlow flow2) throws Exception {
82  		CorrelationData corr = new CorrelationData("foo");
83  		flow2.getInputChannel().send(MessageBuilder.withPayload("test")
84  				.setHeader("rk", "junkjunk")
85  				.setHeader(AmqpHeaders.PUBLISH_CONFIRM_CORRELATION, corr)
86  				.build());
87  		assertThat(corr.getFuture().get(10, TimeUnit.SECONDS).isAck()).isTrue();
88  		assertThat(corr.getReturned()).isNotNull();
89  	}
90  
91  	@Test
92  	void testWithReject(@Autowired IntegrationFlow flow, @Autowired RabbitAdmin admin,
93  			@Autowired RabbitTemplate template) {
94  
95  		Queue queue = QueueBuilder.nonDurable().autoDelete().maxLength(1).overflow(Overflow.rejectPublish).build();
96  		admin.declareQueue(queue);
97  		flow.getInputChannel().send(new GenericMessage<>("test", Collections.singletonMap("rk", queue.getName())));
98  		assertThatThrownBy(() -> flow.getInputChannel()
99  				.send(new GenericMessage<>("test", Collections.singletonMap("rk", queue.getName()))))
100 				.hasCauseInstanceOf(AmqpException.class)
101 				.extracting(Throwable::getCause)
102 				.extracting(Throwable::getMessage)
103 				.matches(msg -> msg.matches("Negative publisher confirm received: .*"));
104 		assertThat(template.receive(queue.getName())).isNotNull();
105 		admin.deleteQueue(queue.getName());
106 	}
107 
108 	@Configuration(proxyBeanMethods = false)
109 	@EnableIntegration
110 	public static class Config {
111 
112 		@Bean
113 		public IntegrationFlow flow(RabbitTemplate template) {
114 			return f -> f.handle(Amqp.outboundAdapter(template)
115 					.exchangeName("")
116 					.routingKeyFunction(msg -> msg.getHeaders().get("rk", String.class))
117 					.confirmCorrelationFunction(msg -> msg)
118 					.waitForConfirm(true));
119 		}
120 
121 		@Bean
122 		public IntegrationFlow flow2(RabbitTemplate template) {
123 			return f -> f.handle(Amqp.outboundAdapter(template)
124 					.exchangeName("")
125 					.routingKeyFunction(msg -> msg.getHeaders().get("rk", String.class)));
126 		}
127 
128 		@Bean
129 		public CachingConnectionFactory cf() {
130 			CachingConnectionFactory ccf = new CachingConnectionFactory(
131 					RabbitAvailableCondition.getBrokerRunning().getConnectionFactory());
132 			ccf.setPublisherConfirmType(CachingConnectionFactory.ConfirmType.CORRELATED);
133 			ccf.setPublisherReturns(true);
134 			return ccf;
135 		}
136 
137 		@Bean
138 		public RabbitTemplate template(ConnectionFactory cf) {
139 			RabbitTemplate rabbitTemplate = new RabbitTemplate(cf);
140 			rabbitTemplate.setMandatory(true);
141 			rabbitTemplate.setReceiveTimeout(10_000);
142 			return rabbitTemplate;
143 		}
144 
145 		@Bean
146 		public RabbitAdmin admin(ConnectionFactory cf) {
147 			return new RabbitAdmin(cf);
148 		}

150 	}
