44  /**
45   * An instance of {@link org.springframework.integration.core.MessageSource} which returns
46   * a {@link org.springframework.messaging.Message} with a payload which is the result of
47   * execution of query. When {@code expectSingleResult} is false (default), the R2DBC
48   * query is executed returning a {@link reactor.core.publisher.Flux}.
49   * The returned {@link reactor.core.publisher.Flux} will be used as the payload of the
50   * {@link org.springframework.messaging.Message} returned by the {@link #receive()}
51   * method.
52   * <p>
53   * When {@code expectSingleResult} is true, the query is executed returning a {@link reactor.core.publisher.Mono}
54   * for the single object returned from the query.
55   *
56   * @author Rohan Mukesh
57   * @author Artem Bilan
58   *
59   * @since 5.4
60   */
61  public class R2dbcMessageSource extends AbstractMessageSource<Publisher<?>> {
62  
63  	private final R2dbcEntityOperations r2dbcEntityOperations;
64  
65  	private final DatabaseClient databaseClient;
66  
67  	private final StatementMapper statementMapper;
68  
69  	private final SelectCreator selectCreator = new SelectCreator();
70  
71  	private final Expression queryExpression;
72  
73  	private Class<?> payloadType = Map.class;
74  
75  	private BiFunction<Row, RowMetadata, ?> rowMapper = ColumnMapRowMapper.INSTANCE;
76  
77  	private boolean expectSingleResult = false;
78  
79  	private StandardEvaluationContext evaluationContext;
80  
81  	private String updateSql;
82  
83  	private BiFunction<DatabaseClient.GenericExecuteSpec, Object, DatabaseClient.GenericExecuteSpec> bindFunction;
84  
85  	private volatile boolean initialized = false;
86  
87  	/**
88  	 * Create an instance with the provided {@link R2dbcEntityOperations} and SpEL expression
89  	 * which should resolve to a Relational 'query' string.
90  	 * It assumes that the {@link R2dbcEntityOperations} is fully initialized and ready to be used.
91  	 * The 'query' will be evaluated on every call to the {@link #receive()} method.
92  	 * @param r2dbcEntityOperations The reactive database client for performing database calls.
93  	 * @param query The query String.
94  	 */
95  	public R2dbcMessageSource(R2dbcEntityOperations r2dbcEntityOperations, String query) {
96  		this(r2dbcEntityOperations, new LiteralExpression(query));
97  	}
98  
99  	/**
100 	 * Create an instance with the provided {@link R2dbcEntityOperations} and SpEL expression
101 	 * which should resolve to a query string or {@link StatementMapper.SelectSpec} instance.
102 	 * It assumes that the {@link R2dbcEntityOperations} is fully initialized and ready to be used.
103 	 * The 'queryExpression' will be evaluated on every call to the {@link #receive()} method.
104 	 * @param r2dbcEntityOperations  The reactive for performing database calls.
105 	 * @param queryExpression The query expression. The root object for evaluation context is a {@link SelectCreator}
106 	 * for delegation int the {@link StatementMapper#createSelect} fluent API.
107 	 */
108 	@SuppressWarnings("deprecation")
109 	public R2dbcMessageSource(R2dbcEntityOperations r2dbcEntityOperations, Expression queryExpression) {
110 		Assert.notNull(r2dbcEntityOperations, "'r2dbcEntityOperations' must not be null");
111 		Assert.notNull(queryExpression, "'queryExpression' must not be null");
112 		this.r2dbcEntityOperations = r2dbcEntityOperations;
113 		this.databaseClient = r2dbcEntityOperations.getDatabaseClient();
114 		this.statementMapper = r2dbcEntityOperations.getDataAccessStrategy().getStatementMapper();
115 		this.queryExpression = queryExpression;
116 	}
117 
118 	/**
119 	 * Set the type of the entityClass which is used for the {@link EntityRowMapper}.
120 	 * @param payloadType The class to use.
121 	 */
122 	public void setPayloadType(Class<?> payloadType) {
123 		Assert.notNull(payloadType, "'payloadType' must not be null");
124 		this.payloadType = payloadType;
125 	}
126 
127 	/**
128 	 * Set an update query that will be passed to the {@link DatabaseClient#sql(String)} method.
129 	 * @param updateSql the update query string.
130 	 */
131 	public void setUpdateSql(String updateSql) {
132 		this.updateSql = updateSql;
133 	}
134 
135 	/**
136 	 * Set a {@link BiFunction} which is used to bind parameters into the update query.
137 	 * @param bindFunction the {@link BiFunction} to use.
138 	 */
139 	@SuppressWarnings("unchecked")
140 	public void setBindFunction(
141 			BiFunction<DatabaseClient.GenericExecuteSpec, ?, DatabaseClient.GenericExecuteSpec> bindFunction) {
142 
143 		this.bindFunction =
144 				(BiFunction<DatabaseClient.GenericExecuteSpec, Object, DatabaseClient.GenericExecuteSpec>) bindFunction;
145 	}
146 
147 	/**
148 	 * The flag to manage which find* method to invoke on {@link R2dbcEntityOperations}.
149 	 * Default is 'false', which means the {@link #receive()} method will use
150 	 * the {@link DatabaseClient#sql(String)} method and will fetch all. If set
151 	 * to 'true'{@link #receive()} will use {@link DatabaseClient#sql(String)}
152 	 * and will fetch one and the payload of the returned {@link org.springframework.messaging.Message}
153 	 * will be the returned target Object of type
154 	 * identified by {@link #payloadType} instead of a List.
155 	 * @param expectSingleResult true if a single result is expected.
156 	 */
157 	public void setExpectSingleResult(boolean expectSingleResult) {
158 		this.expectSingleResult = expectSingleResult;
159 	}
160 
161 	@Override
162 	public String getComponentType() {
163 		return "r2dbc:inbound-channel-adapter";
164 	}
165 
166 	@Override
167 	protected void onInit() {
168 		this.evaluationContext = ExpressionUtils.createStandardEvaluationContext(getBeanFactory());
169 		if (!Map.class.isAssignableFrom(this.payloadType)) {
170 			this.rowMapper = new EntityRowMapper<>(this.payloadType, this.r2dbcEntityOperations.getConverter());
171 		}
172 		this.initialized = true;
173 	}
174 
175 	/**
176 	 * Execute a query returning its results as the Message payload.
177 	 * The payload can be either {@link reactor.core.publisher.Flux} or
178 	 * {@link reactor.core.publisher.Mono} of objects of type identified by {@link #payloadType},
179 	 * or a single element of type identified by {@link #payloadType}
180 	 * based on the value of {@link #expectSingleResult} attribute which defaults to 'false' resulting
181 	 * {@link org.springframework.messaging.Message} with payload of type
182 	 * {@link reactor.core.publisher.Flux}.
183 	 */
184 	@Override
185 	protected Object doReceive() {
186 		Assert.isTrue(this.initialized, "This class is not yet initialized. Invoke its afterPropertiesSet() method");
187 		Mono<RowsFetchSpec<?>> queryMono =
188 				Mono.fromSupplier(() ->
189 						this.queryExpression.getValue(this.evaluationContext, this.selectCreator))
190 						.map(this::prepareFetch);
191 		if (this.expectSingleResult) {
192 			return queryMono.flatMap(RowsFetchSpec::one)
193 					.flatMap(this::executeUpdate);
194 		}
195 
196 		return queryMono.flatMapMany(RowsFetchSpec::all)
197 				.flatMap(this::executeUpdate);
198 	}
199 
200 	private Mono<Object> executeUpdate(Object result) {
201 		if (this.updateSql != null) {
202 			DatabaseClient.GenericExecuteSpec genericExecuteSpec = this.databaseClient.sql(this.updateSql);
203 			if (this.bindFunction != null) {
204 				genericExecuteSpec = this.bindFunction.apply(genericExecuteSpec, result);
205 			}
206 			return genericExecuteSpec.then()
207 					.thenReturn(result);
208 		}
209 		return Mono.just(result);
210 	}
211 
212 	private RowsFetchSpec<?> prepareFetch(Object queryObject) {
213 		Supplier<String> query = evaluateQueryObject(queryObject);
214 		return this.databaseClient.sql(query)
215 				.map(this.rowMapper);
216 	}
217 
218 	private Supplier<String> evaluateQueryObject(Object queryObject) {
219 		if (queryObject instanceof String) {
220 			return () -> (String) queryObject;
221 		}
222 		else if (queryObject instanceof StatementMapper.SelectSpec) {
223 			return this.statementMapper.getMappedObject((StatementMapper.SelectSpec) queryObject);
224 		}
225 		throw new IllegalStateException("'queryExpression' must evaluate to String " +
226 				"or org.springframework.data.r2dbc.core.StatementMapper.SelectSpec, but not: " + queryObject);
227 	}
228 
229 	/**
230 	 * An instance of this class is used as a root object for query expression
231 	 * to give a limited access only to the {@link StatementMapper#createSelect} fluent API.
232 	 */
233 	public class SelectCreator {
234 
235 		SelectCreator() {
236 		}
237 
238 		public StatementMapper.SelectSpec createSelect(String table) {
239 			return R2dbcMessageSource.this.statementMapper.createSelect(table);
240 		}
241 
242 		public StatementMapper.SelectSpec createSelect(SqlIdentifier table) {
243 			return R2dbcMessageSource.this.statementMapper.createSelect(table);
244 		}
245 
246 	}
