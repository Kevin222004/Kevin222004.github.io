362 			}

364 				((BeanFactoryAware) this.requestMapper).setBeanFactory(beanFactory);
365 			}
366 			this.messageConverter.setBeanFactory(beanFactory);
367 		}
368 		this.initialized = true;
369 	}

417 	protected void send(Object object) {
418 		initializeIfNecessary();
419 		Assert.notNull(object, "request must not be null");
420 		MessageChannel channel = getRequestChannel();
421 		Assert.state(channel != null,
422 				"send is not supported, because no request channel has been configured");
423 		SampleFacade sample = null;
424 		if (this.metricsCaptor != null) {
425 			sample = this.metricsCaptor.start();
426 		}
427 		try {
428 			this.messagingTemplate.convertAndSend(channel, object, this.historyWritingPostProcessor);
429 			if (sample != null) {
430 				sample.stop(sendTimer());
431 			}
432 		}
433 		catch (Exception e) {
434 			if (sample != null) {
435 				sample.stop(buildSendTimer(false, e.getClass().getSimpleName()));
436 			}
437 			MessageChannel errorChan = getErrorChannel();
438 			if (errorChan != null) {
439 				this.messagingTemplate.send(errorChan, new ErrorMessage(e));
440 			}
441 			else {
442 				rethrow(e, "failed to send message");
443 			}
444 		}
445 	}

494 	@Nullable // NOSONAR
495 	private Object doSendAndReceive(Object object, boolean shouldConvert) {
496 		initializeIfNecessary();
497 		Assert.notNull(object, "request must not be null");
498 		MessageChannel channel = getRequestChannel();
499 		if (channel == null) {
500 			throw new MessagingException("No request channel available. Cannot send request message.");
501 		}
502 
503 		registerReplyMessageCorrelatorIfNecessary();
504 
505 		Object reply;
506 		Message<?> requestMessage = null;
507 		SampleFacade sample = null;
508 		try {
509 			if (this.metricsCaptor != null) {
510 				sample = this.metricsCaptor.start();
511 			}
512 			if (shouldConvert) {
513 				reply = this.messagingTemplate.convertSendAndReceive(channel, object, Object.class,
514 						this.historyWritingPostProcessor);
515 			}
516 			else {
517 				requestMessage = (object instanceof Message<?>)
518 						? (Message<?>) object : this.requestMapper.toMessage(object);
519 				Assert.state(requestMessage != null, () -> "request mapper resulted in no message for " + object);
520 				requestMessage = this.historyWritingPostProcessor.postProcessMessage(requestMessage);
521 				reply = this.messagingTemplate.sendAndReceive(channel, requestMessage);
522 			}
523 
524 			if (reply == null && this.errorOnTimeout) {
525 				throwMessageTimeoutException(object, "No reply received within timeout");
526 			}
527 			if (sample != null) {
528 				sample.stop(sendTimer());
529 			}
530 		}
531 		catch (Throwable ex) { // NOSONAR (catch throwable)
532 			if (logger.isDebugEnabled()) {
533 				logger.debug("failure occurred in gateway sendAndReceive: " + ex.getMessage());
534 			}
535 			reply = ex;
536 			if (sample != null) {
537 				sample.stop(buildSendTimer(false, ex.getClass().getSimpleName()));
538 			}
539 		}
540 
541 		if (reply instanceof Throwable || reply instanceof ErrorMessage) {
542 			Throwable error =
543 					reply instanceof ErrorMessage
544 							? ((ErrorMessage) reply).getPayload()
545 							: (Throwable) reply;
546 			return handleSendAndReceiveError(object, requestMessage, error, shouldConvert);
547 		}
548 		return reply;
549 	}
550 
551 	@Nullable
552 	private Object handleSendAndReceiveError(Object object, @Nullable Message<?> requestMessage, Throwable error,
553 			boolean shouldConvert) {
554 
555 		MessageChannel errorChan = getErrorChannel();
556 		if (errorChan != null) {
557 			ErrorMessage errorMessage = buildErrorMessage(requestMessage, error);
558 			Message<?> errorFlowReply = sendErrorMessageAndReceive(errorChan, errorMessage);
559 			if (errorFlowReply == null && this.errorOnTimeout) {
560 				throwMessageTimeoutException(object, "No reply received from error channel within timeout");
561 				return null; // unreachable
562 			}
563 			else {
564 				return shouldConvert && errorFlowReply != null
565 						? errorFlowReply.getPayload()
566 						: errorFlowReply;
567 			}
568 		}
569 		else if (error instanceof Error) {
570 			throw (Error) error;
571 		}
572 		else {
573 			Throwable errorToReThrow = error;
574 			if (error instanceof MessagingException &&
575 					requestMessage != null && requestMessage.getHeaders().getErrorChannel() != null) {
576 				// We are in nested flow where upstream expects errors in its own errorChannel header.
577 				errorToReThrow = new MessageHandlingException(requestMessage, error);
578 			}
579 			rethrow(errorToReThrow, "gateway received checked Exception");
580 			return null; // unreachable
581 		}
582 	}

584 	@Nullable
585 	private Message<?> sendErrorMessageAndReceive(MessageChannel errorChan, ErrorMessage errorMessage) {
586 		Message<?> errorFlowReply;
587 		try {
588 			errorFlowReply = this.messagingTemplate.sendAndReceive(errorChan, errorMessage);
589 		}
590 		catch (Exception errorFlowFailure) {
591 			throw new MessagingException(errorMessage, "failure occurred in error-handling flow",
592 					errorFlowFailure);
593 		}
594 		if (errorFlowReply != null && errorFlowReply.getPayload() instanceof Throwable) {
595 			rethrow((Throwable) errorFlowReply.getPayload(), "error flow returned an Error Message");
596 		}
597 		return errorFlowReply;
598 	}

609 	protected Mono<Message<?>> sendAndReceiveMessageReactive(Object object) {
610 		initializeIfNecessary();
611 		Assert.notNull(object, "request must not be null");
612 		MessageChannel channel = getRequestChannel();
613 		if (channel == null) {
614 			throw new MessagingException("No request channel available. Cannot send request message.");
615 		}
616 
617 		registerReplyMessageCorrelatorIfNecessary();
618 
619 		return doSendAndReceiveMessageReactive(channel, object, false);
620 	}
621 
622 	private Mono<Message<?>> doSendAndReceiveMessageReactive(MessageChannel requestChannel, Object object,
623 			boolean error) {
624 
625 		Message<?> requestMessage;
626 		try {
627 			Message<?> message =
628 					object instanceof Message<?>
629 							? (Message<?>) object
630 							: this.requestMapper.toMessage(object);
631 			Assert.state(message != null, () -> "request mapper resulted in no message for " + object);
632 			message = this.historyWritingPostProcessor.postProcessMessage(message);
633 			requestMessage = message;
634 		}
635 		catch (Exception e) {
636 			throw new MessageMappingException("Cannot map to message: " + object, e);
637 		}
638 
639 		return Mono.defer(() -> {
640 			Object originalReplyChannelHeader = requestMessage.getHeaders().getReplyChannel();
641 			Object originalErrorChannelHeader = requestMessage.getHeaders().getErrorChannel();
642 
643 			MonoReplyChannel replyChan = new MonoReplyChannel();
644 
645 			Message<?> messageToSend = MutableMessageBuilder.fromMessage(requestMessage)
646 					.setReplyChannel(replyChan)
647 					.setHeader(this.messagingTemplate.getSendTimeoutHeader(), null)
648 					.setHeader(this.messagingTemplate.getReceiveTimeoutHeader(), null)
649 					.setErrorChannel(replyChan)
650 					.build();
651 
652 			sendMessageForReactiveFlow(requestChannel, messageToSend);
653 
654 			return buildReplyMono(requestMessage, replyChan.replyMono.asMono(), error, originalReplyChannelHeader,
655 					originalErrorChannelHeader);
656 		})
657 				.onErrorResume(t -> error ? Mono.error(t) : handleSendError(requestMessage, t));
658 	}

660 	private void sendMessageForReactiveFlow(MessageChannel requestChannel, Message<?> requestMessage) {
661 		if (requestChannel instanceof ReactiveStreamsSubscribableChannel) {
662 			((ReactiveStreamsSubscribableChannel) requestChannel)
663 					.subscribeTo(Mono.just(requestMessage));
664 		}
665 		else {
666 			long sendTimeout = sendTimeout(requestMessage);
667 
668 			boolean sent =
669 					sendTimeout >= 0
670 							? requestChannel.send(requestMessage, sendTimeout)
671 							: requestChannel.send(requestMessage);
672 
673 			if (!sent) {
674 				throw new MessageDeliveryException(requestMessage,
675 						"Failed to send message to channel '" + requestChannel +
676 								"' within timeout: " + sendTimeout);
677 			}
678 		}
679 	}

681 	private Mono<Message<?>> buildReplyMono(Message<?> requestMessage, Mono<Message<?>> reply, boolean error,
682 			@Nullable Object originalReplyChannelHeader, @Nullable Object originalErrorChannelHeader) {
683 
684 		MetricsCaptor captor = this.metricsCaptor;
685 		return reply
686 				.doOnSubscribe(s -> {
687 					if (!error && captor != null) {
688 						captor.start().stop(sendTimer());
689 					}
690 				})
691 				.<Message<?>>map(replyMessage -> {
692 					if (!error && replyMessage instanceof ErrorMessage) {
693 						ErrorMessage em = (ErrorMessage) replyMessage;
694 						if (em.getPayload() instanceof MessagingException) {
695 							throw (MessagingException) em.getPayload();
696 						}
697 						else {
698 							throw new MessagingException(requestMessage, em.getPayload());
699 						}
700 					}
701 					else {
702 						return MessageBuilder.fromMessage(replyMessage)
703 								.setHeader(MessageHeaders.REPLY_CHANNEL, originalReplyChannelHeader)
704 								.setHeader(MessageHeaders.ERROR_CHANNEL, originalErrorChannelHeader)
705 								.build();
706 					}
707 				});
708 	}

710 	private Mono<Message<?>> handleSendError(Message<?> requestMessage, Throwable exception) {
711 		if (logger.isDebugEnabled()) {
712 			logger.debug("failure occurred in gateway sendAndReceiveReactive: " + exception.getMessage());
713 		}
714 		MessageChannel channel = getErrorChannel();
715 		if (channel != null) {
716 			ErrorMessage errorMessage = buildErrorMessage(requestMessage, exception);
717 			try {
718 				return doSendAndReceiveMessageReactive(channel, errorMessage, true);
719 			}
720 			catch (Exception errorFlowFailure) {
721 				throw new MessagingException(errorMessage, "failure occurred in error-handling flow",
722 						errorFlowFailure);
723 			}
724 		}
725 		else {
726 			// no errorChannel so we'll propagate
727 			throw wrapExceptionIfNecessary(exception, "gateway received checked Exception");
728 		}
729 	}

732 		if (this.successTimer == null) {
733 			this.successTimer = buildSendTimer(true, "none");
734 		}
735 		return this.successTimer;
736 	}

738 	protected TimerFacade buildSendTimer(boolean success, String exception) {
739 		TimerFacade timer = this.metricsCaptor.timerBuilder(SEND_TIMER_NAME)
740 				.tag("type", "source")
741 				.tag("name", getComponentName() == null ? "unknown" : getComponentName())
742 				.tag("result", success ? "success" : "failure")
743 				.tag("exception", exception)
744 				.description("Send processing time")
745 				.build();
746 		this.timers.add(timer);
747 		return timer;

750 	private long sendTimeout(Message<?> requestMessage) {
751 		Long sendTimeout = headerToLong(requestMessage.getHeaders()
752 				.get(this.messagingTemplate.getSendTimeoutHeader()));
753 		return (sendTimeout != null ? sendTimeout : this.messagingTemplate.getSendTimeout());
754 	}

756 	@Nullable
757 	private Long headerToLong(@Nullable Object headerValue) {
758 		if (headerValue instanceof Number) {
759 			return ((Number) headerValue).longValue();
760 		}
761 		else if (headerValue instanceof String) {
762 			return Long.parseLong((String) headerValue);
763 		}
764 		else {
765 			return null;
766 		}
767 	}

806 	protected void registerReplyMessageCorrelatorIfNecessary() {
807 		MessageChannel replyChan = getReplyChannel();
808 		if (replyChan != null && this.replyMessageCorrelator == null) {
809 			synchronized (this.replyMessageCorrelatorMonitor) {
810 				if (this.replyMessageCorrelator != null) {
811 					return;
812 				}
813 				AbstractEndpoint correlator;
814 				BridgeHandler handler = new BridgeHandler();
815 				BeanFactory beanFactory = getBeanFactory();
816 				if (beanFactory != null) {
817 					handler.setBeanFactory(beanFactory);
818 				}
819 				handler.afterPropertiesSet();
820 				if (replyChan instanceof SubscribableChannel) {
821 					correlator = new EventDrivenConsumer((SubscribableChannel) replyChan, handler);
822 				}
823 				else if (replyChan instanceof PollableChannel) {
824 					PollingConsumer endpoint = new PollingConsumer((PollableChannel) replyChan, handler);
825 					endpoint.setReceiveTimeout(this.replyTimeout);
826 					correlator = endpoint;
827 				}
828 				else if (replyChan instanceof ReactiveStreamsSubscribableChannel) {
829 					correlator = new ReactiveStreamsConsumer(replyChan, (Subscriber<Message<?>>) handler);
830 				}
831 				else {
832 					throw new MessagingException("Unsupported 'replyChannel' type [" + replyChan.getClass() + "]."
833 							+ "SubscribableChannel or PollableChannel type are supported.");
834 				}
835 
836 				if (beanFactory != null) {
837 					correlator.setBeanFactory(beanFactory);
838 				}
839 				correlator.afterPropertiesSet();
840 				this.replyMessageCorrelator = correlator;
841 			}
842 			if (isRunning()) {
843 				this.replyMessageCorrelator.start();
844 			}
845 		}
846 	}
