161 	@Test
162 	void testKafkaAdapters() throws Exception {
163 		this.sendToKafkaFlowInput.send(new GenericMessage<>("foo", Collections.singletonMap("foo", "bar")));
164 
165 		assertThat(TestUtils.getPropertyValue(this.kafkaProducer1, "headerMapper")).isSameAs(this.mapper);
166 
167 		for (int i = 0; i < 100; i++) {
168 			Message<?> receive = this.listeningFromKafkaResults1.receive(20000);
169 			assertThat(receive).isNotNull();
170 			assertThat(receive.getPayload()).isEqualTo("FOO");
171 			MessageHeaders headers = receive.getHeaders();
172 			assertThat(headers.containsKey(KafkaHeaders.ACKNOWLEDGMENT)).isTrue();
173 			Acknowledgment acknowledgment = headers.get(KafkaHeaders.ACKNOWLEDGMENT, Acknowledgment.class);
174 			acknowledgment.acknowledge();
175 			assertThat(headers.get(KafkaHeaders.RECEIVED_TOPIC)).isEqualTo(TEST_TOPIC1);
176 			assertThat(headers.get(KafkaHeaders.RECEIVED_MESSAGE_KEY)).isEqualTo(i + 1);
177 			assertThat(headers.get(KafkaHeaders.RECEIVED_PARTITION_ID)).isEqualTo(0);
178 			assertThat(headers.get(KafkaHeaders.OFFSET)).isEqualTo((long) i);
179 			assertThat(headers.get(KafkaHeaders.TIMESTAMP_TYPE)).isEqualTo("CREATE_TIME");
180 			assertThat(headers.get(KafkaHeaders.RECEIVED_TIMESTAMP)).isEqualTo(1487694048633L);
181 			assertThat(headers.get("foo")).isEqualTo("bar");
182 		}
183 
184 		for (int i = 0; i < 100; i++) {
185 			Message<?> receive = this.listeningFromKafkaResults2.receive(20000);
186 			assertThat(receive).isNotNull();
187 			assertThat(receive.getPayload()).isEqualTo("FOO");
188 			MessageHeaders headers = receive.getHeaders();
189 			assertThat(headers.containsKey(KafkaHeaders.ACKNOWLEDGMENT)).isTrue();
190 			Acknowledgment acknowledgment = headers.get(KafkaHeaders.ACKNOWLEDGMENT, Acknowledgment.class);
191 			acknowledgment.acknowledge();
192 			assertThat(headers.get(KafkaHeaders.RECEIVED_TOPIC)).isEqualTo(TEST_TOPIC2);
193 			assertThat(headers.get(KafkaHeaders.RECEIVED_MESSAGE_KEY)).isEqualTo(i + 1);
194 			assertThat(headers.get(KafkaHeaders.RECEIVED_PARTITION_ID)).isEqualTo(0);
195 			assertThat(headers.get(KafkaHeaders.OFFSET)).isEqualTo((long) i);
196 			assertThat(headers.get(KafkaHeaders.TIMESTAMP_TYPE)).isEqualTo("CREATE_TIME");
197 			assertThat(headers.get(KafkaHeaders.RECEIVED_TIMESTAMP)).isEqualTo(1487694048644L);
198 		}
199 
200 		Message<String> message = MessageBuilder.withPayload("BAR").setHeader(KafkaHeaders.TOPIC, TEST_TOPIC2).build();
201 
202 		this.sendToKafkaFlowInput.send(message);
203 
204 		assertThat(this.listeningFromKafkaResults1.receive(10)).isNull();
205 
206 		Message<?> error = this.errorChannel.receive(10000);
207 		assertThat(error).isNotNull();
208 		assertThat(error).isInstanceOf(ErrorMessage.class);
209 		assertThat(error.getPayload()).isInstanceOf(MessageRejectedException.class);
210 
211 		assertThat(this.messageListenerContainer).isNotNull();
212 		assertThat(this.kafkaTemplateTopic1).isNotNull();
213 		assertThat(this.kafkaTemplateTopic2).isNotNull();
214 
215 		this.kafkaTemplateTopic1.send(TEST_TOPIC3, "foo");
216 		assertThat(this.config.sourceFlowLatch.await(10, TimeUnit.SECONDS)).isTrue();
217 		assertThat(this.config.fromSource).isEqualTo("foo");
218 
219 		assertThat(this.config.onPartitionsAssignedCalledLatch.await(10, TimeUnit.SECONDS)).isTrue();
220 	}

274 		public IntegrationFlow topic1ListenerFromKafkaFlow() {
275 			return IntegrationFlows
276 					.from(Kafka.messageDrivenChannelAdapter(consumerFactory(),
277 							KafkaMessageDrivenChannelAdapter.ListenerMode.record, TEST_TOPIC1)
278 							.configureListenerContainer(c ->
279 									c.ackMode(ContainerProperties.AckMode.MANUAL)
280 											.idleEventInterval(100L)
281 											.id("topic1ListenerContainer"))
282 							.recoveryCallback(new ErrorMessageSendingRecoverer(errorChannel(),
283 									new RawRecordHeaderErrorMessageStrategy()))
284 							.retryTemplate(new RetryTemplate())
285 							.filterInRetry(true)
286 							.onPartitionsAssignedSeekCallback((map, callback) ->
287 									ContextConfiguration.this.onPartitionsAssignedCalledLatch.countDown()))
288 					.filter(Message.class, m ->
289 									m.getHeaders().get(KafkaHeaders.RECEIVED_MESSAGE_KEY, Integer.class) < 101,
290 							f -> f.throwExceptionOnRejection(true))
291 					.<String, String>transform(String::toUpperCase)
292 					.channel(c -> c.queue("listeningFromKafkaResults1"))
293 					.get();
294 		}

352 		private KafkaProducerMessageHandlerSpec<Integer, String, ?> kafkaMessageHandler(
353 				ProducerFactory<Integer, String> producerFactory, String topic) {
354 
355 			return Kafka
356 					.outboundChannelAdapter(producerFactory)
357 					.sync(true)
358 					.messageKey(m -> m
359 							.getHeaders()
360 							.get(IntegrationMessageHeaderAccessor.SEQUENCE_NUMBER))
361 					.headerMapper(mapper())
362 					.partitionId(m -> 0)
363 					.topicExpression("headers[kafka_topic] ?: '" + topic + "'")
364 					.configureKafkaTemplate(t -> t.id("kafkaTemplate:" + topic));
365 		}
