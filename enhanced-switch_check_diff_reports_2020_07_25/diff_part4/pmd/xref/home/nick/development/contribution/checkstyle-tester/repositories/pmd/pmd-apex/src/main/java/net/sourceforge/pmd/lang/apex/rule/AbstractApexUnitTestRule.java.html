5   package net.sourceforge.pmd.lang.apex.rule;
6   
7   import net.sourceforge.pmd.lang.apex.ast.ASTModifierNode;
8   import net.sourceforge.pmd.lang.apex.ast.ASTUserClass;
9   import net.sourceforge.pmd.lang.apex.ast.ApexNode;
10  
11  import apex.jorje.services.Version;
12  
13  /**
14   * Do special checks for apex unit test classes and methods
15   * 
16   * @author a.subramanian
17   */
18  public abstract class AbstractApexUnitTestRule extends AbstractApexRule {
19  
20      public AbstractApexUnitTestRule() {
21          setProperty(CODECLIMATE_CATEGORIES, "Bug Risk");
22          setProperty(CODECLIMATE_REMEDIATION_MULTIPLIER, 100);
23          setProperty(CODECLIMATE_BLOCK_HIGHLIGHTING, false);
24      }
25  
26      /**
27       * Don't bother visiting this class if it's not a class with @isTest and
28       * newer than API v24 (V176 internal).
29       */
30      @Override
31      public Object visit(final ASTUserClass node, final Object data) {
32          if (!isTestMethodOrClass(node) && node.getApexVersion() >= Version.V176.getExternal()) {
33              return data;
34          }
35          return super.visit(node, data);
36      }
37  
38      protected boolean isTestMethodOrClass(final ApexNode<?> node) {
39          final ASTModifierNode modifierNode = node.getFirstChildOfType(ASTModifierNode.class);
40          return modifierNode != null && modifierNode.isTest();
41      }
42  }
