60  @Disabled("Requires MongoDb server of version 4.x started with 'replSet' option.")
61  @SpringJUnitConfig
62  @DirtiesContext
63  public class MongoDbChangeStreamMessageProducerTests {
64  
65  	private static final String CONNECTION_STRING =
66  			"mongodb://127.0.0.1:27017/?replicaSet=rs0&w=majority&uuidrepresentation=javaLegacy";
67  
68  	private static MongoClient mongoClient;
69  
70  	@Autowired
71  	ReactiveMongoOperations mongoTemplate;
72  
73  	@Autowired
74  	FluxMessageChannel fluxMessageChannel;
75  
76  	@Autowired
77  	MessageProducerSupport changeStreamMessageProducer;
78  
79  	@BeforeAll
80  	static void setup() {
81  		mongoClient = MongoClients.create(CONNECTION_STRING);
82  	}
83  
84  	@AfterAll
85  	static void tearDown() {
86  		mongoClient.close();
87  	}
88  
89  	@BeforeEach
90  	void prepare() {
91  		this.mongoTemplate.remove(Person.class).all()
92  				.then()
93  				.onErrorResume(ex -> this.mongoTemplate.createCollection(Person.class).then())
94  				.block(Duration.ofSeconds(10));
95  	}
96  
97  	@AfterEach
98  	void cleanUp() {
99  		this.mongoTemplate.remove(Person.class).all()
100 				.block(Duration.ofSeconds(10));
101 	}
102 
103 	@Test
104 	void testChangeStreamMessageProducer() {
105 		Person person1 = new Person("John", 38);
106 		Person person2 = new Person("Josh", 39);
107 		Person person3 = new Person("Jack", 37);
108 
109 		this.changeStreamMessageProducer.start();
110 
111 		StepVerifier stepVerifier =
112 				Flux.from(this.fluxMessageChannel)
113 						.as(StepVerifier::create)
114 						.assertNext((message) -> {
115 							assertThat(message.getPayload())
116 									.isInstanceOf(ChangeStreamEvent.class)
117 									.extracting("body")
118 									.asInstanceOf(InstanceOfAssertFactories.type(Person.class))
119 									.isEqualTo(person1);
120 							assertThat(message.getHeaders())
121 									.containsEntry(MongoHeaders.COLLECTION_NAME, "person")
122 									.containsEntry(MongoHeaders.CHANGE_STREAM_OPERATION_TYPE, OperationType.INSERT)
123 									.containsKeys(MongoHeaders.CHANGE_STREAM_TIMESTAMP,
124 											MongoHeaders.CHANGE_STREAM_RESUME_TOKEN);
125 						})
126 						.assertNext((message) ->
127 								assertThat(message.getPayload())
128 										.extracting("body")
129 										.isEqualTo(person2))
130 						.assertNext((message) ->
131 								assertThat(message.getPayload())
132 										.extracting("body")
133 										.isEqualTo(person3))
134 						.thenCancel()
135 						.verifyLater();
136 
137 		Flux.concat(this.mongoTemplate.insert(person1),
138 				this.mongoTemplate.insert(person2),
139 				this.mongoTemplate.insert(person3))
140 				.as(StepVerifier::create)
141 				.expectNextCount(3)
142 				.verifyComplete();
143 
144 		stepVerifier.verify(Duration.ofSeconds(10));
145 
146 		this.changeStreamMessageProducer.stop();
147 	}
148 
149 	@Configuration
150 	@EnableIntegration
151 	static class Config {
152 
153 		@Bean
154 		ReactiveMongoOperations mongoTemplate() {
155 			return new ReactiveMongoTemplate(mongoClient, "test");
156 		}
157 
158 		@Bean
159 		IntegrationFlow changeStreamFlow() {
160 			return IntegrationFlows.from(
161 					MongoDb.changeStreamInboundChannelAdapter(mongoTemplate())
162 							.domainType(Person.class)
163 							.collection("person")
164 							.extractBody(false)
165 							.autoStartup(false)
166 							.shouldTrack(true))
167 					.channel(MessageChannels.flux())
168 					.get();
169 		}
170 
171 	}
