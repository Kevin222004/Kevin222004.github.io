17  package org.springframework.integration.webflux.config;
18  
19  import org.w3c.dom.Element;
20  
21  import org.springframework.beans.factory.config.RuntimeBeanReference;
22  import org.springframework.beans.factory.support.BeanDefinitionBuilder;
23  import org.springframework.beans.factory.xml.ParserContext;
24  import org.springframework.integration.config.ExpressionFactoryBean;
25  import org.springframework.integration.config.xml.IntegrationNamespaceUtils;
26  import org.springframework.integration.http.config.HttpOutboundChannelAdapterParser;
27  import org.springframework.integration.webflux.outbound.WebFluxRequestExecutingMessageHandler;
28  import org.springframework.util.StringUtils;
29  
30  /**
31   * Parser for the 'outbound-channel-adapter' element of the webflux namespace.
32   *
33   * @author Artem Bilan
34   *
35   * @since 5.0
36   */
37  public class WebFluxOutboundChannelAdapterParser extends HttpOutboundChannelAdapterParser {
38  
39  	@Override
40  	protected BeanDefinitionBuilder getBuilder(Element element, ParserContext parserContext) {
41  		return buildWebFluxRequestExecutingMessageHandler(element, parserContext);
42  	}
43  
44  	static BeanDefinitionBuilder buildWebFluxRequestExecutingMessageHandler(Element element,
45  			ParserContext parserContext) {
46  
47  		BeanDefinitionBuilder builder =
48  				BeanDefinitionBuilder.genericBeanDefinition(WebFluxRequestExecutingMessageHandler.class);
49  
50  		String webClientRef = element.getAttribute("web-client");
51  		if (StringUtils.hasText(webClientRef)) {
52  			builder.getBeanDefinition()
53  					.getConstructorArgumentValues()
54  					.addIndexedArgumentValue(1, new RuntimeBeanReference(webClientRef));
55  		}
56  		else {
57  			IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, "encoding-mode");
58  		}
59  
60  		String type = element.getAttribute("publisher-element-type");
61  		String typeExpression = element.getAttribute("publisher-element-type-expression");
62  
63  		boolean hasType = StringUtils.hasText(type);
64  		boolean hasTypeExpression = StringUtils.hasText(typeExpression);
65  
66  		if (hasType && hasTypeExpression) {
67  			parserContext.getReaderContext()
68  					.error("The 'publisher-element-type' and 'publisher-element-type-expression' " +
69  							"are mutually exclusive. You can only have one or the other", element);
70  		}
71  
72  		if (hasType) {
73  			builder.addPropertyValue("publisherElementType", type);
74  		}
75  		else if (hasTypeExpression) {
76  			builder.addPropertyValue("publisherElementTypeExpression",
77  					BeanDefinitionBuilder.rootBeanDefinition(ExpressionFactoryBean.class)
78  							.addConstructorArgValue(typeExpression)
79  							.getBeanDefinition());
80  		}
81  
82  		return builder;
83  	}
84  
85  }
