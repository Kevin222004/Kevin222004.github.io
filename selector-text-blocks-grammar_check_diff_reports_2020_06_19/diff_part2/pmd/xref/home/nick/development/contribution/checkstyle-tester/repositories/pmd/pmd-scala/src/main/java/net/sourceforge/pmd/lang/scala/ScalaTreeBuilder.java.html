303     // The nodes having children built.
304     private Stack<Node> nodes = new Stack<>();
305 
306     private static <T extends Tree> void register(Class<T> nodeType,
307             Class<? extends ScalaNode<T>> nodeAdapterType) {
308         try {
309             NODE_TYPE_TO_NODE_ADAPTER_TYPE.put(nodeType, nodeAdapterType.getConstructor(nodeType));
310         } catch (SecurityException | NoSuchMethodException e) {
311             throw new RuntimeException(e);
312         }
313     }
314 
315     @SuppressWarnings("unchecked")
316     private static <T extends Tree> ScalaNode<T> createNodeAdapter(T node) {
317         try {
318 
319             Constructor<? extends ScalaNode<T>> constructor = null;
320 
321             // This isInstance is unfortunately necessary as Scala gives us
322             // access to the Interface (Trait) of classes at compile time, but
323             // at runtime only operates using a synthetic Impl class. So at
324             // runtime, Case.class is really CaseImpl.class due to the
325             // translation between Scala Traits and Java Classes
326             for (Class<?> treeClass : NODE_TYPE_TO_NODE_ADAPTER_TYPE.keySet()) {
327                 if (treeClass.isInstance(node)) {
328                     constructor = (Constructor<? extends ScalaNode<T>>) NODE_TYPE_TO_NODE_ADAPTER_TYPE
329                             .get(treeClass);
330                 }
331             }
332 
333             if (constructor == null) {
334                 throw new IllegalArgumentException(
335                         "There is no Node adapter class registered for the Node class: " + node.getClass());
336             }
337             return constructor.newInstance(node);
338 
339         } catch (InstantiationException | IllegalAccessException e) {
340             throw new RuntimeException(e);
341         } catch (InvocationTargetException e) {
342             throw new RuntimeException(e.getTargetException());
343         }
344     }
345 
346     /**
347      * Construct a matching tree that implements the PMD Node interface.
348      *
349      * @param <T>
350      *            the scala node that extends the Tree trait
351      * @param astNode
352      *            the Java node that extends the PMD Node interface
353      * @return a PMD compatible node representing the Scala AST node
354      */
355     <T extends Tree> ScalaNode<T> build(T astNode) {
356         return buildInternal(astNode);
357     }
358 
359     private <T extends Tree> ScalaNode<T> buildInternal(T astNode) {
360         // Create a Node
361         ScalaNode<T> node = createNodeAdapter(astNode);
362         // Append to parent
363         Node parent = nodes.isEmpty() ? null : nodes.peek();
364         if (parent != null) {
365             parent.jjtAddChild(node, parent.getNumChildren());
366             node.jjtSetParent(parent);
367         }
368 
369         // Build the children...
370         nodes.push(node);
371         int childrenNum = astNode.children().size();
372         for (int i = 0; i < childrenNum; i++) {
373             buildInternal(astNode.children().apply(i));
374         }
375         nodes.pop();
376 
377         return node;
378     }
