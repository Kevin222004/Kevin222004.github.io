46      /**
47       * Add elements from the source to the target as long as they don't already
48       * exist there. Return the number of items actually added.
49       *
50       * @param source
51       * @param target
52       * @return int
53       */
54      public static <T> int addWithoutDuplicates(Collection<T> source, Collection<T> target) {
55  
56          int added = 0;
57  
58          for (T item : source) {
59              if (target.contains(item)) {
60                  continue;
61              }
62              target.add(item);
63              added++;
64          }
65  
66          return added;
67      }

69      /**
70       * Returns the collection type if we recognize it by its short name.
71       *
72       * @param shortName
73       *            String
74       * @return Class
75       */
76      public static Class<?> getCollectionTypeFor(String shortName) {
77          Class<?> cls = COLLECTION_CLASSES_BY_NAMES.typeFor(shortName);
78          if (cls != null) {
79              return cls;
80          }
81  
82          return COLLECTION_INTERFACES_BY_NAMES.typeFor(shortName);
83      }

87       * class or interface as specified.
88       *
89       * @param typeName
90       *            String
91       * @param includeInterfaces
92       *            boolean
93       * @return boolean
94       */
95      public static boolean isCollectionType(String typeName, boolean includeInterfaces) {
96  
97          if (COLLECTION_CLASSES_BY_NAMES.contains(typeName)) {

135     /**
136      * Creates and returns a map populated with the keyValuesSets where the
137      * value held by the tuples are they key and value in that order.
138      *
139      * @param keys
140      *            K[]
141      * @param values
142      *            V[]
143      * @return Map
144      */
145     public static <K, V> Map<K, V> mapFrom(K[] keys, V[] values) {
146         if (keys.length != values.length) {
147             throw new RuntimeException("mapFrom keys and values arrays have different sizes");
148         }
149         Map<K, V> map = new HashMap<>(keys.length);
150         for (int i = 0; i < keys.length; i++) {
151             map.put(keys[i], values[i]);
152         }
153         return map;
154     }

156     /**
157      * Returns a map based on the source but with the key &amp; values swapped.
158      *
159      * @param source
160      *            Map
161      * @return Map
162      */
163     public static <K, V> Map<V, K> invertedMapFrom(Map<K, V> source) {
164         Map<V, K> map = new HashMap<>(source.size());
165         for (Map.Entry<K, V> entry : source.entrySet()) {
166             map.put(entry.getValue(), entry.getKey());
167         }
168         return map;
169     }

172     /**
173      * Consumes all the elements of the iterator and
174      * returns a list containing them. The iterator is
175      * then unusable
176      *
177      * @param it An iterator
178      *
179      * @return a list containing the elements remaining
180      * on the iterator
181      */
182     public static <T> List<T> toList(Iterator<T> it) {
183         List<T> list = new ArrayList<>();
184         while (it.hasNext()) {
185             list.add(it.next());
186         }
187         return list;
188     }

213     /**
214      * Returns whether the arrays are equal by examining each of their elements,
215      * even if they are arrays themselves.
216      *
217      * @param thisArray
218      *            Object[]
219      * @param thatArray
220      *            Object[]
221      * @return boolean
222      * @deprecated {@link Arrays#deepEquals(Object[], Object[])}
223      */
224     @Deprecated
225     public static boolean valuesAreTransitivelyEqual(Object[] thisArray, Object[] thatArray) {
226         if (thisArray == thatArray) {
227             return true;
228         }
229         if (thisArray == null || thatArray == null) {
230             return false;
231         }
232         if (thisArray.length != thatArray.length) {
233             return false;
234         }
235         for (int i = 0; i < thisArray.length; i++) {
236             if (!areEqual(thisArray[i], thatArray[i])) {
237                 return false; // recurse if req'd
238             }
239         }
240         return true;
241     }

243     /**
244      * A comprehensive isEqual method that handles nulls and arrays safely.
245      *
246      * @param value
247      *            Object
248      * @param otherValue
249      *            Object
250      * @return boolean
251      * @deprecated {@link Objects#deepEquals(Object, Object)}
252      */
253     @Deprecated
254     @SuppressWarnings("PMD.CompareObjectsWithEquals")
255     public static boolean areEqual(Object value, Object otherValue) {
256         if (value == otherValue) {
257             return true;
258         }
259         if (value == null) {
260             return false;
261         }
262         if (otherValue == null) {
263             return false;
264         }
265 
266         if (value.getClass().getComponentType() != null) {
267             return arraysAreEqual(value, otherValue);
268         }
269         return value.equals(otherValue);
270     }

292     /**
293      * Returns true if both arrays are if both are null or have zero-length,
294      * otherwise return the false if their respective elements are not equal by
295      * position.
296      *
297      * @param <T>
298      * @param a
299      * @param b
300      * @return boolean
301      * @deprecated {@link Arrays#deepEquals(Object[], Object[])}
302      */
303     @Deprecated
304     public static <T> boolean areSemanticEquals(T[] a, T[] b) {
305         if (a == null) {
306             return b == null || b.length == 0;
307         }
308         if (b == null) {
309             return a.length == 0;
310         }
311 
312         if (a.length != b.length) {
313             return false;
314         }
315 
316         for (int i = 0; i < a.length; i++) {
317             if (!areEqual(a[i], b[i])) {
318                 return false;
319             }
320         }
321 
322         return true;
323     }

326     /**
327      * If the newValue is already held within the values array then the values
328      * array is returned, otherwise a new array is created appending the
329      * newValue to the end.
330      *
331      * @param <T>
332      * @param values
333      * @param newValue
334      * @return an array containing the union of values and newValue
335      */
336     @Deprecated
337     public static <T> T[] addWithoutDuplicates(T[] values, T newValue) {
338 
339         for (T value : values) {
340             if (value.equals(newValue)) {
341                 return values;
342             }
343         }
344 
345         T[] largerOne = (T[]) Array.newInstance(values.getClass().getComponentType(), values.length + 1);
346         System.arraycopy(values, 0, largerOne, 0, values.length);
347         largerOne[values.length] = newValue;
348         return largerOne;
349     }

351     /**
352      * Returns an array of values as a union set of the two input arrays.
353      *
354      * @param <T>
355      * @param values
356      * @param newValues
357      * @return the union of the two arrays
358      */
359     @Deprecated
360     public static <T> T[] addWithoutDuplicates(T[] values, T[] newValues) {
361 
362         Set<T> originals = new HashSet<>(values.length);
363         for (T value : values) {
364             originals.add(value);
365         }
366         List<T> newOnes = new ArrayList<>(newValues.length);
367         for (T value : newValues) {
368             if (originals.contains(value)) {
369                 continue;
370             }
371             newOnes.add(value);
372         }
373 
374         T[] largerOne = (T[]) Array.newInstance(values.getClass().getComponentType(), values.length + newOnes.size());
375         System.arraycopy(values, 0, largerOne, 0, values.length);
376         for (int i = values.length; i < largerOne.length; i++) {
377             largerOne[i] = newOnes.get(i - values.length);
378         }
379         return largerOne;
380     }
