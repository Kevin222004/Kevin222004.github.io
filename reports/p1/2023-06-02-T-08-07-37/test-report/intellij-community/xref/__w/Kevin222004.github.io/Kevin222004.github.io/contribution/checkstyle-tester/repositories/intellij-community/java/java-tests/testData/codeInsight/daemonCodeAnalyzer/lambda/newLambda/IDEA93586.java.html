1   interface Eff<A, B> {
2     B f(A a);
3   }

5   class Disfunction {
6     public static <A, B, C> Eff<C, B> apply(final Eff<C, Eff<A, B>> cab, final Eff<C, A> ca) {
7       return bind(cab, f -> compose(a -> f.f(a), ca));
8     }
9   
10    public static <A, B, C> Eff<C, B> bind(final Eff<C, A> ma, final Eff<A, Eff<C, B>> f) {
11      return m -> f.f(ma.f(m)).f(m);
12    }
13  
14    public static <A, B, C> Eff<A, C> compose(final Eff<B, C> f, final Eff<A, B> g) {
15      return a -> f.f(g.f(a));
16    }
17  }
