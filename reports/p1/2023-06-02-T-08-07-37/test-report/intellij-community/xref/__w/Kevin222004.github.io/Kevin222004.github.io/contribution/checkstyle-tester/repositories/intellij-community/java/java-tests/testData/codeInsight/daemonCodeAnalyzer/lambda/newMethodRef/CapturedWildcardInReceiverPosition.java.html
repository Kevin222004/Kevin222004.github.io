10  class Test {
11    static <T> Map<T, Integer> combine(Collection<Map<? extends T, Integer>> pMaps) {
12      Function<Map<? extends T, Integer>, Set<? extends Map.Entry<? extends T, Integer>>> entrySet = Map::entrySet;
13      return pMaps.stream()
14        .map(entrySet)
15        .flatMap(Collection::stream)
16        .collect(Collectors.toMap(Map.Entry::getKey, Map.Entry::getValue,
17                                  (pEntry1, pEntry2) -> pEntry1 + pEntry2,
18                                  (Supplier<HashMap<T, Integer>>) HashMap::new
19                 )
20        );
21    }
