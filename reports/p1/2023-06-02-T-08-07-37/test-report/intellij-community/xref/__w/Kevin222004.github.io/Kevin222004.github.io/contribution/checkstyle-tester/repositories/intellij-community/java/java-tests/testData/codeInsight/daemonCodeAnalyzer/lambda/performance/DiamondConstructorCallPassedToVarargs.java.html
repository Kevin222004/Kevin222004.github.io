4   interface Graph<T> extends Iterable<T> {
5     double distance(T from, T to);
6     T startVertex();
7     default Stream<T> stream() {
8       return StreamSupport.stream(spliterator(), false);
9     }
10  }
11  
12  class GeometricGraph<T extends Comparable<T>> implements Graph<GeometricGraph.Point<T>> {
13    private Point<T>[] vertices;
14    @SafeVarargs
15    public GeometricGraph(Point<T>... vertices) {
16      this.vertices = vertices;
17    }
18    @Override public double distance(Point<T> from, Point<T> to) {
19      double dx = to.x - from.x;
20      double dy = to.y - from.y;
21      return Math.sqrt(dx * dx + dy * dy);
22    }
23    @Override public Point<T> startVertex() {
24      return vertices[0];
25    }
26    @Override public Iterator<Point<T>> iterator() {
27      return Arrays.asList(vertices).iterator();
28    }
29    public static class Point<T extends Comparable<T>> implements Comparable<Point<T>> {
30      final T label;
31      final double x;
32      final double y;
33      public Point(T label, double x, double y) {
34        this.label = label;
35        this.x = x;
36        this.y = y;
37      }
38      @Override public String toString() {
39        return String.valueOf(label);
40      }
41      @Override public int compareTo(Point<T> o) {
42        return label.compareTo(o.label);
43      }
44    }
45  }
