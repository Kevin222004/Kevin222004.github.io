2   import java.util.function.Function;
3   import java.util.function.Consumer;
4   
5   import lombok.experimental.ExtensionMethod;
6   
7   @ExtensionMethod(ExtensionMethodFunctional.Extensions.class)
8   class ExtensionMethodFunctional {
9   	public void test() {
10  		String test = "test";
11  		test = test.map(s -> s.reverse());
12  		
13  		test.consume(s -> System.out.println("1: " + s), s -> System.out.println("2: " + s));
14  		test.consume(System.out::println, System.out::println);
15  	}
16  	
17  	static class Extensions {
18  		public static <T, R> R map(T value, Function<T, R> mapper) {
19  			return mapper.apply(value);
20  		}
21  		
22  		public static String reverse(String string) {
23  			return new StringBuilder(string).reverse().toString();
24  		}
25  
26  		@SafeVarargs
27  		public static <T> void consume(T o, Consumer<T>... consumer) {
28  			for (int i = 0; i < consumer.length; i++) {
29  				consumer[i].accept(o);
30  			}
31  		}
32  	}
