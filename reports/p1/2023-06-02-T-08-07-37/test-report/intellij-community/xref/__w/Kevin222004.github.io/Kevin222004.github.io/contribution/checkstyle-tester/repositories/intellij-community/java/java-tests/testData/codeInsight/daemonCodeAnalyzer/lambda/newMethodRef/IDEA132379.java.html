13  
14      Map<String, Set<Integer>> map2 =
15        map1.entrySet().stream()
16          .collect(Collectors.toMap(
17                     Map.Entry::getKey,
18                     entry -> entry.getValue().stream().map(String::length).collect(Collectors.toSet()),
19                     (s1, s2) -> {
20                       if (s1.equals(s2)) {
21                         throw new IllegalArgumentException("duplicate not allowed");
22                       }
23                       return null;
24                     },
25                     ConcurrentHashMap::new
26                   )
27          );
28  
29      System.out.println("map1: " + map1);
