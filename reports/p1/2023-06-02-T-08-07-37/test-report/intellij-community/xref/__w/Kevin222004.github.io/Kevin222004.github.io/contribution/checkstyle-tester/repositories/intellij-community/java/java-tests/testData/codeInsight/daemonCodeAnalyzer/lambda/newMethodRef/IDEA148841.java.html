16    public static <T, K, D, A, M extends Map<K, D>> Collector<T, ?, M> groupingByFlatKeys(Function<? super T, Stream<? extends K>> classifier,
17                                                                                          Supplier<M> mapFactory,
18                                                                                          Collector<? super T, A, D> downstream) {
19      return flatMapping(t -> classifier.apply(t).map(k -> new AbstractMap.SimpleEntry<>(k, t)),
20                         groupingBy(Map.Entry::getKey, mapFactory, mapping(Map.Entry::getValue, downstream)));
21    }
22  
23    public static <T, K, U> Collector<T, ?, Map<K, U>> toMapByFlatKeys(Function<? super T, Stream<? extends K>> keyMapper,
24                                                                       Function<? super T, ? extends U> valueMapper) {
25      return flatMapping(t -> keyMapper.apply(t).map(k -> new AbstractMap.SimpleEntry<>(k, t)),
26                         toMap(Map.Entry::getKey, valueMapper.compose(Map.Entry::getValue)));
27    }
28  
29    public static <T, U, A, R> Collector<T, ?, R> flatMapping(Function<? super T, Stream<? extends U>> mapper,
30                                                              Collector<? super U, A, R> downstream) {
31      BiConsumer<A, ? super U> downstreamAccumulator = downstream.accumulator();
32      return Collector.of(downstream.supplier(),
33                          (r, t) -> mapper.apply(t).forEach(v -> downstreamAccumulator.accept(r, v)),
34                          downstream.combiner(),
35                          downstream.finisher(),
36                          downstream.characteristics().toArray(new Collector.Characteristics[downstream.characteristics().size()]));
37    }
38  }
