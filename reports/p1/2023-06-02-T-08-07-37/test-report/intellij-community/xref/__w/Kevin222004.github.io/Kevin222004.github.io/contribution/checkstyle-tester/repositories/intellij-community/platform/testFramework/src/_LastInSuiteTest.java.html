29  @RunWith(JUnit38AssumeSupportRunner.class)
30  public class _LastInSuiteTest extends TestCase {
31  
32    @Override
33    protected void setUp() throws Exception {
34      super.setUp();
35      Disposer.setDebugMode(true);
36    }
37  
38    @Override
39    public String getName() {
40      return getTestName(super.getName());
41    }
42  
43    private static String getTestName(String name) {
44      String buildConf = System.getProperty("teamcity.buildConfName");
45      return buildConf == null ? name : name + "[" + buildConf + "]";
46    }
47  
48    public void testDynamicExtensions() {
49      boolean testDynamicExtensions = SystemProperties.getBooleanProperty("intellij.test.all.dynamic.extension.points", false);
50      Assume.assumeTrue("intellij.test.all.dynamic.extension.points is off, no dynamic extensions to test",
51                        !DynamicExtensionPointsTester.EXTENSION_POINTS_WHITE_LIST.isEmpty() || testDynamicExtensions);
52      DynamicExtensionPointsTester.checkDynamicExtensionPoints(_LastInSuiteTest::getTestName);
53    }
54  
55    public void testProjectLeak() {
56      TestApplicationManager.testProjectLeak();
57    }
58  
59    // should be run as late as possible to give the Languages the chance to instantiate as many of them as possible
60    public void testLanguagesHaveDifferentDisplayNames() {
61      Collection<Language> languages = Language.getRegisteredLanguages();
62      Map<String, Language> displayNames = new HashMap<>();
63      for (Language language : languages) {
64        Language prev = displayNames.put(language.getDisplayName(), language);
65        if (prev != null) {
66          fail("The languages '%s' (%s) and '%s' (%s) have the same display name '%s'"
67                 .formatted(prev, prev.getClass().getName(), language, language.getClass().getName(), language.getDisplayName()));
68        }
69      }
70    }
71  
72    public void testStatistics() {
73      long started = _FirstInSuiteTest.getSuiteStartTime();
74      if (started != 0) {
75        long testSuiteDuration = System.nanoTime() - started;
76        System.out.printf("##teamcity[buildStatisticValue key='ideaTests.totalTimeMs' value='%d']%n", testSuiteDuration / 1_000_000);
77      }
78      LightPlatformTestCase.reportTestExecutionStatistics();
79    }

83    }
