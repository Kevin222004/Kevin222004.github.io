1   import java.io.Writer;
2   import java.lang.reflect.ParameterizedType;
3   import java.lang.reflect.Type;
4   import java.util.List;
5   import java.util.Map;
6   import java.util.Set;
7   
8   abstract class Example {
9     @FunctionalInterface
10    public interface ObjectWriter<T> {
11      void writeObject(Writer writer, T object);
12    }
13  
14    protected abstract ObjectWriter<?> getWriter(Type type);
15  
16    protected abstract <T> void writeList(Writer out, List<T> list, ObjectWriter<T> valueWriter);
17  
18    protected abstract <T> void writeSet(Writer out, Set<T> set, ObjectWriter<T> valueWriter);
19  
20    protected abstract <K, V> void writeMap(Writer out, Map<K, V> map, ObjectWriter<K> keyWriter, ObjectWriter<V> valueWriter);
21  
22    @SuppressWarnings({"unchecked", "rawtypes"})
23    protected ObjectWriter<?> getCollectionWriter(ParameterizedType parameterizedType) {
24      Type rawType = parameterizedType.getRawType();
25      if (rawType instanceof Class) {
26        Class clazz = (Class)rawType;
27        if (List.class.isAssignableFrom(clazz)) {
28          Type elementType = parameterizedType.getActualTypeArguments()[0];
29          ObjectWriter elementWriter = getWriter(elementType);
30          //(List) is marked redundant
31          return (writer, object) -> writeList(writer, (List)object, elementWriter);
32        } else if (Set.class.isAssignableFrom(clazz)) {
33          Type elementType = parameterizedType.getActualTypeArguments()[0];
34          ObjectWriter elementWriter = getWriter(elementType);
35          //(Set) is marked redundant
36          return (writer, object) -> writeSet(writer, (Set)object, elementWriter);
37        } else if (Map.class.isAssignableFrom(clazz)) {
38          Type keyType = parameterizedType.getActualTypeArguments()[0];
39          ObjectWriter keyWriter = getWriter(keyType);
40          Type valueType = parameterizedType.getActualTypeArguments()[1];
41          ObjectWriter valueWriter = getWriter(valueType);
42          //(Map) is marked redundant
43          return (writer, object) -> writeMap(writer, (Map)object, keyWriter, valueWriter);
44        } else {
45          throw new RuntimeException("Unable to create writer for generic type: " + clazz);
46        }
47      } else {
48        throw new RuntimeException("Unable to create writer for raw type: " + rawType);
49      }
50    }
51  }
52  
53  abstract class SimplifiedExample {
54    @FunctionalInterface
55    public interface ObjectWriter<T> {
56      void writeObject(T object);
57    }
58  
59    protected abstract <T> void writeList(List<T> list);
60  
61    @SuppressWarnings({"unchecked", "rawtypes"})
62    protected ObjectWriter<?> getCollectionWriter() {
63      return (Object object) -> writeList((List)object);
64    }
65  }
