31    public interface Coproduct2<A, B> {
32      <R> R match(Function<? super A, ? extends R> aFn, Function<? super B, ? extends R> bFn);
33    }
34  
35    public static abstract class NatF<A> implements Functor<A, NatF>, Coproduct2<NatF.Z<A>, NatF.S<A>> {
36  
37      private NatF() {
38      }
39  
40      public static <A> NatF<A> z() {
41        return new Z<>();
42      }
43  
44      public static <A> NatF<A> s(A a) {
45        return new S<>(a);
46      }
47  
48      public static final class Z<A> extends NatF<A> {
49        private Z() {
50        }
51  
52        @Override
53        @SuppressWarnings("unchecked")
54        public <B> Z<B> fmap(Function<? super A, ? extends B> fn) {
55          return (Z<B>) this;
56        }
57  
58        @Override
59        public <R> R match(Function<? super Z<A>, ? extends R> aFn,
60                           Function<? super S<A>, ? extends R> bFn) {
61          return aFn.apply(this);
62        }
63      }
64  
65      public static final class S<A> extends NatF<A> {
66        private final A a;
67  
68        private S(A a) {
69          this.a = a;
70        }
71  
72        public A carrier() {
73          return a;
74        }
75  
76        @Override
77        public <B> S<B> fmap(Function<? super A, ? extends B> fn) {
78          return new S<>(fn.apply(carrier()));
79        }
80  
81        @Override
82        public <R> R match(Function<? super Z<A>, ? extends R> aFn,
83                           Function<? super S<A>, ? extends R> bFn) {
84          return bFn.apply(this);
85        }
86      }
87    }
