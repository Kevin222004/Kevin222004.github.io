5   interface Mappable<T> {
6       T map(Mapper mapper);
7       class Foo<B extends Foo.FooBuilder<B, U>, U extends Foo<B, U>> implements Mappable<Foo<B, U>> {
8           static class FooBuilder<B extends FooBuilder<B, U>, U extends Foo<B, U>> {}
9           @Override
10          public Foo<B, U> map(Mapper mapper) { return this; }
11      }
12  
13      class Mapper {
14          <E extends Mappable<? extends E>> List<E> map(List<E> in) { return in; }
15          <E extends Mappable<? extends E>> Collection<E> map(Collection<E> in) { return in; }
16      }
17      class Counter implements Mappable<Counter> {
18          private final List<Foo<?, ?>> in;
19          Counter(List<Foo<?, ?>> in) { this.in = in; }
20          @Override
21          public Counter map(Mapper mapper) {
22              return new Counter(mapper.map(in));
23          }
24      }
