8   class InferenceTest {
9   
10    public static void main(String[] args) {
11      List<SortedMap<String, Double>> results = codeUnderTest();
12  
13      assertThat(results, contains(Stream.of(entry("A", 1.0), entry("B", 2.0), entry("C", 3.0))
14                                     .collect(Collectors.toMap(Map.Entry::getKey, Map.Entry::getValue, (a, b) -> a, TreeMap::new))));
15  
16      System.out.println("ok!");
17    }
18  
19    private static List<SortedMap<String, Double>> codeUnderTest() {
20      SortedMap<String, Double> result = new TreeMap<>();
21      result.put("A", 1.0);
22      result.put("B", 2.0);
23      result.put("C", 3.0);
24      return Collections.singletonList(result);
25    }
26  
27    private static <K, V> Map.Entry<K, V> entry(K key, V value) {
28      return new AbstractMap.SimpleEntry<>(key, value);
29    }
30  
31    public static <T> void assertThat(T actual, Predicate<? super T> matcher) {
32      if (!matcher.test(actual)) throw new AssertionError(String.valueOf(actual));
33    }
34  
35    @SafeVarargs
36    public static <E> Predicate<Iterable<? extends E>> contains(E... items) {
37      return Predicate.isEqual(Arrays.asList(items));
38    }
39  
40  }
