23  /**
24   * This should be the first test in all test runs,
25   * so we can measure how long the tests are starting up.
26   *
27   * @author max
28   */
29  @SuppressWarnings({"JUnitTestClassNamingConvention", "UseOfSystemOutOrSystemErr"})
30  public class _FirstInSuiteTest extends TestCase {
31    private static long suiteStarted;
32    private static boolean nothingIsCalled;
33  
34    static long getSuiteStartTime() {
35      return suiteStarted;
36    }
37  
38    public void testReportClassLoadingProblems() {
39      List<Throwable> problems = TestAll.getLoadingClassProblems();
40      if (problems.isEmpty()) return;
41  
42      StringBuilder builder = new StringBuilder("The following test classes were not loaded:\n");
43      for (Throwable each : problems) {
44        builder.append(ExceptionUtil.getThrowableText(each)).append("\n");
45        each.printStackTrace(System.out);
46      }
47  
48      throw new AssertionError(builder.toString());
49    }
50  
51    @SuppressWarnings("AssignmentToStaticFieldFromInstanceMethod")
52    public void testNothing() throws Exception {
53      if (nothingIsCalled) return;
54  
55      nothingIsCalled = true;
56  
57      suiteStarted = System.nanoTime();
58      IdeaForkJoinWorkerThreadFactory.setupForkJoinCommonPool(true);
59      SwingUtilities.invokeAndWait(() -> System.out.println("EDT is " + Thread.currentThread()));
60      // in tests EDT inexplicably shuts down sometimes during the first access,
61      // which leads to nasty problems in ApplicationImpl which assumes there is only one EDT.
62      // so we try to forcibly terminate EDT here to urge JVM to re-spawn new shiny permanent EDT-1
63      UITestUtil.replaceIdeEventQueueSafely();
64      SwingUtilities.invokeAndWait(() -> System.out.println("EDT is " + Thread.currentThread()));
65  
66      // force platform JNA load
67      Class.forName("com.sun.jna.Native");
68  
69      String tempDirectory = FileUtilRt.getTempDirectory();
70      String[] list = new File(tempDirectory).list();
71      assert list != null;
72      System.out.println("FileUtil.getTempDirectory() = " + tempDirectory + " (" + list.length + " files)");
73  
74      Preferences.userRoot(); // starts (anonymous!) timer deep in JDK bowels. helps against thread leaks
75    }
76  
77    // performance tests
78    public void testNothingPerformance() throws Exception {
79      System.out.println(Timings.getStatistics());
80      testNothing();
81    }
82  
83    // agents where this test is failing should be disabled and configured properly
84    public void testFileEncoding() {
85      assertEncoding("file.encoding");
86      assertEncoding("sun.jnu.encoding");
87    }
88  
89    private static void assertEncoding(@NotNull String property) {
90      String encoding = System.getProperty(property);
91      System.out.println("** " + property + "=" + encoding);
92      assertNotNull("The property '" + property + "' is 'null'. Please check build configuration settings.", encoding);
93      assertFalse(
94        "The property '" + property + "' is set to a default value. Please make sure the build agent has sane locale settings.",
95        Charset.forName(encoding).aliases().contains("default"));
96    }
97  
98    // agents where this test is failing should be disabled and configured properly
99    @ReviseWhenPortedToJDK("13")
100   public void testSymlinkAbility() {
101     assertTrue(
102       String.format("Symlink creation not supported for %s on %s (%s)",
103                     SystemProperties.getUserName(), SystemInfo.OS_NAME, SystemInfo.OS_VERSION),
104       IoTestUtil.isSymLinkCreationSupported);
105     assertEquals(
106       "The `sun.io.useCanonCaches` makes `File#getCanonical*` methods unreliable and should be set to `false`",
107       "false", System.getProperty("sun.io.useCanonCaches", Runtime.version().feature() >= 13 ? "false" : ""));
108   }
