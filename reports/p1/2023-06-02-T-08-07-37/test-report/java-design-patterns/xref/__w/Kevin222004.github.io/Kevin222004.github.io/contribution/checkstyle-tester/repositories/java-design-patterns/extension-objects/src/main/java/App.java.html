25  import abstractextensions.CommanderExtension;
26  import abstractextensions.SergeantExtension;
27  import abstractextensions.SoldierExtension;
28  import java.util.Optional;
29  import java.util.function.Function;
30  import org.slf4j.LoggerFactory;
31  import units.CommanderUnit;
32  import units.SergeantUnit;
33  import units.SoldierUnit;
34  import units.Unit;
35  
36  /**
37   * Anticipate that an object’s interface needs to be extended in the future. Additional interfaces
38   * are defined by extension objects.
39   */
40  public class App {
41  
42    /**
43     * Program entry point.
44     *
45     * @param args command line args
46     */
47    public static void main(String[] args) {
48  
49      //Create 3 different units
50      var soldierUnit = new SoldierUnit("SoldierUnit1");
51      var sergeantUnit = new SergeantUnit("SergeantUnit1");
52      var commanderUnit = new CommanderUnit("CommanderUnit1");
53  
54      //check for each unit to have an extension
55      checkExtensionsForUnit(soldierUnit);
56      checkExtensionsForUnit(sergeantUnit);
57      checkExtensionsForUnit(commanderUnit);
58  
59    }
60  
61    private static void checkExtensionsForUnit(Unit unit) {
62      final var logger = LoggerFactory.getLogger(App.class);
63  
64      var name = unit.getName();
65      Function<String, Runnable> func = (e) -> () -> logger.info(name + " without " + e);
66  
67      var extension = "SoldierExtension";
68      Optional.ofNullable(unit.getUnitExtension(extension))
69          .map(e -> (SoldierExtension) e)
70          .ifPresentOrElse(SoldierExtension::soldierReady, func.apply(extension));
71  
72      extension = "SergeantExtension";
73      Optional.ofNullable(unit.getUnitExtension(extension))
74          .map(e -> (SergeantExtension) e)
75          .ifPresentOrElse(SergeantExtension::sergeantReady, func.apply(extension));
76  
77      extension = "CommanderExtension";
78      Optional.ofNullable(unit.getUnitExtension(extension))
79          .map(e -> (CommanderExtension) e)
80          .ifPresentOrElse(CommanderExtension::commanderReady, func.apply(extension));
81    }
82  }
