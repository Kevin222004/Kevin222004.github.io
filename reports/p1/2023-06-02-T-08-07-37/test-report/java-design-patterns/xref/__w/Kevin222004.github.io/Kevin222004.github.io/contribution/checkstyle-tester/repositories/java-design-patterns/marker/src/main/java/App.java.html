25  import org.slf4j.Logger;
26  import org.slf4j.LoggerFactory;
27  
28  /**
29   * Created by Alexis on 28-Apr-17. With Marker interface idea is to make empty interface and extend
30   * it. Basically it is just to identify the special objects from normal objects. Like in case of
31   * serialization , objects that need to be serialized must implement serializable interface (it is
32   * empty interface) and down the line writeObject() method must be checking if it is a instance of
33   * serializable or not.
34   *
35   * <p>Marker interface vs annotation Marker interfaces and marker annotations both have their uses,
36   * neither of them is obsolete or always better then the other one. If you want to define a type
37   * that does not have any new methods associated with it, a marker interface is the way to go. If
38   * you want to mark program elements other than classes and interfaces, to allow for the possibility
39   * of adding more information to the marker in the future, or to fit the marker into a framework
40   * that already makes heavy use of annotation types, then a marker annotation is the correct choice
41   */
42  public class App {
43  
44    /**
45     * Program entry point.
46     *
47     * @param args command line args
48     */
49    public static void main(String[] args) {
50      final var logger = LoggerFactory.getLogger(App.class);
51      var guard = new Guard();
52      var thief = new Thief();
53  
54      //noinspection ConstantConditions
55      if (guard instanceof Permission) {
56        guard.enter();
57      } else {
58        logger.info("You have no permission to enter, please leave this area");
59      }
60  
61      //noinspection ConstantConditions
62      if (thief instanceof Permission) {
63        thief.steal();
64      } else {
65        thief.doNothing();
66      }
67    }
68  }
