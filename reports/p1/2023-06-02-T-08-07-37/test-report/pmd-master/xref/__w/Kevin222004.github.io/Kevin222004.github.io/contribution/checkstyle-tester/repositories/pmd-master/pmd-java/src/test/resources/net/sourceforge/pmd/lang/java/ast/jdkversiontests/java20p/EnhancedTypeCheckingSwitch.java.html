5   /**
6    * @see <a href="https://openjdk.org/jeps/433">JEP 433: Pattern Matching for switch (Fourth Preview)</a>
7    */
8   public class EnhancedTypeCheckingSwitch {
9   
10  
11      static void typeTester(Object o) {
12          switch (o) {
13              case null     -> System.out.println("null");
14              case String s -> System.out.println("String");
15              case Color c  -> System.out.println("Color with " + c.values().length + " values");
16              case Point p  -> System.out.println("Record class: " + p.toString());
17              case int[] ia -> System.out.println("Array of ints of length " + ia.length);
18              default       -> System.out.println("Something else");
19          }
20      }
21  
22      public static void main(String[] args) {
23          Object o = "test";
24          typeTester(o);
25          typeTester(Color.BLUE);
26  
27          o = new int[] {1, 2, 3, 4};
28          typeTester(o);
29  
30          o = new Point(7, 8);
31          typeTester(o);
32  
33          o = new Object();
34          typeTester(o);
35      }
36  }
37  
38  record Point(int i, int j) {}
39  enum Color { RED, GREEN, BLUE; }
