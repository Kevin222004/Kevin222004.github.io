5   /**
6    * @see <a href="https://openjdk.org/jeps/432">JEP 432: Record Patterns (Second Preview)</a>
7    */
8   public class RecordPatterns {
9   
10      record Point(int x, int y) {}
11      enum Color { RED, GREEN, BLUE }
12      record ColoredPoint(Point p, Color c) {}
13      record Rectangle(ColoredPoint upperLeft, ColoredPoint lowerRight) {}
14  
15      void printSum1(Object o) {
16          if (o instanceof Point p) {
17              int x = p.x();
18              int y = p.y();
19              System.out.println(x+y);
20          }
21      }
22  
23      // record pattern
24      void printSum2(Object o) {
25          if (o instanceof Point(int x, int y)) {
26              System.out.println(x+y);
27          }
28      }
29  
30      void printUpperLeftColoredPoint(Rectangle r) {
31          if (r instanceof Rectangle(ColoredPoint ul, ColoredPoint lr)) {
32              System.out.println(ul.c());
33          }
34      }
35  
36      // nested record pattern
37      void printColorOfUpperLeftPoint(Rectangle r) {
38          if (r instanceof Rectangle(ColoredPoint(Point p, Color c),
39                  ColoredPoint lr)) {
40              System.out.println(c);
41          }
42      }
43  
44      Rectangle createRectangle(int x1, int y1, Color c1, int x2, int y2, Color c2) {
45          Rectangle r = new Rectangle(new ColoredPoint(new Point(x1, y1), c1),
46                  new ColoredPoint(new Point(x2, y2), c2));
47          return r;
48      }
49  
50      // fully nested record pattern, also using "var"
51      void printXCoordOfUpperLeftPointWithPatterns(Rectangle r) {
52          if (r instanceof Rectangle(ColoredPoint(Point(var x, var y), var c),
53                  var lr)) {
54              System.out.println("Upper-left corner: " + x);
55          }
56      }
57  
58      record Pair(Object x, Object y) {}
59      void nestedPatternsCanFailToMatch() {
60          Pair p = new Pair(42, 42);
61          if (p instanceof Pair(String s, String t)) {
62              System.out.println(s + ", " + t);
63          } else {
64              System.out.println("Not a pair of strings");
65          }
66      }
67  
68      // record patterns with generic types
69      record Box<T>(T t) {}
70      void test1a(Box<Object> bo) {
71          if (bo instanceof Box<Object>(String s)) {
72              System.out.println("String " + s);
73          }
74      }
75      void test1(Box<String> bo) {
76          if (bo instanceof Box<String>(var s)) {
77              System.out.println("String " + s);
78          }
79      }
80  
81      // type argument is inferred
82      void test2(Box<String> bo) {
83          if (bo instanceof Box(var s)) {    // Inferred to be Box<String>(var s)
84              System.out.println("String " + s);
85          }
86      }
87  
88      // nested record patterns
89      void test3(Box<Box<String>> bo) {
90          if (bo instanceof Box<Box<String>>(Box(var s))) {
91              System.out.println("String " + s);
92          }
93      }
94  
95      void test4(Box<Box<String>> bo) {
96          if (bo instanceof Box(Box(var s))) {
97              System.out.println("String " + s);
98          }
99      }
100 }
