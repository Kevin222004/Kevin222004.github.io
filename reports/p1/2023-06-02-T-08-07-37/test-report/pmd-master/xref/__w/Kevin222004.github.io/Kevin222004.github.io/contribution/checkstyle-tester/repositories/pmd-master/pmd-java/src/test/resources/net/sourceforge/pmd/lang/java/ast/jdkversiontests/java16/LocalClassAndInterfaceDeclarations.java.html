8   public class LocalClassAndInterfaceDeclarations {
9   
10      {
11          class MyLocalClass {
12              // constant fields are always allowed
13              static final int constantField = 1;
14              // static members in local classes are allowed with Java16
15              static int staticField;
16              static void staticMethod() { }
17          }
18  
19          // static local classes are not allowed (neither Java16 nor Java16 Preview)
20          // Note: PMD's parser allows this, but it would actually be a compile error
21          //static class MyLocalStaticClass {}
22  
23          // local interfaces are allowed with Java16
24          interface MyLocalInterface {}
25  
26          // local enums are allowed with Java16
27          enum MyLocalEnum { A }
28  
29          // local annotation types are not allowed in Java16 (have been with Java15 Preview)
30          //@interface MyLocalAnnotation {}
31      }
