5   /**
6    * @see <a href="https://openjdk.org/jeps/405">JEP 405: Record Patterns (Preview)</a>
7    */
8   public class RecordPatterns {
9   
10      record Point(int x, int y) {}
11      enum Color { RED, GREEN, BLUE }
12      record ColoredPoint(Point p, Color c) {}
13      record Rectangle(ColoredPoint upperLeft, ColoredPoint lowerRight) {}
14  
15      void printSum1(Object o) {
16          if (o instanceof Point p) {
17              int x = p.x();
18              int y = p.y();
19              System.out.println(x+y);
20          }
21      }
22  
23      // record pattern
24      void printSum2(Object o) {
25          if (o instanceof Point(int x, int y)) {
26              System.out.println(x+y);
27          }
28      }
29  
30      void printUpperLeftColoredPoint(Rectangle r) {
31          if (r instanceof Rectangle(ColoredPoint ul, ColoredPoint lr)) {
32              System.out.println(ul.c());
33          }
34      }
35  
36      // nested record pattern
37      void printColorOfUpperLeftPoint(Rectangle r) {
38          if (r instanceof Rectangle(ColoredPoint(Point p, Color c),
39                  ColoredPoint lr)) {
40              System.out.println(c);
41          }
42      }
43  
44      // fully nested record pattern, also using "var"
45      void printXCoordOfUpperLeftPointWithPatterns(Rectangle r) {
46          if (r instanceof Rectangle(ColoredPoint(Point(var x, var y), var c),
47                  var lr)) {
48              System.out.println("Upper-left corner: " + x);
49          }
50      }
51  
52      // record patterns with generic types
53      record Box<T>(T t) {}
54      void test1(Box<Object> bo) {
55          if (bo instanceof Box<Object>(String s)) {
56              System.out.println("String " + s);
57          }
58      }
59      void test2(Box<String> bo) {
60          if (bo instanceof Box<String>(var s)) {
61              System.out.println("String " + s);
62          }
63      }
64  }
