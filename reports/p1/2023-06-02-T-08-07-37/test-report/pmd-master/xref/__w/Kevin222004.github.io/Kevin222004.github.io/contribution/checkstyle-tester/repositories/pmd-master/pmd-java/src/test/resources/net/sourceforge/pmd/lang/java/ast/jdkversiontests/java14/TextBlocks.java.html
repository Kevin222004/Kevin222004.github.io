1   import javax.script.ScriptEngine;

4   /**

7   public class TextBlocks {

9   
10      public static void main(String[] args) throws Exception {
11          // note: there is trailing whitespace!!
12          String html = """
13                        <html>   
14                            <body>
15                                <p>Hello, world</p>    
16                            </body> 
17                        </html>   
18                        """;
19          System.out.println(html);
20  
21          String query = """
22                         SELECT `EMP_ID`, `LAST_NAME` FROM `EMPLOYEE_TB`
23                         WHERE `CITY` = 'INDIANAPOLIS'
24                         ORDER BY `EMP_ID`, `LAST_NAME`;
25                         """;
26          System.out.println(query);
27  
28          ScriptEngine engine = new ScriptEngineManager().getEngineByName("js");
29          Object obj = engine.eval("""
30                                   function hello() {
31                                       print('"Hello, world"');
32                                   }
33                                   
34                                   hello();
35                                   """);
36  
37          // Escape sequences
38          String htmlWithEscapes = """
39                        <html>\r
40                            <body>\r
41                                <p>Hello, world</p>\r
42                            </body>\r
43                        </html>\r
44                        """;
45          System.out.println(htmlWithEscapes);
46  
47          String season = """
48                  winter""";    // the six characters w i n t e r
49  
50          String period = """
51                          winter
52                          """;          // the seven characters w i n t e r LF
53  
54          String greeting =
55              """
56              Hi, "Bob"
57              """;        // the ten characters H i , SP " B o b " LF
58  
59          String salutation =
60              """
61              Hi,
62               "Bob"
63              """;        // the eleven characters H i , LF SP " B o b " LF
64  
65          String empty = """
66                         """;      // the empty string (zero length)
67  
68          String quote = """
69                         "
70                         """;      // the two characters " LF
71  
72          String backslash = """
73                             \\
74                             """;  // the two characters \ LF
75  
76          String normalStringLiteral = "test";
77  
78          String code =
79              """
80              String text = \"""
81                  A text block inside a text block
82              \""";
83              """;
84  
85          // new escape sequences
86          String text = """
87                        Lorem ipsum dolor sit amet, consectetur adipiscing \
88                        elit, sed do eiusmod tempor incididunt ut labore \
89                        et dolore magna aliqua.\
90                        """;
91          System.out.println(text);
92  
93          String colors = """
94                          red  \s
95                          green\s
96                          blue \s
97                          """;
98          System.out.println(colors);
99  
100         // empty new line as first content
101         String emptyLine = """
102 
103 test
104 """;
105         System.out.println(emptyLine.replaceAll("\n", "<LF>"));
106 
107         // backslash escapes
108         String bs = """
109                 \\test
110                 """;
111         System.out.println(bs.replaceAll("\n", "<LF>"));
112     }
