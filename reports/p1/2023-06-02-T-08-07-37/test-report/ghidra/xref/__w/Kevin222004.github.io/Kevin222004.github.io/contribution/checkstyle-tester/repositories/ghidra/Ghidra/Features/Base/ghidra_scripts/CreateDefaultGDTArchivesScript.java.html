59  	private void parseHeaderFilesToGDT(File outputDir, String gdtName, String languageID, String compiler,
60  			String[] filenames, String includePaths[], String[] args)
61  			throws ParseException, ghidra.app.util.cparser.CPP.ParseException, IOException {
62  		DataTypeManager openTypes[] = null;
63  		
64  		parseHeaderFilesToGDT(openTypes, outputDir, gdtName, languageID, compiler, filenames, includePaths, args);
65  	}
66  
67  	private void parseHeaderFilesToGDT(DataTypeManager openTypes[], File outputDir, String gdtName, String languageID, String compiler,
68  			String[] filenames, String[] includePaths, String[] args)
69  			throws ParseException, ghidra.app.util.cparser.CPP.ParseException, IOException {
70  		
71  		String dataTypeFile = outputDir + File.separator + gdtName + ".gdt";
72  		
73  		File f = getArchiveFile(dataTypeFile);
74  		
75          FileDataTypeManager dtMgr = FileDataTypeManager.createFileArchive(f);
76          
77  		CParseResults results = CParserUtils.parseHeaderFiles(openTypes, filenames, includePaths, args, dtMgr, languageID, compiler, monitor);
78  		
79  		Msg.info(this, results.getFormattedParseMessage(null));
80  
81  		dtMgr.save();
82  		dtMgr.close();
83  	}
84  
85  	/**
86  	 * Turn string into a file, delete old archive if it exists
87  	 * 
88  	 * @param dataTypeFile
89  	 * 
90  	 * @return file
91  	 */
92  	private File getArchiveFile(String dataTypeFile) {
93  		File f = new File(dataTypeFile);
94  		if (f.exists()) {
95  			f.delete();
96  		}
97  		String lockFile = dataTypeFile + ".ulock";
98  		File lf = new File(lockFile);
99  		if (lf.exists()) {
100 			lf.delete();
101 		}
102 		return f;
103 	}
