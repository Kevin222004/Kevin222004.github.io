16  //Assemble hard-coded block of instructions.
17  //@category Assembly
18  
19  import ghidra.app.plugin.assembler.Assembler;
20  import ghidra.app.plugin.assembler.Assemblers;
21  import ghidra.app.script.GhidraScript;
22  
23  /**
24   * This is a demonstration script to show how to use the assembler to assemble a block of
25   * instructions.
26   */
27  public class AssembleBlockScript extends GhidraScript {
28  	@Override
29  	protected void run() throws Exception {
30  		monitor.setMessage("Constructing Assembler");
31  		// First, obtain an assembler bound to the current program.
32  		Assembler asm = Assemblers.getAssembler(currentProgram);
33  
34  		monitor.setMessage("Assembling");
35  		// Now assemble a block. A block can be given as an array of strings, or a string of
36  		// newline-separated instructions.
37  		// This will patch each resulting instruction into the bound program in sequence.
38  		asm.assemble(currentAddress, //
39  			"ADD [RBX],BL", //
40  			"JMP 0x34", //
41  			"SCASB.REPE RDI");
42  	}
43  }
