43  	
44  	// location of header files base directory
45  	private static String headerFilePath = "/data/HeaderFiles";
46  
47  	@Override
48  	protected void run() throws Exception {
49  		outputDirectory = askDirectory("Select Directory for GDT files", "Select GDT Output Dir");
50  		
51  		parseGDT_DirectX();
52  		
53  		parseGDT_WinVS22();
54  	}
55  	
56  	private void parseHeaderFilesToGDT(File outputDir, String gdtName, String languageID, String compiler,
57  			String[] filenames, String includePaths[], String[] args)
58  			throws ParseException, ghidra.app.util.cparser.CPP.ParseException, IOException {
59  		DataTypeManager openTypes[] = null;
60  		
61  		parseHeaderFilesToGDT(openTypes, outputDir, gdtName, languageID, compiler, filenames, includePaths, args);
62  	}
63  
64  	private void parseHeaderFilesToGDT(DataTypeManager openTypes[], File outputDir, String gdtName, String languageID, String compiler,
65  			String[] filenames, String[] includePaths, String[] args)
66  			throws ParseException, ghidra.app.util.cparser.CPP.ParseException, IOException {
67  		
68  		String dataTypeFile = outputDir + File.separator + gdtName + ".gdt";
69  		
70  		File f = getArchiveFile(dataTypeFile);
71  		
72          FileDataTypeManager dtMgr = FileDataTypeManager.createFileArchive(f);
73          
74  		CParseResults results = CParserUtils.parseHeaderFiles(openTypes, filenames, includePaths, args, dtMgr, languageID, compiler, monitor);
75  		
76  		Msg.info(this, results.getFormattedParseMessage(null));
77  
78  		dtMgr.save();
79  		dtMgr.close();
80  	}
81  
82  	/**
83  	 * Turn string into a file, delete old archive if it exists
84  	 * 
85  	 * @param dataTypeFile name of archive file
86  	 * 
87  	 * @return file
88  	 */
89  	private File getArchiveFile(String dataTypeFile) {
90  		File f = new File(dataTypeFile);
91  		if (f.exists()) {
92  			f.delete();
93  		}
94  		String lockFile = dataTypeFile + ".ulock";
95  		File lf = new File(lockFile);
96  		if (lf.exists()) {
97  			lf.delete();
98  		}
99  		return f;
100 	}
