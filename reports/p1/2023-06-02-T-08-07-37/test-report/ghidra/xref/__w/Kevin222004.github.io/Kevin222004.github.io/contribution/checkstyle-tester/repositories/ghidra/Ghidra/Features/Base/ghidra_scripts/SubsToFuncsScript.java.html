16  //Converts subroutines to functions.
17  //Subroutines are located using the active
18  //subroutine model on the BlockModelService.
19  //@category Functions
20  
21  import ghidra.app.script.GhidraScript;
22  import ghidra.app.services.BlockModelService;
23  import ghidra.program.model.address.AddressSetView;
24  import ghidra.program.model.block.*;
25  import ghidra.program.model.listing.FunctionIterator;
26  import ghidra.program.model.listing.Listing;
27  import ghidra.program.model.symbol.SourceType;
28  import ghidra.program.model.symbol.Symbol;
29  
30  public class SubsToFuncsScript extends GhidraScript {
31  
32  	@Override
33  	public void run() throws Exception {
34  		BlockModelService blockModelService = state.getTool().getService(BlockModelService.class);
35  		Listing listing = currentProgram.getListing();
36  		StringBuffer errorBuf = new StringBuffer();
37  		CodeBlockModel cbm = blockModelService.getActiveSubroutineModel();
38  		AddressSetView addrset =
39  			currentSelection == null ? (AddressSetView) currentProgram.getMemory()
40  					: currentSelection;
41  		CodeBlockIterator cbIter = cbm.getCodeBlocksContaining(addrset, monitor);
42  		while (cbIter.hasNext()) {
43  			CodeBlock block = cbIter.next();
44  			FunctionIterator fIter = listing.getFunctions(block, true);
45  			if (!fIter.hasNext()) {
46  				try {
47  					String name = "DEAD_" + block.getFirstStartAddress();
48  					Symbol symbol = getSymbolAt(block.getFirstStartAddress());
49  					if (symbol != null && !symbol.isDynamic()) {
50  						name = symbol.getName();
51  					}
52  					listing.createFunction(name, block.getFirstStartAddress(), block,
53  						SourceType.USER_DEFINED);
54  				}
55  				catch (Exception e) {
56  					errorBuf.append(e.toString() + "\n");
57  				}
58  			}
59  		}
60  		if (errorBuf.length() > 0) {
61  			println(errorBuf.toString());
62  		}
63  	}
64  
65  }
