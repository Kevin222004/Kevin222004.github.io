509 	@Override
510 	protected void run() throws Exception {
511 		FidFileManager fidFileManager = FidFileManager.getInstance();
512 
513 		List<FidFile> allKnownFidFiles = fidFileManager.getFidFiles();
514 		ArrayList<String> dbfiles = new ArrayList<>();
515 		HashMap<String, FidFile> fidMap = new HashMap<>();
516 		for (FidFile fidFile : allKnownFidFiles) {
517 			if (!fidFile.isInstalled()) {
518 				fidMap.put(fidFile.getName(), fidFile);
519 				dbfiles.add(fidFile.getName());
520 			}
521 		}
522 		String[] nameArray = new String[dbfiles.size()];
523 		dbfiles.toArray(nameArray);
524 		String askChoice = null;
525 		try {
526 			askChoice = askChoice("RemoveFunctions script", "Choose FID database: ",
527 				Arrays.asList(nameArray), nameArray[0]);
528 		}
529 		catch (CancelledException ex) {
530 			runSearch();
531 			return;
532 		}
533 		FidFile fidFile = fidMap.get(askChoice);
534 		FidDB modifiableFidDB = fidFile.getFidDB(true);
535 		List<LibraryRecord> allLibraries = modifiableFidDB.getAllLibraries();
536 		boolean is64bit = false;
537 		boolean is32bit = false;
538 		for (LibraryRecord rec : allLibraries) {
539 			if (rec.getGhidraLanguageID().getIdAsString().startsWith("x86:LE:64")) {
540 				is64bit = true;
541 			}
542 			else if (rec.getGhidraLanguageID().getIdAsString().startsWith("x86:LE:32")) {
543 				is32bit = true;
544 			}
545 		}
546 		if (is32bit && is64bit) {
547 			throw new IOException(
548 				"Script can only process code of one type (32-bit or 64-bit) at a time");
549 		}
550 		if ((!is32bit) && (!is64bit)) {
551 			throw new IOException(
552 				"Script is designed to run only on a FID database generated from x86 code");
553 		}
554 		if (is64bit) {
555 			buildKnownHashes64();
556 		}
557 		else {
558 			buildKnownHashes32();
559 		}
560 		try {
561 			monitor.setMaximum(AUTO_FAIL_REGEX.size());
562 			monitor.setProgress(0);
563 			for (String regex : AUTO_FAIL_REGEX) {
564 				monitor.checkCancelled();
565 				monitor.incrementProgress(1);
566 				List<FunctionRecord> recordList = modifiableFidDB.findFunctionsByNameRegex(regex);
567 				for (FunctionRecord record : recordList) {
568 					modifiableFidDB.setAutoFailOnFunction(record, true);
569 				}
570 			}
571 			monitor.setMaximum(SPECIAL_PARENT.size());
572 			monitor.setProgress(0);
573 			for (Pair<String, Pair<Long, Long>> pair : SPECIAL_PARENT) {
574 				monitor.checkCancelled();
575 				monitor.incrementProgress(1);
576 				List<FunctionRecord> childFunctions =
577 					modifiableFidDB.findFunctionsByFullHash(pair.second.first);
578 				List<FunctionRecord> parentFunctions =
579 					modifiableFidDB.findFunctionsByFullHash(pair.second.second);
580 				if (parentFunctions.isEmpty()) {
581 					continue;
582 				}
583 				FunctionRecord parentFunction = parentFunctions.get(0);
584 				for (FunctionRecord childFunction : childFunctions) {
585 					if (!childFunction.getName().equals(pair.first)) {
586 						continue;
587 					}
588 					modifiableFidDB.createInferiorRelation(parentFunction, childFunction);
589 				}
590 			}
591 			monitor.setMaximum(REMOVE_HASHES.size());
592 			monitor.setProgress(0);
593 			for (Pair<Short, Long> pair : REMOVE_HASHES) {
594 				modifiableFidDB.setAutoFailByFullHash(pair.second.longValue(), true);
595 				monitor.checkCancelled();
596 				monitor.incrementProgress(1);
597 			}
598 			monitor.setMaximum(FORCE_RELATION.size());
599 			monitor.setProgress(0);
600 			for (Pair<Short, Long> pair : FORCE_RELATION) {
601 				modifiableFidDB.setForceRelationByFullHash(pair.second.longValue(), true);
602 				monitor.checkCancelled();
603 				monitor.incrementProgress(1);
604 			}
605 			monitor.setMaximum(FORCE_SPECIFIC.size());
606 			monitor.setProgress(0);
607 			for (Pair<Short, Long> pair : FORCE_SPECIFIC) {
608 				modifiableFidDB.setForceSpecificByFullHash(pair.second.longValue(), true);
609 				monitor.checkCancelled();
610 				monitor.incrementProgress(1);
611 			}
612 			monitor.setMaximum(AUTO_PASS.size());
613 			monitor.setProgress(0);
614 			for (Pair<Short, Long> pair : AUTO_PASS) {
615 				modifiableFidDB.setAutoPassByFullHash(pair.second.longValue(), true);
616 				monitor.checkCancelled();
617 				monitor.incrementProgress(1);
618 			}
619 
620 			monitor.setMaximum(REMOVE_SPECHASHES.size());
621 			monitor.setProgress(0);
622 			for (Pair<Short, Long> pair : REMOVE_SPECHASHES) {
623 				List<FunctionRecord> listSpecHash =
624 					modifiableFidDB.findFunctionsBySpecificHash(pair.second.longValue());
625 				for (FunctionRecord funcRec : listSpecHash) {
626 					modifiableFidDB.setAutoFailOnFunction(funcRec, true);
627 				}
628 				monitor.checkCancelled();
629 				monitor.incrementProgress(1);
630 			}
631 			modifiableFidDB.saveDatabase("", monitor);
632 		}
633 		finally {
634 			modifiableFidDB.close();
635 		}
