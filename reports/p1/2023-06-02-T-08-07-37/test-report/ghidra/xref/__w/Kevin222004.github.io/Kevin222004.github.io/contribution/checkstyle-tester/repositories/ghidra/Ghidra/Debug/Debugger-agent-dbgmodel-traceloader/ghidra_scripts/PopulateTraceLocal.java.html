44  /**
45   * This script populates a trace database for demonstrations purposes and opens it in the current
46   * tool.
47   * 
48   * <p>
49   * Your current tool had better be the "TraceBrowser"! The demonstration serves two purposes. 1) It
50   * puts interesting data into the TraceBrowser and leaves some annotations as an exercise. 2) It
51   * demonstrates how a decent portion the Trace API works.
52   * 
53   * <p>
54   * A Trace is basically a collection of observations of memory and registers over the lifetime of an
55   * application or computer system. In Ghidra, the Trace object also supports many of the same
56   * annotations as does Program. In the same way that Program brings knowledge markup to an image of
57   * bytes, Trace brings knowledge markup to bytes observed over time.
58   * 
59   * <p>
60   * Effectively, if you take the cross-product of Program with time and add Threads, Breakpoints,
61   * etc., you get Trace. It's a lot. In order to use all the UI components which take a Program,
62   * Trace can present itself as a Program at a particular point in time.
63   * 
64   * <p>
65   * Each particular component will be introduced as its used in the script below, but for now some
66   * core concepts:
67   * 
68   * <ul>
69   * <li>A point in time is called a "tick." These don't necessarily correspond to any real unit of
70   * time, though they may. The only requirement is that they are numbered in chronological
71   * order.</li>
72   * <li>Every annotation has a "lifespan," which is the range of ticks for which the annotation is
73   * effective. Some annotations may overlap, others may not. In general, if the corresponding concept
74   * in Program permits address overlap, then Trace permits both address and time overlap. If not,
75   * then neither is permitted. In essense, Trace defines overlap as the intersection of rectangles,
76   * where an annotation's X dimension is it's address range, and its Y dimension is its lifespan.
77   * </li>
78   * <li>Observations in memory happen at a particular tick and are assumed in effect until another
79   * observation changes that. To record the "freshness" of observations, the memory manager tags
80   * regions as KNOWN, UNKNOWN, or ERROR. An observation implicitly marks the affected region as
81   * KNOWN. The intent is to grey the background for regions where memory is UNKNOWN for the current
82   * tick.</li>
83   * <li>Observations of registers behave exactly the same as observations for memory, by leveraging
84   * Ghidra's "register space." The only difference is that those observations must be recorded with
85   * respect to a given thread. Each thread is effectively allocated its own copy of the register
86   * space. Most the the API components require you to obtain a special "register space" for a given
87   * thread before recording observations of or applying annotations to that thread.</li>
88   * </ul>
89   * 
90   * <p>
91   * After you've run this script, a trace should appear in the UI. Note that there is not yet a way
92   * to save a trace in the UI. As an exercise, try adding data units to analyze the threads' stacks.
93   * It may take some getting accustomed to, but the rules for laying down units should be very
94   * similar to those in a Program. However, the Trace must take the applied units and decide how far
95   * into the future they are effective. In general, it defaults to "from here on out." However, two
96   * conditions may cause the trace to choose an ending tick: 1) The underlying bytes change sometime
97   * in the future, and 2) There is an overlapping code unit sometime in the future.
98   * 
99   * <p>
100  * The trace chooses the latest tick possible preceding any byte change or existing code unit, so
101  * that the unit's underlying bytes remain constant for its lifespan, and the unit does not overlap
102  * any existing unit. This rule causes some odd behavior for null-terminated strings. I intend to
103  * adjust this rule slightly for static data types wrt/ byte changes. For those, the placed unit
104  * should be truncated as described above, however, another data unit of the same type can be placed
105  * at the change. The same rule is then applied iteratively into the future until an overlapping
106  * unit is encountered, or there are no remaining byte changes.
107  */

136 	/**
137 	 * Create an address in the processor's (x86_64) default space.
138 	 * 
139 	 * @param offset the byte offset
140 	 * @return the address
141 	 */
142 	protected Address addr(long offset) {
143 		return defaultSpace.getAddress(offset);
144 	}
145 
146 	/**
147 	 * Create an address range in the processor's default space.
148 	 * 
149 	 * @param min the minimum byte offset
150 	 * @param max the maximum (inclusive) byte offset
151 	 * @return the range
152 	 */
153 	protected AddressRange rng(long min, long max) {
154 		return new AddressRangeImpl(addr(min), addr(max));
155 	}
156 
157 	/**
158 	 * Get a register by name
159 	 * 
160 	 * @param name the name
161 	 * @return the register
162 	 */
163 	protected Register reg(String name) {
164 		return lang.getRegister(name);
165 	}

204 				Map<String, ModelObject> eventMap = event.getKeyValueMap();
205 				ModelObject pos = eventMap.get("Position");
206 				ModelObject seq = pos.getKeyValue("Sequence");
207 				//ModelObject step = pos.getKeyValue("Steps");
208 				ModelObject type = eventMap.get("Type");
209 				String display = type.getValueString();
210 				Long snap = (Long) seq.getValue();
211 				if (display.contains("ModuleLoaded") || display.contains("ModuleUnloaded")) {
212 					ModelObject module = eventMap.get("Module");
213 					Map<String, ModelObject> moduleMap = module.getKeyValueMap();
214 					ModelObject name = moduleMap.get("Name");
215 					ModelObject address = moduleMap.get("Address");
216 					ModelObject size = moduleMap.get("Size");

226 						modules.addLoadedModule(moduleId, moduleId, rng, snap);
227 						memory.addRegion(moduleId, Lifespan.nowOn(snap), rng,
228 							TraceMemoryFlag.READ, TraceMemoryFlag.WRITE, TraceMemoryFlag.EXECUTE);
229 						try {
230 							int read =
231 								client.getDataSpaces().readVirtual(start, buf, sz.intValue());
232 							int written = memory.putBytes(snap, rng.getMinAddress(), buf.flip());

235 									"read:written=" + read + ":" + written + " at " + start);
236 							}
237 						}
238 						catch (Exception e) {
239 							System.err.println("Unable to read " + moduleId + " at " + snap);
240 						}
241 					}
242 					else {
243 						if (snap >= 0) {
244 							Collection<? extends TraceModule> mods =
245 								modules.getModulesByPath(moduleId);
246 							Iterator<? extends TraceModule> iterator = mods.iterator();
247 							while (iterator.hasNext()) {
248 								TraceModule next = iterator.next();
249 								next.setUnloadedSnap(snap);
250 							}
251 						}
252 					}
253 				}
254 				else if (display.contains("ThreadCreated") ||
255 					display.contains("ThreadTerminated")) {
256 					ModelObject thread = eventMap.get("Thread");
257 					//ModelObject uid = thread.getKeyValue("UniqueId");
258 					ModelObject id = thread.getKeyValue("Id");
259 					String threadId = id.getValueString();
260 					display += " " + threadId;
261 					if (display.contains("ThreadCreated")) {
262 						threads.addThread(threadId, Lifespan.nowOn(snap));
263 					}
264 					else {
265 						if (snap >= 0) {
266 							Collection<? extends TraceThread> thrs =
267 								threads.getThreadsByPath(threadId);
268 							Iterator<? extends TraceThread> iterator = thrs.iterator();
269 							while (iterator.hasNext()) {
270 								TraceThread next = iterator.next();
271 								next.setDestructionSnap(snap);
272 							}
273 						}
274 					}
275 				}
276 				if (snap < 0) {
277 					snap = ++max;
278 				}
279 				timeManager.getSnapshot(snap, true).setDescription(display);
280 				eventSnaps.add(snap);
281 			}
282 		}
283 
284 		try (Transaction tx = trace.openTransaction("Populate Registers")) {
285 			//for (Long tick : tickManager.getAllTicks()) {
286 			for (Long snap : eventSnaps) {
287 				control.execute("!tt " + Long.toHexString(snap) + ":0");
288 				control.waitForEvent();
289 
290 				List<ModelObject> modelThreads =
291 					util.getElements(
292 						List.of("Debugger", "State", "DebuggerVariables", "curprocess", "Threads"));
293 				Map<String, ModelObject> modelThreadMap = new HashMap<String, ModelObject>();
294 				for (ModelObject obj : modelThreads) {
295 					modelThreadMap.put(obj.getSearchKey(), obj);
296 				}
297 
298 				Collection<? extends TraceThread> liveThreads = threads.getLiveThreads(snap);
299 				for (TraceThread thread : liveThreads) {
300 					TraceMemorySpace regspace = memory.getMemoryRegisterSpace(thread, true);
301 					ModelObject modelThread = modelThreadMap.get("0x" + thread.getName());
302 					if (modelThread != null) {
303 						Map<String, ModelObject> registers =
304 							modelThread.getKeyValue("Registers")
305 									.getKeyValue(
306 										"User")
307 									.getKeyValueMap();
308 						for (String rname : registers.keySet()) {
309 							ModelObject r = registers.get(rname);
310 							Register reg = reg(rname.toUpperCase());
311 							if (reg != null) {
312 								try {
313 									regspace.setValue(snap,
314 										new RegisterValue(reg, BigInteger.valueOf(
315 											Long.parseUnsignedLong(r.getValueString(), 16))));
316 								}
317 								catch (Exception e) {
318 									System.err.println(rname + "<--" + r.getValueString());
319 								}
320 							}
321 						}
322 					}
323 					else {
324 						System.err.println(thread.getName() + " not found!");
325 					}
326 
327 				}
328 			}
329 		}
