16  //An example emulation script that integrates well with the Debgger UI.
17  //It provides the set-up code and then demonstrates some use cases.
18  //It should work with any x64 program, but some snippets may require specific conditions.
19  //It should be easily ported to other platforms just by adjusting register names.
20  //@author 
21  //@category Emulation
22  //@keybinding
23  //@menupath
24  //@toolbar
25  
26  import java.nio.charset.Charset;
27  
28  import db.Transaction;
29  import ghidra.app.plugin.assembler.Assembler;
30  import ghidra.app.plugin.assembler.Assemblers;
31  import ghidra.app.plugin.core.debug.service.emulation.BytesDebuggerPcodeEmulator;
32  import ghidra.app.plugin.core.debug.service.emulation.ProgramEmulationUtils;
33  import ghidra.app.plugin.core.debug.service.emulation.data.DefaultPcodeDebuggerAccess;
34  import ghidra.app.plugin.processors.sleigh.SleighLanguage;
35  import ghidra.app.script.GhidraScript;
36  import ghidra.app.services.DebuggerTraceManagerService;
37  import ghidra.app.services.ProgramManager;
38  import ghidra.framework.plugintool.PluginTool;
39  import ghidra.pcode.emu.PcodeThread;
40  import ghidra.pcode.exec.*;
41  import ghidra.pcode.exec.PcodeExecutorStatePiece.Reason;
42  import ghidra.pcode.exec.trace.TraceSleighUtils;
43  import ghidra.pcode.utils.Utils;
44  import ghidra.program.database.ProgramDB;
45  import ghidra.program.model.address.Address;
46  import ghidra.program.model.address.AddressSpace;
47  import ghidra.program.model.lang.LanguageID;
48  import ghidra.program.model.listing.InstructionIterator;
49  import ghidra.program.model.listing.Program;
50  import ghidra.program.model.mem.Memory;
51  import ghidra.trace.model.Trace;
52  import ghidra.trace.model.guest.TracePlatform;
53  import ghidra.trace.model.thread.TraceThread;
54  import ghidra.trace.model.time.TraceSnapshot;
55  import ghidra.trace.model.time.TraceTimeManager;
56  
57  public class DebuggerEmuExampleScript extends GhidraScript {
58  	private final static Charset UTF8 = Charset.forName("utf8");
59  
60  	@Override
61  	protected void run() throws Exception {
62  		/*
63  		 * First, get all the services and stuff:
64  		 */
65  		PluginTool tool = state.getTool();
66  		ProgramManager programManager = tool.getService(ProgramManager.class);
67  		DebuggerTraceManagerService traceManager =
68  			tool.getService(DebuggerTraceManagerService.class);
69  		SleighLanguage language = (SleighLanguage) getLanguage(new LanguageID("x86:LE:64:default"));
70  
71  		/*
72  		 * I'll generate a new program, because I don't want to require the user to pick something
73  		 * specific.
74  		 */
75  		Address entry;
76  		Address injectHere;
77  		Program program = null;
78  		try {
79  			program =
80  				new ProgramDB("emu_example", language, language.getDefaultCompilerSpec(), this);
81  			// Save the program into the project so it has a URL for the trace's static mapping
82  			tool.getProject()
83  					.getProjectData()
84  					.getRootFolder()
85  					.createFile("emu_example", program, monitor);
86  			try (Transaction tx = program.openTransaction("Init")) {
87  				AddressSpace space = program.getAddressFactory().getDefaultAddressSpace();
88  				entry = space.getAddress(0x00400000);
89  				Address dataEntry = space.getAddress(0x00600000);
90  				Memory memory = program.getMemory();
91  				memory.createInitializedBlock(".text", entry, 0x1000, (byte) 0, monitor, false);
92  				Assembler asm = Assemblers.getAssembler(program);
93  				InstructionIterator ii = asm.assemble(entry,
94  					"MOV RCX, 0x" + dataEntry,
95  					"MOV RAX, 1",
96  					"SYSCALL",
97  					"MOV RAX, 2",
98  					"SYSCALL");
99  				ii.next(); // drop MOV RCX
100 				injectHere = ii.next().getAddress();
101 				memory.createInitializedBlock(".data", dataEntry, 0x1000, (byte) 0, monitor, false);
102 				memory.setBytes(dataEntry, "Hello, World!\n".getBytes(UTF8));
103 			}
104 			program.save("Init", monitor);
105 			// Display the program in the UI
106 			programManager.openProgram(program);
107 		}
108 		finally {
109 			if (program != null) {
110 				program.release(this);
111 			}
112 		}
113 
114 		/*
115 		 * Now, load the program into a trace. This doesn't copy any bytes, it just sets up a static
116 		 * mapping. The emulator will know how to read through to the mapped program. We use a
117 		 * utility, which is the same used by the "Emulate Program" action in the UI. It will load
118 		 * the program, allocate a stack, and initialize the first thread to the given entry.
119 		 */
120 		Trace trace = null;
121 		try {
122 			trace = ProgramEmulationUtils.launchEmulationTrace(program, entry, this);
123 			// Display the trace in the UI
124 			traceManager.openTrace(trace);
125 			traceManager.activateTrace(trace);
126 		}
127 		finally {
128 			if (trace != null) {
129 				trace.release(this);
130 			}
131 		}
132 		// Get the initial thread
133 		TraceThread traceThread = trace.getThreadManager().getAllThreads().iterator().next();
134 		traceManager.activateThread(traceThread);
135 
136 		/*
137 		 * Instead of using the UI's emulator, this script will create its own with a custom
138 		 * library. This emulator will still know how to integrate with the UI, reading through to
139 		 * open programs and writing state back into the trace.
140 		 */
141 		TracePlatform host = trace.getPlatformManager().getHostPlatform();
142 		DefaultPcodeDebuggerAccess access = new DefaultPcodeDebuggerAccess(tool, null, host, 0);
143 		BytesDebuggerPcodeEmulator emulator = new BytesDebuggerPcodeEmulator(access) {
144 			@Override
145 			protected PcodeUseropLibrary<byte[]> createUseropLibrary() {
146 				return new DemoPcodeUseropLibrary(language, DebuggerEmuExampleScript.this);
147 			}
148 		};
149 		// Conventionally, emulator threads are named after their trace thread's path.
150 		PcodeThread<byte[]> thread = emulator.getThread(traceThread.getPath(), true);
151 
152 		/*
153 		 * Inject a call to our custom print userop. Otherwise, the language itself will never
154 		 * invoke it.
155 		 */
156 		emulator.inject(injectHere, """
157 				print_utf8(RCX);
158 				emu_exec_decoded();
159 				""");
160 
161 		/*
162 		 * Run the experiment: This should interrupt on the second SYSCALL, because any value other
163 		 * than 1 calls emu_swi.
164 		 * 
165 		 * For demonstration, we'll record a trace snapshot for every step of emulation. This is not
166 		 * ordinarily recommended except for very small experiments. A more reasonable approach in
167 		 * practice may be to snapshot on specific breakpoints.
168 		 */
169 		TraceTimeManager time = trace.getTimeManager();
170 		TraceSnapshot snapshot = time.getSnapshot(0, true);
171 		try (Transaction tx = trace.openTransaction("Emulate")) {
172 			for (int i = 0; i < 10; i++) {
173 				println("Executing: " + thread.getCounter());
174 				thread.stepInstruction();
175 				snapshot =
176 					time.createSnapshot("Stepped to " + thread.getCounter());
177 				emulator.writeDown(host, snapshot.getKey(), 0);
178 			}
179 			printerr("We should not have completed 10 steps!");
180 		}
181 		catch (InterruptPcodeExecutionException e) {
182 			println("Terminated via interrupt. Good.");
183 		}
184 		// Display the final snapshot in the UI
185 		traceManager.activateSnap(snapshot.getKey());
186 
187 		/*
188 		 * Inspect the machine. You can always do this by accessing the state directly, but for
189 		 * anything other than simple variables, you may find compiling an expression more
190 		 * convenient.
191 		 * 
192 		 * This works the same as in the stand-alone case.
193 		 */
194 		println("RCX = " +
195 			Utils.bytesToLong(thread.getState().getVar(language.getRegister("RCX"), Reason.INSPECT),
196 				8, language.isBigEndian()));
197 
198 		println("RCX = " + Utils.bytesToLong(
199 			SleighProgramCompiler.compileExpression(language, "RCX").evaluate(thread.getExecutor()),
200 			8, language.isBigEndian()));
201 
202 		println("RCX+4 = " +
203 			Utils.bytesToLong(SleighProgramCompiler.compileExpression(language, "RCX+4")
204 					.evaluate(thread.getExecutor()),
205 				8, language.isBigEndian()));
206 
207 		/*
208 		 * To evaluate a Sleigh expression against the trace: The result is the same as evaluating
209 		 * directly against the emulator, but these work with any trace, no matter the original data
210 		 * source (live target, emulated, imported, etc.) It's also built into utilities, making it
211 		 * easier to use.
212 		 */
213 		println("RCX+4 (trace) = " +
214 			TraceSleighUtils.evaluate("RCX+4", trace, snapshot.getKey(), traceThread, 0));
215 	}
216 }
