16  //Trivial search for
17  //embedded binaries
18  //@category Binary
19  
20  import java.util.*;
21  
22  import ghidra.app.script.GhidraScript;
23  import ghidra.program.model.address.Address;
24  import ghidra.program.model.mem.Memory;
25  
26  /**
27   * EmbeddedFinder runs a trivial byte search across a file (input) and searches for potential embedded PE files (targets)
28   * <p>
29   * It will return an identified target if the target's NT header is where the DOS header says it should be
30   * <p>
31   * Currently this is the only sanity check it runs
32   */
33  public class EmbeddedFinderScript extends GhidraScript {
34  
35  	@Override
36      public void run() throws Exception {
37  		byte[] MAGIC_DOS_HEADER = new byte[] { 0x4d, 0x5a };				// M Z
38  		byte[] MAGIC_NT_HEADER  = new byte[] { 0x50, 0x45, 0x00, 0x00 };	// P E 0x00 0x00
39  
40  		List<Address> allFound = new ArrayList<Address>();
41  
42  		Memory memory = currentProgram.getMemory();
43  		Address baseAddr = memory.getMinAddress();
44  		Address currAddr = baseAddr;
45  
46  		while (currAddr != null) {
47  			// The purpose of breaking each check into small segments (where they could be combined)
48  			// is to make way for future file type support, keep code clean, and to encourage readability.
49  			boolean DOSExists = false;
50  			boolean NTExists = false;
51  			boolean DOSAgreesWithNT = false;
52  
53  			Address DOS = memory.findBytes(currAddr, MAGIC_DOS_HEADER, null, true, getMonitor());
54  			if (DOS != null) {
55  				// IMAGE_DOS_HEADER is 128 bytes in length, so let's check if that much memory is available
56  				if (memory.contains(DOS.add(128)))
57  					DOSExists = true;
58  			}
59  
60  			Address NT = memory.findBytes(DOS, MAGIC_NT_HEADER, null, true, getMonitor());
61  			if (NT != null) {
62  				// IMAGE_NT_HEADERS32 is 80 bytes in length, so let's check if that much memory is available
63  				if (memory.contains(NT.add(80)))
64  					NTExists = true;
65  			}
66  
67  			if (DOSExists && NTExists) {
68  				// It would be better to import the proper structs rather than hard coding offsets.
69  				// However I'm unsure of what the best way of doing this would be. It's possible to include WINNT.h
70  				// but this requires the non-development environment to have access to it which makes things
71  				// less flexible and renders it brittle for future embedded target-type searches.
72  				// IMAGE_DOS_HEADER + 0x3c is the IMAGE_NT_HEADERS32 offset
73  				long impliedOffset = memory.getShort(DOS.add(0x3c));
74  				long actualOffset = NT.getAddressableWordOffset() - DOS.getAddressableWordOffset();
75  				if (impliedOffset == actualOffset)
76  					DOSAgreesWithNT = true;
77  			}
78  
79  			if (DOSAgreesWithNT) {
80  				byte[] MAGIC_NT_HEADER_TEST = new byte[4];	// [TODO] Get this to dynamically pull correct size, not hardcoded
81  				memory.getBytes(NT, MAGIC_NT_HEADER_TEST);
82  
83  				if (Arrays.equals(MAGIC_NT_HEADER, MAGIC_NT_HEADER_TEST)) {
84  					if (DOS != baseAddr)
85  						allFound.add(DOS);		// We only care about targets that are not also the parent file
86  				}
87  			}
88  
89  			if (DOS != null)
90  				currAddr = DOS.add(1);	// Ensure next search doesn't overlap with current target
91  			else
92  				currAddr = null;
93  		}
94  
95  		// Present user with target discovery(s)
96  
97  		if (allFound.isEmpty())
98  			println("No embedded targets identified");
99  		else {
100 			println("Embedded targets identified");
101 			for (Address found : allFound)
102 				println("\t" + found.toString());
103 		}
104     }
105 }
