37  public class RecursiveRecursiveMSLibImport extends GhidraScript {
38  	final static Predicate<Loader> LOADER_FILTER = new SingleLoaderFilter(MSCoffLoader.class);
39  	final static LoadSpecChooser LOADSPEC_CHOOSER = new CsHintLoadSpecChooser("windows");
40  
41  	@Override
42  	protected void run() throws Exception {
43  		DomainFolder non_debug_root =
44  			askProjectFolder("Choose a top-level domain folder for NON-DEBUG libraries");
45  		DomainFolder debug_root =
46  			askProjectFolder("Choose a top-level domain folder for DEBUG libraries");
47  
48  		// ALL .LIB files under this directory will be inspected/imported as Win COFF
49  		File directory = askDirectory("Choose the top-level import directory", "Choose");
50  		String directoryPath = directory.getAbsolutePath();
51  
52  		ArrayList<File> non_debug_files = new ArrayList<File>();
53  		ArrayList<File> debug_files = new ArrayList<File>();
54  
55  		findFiles(non_debug_files, debug_files, directory);
56  
57  		for (File file : non_debug_files) {
58  			MessageLog log = new MessageLog();
59  			importLibrary(formLibraryPath(non_debug_root, directoryPath, file), file, log);
60  		}
61  
62  		for (File file : debug_files) {
63  			MessageLog log = new MessageLog();
64  			importLibrary(formLibraryPath(debug_root, directoryPath, file), file, log);
65  		}
66  	}
67  
68  	private void findFiles(ArrayList<File> non_debug_files, ArrayList<File> debug_files,
69  			File directory) {
70  		ArrayList<File> subdirs = new ArrayList<File>();
71  		ArrayList<File> my_non_debug = new ArrayList<File>();
72  		ArrayList<File> my_debug = new ArrayList<File>();
73  		File[] files = directory.listFiles();
74  
75  		if (files != null) {
76  			for (File file : files) {
77  				if (file.isFile()) {
78  					String lowerName = file.getName().toLowerCase();
79  					if (lowerName.endsWith("d.lib")) {
80  						my_debug.add(file);
81  					}
82  					else if (lowerName.endsWith(".lib")) {
83  						my_non_debug.add(file);
84  					}
85  				}
86  				else if (file.isDirectory()) {
87  					subdirs.add(file);
88  				}
89  			}
90  		}
91  
92  		for (File file : my_debug) {
93  			String lowerName = file.getName().toLowerCase();
94  			String non_debug_name = lowerName.substring(0, lowerName.length() - 5) + ".lib";
95  			boolean notfound = true;
96  			int ii = 0;
97  			while (notfound && ii < my_non_debug.size()) {
98  				File non_debug = my_non_debug.get(ii);
99  				if (non_debug.getName().toLowerCase().equals(non_debug_name)) {
100 					notfound = false;
101 				}
102 				++ii;
103 			}
104 			if (notfound) {
105 				non_debug_files.add(file);
106 			}
107 			else {
108 				debug_files.add(file);
109 			}
110 		}
111 
112 		non_debug_files.addAll(my_non_debug);
113 
114 		for (File subdir : subdirs) {
115 			findFiles(non_debug_files, debug_files, subdir);
116 		}
117 	}

119 	private DomainFolder formLibraryPath(DomainFolder root, String directoryPath, File file)
120 			throws InvalidNameException, IOException {
121 		String filePath = file.getAbsolutePath();
122 		String rest = filePath.substring(directoryPath.length() + 1);
123 		Pair<DomainFolder, String> pair = establishProgramFolder(root, rest);
124 		DomainFolder folder =
125 			pair.first.createFolder(mangleNameBecauseDomainFoldersAreSoRetro(pair.second));
126 		return folder;
127 	}
128 
129 	private void importLibrary(DomainFolder currentLibrary, File file, MessageLog log)
130 			throws CancelledException, DuplicateNameException, InvalidNameException,
131 			VersionException, IOException {
132 
133 		try (RandomAccessByteProvider provider = new RandomAccessByteProvider(file)) {
134 			if (!CoffArchiveHeader.isMatch(provider)) {
135 				return;
136 			}
137 			CoffArchiveHeader coffArchiveHeader =
138 				CoffArchiveHeader.read(provider, TaskMonitor.DUMMY);
139 			HashSet<Long> offsetsSeen = new HashSet<Long>();
140 			for (CoffArchiveMemberHeader archiveMemberHeader : coffArchiveHeader.getArchiveMemberHeaders()) {
141 				if (offsetsSeen.contains(archiveMemberHeader.getPayloadOffset())) {
142 					continue;
143 				}
144 				offsetsSeen.add(archiveMemberHeader.getPayloadOffset());
145 				if (archiveMemberHeader.isCOFF()) {
146 					try (ByteProvider coffProvider = new ByteProviderWrapper(provider,
147 						archiveMemberHeader.getPayloadOffset(), archiveMemberHeader.getSize())) {
148 						CoffFileHeader header = new CoffFileHeader(coffProvider);
149 						if (CoffMachineType.isMachineTypeDefined(header.getMagic())) {
150 							String preferredName = archiveMemberHeader.getName();
151 
152 							Pair<DomainFolder, String> pair =
153 								establishProgramFolder(currentLibrary, preferredName);
154 
155 							LoadResults<? extends DomainObject> loadResults =
156 								AutoImporter.importFresh(coffProvider, state.getProject(),
157 									pair.first.getPathname(), this, log, monitor, LOADER_FILTER,
158 									LOADSPEC_CHOOSER,
159 									mangleNameBecauseDomainFoldersAreSoRetro(pair.second),
160 									OptionChooser.DEFAULT_OPTIONS);
161 
162 							try {
163 								loadResults.save(state.getProject(), this, log, monitor);
164 							}
165 							finally {
166 								loadResults.release(this);
167 							}
168 						}
169 					}
170 				}
171 			}
172 		}
173 		catch (CoffException e) {
174 			//TODO
175 		}
176 	}
177 
178 	private Pair<DomainFolder, String> establishProgramFolder(DomainFolder current,
179 			String preferredName) throws InvalidNameException, IOException {
180 		String[] splits = preferredName.split("[/\\\\]");
181 		for (int ii = 0; ii < splits.length - 1; ++ii) {
182 			String nextName = splits[ii];
183 			DomainFolder next = current.getFolder(nextName);
184 			if (next == null) {
185 				next = current.createFolder(mangleNameBecauseDomainFoldersAreSoRetro(nextName));
186 			}
187 			current = next;
188 		}
189 		return new Pair<DomainFolder, String>(current, splits[splits.length - 1]);
190 	}
191 
192 	private String mangleNameBecauseDomainFoldersAreSoRetro(String name) {
193 		StringBuilder sb = new StringBuilder();
194 		char[] charArray = name.toCharArray();
195 		for (char c : charArray) {
196 			if (!LocalFileSystem.isValidNameCharacter(c)) {
197 				c = '_';
198 			}
199 			sb.append(c);
200 		}
201 		return sb.toString();
202 	}
