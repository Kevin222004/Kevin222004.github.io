16  //Finds PNG and GIF images and applies data type if not already applied
17  //@category Images
18  import ghidra.app.script.GhidraScript;
19  import ghidra.program.model.address.Address;
20  import ghidra.program.model.data.GifDataType;
21  import ghidra.program.model.data.PngDataType;
22  import ghidra.program.model.listing.Data;
23  import ghidra.program.model.mem.Memory;
24  import ghidra.program.model.mem.MemoryBlock;
25  
26  import java.util.ArrayList;
27  import java.util.List;
28  
29  public class FindImagesScript extends GhidraScript {
30  
31  	@Override
32  	public void run() throws Exception {
33  
34  		int numValidImagesFound = 0;
35  		//Look for potential GIF images in binary using image header byte patterns
36  		println("Looking for GIF and PNG images in " + currentProgram.getName());
37  		List<Address> foundGIFS = scanForGIF87aImages();
38  		foundGIFS.addAll(scanForGIF89aImages());
39  
40  		//Loop over all potential found GIFs
41  		for (int i = 0; i < foundGIFS.size(); i++) {
42  			boolean foundGIFImage = false;
43  			//See if already applied GIF
44  			Data data = getDataAt(foundGIFS.get(i));
45  			//If not already applied, try to apply GIF data type
46  			if (data == null) {
47  				println("Trying to apply GIF datatype at " + foundGIFS.get(i).toString());
48  				try {
49  					Data newGIF = createData(foundGIFS.get(i), new GifDataType());
50  					if (newGIF != null) {
51  						println("Applied GIF at " + newGIF.getAddressString(false, true));
52  						foundGIFImage = true;
53  					}
54  				}
55  				//If GIF does not apply correctly then it is not really a GIF data type
56  				catch (Exception e) {
57  					println("Invalid GIF at " + foundGIFS.get(i).toString());
58  				}
59  			}
60  			else if (data.getMnemonicString().equals("GIF")) {
61  				println("GIF already applied at " + data.getAddressString(false, true));
62  				foundGIFImage = true;
63  			}
64  
65  			//print found message only for those that apply corrrectly or were already applied
66  			if (foundGIFImage) {
67  				println("Found GIF in program " + currentProgram.getExecutablePath() +
68  					" at address " + foundGIFS.get(i).toString());
69  				numValidImagesFound++;
70  			}
71  
72  		}
73  
74  		//Look for potential PNG images in binary using image header byte patterns
75  		List<Address> foundPNGS = scanForPNGs();
76  
77  		//Loop over all potential found PNGs
78  		for (int i = 0; i < foundPNGS.size(); i++) {
79  			boolean foundPNGImage = false;
80  			//See if already applied PNG
81  			Data data = getDataAt(foundPNGS.get(i));
82  
83  			//If not already applied, try to apply PNG data type
84  			if (data == null) {
85  				println("Trying to apply PNG datatype at " + foundPNGS.get(i).toString());
86  				try {
87  					Data newPNG = createData(foundPNGS.get(i), new PngDataType());
88  					if (newPNG != null) {
89  						println("Applied PNG at " + newPNG.getAddressString(false, true));
90  						foundPNGImage = true;
91  					}
92  				}
93  				//If PNG does not apply correctly then it is not really a PNG data type
94  				catch (Exception e) {
95  					println("Invalid PNG at " + foundPNGS.get(i).toString());
96  				}
97  			}
98  			else if (data.getMnemonicString().equals("PNG")) {
99  				println("PNG already applied at " + data.getAddressString(false, true));
100 				foundPNGImage = true;
101 			}
102 
103 			//print found message only for those that apply corrrectly or were already applied
104 			if (foundPNGImage) {
105 				println("Found PNG in program " + currentProgram.getExecutablePath() +
106 					" at address " + foundPNGS.get(i).toString());
107 				numValidImagesFound++;
108 			}
109 		}
110 		if (numValidImagesFound == 0) {
111 			println("No PNG or GIF images found in " + currentProgram.getName());
112 			if (this.isRunningHeadless()) {
113 				currentProgram.setTemporary(true);
114 			}
115 		}
116 
117 	}
118 
119 	List<Address> scanForGIF87aImages() {
120 
121 		byte gifBytes[] = new byte[6];
122 		gifBytes[0] = (byte) 0x47;
123 		gifBytes[1] = (byte) 0x49;
124 		gifBytes[2] = (byte) 0x46;
125 		gifBytes[3] = (byte) 0x38;
126 		gifBytes[4] = (byte) 0x37;
127 		gifBytes[5] = (byte) 0x61;
128 
129 		List<Address> foundGIFS = scanForImages(gifBytes);
130 		return foundGIFS;
131 	}
132 
133 	List<Address> scanForGIF89aImages() {
134 
135 		byte gifBytes[] = new byte[6];
136 		gifBytes[0] = (byte) 0x47;
137 		gifBytes[1] = (byte) 0x49;
138 		gifBytes[2] = (byte) 0x46;
139 		gifBytes[3] = (byte) 0x38;
140 		gifBytes[4] = (byte) 0x39;
141 		gifBytes[5] = (byte) 0x61;
142 
143 		List<Address> foundGIFS = scanForImages(gifBytes);
144 		return foundGIFS;
145 	}
146 
147 	List<Address> scanForPNGs() {
148 
149 		byte pngBytes[] = new byte[8];
150 		pngBytes[0] = (byte) 0x89;
151 		pngBytes[1] = (byte) 0x50;
152 		pngBytes[2] = (byte) 0x4e;
153 		pngBytes[3] = (byte) 0x47;
154 		pngBytes[4] = (byte) 0x0d;
155 		pngBytes[5] = (byte) 0x0a;
156 		pngBytes[6] = (byte) 0x1a;
157 		pngBytes[7] = (byte) 0x0a;
158 
159 		List<Address> foundPNGs = scanForImages(pngBytes);
160 		return foundPNGs;
161 
162 	}
163 
164 	List<Address> scanForImages(byte[] imageBytes) {
165 		Memory memory = currentProgram.getMemory();
166 		MemoryBlock[] blocks = memory.getBlocks();
167 
168 		byte maskBytes[] = null;
169 
170 		List<Address> foundImages = new ArrayList<Address>();
171 
172 		for (int i = 0; i < blocks.length; i++) {
173 			if (blocks[i].isInitialized()) {
174 				Address start = blocks[i].getStart();
175 				Address found = null;
176 				while (true) {
177 					if (monitor.isCancelled()) {
178 						break;
179 					}
180 					found =
181 						memory.findBytes(start, blocks[i].getEnd(), imageBytes, maskBytes, true,
182 							monitor);
183 					if (found != null) {
184 						foundImages.add(found);
185 						start = found.add(1);
186 					}
187 					else
188 						break;
189 				}
190 			}
191 		}
192 		return foundImages;
193 	}
