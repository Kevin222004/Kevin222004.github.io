35  /**
36   * This script launches the {@link InstructionSearchDialog} and populates it
37   * with the currently- selected bytes in the listing. The YARA search string
38   * matching the instructions selected is displayed in a text window and may be
39   * modified in two ways:
40   * 
41   * 1. By directly editing the text 2. By toggling the cells in the instruction
42   * table, which masks the appropriate mnemonic/instruction
43   * 
44   * The generated string will appear similar to the following:
45   * 
46   * rule <rule name> { strings: $STR1 = { 4? 56 4? 55 4? 54 ?? 50 4? 8b ?? c7 4d
47   * 00 00 4?8d 1d 08 4e 00 00 }
48   *
49   * condition: $STR1 }
50   * 
51   * Note that this script uses only a portion of the existing search dialog
52   * mentioned above; that entire dialog is far too bulky for what is needed here,
53   * so we just launch the part that allows users to toggle the mnemonics and
54   * operands on/off (masking).
55   * 
56   * In addition to this existing dialog, some custom components are added to it
57   * in order to display the resulting Yara string. {@link #YaraDialog}.
58   * 
59   */
60  public class YaraGhidraGUIScript extends GhidraScript {
61  
62  	/**
63  	 * The plugin provides all access to the {@link InstructionSearchDialog} and
64  	 * its components.
65  	 */
66  	private InstructionSearchPlugin plugin;
67  
68  	private InstructionSearchDialog dialog;
69  
70  	/*********************************************************************************************
71  	 * PROTECTED METHODS
72  	 ********************************************************************************************/
73  
74  	@Override
75  	protected void run() throws Exception {
76  
77  		// First we have to find the plugin - if this hasn't been installed it won't be available,
78  		// hence the null check.
79  		plugin = InstructionSearchUtils.getInstructionSearchPlugin(state.getTool());
80  
81  		// Now check some error conditions and notify the user if there are issues.
82  		if (plugin == null) {
83  			popup("Instruction Pattern Search plugin not installed! Please install and " +
84  				"re-run script.");
85  			return;
86  		}
87  
88  		if (currentProgram == null) {
89  			popup("Please open a program before running this script.");
90  			return;
91  		}
92  
93  		if (currentSelection == null) {
94  			popup(
95  				"Please make a valid selection in the program and select 'reload'. Or select the " +
96  					"'manual entry' option from the toolbar.");
97  		}
98  
99  		// Next, create and open a new Yara dialog.
100 		dialog = new YaraDialog();
101 		state.getTool().showDialog(dialog);
102 
103 		// Finally, load whatever instructions are selected in the listing.
104 		dialog.loadInstructions(plugin);
105 	}

107 	/*********************************************************************************************
108 	 * PRIVATE METHODS

111 	/**
112 	 * Creates a correctly-formatted Yara string with masking where appropriate.
113 	 * Yara strings have the following general format:
114 	 * 
115 	 * rule <rule name> { strings: $STR1 = { 4? 56 4? 55 4? 54 ?? 50 4? 8b ?? c7
116 	 * 4d 00 00 4?8d 1d 08 4e 00 00 }
117 	 *
118 	 * condition: $STR1 }
119 	 * 
120 	 * @param ruleName
121 	 * @return
122 	 */
123 	private String generateYaraString(String ruleName) {
124 
125 		StringBuilder yaraString = new StringBuilder("\n\nrule " + ruleName + "\n");
126 		yaraString.append("{\n\tstrings:\n");
127 
128 		String fullStr = "";
129 
130 		// Get the "combined string" from the search data object; this is the ENTIRE set of 
131 		// instructions in one string, with all masking applied.
132 		if (dialog == null || dialog.getSearchData() == null) {
133 			return null;
134 		}
135 		String instrStr = dialog.getSearchData().getCombinedString();
136 
137 		// Loop over the combined string, converting each nibble to hex. If we don't have enough
138 		// bytes to create a nibble, or if part of the nibble is masked, fill with a '?' char. This
139 		// is a Yara constraint - it can only handle displaying data down to the nibble level.
140 		//
141 		// ie: 10100110 -> A6
142 		//     101001.. -> A?
143 		//     ..11.001 -> ??
144 		//
145 		for (int i = 0; i < instrStr.length(); i += 8) {
146 
147 			String curByte =
148 				instrStr.length() >= 8 ? instrStr.substring(i, i + 8) : instrStr.substring(i);
149 			String nibble1 = curByte.length() >= 4 ? curByte.substring(0, 4) : curByte.substring(0);
150 			String nibble2 = curByte.length() >= 8 ? curByte.substring(4, 8)
151 					: curByte.length() >= 4 ? curByte.substring(4) : "";
152 
153 			if (nibble1.contains(".")) {
154 				fullStr += "?";
155 			}
156 			else {
157 				fullStr += InstructionSearchUtils.toHex(nibble1, false).trim();
158 			}
159 
160 			if (nibble2.contains(".")) {
161 				fullStr += "?";
162 			}
163 			else {
164 				fullStr += InstructionSearchUtils.toHex(nibble2, false).trim();
165 			}
166 
167 			fullStr += " ";
168 		}
169 
170 		// Add the formatted string to our final output, and add some boilerplate Yara 
171 		// stuff.
172 		yaraString.append("\t\t$STR" + 1 + " = { " + fullStr + " }\n");
173 		yaraString.append("\n\tcondition:\n");
174 		yaraString.append("\t\t$STR1");
175 		yaraString.append("\n}\n");
176 
177 		return yaraString.toString();
178 	}

184 	/**
185 	 * This dialog is a hybrid, containing parts of the
186 	 * {@link InstructionTablePanel}, which allows users to mask
187 	 * mnemonics/operands in an instruction set, and some custom pieces for
188 	 * displaying the Yara string.
189 	 * 
190 	 * The layout:
191 	 * 
192 	 * --------------------------- | | | Instruction Table Panel | | |
193 	 * |-------------------------| | | | YARA Text | | |
194 	 * ---------------------------
195 	 */
196 	private class YaraDialog extends InstructionSearchDialog {
197 
198 		// The area where the yara search string is displayed.
199 		private JTextArea yaraTA;
200 		JScrollPane scrollPane;
201 
202 		// Use a splitter to separate the masking panel from the yara text area.
203 		private JSplitPane verticalSplitter;
204 
205 		// Keep track of the splitter location so it can be restored when the dialog is
206 		// refreshed.  
207 		private int splitterSave = 200;
208 
209 		/**
210 		 * Constructor.
211 		 */
212 		private YaraDialog() {
213 			super(plugin, "Yara Search String Generator", null);
214 			revalidate();
215 			setPreferredSize(500, 400);
216 		}
217 
218 		/**
219 		 * The dialog we're using here is a modified form of the
220 		 * {@link InstructionSearchDialog}; this one contains only the
221 		 * {@link InstructionTablePanel} portion, which provides the operand
222 		 * masking capability.
223 		 * 
224 		 * To accomplish this we override the method that constructs the UI
225 		 * components and add just the components we need.
226 		 * 
227 		 * @return
228 		 */
229 		@Override
230 		protected JPanel createWorkPanel() {
231 
232 			// Create the main text area and give it a scroll bar. 
233 			yaraTA = new JTextArea(12, 0);
234 			scrollPane = new JScrollPane(yaraTA);
235 			yaraTA.setWrapStyleWord(true);
236 			yaraTA.setLineWrap(true);
237 
238 			// Create the instruction table and set it as a listener of the table model, so 
239 			// this gui will be notified when changes have been made (when the user has adjusted
240 			// the mask settings).  This allows us to dynamically update the yara string as 
241 			// the user is changing things.
242 			InstructionTablePanel instructionTablePanel =
243 				new InstructionTablePanel(searchData.getMaxNumOperands(), plugin, this);
244 			instructionTablePanel.getTable().getModel().addTableModelListener(e -> {
245 				generateYara();
246 			});
247 
248 			// Finally, set up the main panel and create a split pane so the user can adjust
249 			// the dimensions of the masking table and the yara text display.
250 			JPanel mainPanel = new JPanel();
251 			mainPanel.setLayout(new BorderLayout());
252 			verticalSplitter = new JSplitPane(JSplitPane.VERTICAL_SPLIT,
253 				instructionTablePanel.getWorkPanel(), scrollPane);
254 			mainPanel.add(verticalSplitter);
255 
256 			// Tell the data model to listen for changes the table; when the GUI is updated, the
257 			// model needs to be told to update itself to reflect the new mask settings. 
258 			searchData.registerForGuiUpdates(instructionTablePanel.getTable());
259 
260 			// Now restore the splitter location to whatever it was before.
261 			verticalSplitter.setDividerLocation(splitterSave);
262 
263 			return mainPanel;
264 		}
265 
266 		/**
267 		 * Creates a properly-formatted yara string and displays it in the text
268 		 * area.
269 		 */
270 		private void generateYara() {
271 			try {
272 				yaraTA.setText(generateYaraString("<insert name>"));
273 			}
274 			catch (Exception e1) {
275 				Msg.error(this, "Error generating yara string: " + e1);
276 			}
277 		}
278 
279 		/**
280 		 * Part of the {@link Observer} structure. This is invoked whenever the
281 		 * {@link InstructionSearchData} class is updated, indicating that the
282 		 * user has selected a new set of instructions, or changed mask
283 		 * settings. When this happens we need to save off the splitter location
284 		 * and reload the dialog if necessary.
285 		 * 
286 		 * @param o
287 		 * @param arg
288 		 */
289 		@Override
290 		public void update(Observable o, Object arg) {
291 
292 			// Before rebuilding the UI, remember the splitter location so we can reset it
293 			// afterwards.
294 			if (verticalSplitter != null) {
295 				splitterSave = verticalSplitter.getDividerLocation();
296 			}
297 
298 			if (arg instanceof UpdateType) {
299 				UpdateType type = (UpdateType) arg;
300 				switch (type) {
301 					case RELOAD:
302 						revalidate();
303 						break;
304 					case UPDATE:
305 						// do nothing
306 				}
307 			}
308 		}
309 
310 		/**
311 		 * Updates the GUI when the user has made a new selection.
312 		 */
313 		@Override
314 		protected void revalidate() {
315 			removeWorkPanel();
316 			addWorkPanel(createWorkPanel());
317 			generateYara();
318 		}
319 	}
