16  import java.util.Set;
17  
18  import ghidra.app.plugin.core.debug.DebuggerCoordinates;
19  import ghidra.app.script.GhidraScript;
20  import ghidra.app.services.*;
21  import ghidra.dbg.DebuggerObjectModel;
22  import ghidra.dbg.DebuggerObjectModel.RefreshBehavior;
23  import ghidra.dbg.target.TargetObject;
24  import ghidra.dbg.target.TargetRegisterBank;
25  import ghidra.dbg.util.PathMatcher;
26  
27  public class RefreshRegistersScript extends GhidraScript {
28  	@Override
29  	protected void run() throws Exception {
30  		// There is no need to fish this from the ObjectUpdateService, you can get it directly
31  		DebuggerModelService modelService = state.getTool().getService(DebuggerModelService.class);
32  		// The current model is retrieved with one method, no need to stream or filter
33  		DebuggerObjectModel model = modelService.getCurrentModel();
34  
35  		/**
36  		 * Navigating a model generically requires some introspection. Use the schema. We used to
37  		 * decouple the descriptions (RegisterContainer) from the values (RegisterBank), but we
38  		 * always realize the two interfaces on the same no it seems. Still, we need to work with
39  		 * values, so we search for all the Banks.
40  		 */
41  		PathMatcher allBanksMatcher =
42  			model.getRootSchema().searchFor(TargetRegisterBank.class, true);
43  		for (TargetObject objBank : allBanksMatcher.fetchSuccessors(model.fetchModelRoot().get())
44  				.get()
45  				.values()) {
46  			// Because of a bug in our path search, this type check is still necessary :(
47  			if (!(objBank instanceof TargetRegisterBank)) {
48  				continue;
49  			}
50  			TargetRegisterBank bank = (TargetRegisterBank) objBank;
51  			// This is equivalent to hitting the "Flush Caches" button in the Target Provider
52  			bank.invalidateCaches().get();
53  			// If you know the names of the registers to read
54  			bank.readRegistersNamed("rax", "rbx").get();
55  			// Values are coupled to elements, so we can also just refresh them to re-read all
56  			bank.fetchElements(RefreshBehavior.REFRESH_ALWAYS).get();
57  		}
58  
59  		/**
60  		 * Alternatively, to refresh just the bank for the current thread or frame, we need to
61  		 * determine the active frame. We'll do that by asking the TraceManagerService. Then,
62  		 * because that's in "trace land," we'll use the TraceRecorder to map that into "target
63  		 * land." Generally, you'd then need to navigate the schema as before, but relative to the
64  		 * target object. Because fetching registers is so common, this is already built in to the
65  		 * recorder.
66  		 */
67  		DebuggerTraceManagerService traceManager =
68  			state.getTool().getService(DebuggerTraceManagerService.class);
69  		// There are also getCurreentTrace(), etc., if you want just the one thing
70  		DebuggerCoordinates current = traceManager.getCurrent();
71  
72  		// Now, we need to get the relevant recorder
73  		TraceRecorder recorder = modelService.getRecorder(current.getTrace());
74  		// There's a chance of an NPE here if there is no "current frame"
75  		Set<TargetRegisterBank> banks =
76  			recorder.getTargetRegisterBanks(current.getThread(), current.getFrame());
77  		for (TargetRegisterBank bank : banks) {
78  			// Now do the same to the bank as before
79  			bank.invalidateCaches().get();
80  			bank.fetchElements(RefreshBehavior.REFRESH_ALWAYS).get();
81  		}
82  	}
83  }
