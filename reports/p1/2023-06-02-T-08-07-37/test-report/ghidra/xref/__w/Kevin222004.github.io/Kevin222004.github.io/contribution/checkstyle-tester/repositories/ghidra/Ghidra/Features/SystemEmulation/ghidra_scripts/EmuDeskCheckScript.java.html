51  		Var RAX = lang("RAX", type("void *"));
52  		Var RSP = lang("RSP", type("void *"));

54  		protected Injects() {
55  			super(currentProgram);

58  		public Var POP() {
59  			Var tgt = local("tgt", RSP.cast(type("void **")).deref());
60  			RSP.set(RSP.addi(8));
61  			return tgt;
62  		}
63  
64  		public void RET() {
65  			_goto(POP());
66  		}
67  
68  		public void RET(RVal val) {
69  			RAX.set(val);
70  			RET();
71  		}

74  		 * TODO: A framework for stubbing the functions. This is close, and the system calls stuff
75  		 * can get us closer in its handling of calling conventions. We need either to generate
76  		 * Sleigh that gets the parameters in place, or if we're going to use the aliasing idea that
77  		 * the syscall stuff does, then we need to allow injection of the already-compiled Sleigh
78  		 * program. For now, we'll have to declare the parameter-holding register as a language
79  		 * variable.
80  		 */
81  		@StructuredUserop
82  		public void strlen(/*@Param(name = "RDI", type = "char *") Var s*/) {
83  			Var s = lang("RDI", type("char *"));
84  			Var t = temp(type("char *"));
85  			_for(t.set(s), t.deref().neq(0), t.inc(), () -> {

87  			RET(t.subi(s));

91  	public final List<Watch> watches = List.of(
92  		watch("RAX", type("int")),
93  		watch("RCX", type("int"),
94  			set(FormatSettingsDefinition.DEF, FormatSettingsDefinition.DECIMAL)),
95  		watch("RSP", type("void *")));

98  	@Override
99  	protected void run() throws Exception {
100 		Trace trace = emulateLaunch(currentProgram, currentAddress);
101 		TracePlatform platform = trace.getPlatformManager().getHostPlatform();
102 		long snap = 0;
103 
104 		TableChooserDialog tableDialog =
105 			createTableChooserDialog("Desk Check", new CheckRowChooser());
106 
107 		tableDialog.show();
108 
109 		tableDialog.addCustomColumn(new CheckRowScheduleDisplay());
110 		tableDialog.addCustomColumn(new CheckRowCounterDisplay());
111 
112 		List<PcodeExpression> compiled = new ArrayList<>();
113 		TypeLoader loader = new TypeLoader(currentProgram);
114 		for (Watch w : watches) {
115 			PcodeExpression ce = SleighProgramCompiler
116 					.compileExpression((SleighLanguage) platform.getLanguage(), w.expression);
117 			tableDialog.addCustomColumn(new CheckRowWatchDisplay(loader, w, compiled.size()));
118 			compiled.add(ce);
119 		}
120 
121 		TraceSchedule schedule;
122 		while (true) {
123 			try {
124 				schedule = TraceSchedule
125 						.parse(askString("Schedule", "Enter the steping schedule", "0:t0-1000"));
126 				break;
127 			}
128 			catch (CancelledException e) {
129 				throw e;
130 			}
131 			catch (Exception e) {
132 				Msg.showError(this, null, "Schedule", "Error: " + e);
133 			}
134 		}
135 
136 		BytesDebuggerPcodeEmulator emu = new BytesDebuggerPcodeEmulator(
137 			new DefaultPcodeDebuggerAccess(state.getTool(), null, platform, snap)) {
138 			TraceSchedule position = TraceSchedule.snap(snap);
139 
140 			@Override
141 			protected BytesPcodeThread createThread(String name) {
142 				return new BytesPcodeThread(name, this) {
143 					TraceThread thread = trace.getThreadManager().getLiveThreadByPath(snap, name);
144 					PcodeExecutor<Pair<byte[], ValueLocation>> inspector =
145 						new PcodeExecutor<>(language,
146 							new PairedPcodeArithmetic<>(arithmetic,
147 								LocationPcodeArithmetic.forEndian(language.isBigEndian())),
148 							state.paired(new LocationPcodeExecutorStatePiece(language)),
149 							Reason.INSPECT);
150 
151 					{
152 						tableDialog.add(createRow());
153 					}
154 
155 					@Override
156 					public void stepInstruction() {
157 						super.stepInstruction();
158 						position = position.steppedForward(thread, 1);
159 						tableDialog.add(createRow());
160 					}
161 
162 					@Override
163 					public void stepPcodeOp() {
164 						super.stepPcodeOp();
165 						position = position.steppedPcodeForward(thread, 1);
166 						tableDialog.add(createRow());
167 					}
168 
169 					public CheckRow createRow() {
170 						List<Pair<byte[], ValueLocation>> values = new ArrayList<>();
171 						for (PcodeExpression exp : compiled) {
172 							values.add(exp.evaluate(inspector));
173 						}
174 						return new CheckRow(position, getCounter(), values);
175 					}
176 				};
177 			}
178 		};
179 
180 		for (SleighPcodeUseropDefinition<?> inject : new Injects().generate().values()) {
181 			String source = inject.getBody();
182 			println("Injecting " + inject.getName() + ":\n" + source);
183 			for (Symbol sym : currentProgram.getSymbolTable()
184 					.getExternalSymbols(inject.getName())) {
185 				if (sym.getObject() instanceof Function fun) {
186 					Set<Address> addresses =
187 						new HashSet<>(List.of(fun.getFunctionThunkAddresses(true)));
188 					addresses.add(fun.getEntryPoint());
189 					for (Address sEntry : addresses) {
190 						Address dEntry = translateStaticToDynamic(sEntry);
191 						println("  " + sEntry + " ( -> " + dEntry + ")");
192 						emu.inject(dEntry, source);
193 					}
194 				}
195 			}
196 		}
197 
198 		schedule.execute(trace, emu, monitor);
199 	}

231 		DataType get(TypeLoader loader) {
232 			return loader.type(path);
233 		}
234 	}
235

237 		return new TypeRec(path);
238 	}

240 	static Setting set(EnumSettingsDefinition def, int value) {
241 		return new EnumSetting(def, value);
242 	}

244 	static Watch watch(String expression, TypeRec type, Setting... settings) {
245 		Settings settingsImpl = new SettingsImpl();
246 		for (Setting set : settings) {
247 			set.set(settingsImpl);
248 		}
249 		return new Watch(expression, type, settingsImpl);
250 	}

252 	class CheckRow implements AddressableRowObject {
253 		private final TraceSchedule schedule;
254 		private final Address pc;
255 		private final List<Pair<byte[], ValueLocation>> values;
256 
257 		public CheckRow(TraceSchedule schedule, Address pc,
258 				List<Pair<byte[], ValueLocation>> values) {
259 			this.schedule = schedule;
260 			this.pc = pc;
261 			this.values = values;
262 		}
263 
264 		@Override
265 		public Address getAddress() { // Instruction address
266 			TraceProgramView view = getCurrentView();
267 			if (view == null) {
268 				return Address.NO_ADDRESS;
269 			}
270 			Address st = translateDynamicToStatic(pc);
271 			return st == null ? Address.NO_ADDRESS : st;
272 		}
273 	}

337 	public class CheckRowWatchDisplay implements TypedDisplay<CheckRow, Object> {
338 		private final boolean isBigEndian = currentProgram.getLanguage().isBigEndian();
339 		private final Watch watch;
340 		private final DataType type;
341 		private final int index;
342 		private final Class<?> valueClass;
343 
344 		public CheckRowWatchDisplay(TypeLoader loader, Watch watch, int index) {
345 			this.watch = watch;
346 			this.type = watch.type.get(loader);
347 			this.index = index;
348 			this.valueClass = type.getValueClass(watch.settings);
349 		}
350 
351 		private Object getObjectValue(CheckRow r) {
352 			try {
353 				Pair<byte[], ValueLocation> p = r.values.get(index);
354 				Address addr = p.getRight() == null ? null : p.getRight().getAddress();
355 				byte[] bytes = p.getLeft();
356 				return type.getValue(new ByteMemBufferImpl(addr, bytes, isBigEndian),
357 					watch.settings, bytes.length);
358 			}
359 			catch (Exception e) {
360 				return "Err: " + e.getMessage();
361 			}
362 		}
363 
364 		private String getStringValue(CheckRow r) {
365 			try {
366 				Pair<byte[], ValueLocation> p = r.values.get(index);
367 				Address addr = p.getRight() == null ? null : p.getRight().getAddress();
368 				byte[] bytes = p.getLeft();
369 				return type.getRepresentation(new ByteMemBufferImpl(addr, bytes, isBigEndian),
370 					watch.settings, bytes.length);
371 			}
372 			catch (Exception e) {
373 				return "Err: " + e.getMessage();
374 			}
375 		}
376 
377 		@Override
378 		@SuppressWarnings({ "unchecked", "rawtypes" })
379 		public int compareTyped(CheckRow r1, CheckRow r2) {
380 			if (Comparable.class.isAssignableFrom(valueClass)) {
381 				Object v1 = getObjectValue(r1);
382 				Object v2 = getObjectValue(r2);
383 				return ((Comparable) v1).compareTo(v2);
384 			}
385 			String s1 = getStringValue(r1);
386 			String s2 = getStringValue(r2);
387 			return s1.compareTo(s2);
388 		}
389 
390 		@Override
391 		public String getTypedValue(CheckRow row) {
392 			return getStringValue(row);
393 		}
394 
395 		@Override
396 		public String getColumnName() {
397 			return watch.expression;
398 		}
399 
400 		@Override
401 		@SuppressWarnings("unchecked")
402 		public Class<Object> getColumnClass() {
403 			return (Class<Object>) valueClass;
404 		}
405 	}
