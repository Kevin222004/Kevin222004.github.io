16  //An example emulation script that uses a stand-alone emulator.
17  //It provides the set-up code and then demonstrates some use cases.
18  //@author 
19  //@category Emulation
20  //@keybinding
21  //@menupath
22  //@toolbar
23  
24  import java.nio.charset.Charset;
25  
26  import ghidra.app.plugin.assembler.*;
27  import ghidra.app.plugin.processors.sleigh.SleighLanguage;
28  import ghidra.app.script.GhidraScript;
29  import ghidra.pcode.emu.PcodeEmulator;
30  import ghidra.pcode.emu.PcodeThread;
31  import ghidra.pcode.exec.*;
32  import ghidra.pcode.exec.PcodeExecutorStatePiece.Reason;
33  import ghidra.pcode.utils.Utils;
34  import ghidra.program.model.address.Address;
35  import ghidra.program.model.address.AddressSpace;
36  import ghidra.program.model.lang.LanguageID;
37  
38  public class StandAloneEmuExampleScript extends GhidraScript {
39  	private final static Charset UTF8 = Charset.forName("utf8");
40  	private SleighLanguage language;
41  	private PcodeEmulator emulator;
42  
43  	@Override
44  	protected void run() throws Exception {
45  		/*
46  		 * Create an emulator and start a thread
47  		 */
48  		language = (SleighLanguage) getLanguage(new LanguageID("x86:LE:64:default"));
49  		emulator = new PcodeEmulator(language) {
50  			@Override
51  			protected PcodeUseropLibrary<byte[]> createUseropLibrary() {
52  				return new DemoPcodeUseropLibrary(language, StandAloneEmuExampleScript.this);
53  			}
54  
55  			// Uncomment this to see instructions printed as they are decoded
56  			/*
57  			protected BytesPcodeThread createThread(String name) {
58  				return new BytesPcodeThread(name, this) {
59  					@Override
60  					protected SleighInstructionDecoder createInstructionDecoder(
61  							PcodeExecutorState<byte[]> sharedState) {
62  						return new SleighInstructionDecoder(language, sharedState) {
63  							@Override
64  							public Instruction decodeInstruction(Address address,
65  									RegisterValue context) {
66  								Instruction instruction = super.decodeInstruction(address, context);
67  								println("Decoded " + address + ": " + instruction);
68  								return instruction;
69  							}
70  						};
71  					}
72  				};
73  			}
74  			*/
75  		};
76  		PcodeThread<byte[]> thread = emulator.newThread();
77  		// The emulator composes the full library for each thread
78  		PcodeUseropLibrary<byte[]> library = thread.getUseropLibrary();
79  		AddressSpace dyn = language.getDefaultSpace();
80  
81  		/*
82  		 * Assemble a little test program and write it into the emulator
83  		 * 
84  		 * We're not really going to implement system calls here. We're just using it to demonstrate
85  		 * the implementation of a language-defined userop.
86  		 */
87  		Address entry = dyn.getAddress(0x00400000);
88  		Assembler asm = Assemblers.getAssembler(language);
89  		AssemblyBuffer buffer = new AssemblyBuffer(asm, entry);
90  		buffer.assemble("MOV RCX, 0xdeadbeef");
91  		Address injectHere = buffer.getNext();
92  		buffer.assemble("MOV RAX, 1");
93  		buffer.assemble("SYSCALL");
94  		buffer.assemble("MOV RAX, 2"); // Induce the interrupt we need to terminate
95  		buffer.assemble("SYSCALL");
96  		byte[] code = buffer.getBytes();
97  		emulator.getSharedState().setVar(dyn, entry.getOffset(), code.length, true, code);
98  
99  		/*
100 		 * Initialize other parts of the emulator and thread state. Note the use of the L suffix on
101 		 * 0xdeadbeefL, because Java with sign extend the (negative) int to a long otherwise.
102 		 */
103 		byte[] hw = "Hello, World!\n".getBytes(UTF8);
104 		emulator.getSharedState().setVar(dyn, 0xdeadbeefL, hw.length, true, hw);
105 		PcodeProgram init = SleighProgramCompiler.compileProgram(language, "init", String.format("""
106 				RIP = 0x%s;
107 				RSP = 0x00001000;
108 				""", entry), library);
109 		thread.getExecutor().execute(init, library);
110 		thread.overrideContextWithDefault();
111 		thread.reInitialize();
112 
113 		/*
114 		 * Inject a call to our custom print userop. Otherwise, the language itself will never
115 		 * invoke it.
116 		 */
117 		emulator.inject(injectHere, """
118 				print_utf8(RCX);
119 				emu_exec_decoded();
120 				""");
121 
122 		/*
123 		 * Run the experiment: This should interrupt on the second SYSCALL, because any value other
124 		 * than 1 calls emu_swi.
125 		 */
126 		try {
127 			thread.stepInstruction(10);
128 			printerr("We should not have completed 10 steps!");
129 		}
130 		catch (InterruptPcodeExecutionException e) {
131 			println("Terminated via interrupt. Good.");
132 		}
133 
134 		/*
135 		 * Inspect the machine. You can always do this by accessing the state directly, but for
136 		 * anything other than simple variables, you may find compiling an expression more
137 		 * convenient.
138 		 */
139 		println("RCX = " +
140 			Utils.bytesToLong(thread.getState().getVar(language.getRegister("RCX"), Reason.INSPECT),
141 				8, language.isBigEndian()));
142 
143 		println("RCX = " + Utils.bytesToLong(
144 			SleighProgramCompiler.compileExpression(language, "RCX").evaluate(thread.getExecutor()),
145 			8, language.isBigEndian()));
146 
147 		println("RCX+4 = " +
148 			Utils.bytesToLong(SleighProgramCompiler.compileExpression(language, "RCX+4")
149 					.evaluate(thread.getExecutor()),
150 				8, language.isBigEndian()));
151 	}
152 }
