16  //Example script for training random forests to find function starts
17  //@category    machineLearning
18  
19  import java.util.*;
20  import java.util.Map.Entry;
21  import java.util.stream.Collectors;
22  
23  import ghidra.app.cmd.disassemble.DisassembleCommand;
24  import ghidra.app.cmd.function.CreateFunctionCmd;
25  import ghidra.app.script.GhidraScript;
26  import ghidra.machinelearning.functionfinding.*;
27  import ghidra.program.model.address.Address;
28  import ghidra.program.model.address.AddressSet;
29  import ghidra.program.model.block.BasicBlockModel;
30  
31  //NOTE: This script is referenced by name in the help for the 
32  //RandomForestFunctionFinderPlugin.  If you change the name be
33  //sure to update the help.
34  public class FindFunctionsRFExampleScript extends GhidraScript {
35  
36  	@Override
37  	protected void run() throws Exception {
38  
39  		//get the parameters controlling how many models are trained and
40  		//what data is used to train/test them
41  		FunctionStartRFParams params = new FunctionStartRFParams(currentProgram);
42  
43  		//maximum number of function starts to use in the training set
44  		//a warning will be issued if there are no functions left over for the test set
45  		params.setMaxStarts(1000);
46  
47  		//minimum size of a function to be included in the training/test sets
48  		params.setMinFuncSize(16);
49  
50  		//number of bytes before a function start
51  		params.setPreBytes(Arrays.asList(new Integer[] { 2, 8 }));
52  
53  		//number of bytes after (and including) a function start
54  		params.setInitialBytes(Arrays.asList(new Integer[] { 8, 16 }));
55  
56  		//number of non-starts to sample for each start in the training set
57  		params.setFactors(Arrays.asList(new Integer[] { 10, 50 }));
58  
59  		//for every function start in the (test,training) set, add the code
60  		//units immediately before and immediately after to the (test,training) set
61  		//as non-starts. 
62  		params.setIncludePrecedingAndFollowing(true);
63  
64  		//uncomment to include features for each bit rather than just byte-level features
65  		//params.setIncludeBitFeatures(true);
66  
67  		//bound for reducing the size of the test sets
68  		long testSetMax = 1000000l;
69  
70  		//this is where the trained models will go
71  		List<RandomForestRowObject> trainedModels = new ArrayList<>();
72  
73  		RandomForestTrainingTask trainingTask = new RandomForestTrainingTask(currentProgram, params,
74  			r -> trainedModels.add(r), testSetMax);
75  
76  		//launch the task to train the models in parallel
77  		trainingTask.run(monitor);
78  
79  		//sort the models by the number of false positives (ascending)
80  		//if you actually need *unsigned* comparison it's likely that something has gone
81  		//horribly wrong
82  		Collections.sort(trainedModels,
83  			(x, y) -> Integer.compareUnsigned(x.getNumFalsePositives(), y.getNumFalsePositives()));
84  
85  		//grab the model with the fewest false positives
86  		//note: there could be ties; could sort the winners by recall
87  		RandomForestRowObject best = trainedModels.get(0);
88  
89  		printf(
90  			"Best model: pre-bytes: %d, initialBytes: %d, sampling factor: %d, false positives: %d," +
91  				" precision: %g, recall: %g\n",
92  			best.getNumPreBytes(), best.getNumInitialBytes(), best.getSamplingFactor(),
93  			best.getNumFalsePositives(), best.getPrecision(), best.getRecall());
94  
95  		//to get more information about the test set errors, apply the model to the error set
96  
97  		FunctionStartClassifier classifier = new FunctionStartClassifier(currentProgram, best,
98  			RandomForestFunctionFinderPlugin.FUNC_START);
99  		//uncomment to see false negatives as well
100 		//classifier.setProbabilityThreshold(0.0);
101 
102 		Map<Address, Double> errors = classifier.classify(best.getTestErrors(), monitor);
103 
104 		List<Entry<Address, Double>> falsePositives = errors.entrySet()
105 				.stream()
106 				.filter(x -> currentProgram.getFunctionManager().getFunctionAt(x.getKey()) == null)
107 				.sorted((x, y) -> Double.compare(y.getValue(), x.getValue()))
108 				.toList();
109 
110 		//print out addresses of false positives
111 		printf("False positives:\n");
112 		falsePositives.forEach(x -> printf("  %s %g\n", x.getKey().toString(), x.getValue()));
113 
114 		//show the true function starts most similar to one of the false positives
115 		if (!falsePositives.isEmpty()) {
116 			SimilarStartsFinder finder =
117 				new SimilarStartsFinder(currentProgram, currentProgram, best);
118 			List<SimilarStartRowObject> neighbors =
119 				finder.getSimilarFunctionStarts(falsePositives.get(0).getKey(), 10);
120 			printf("\nClosest function starts to false positive at %s :\n",
121 				falsePositives.get(0).getKey());
122 			neighbors.forEach(n -> printf("  %s %d\n", n.funcStart(), n.numAgreements()));
123 		}
124 
125 		//grab the set of bytes in executable memory which are undefined (i.e., initialized but
126 		//not yet assigned to be code or data) or instructions which are not assigned to a function 
127 		//body
128 		//don't bother looking in small undefined ranges
129 		long minUndefinedRange = 16;
130 		GetAddressesToClassifyTask getAddressTask =
131 			new GetAddressesToClassifyTask(currentProgram, minUndefinedRange);
132 		getAddressTask.run(monitor);
133 
134 		AddressSet toClassify = getAddressTask.getAddressesToClassify();
135 
136 		Map<Address, Double> potentialStarts = classifier.classify(toClassify, monitor);
137 
138 		//grab all of the addresses with probability of being a function start >= .7
139 		//and disassemble any that are currently undefined
140 
141 		//block model is needed to get the interpretation of an address
142 		//e.g., undefined, block start, within block,...
143 		BasicBlockModel blockModel = new BasicBlockModel(currentProgram);
144 
145 //@formatter:off
146 		List<Address> addresses = potentialStarts.entrySet()
147 			.stream()
148 			.filter(x -> {return x.getValue() >= 0.7d;})
149 			.map(x -> x.getKey())
150 			.collect(Collectors.toList());
151 //@formatter:on
152 
153 		AddressSet toDisassemble = new AddressSet();
154 		for (Address addr : addresses) {
155 			Interpretation inter =
156 				Interpretation.getInterpretation(currentProgram, addr, blockModel, monitor);
157 			if (inter.equals(Interpretation.UNDEFINED)) {
158 				toDisassemble.add(addr);
159 			}
160 		}
161 
162 		//see DisassemblyAndApplyContextAction#actionPerfomed if you need to
163 		//apply context register values
164 		printf("Found %d addresses to disassemble\n", toDisassemble.getNumAddresses());
165 		DisassembleCommand cmd = new DisassembleCommand(toDisassemble, null, true);
166 		cmd.applyTo(currentProgram);
167 
168 		//create functions at any address with probability of being a function start > .8, 
169 		//where an instruction exists,
170 		//which is defined as a BLOCK_START (so not already a function start)
171 		//and with no conditional flow references to it
172 		//FunctionStartRowObject represents a row in a table in the gui, but you don't
173 		//actually need the gui and it's a convenient container for information about an address
174 
175 //@formatter:off
176 		List<FunctionStartRowObject> funcRows  = potentialStarts.entrySet()
177 			.stream()
178 			.filter(x -> {return x.getValue() >= 0.8d;})
179 			.map(x -> new FunctionStartRowObject(x.getKey(),x.getValue()))
180 			.collect(Collectors.toList());
181 //@formatter:on
182 
183 		for (FunctionStartRowObject funcRow : funcRows) {
184 			funcRow.setCurrentInterpretation(Interpretation.getInterpretation(currentProgram,
185 				funcRow.getAddress(), blockModel, monitor));
186 			FunctionStartRowObject.setReferenceData(funcRow, currentProgram);
187 		}
188 
189 		AddressSet entries = new AddressSet();
190 		funcRows.stream()
191 				.filter(x -> x.getCurrentInterpretation().equals(Interpretation.BLOCK_START))
192 				.filter(x -> x.getNumConditionalFlowRefs() == 0)
193 				.forEach(x -> entries.add(x.getAddress()));
194 
195 		printf("Found %d addresses to create functions\n", entries.getNumAddresses());
196 
197 		CreateFunctionCmd createCmd = new CreateFunctionCmd(entries);
198 		createCmd.applyTo(currentProgram);
199 	}
200 
201 }
