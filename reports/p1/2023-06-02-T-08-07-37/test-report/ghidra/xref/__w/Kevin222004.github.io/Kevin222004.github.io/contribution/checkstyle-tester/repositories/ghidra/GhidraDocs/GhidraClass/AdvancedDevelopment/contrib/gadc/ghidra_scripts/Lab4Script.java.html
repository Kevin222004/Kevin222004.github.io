16  //Ask for an integer and print the current 
17  //program?s name that many times to the console
18  //@category GADC
19  
20  import ghidra.app.script.GhidraScript;
21  
22  public class Lab4Script extends GhidraScript {
23  	@Override
24  	public void run() throws Exception {
25  		int n = askInt("How Many Times?", "N");
26  		for (int i = 0; i < n; ++i) {
27  			if (monitor.isCancelled()) {
28  				break;
29  			}
30  			println(i + ". " + currentProgram.getName());
31  			Thread.sleep(1000);
32  		}
33  	}
34  }
