33  	@Override
34  	public void run() throws Exception {
35  		ByteProvider provider =
36  			MemoryByteProvider.createProgramHeaderByteProvider(currentProgram, false);
37  
38  		BinaryReader reader = new BinaryReader(provider, false);
39  
40  		BTreeRootNodeDescriptor root = new BTreeRootNodeDescriptor(reader);
41  
42  		markupRecordOffsets(currentProgram, root, 0, root);
43  
44  		Data headerNodeData = createBTreeNode(currentProgram, root, 0);
45  
46  		Data headerRecordData =
47  			createBTreeHeaderRecord(currentProgram, root.getHeaderRecord(),
48  				headerNodeData.getLength());
49  
50  		Data userDataRecordData =
51  			createUserDataRecord(currentProgram, root.getUserDataRecord(),
52  				headerNodeData.getLength() + headerRecordData.getLength());
53  
54  		Data mapRecordData =
55  			createMapRecord(currentProgram, root.getMapRecord(), headerNodeData.getLength() +
56  				headerRecordData.getLength() + userDataRecordData.getLength());
57  
58  		if (mapRecordData == null) {
59  			printerr("mapRecordData == null ????");
60  		}
61  
62  		processNodes(currentProgram, root);
63  	}
64  
65  	@Override
66  	public AnalysisMode getScriptAnalysisMode() {
67  		return AnalysisMode.DISABLED;
68  	}

70  	private int processNodes(Program program, BTreeRootNodeDescriptor root) throws Exception {
71  
72  		int nodeIndex = 1;
73  		int min = (int) program.getMinAddress().getOffset();
74  		int max = (int) program.getMaxAddress().getOffset();
75  		monitor.setMaximum(max - min);
76  		monitor.setMessage("Applying node descriptors...");
77  
78  		int nodeSize = root.getHeaderRecord().getNodeSize() & 0xffff;
79  		for (int i = nodeSize; i < program.getMemory().getSize(); i += nodeSize) {
80  			if (monitor.isCancelled()) {
81  				break;
82  			}
83  			monitor.setProgress(min + i);
84  
85  			BTreeNodeDescriptor nodeI = root.getNode(nodeIndex);
86  			createBTreeNode(program, nodeI, i);
87  
88  			StringBuffer buffer = new StringBuffer();
89  			buffer.append("Index:   0x" + Integer.toHexString(nodeIndex) + '\n');
90  			buffer.append("flink:   0x" + Integer.toHexString(nodeI.getFLink()) + '\n');
91  			buffer.append("blink:   0x" + Integer.toHexString(nodeI.getBLink()) + '\n');
92  			buffer.append("kind:      " + nodeI.getKind() + '\n');
93  			buffer.append("Records: 0x" + Integer.toHexString(nodeI.getNumRecords()) + '\n');
94  
95  			setPlateComment(toAddr(i), buffer.toString());
96  
97  			markupBTreeNodeData(program, nodeI);
98  			markupRecordOffsets(program, root, i, nodeI);
99  
100 			++nodeIndex;
101 		}
102 		return nodeIndex;
103 	}

122 	private void markupBTreeNodeData(Program program, BTreeNodeDescriptor descriptor)
123 			throws Exception {
124 
125 		for (BTreeNodeRecord record : descriptor.getRecords()) {
126 
127 			Address address = toAddr(record.getRecordOffset());
128 
129 			DataType recordDataType = record.toDataType();
130 			Data recordData = createData(address, recordDataType);
131 			createFragment(record.getType(), recordData.getMinAddress(), recordData.getLength());
132 			setPlateComment(address,
133 				record.getType() + " 0x" + Integer.toHexString(record.getFileID()));
134 
135 			if (descriptor.getKind() == BTreeNodeKinds.kBTLeafNode) {
136 				if (record.getType().equals(XattrConstants.DECMPFS_XATTR_NAME)) {
137 					markupDecmpfs(program, descriptor, record, recordData.getMaxAddress().add(1));
138 				}
139 				else if (record.getType().equals(XattrConstants.KAUTH_FILESEC_XATTR_NAME)) {
140 					//TODO
141 				}
142 			}
143 		}
144 	}
145 
146 	private void markupDecmpfs(Program program, BTreeNodeDescriptor descriptor,
147 			BTreeNodeRecord record, Address address) throws Exception {
148 
149 		DecmpfsHeader header = record.getDecmpfsHeader();
150 		DataType headerDataType = header.toDataType();
151 		Data headerData = createData(address, headerDataType);
152 		changeEndianSettings(headerData);
153 		createFragment(header.getCompressionMagic(), headerData.getMinAddress(),
154 			headerData.getLength());
155 		StringBuffer buffer = new StringBuffer();
156 		buffer.append(header.getCompressionMagic());
157 		buffer.append('\n');
158 		buffer.append("CompressionType: 0x" + Integer.toHexString(header.getCompressionType()));
159 		buffer.append('\n');
160 		buffer.append("UncompressedSize: 0x" + Long.toHexString(header.getUncompressedSize()));
161 		buffer.append('\n');
162 		setPlateComment(address, buffer.toString());
163 	}

201 	private void changeEndianSettings(Data data) throws Exception {
202 		for (int i = 0; i < data.getNumComponents(); ++i) {
203 			Data component = data.getComponent(i);
204 			SettingsDefinition[] settings = component.getDataType().getSettingsDefinitions();
205 			for (int j = 0; j < settings.length; ++j) {
206 				if (settings[j] instanceof EndianSettingsDefinition) {
207 					EndianSettingsDefinition setting = (EndianSettingsDefinition) settings[j];
208 					setting.setBigEndian(component, false);
209 				}
210 			}
211 		}
