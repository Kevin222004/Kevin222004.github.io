16  //Example script to show how to export the current program in its original binary format by 
17  //looking at the original bytes compared to the current bytes and exporting the original if they 
18  //are the same and the current bytes if they are different. If the changed bytes are relocations, 
19  //the original bytes are used. The script only handles simple use cases.
20  //@category Examples
21  
22  import java.io.*;
23  import java.util.Iterator;
24  import java.util.List;
25  
26  import ghidra.app.script.GhidraScript;
27  import ghidra.program.database.mem.FileBytes;
28  import ghidra.program.model.address.Address;
29  import ghidra.program.model.address.AddressSet;
30  import ghidra.program.model.mem.Memory;
31  import ghidra.program.model.reloc.Relocation;
32  import ghidra.program.model.reloc.RelocationTable;
33  import ghidra.util.exception.CancelledException;
34  
35  public class ExportProgramScript extends GhidraScript {
36  
37  	@Override
38  	public void run() throws Exception {
39  
40  		if (currentProgram == null) {
41  			println("Must have an open program");
42  			return;
43  		}
44  
45  		File outBinaryFile = askFile("Select Binary Output File", "Binary File");
46  
47  		if (outBinaryFile.exists()) {
48  			if (!askYesNo("Binary File Already Exists",
49  				"The binary file already exists.\nDo you want to overwrite it?")) {
50  				return;
51  			}
52  		}
53  
54  		// make address set of relocation byte addresses
55  		AddressSet relocationAddrs = new AddressSet();
56  
57  		RelocationTable relocTable = currentProgram.getRelocationTable();
58  		Iterator<Relocation> iter = relocTable.getRelocations();
59  		while (iter.hasNext()) {
60  			monitor.checkCancelled();
61  
62  			Relocation reloc = iter.next();
63  			Address relocStart = reloc.getAddress();
64  
65  			int rlocLen = reloc.getBytes().length;
66  			relocationAddrs.add(relocStart, relocStart.add(rlocLen));
67  		}
68  
69  		// create an output stream
70  		OutputStream out = new FileOutputStream(outBinaryFile);
71  
72  		// get all program file bytes
73  		Memory memory = currentProgram.getMemory();
74  		List<FileBytes> allFileBytes = memory.getAllFileBytes();
75  		if (allFileBytes.isEmpty()) {
76  			println(
77  				"Cannot access original file bytes. Either the program was imported before Ghidra " +
78  					"started saving original file bytes or the program was not imported directly from the " +
79  					"original binary.");
80  			out.close();
81  			return;
82  		}
83  
84  		//TODO: update to handle multiple case once FileBytes adds new method to support it
85  		if (allFileBytes.size() > 1) {
86  			println(
87  				"*** This program was created using multiple imported programs. This script will currently only work if the program was created with a single imported file.");
88  			out.close();
89  			return;
90  		}
91  
92  		//Get the original import file bytes
93  		FileBytes fileBytes = allFileBytes.get(0);
94  
95  		long size = fileBytes.getSize();
96  		println(
97  			"Exporting current program to a new binary file including any changes to the original " +
98  				"except for relocation changes...");
99  
100 		// compare each original imported byte with the current program byte at the equivalent location
101 		// if the current byte is different and is not a relocation, export that byte instead of
102 		// the original
103 		for (long i = 0; i < size; i++) {
104 
105 			monitor.checkCancelled();
106 
107 			byte originalByte = fileBytes.getOriginalByte(i);
108 			byte currentByte = fileBytes.getModifiedByte(i);
109 
110 			int original = originalByte & 0xff;
111 			int current = currentByte & 0xff;
112 
113 			if (originalByte != currentByte) {
114 
115 				// TODO: once new method is created to use the fileBytes and offset to retrieve the
116 				// address, update this to use the correct method and lift the above restriction 
117 				// on only handling one imported file.
118 				List<Address> addresses = memory.locateAddressesForFileOffset(i);
119 
120 
121 				// NOTE: It is rare that there would be more than one address in the program corresponding
122 				// to a given offset but check anyway and if any of them are a relocation then 
123 				// output the original byte
124 				if (!addressSetContainsAnyInList(relocationAddrs, addresses)) {
125 
126 					println(addresses +
127 						": Writing out changed byte since it is not a relocation. File offset = " +
128 						i + ", originalByte = " + Integer.toHexString(original) +
129 						", changed to = " + Integer.toHexString(current));
130 					println("****** ");
131 					out.write(current);
132 
133 				}
134 				else {
135 					// Not writing out change since it is a relocation
136 					out.write(original);
137 				}
138 
139 			}
140 			// same in current and original
141 			else {
142 				out.write(original);
143 			}
144 		}
145 
146 		out.close();
147 		println("Done!");
148 	}
149 
150 	private boolean addressSetContainsAnyInList(AddressSet set, List<Address> list)
151 			throws CancelledException {
152 
153 		if (list.isEmpty()) {
154 			return false;
155 		}
156 		if (set.isEmpty()) {
157 			return false;
158 		}
159 
160 		for (Address address : list) {
161 			monitor.checkCancelled();
162 			if (set.contains(address)) {
163 				return true;
164 			}
165 		}
166 		return false;
167 	}
168 }
