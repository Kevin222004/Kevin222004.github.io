35  /**
36   * Runs through recursively through a set of folders provided by the user.
37   * Programs are assumed to be "marked", meaning that functions all have their correct labels.
38   * Each function in each program is searched within the active FID databases and various statistics
39   * are calculated.
40   *    1) Match found/not found
41   *    2) Are any matches found correct (as compared to existing symbol)
42   *    3) Are there multiple potential matches (whether correct or not)
43   *
44   * If program symbols are mangled, this script assumes the primary symbol is not demangled.
45   */
46  public class FidStatistics extends GhidraScript {

54  	private float scoreThreshold;

56  	public static class SymbolPair implements Comparable<SymbolPair> {
57  
58  		private String sym1;
59  		private String sym2;
60  
61  		public SymbolPair(String a, String b) {
62  			sym1 = a;
63  			sym2 = b;
64  		}
65  
66  		@Override
67  		public int compareTo(SymbolPair o) {
68  			int val = sym1.compareTo(o.sym1);
69  			if (val != 0) {
70  				return val;
71  			}
72  			return sym2.compareTo(o.sym2);
73  		}
74  
75  	}

77  	public static class MatchRecord {
78  		private String progName;
79  		private String funcName;
80  		private String matchName;
81  		private long fullHash;
82  		private long address;
83  		private float score;
84  		private float childScore;
85  		private float parentScore;
86  
87  		public MatchRecord(FidSearchResult result, String finalMatchName, boolean isFalse) {
88  			this.progName = result.function.getProgram().getDomainFile().getPathname();
89  			this.fullHash = result.hashQuad.getFullHash();
90  			this.funcName = result.function.getName();
91  			this.matchName = finalMatchName;
92  			this.address = result.function.getEntryPoint().getOffset();
93  			FidMatch fidMatch = result.matches.get(0);
94  			this.score = fidMatch.getOverallScore();
95  			this.childScore = fidMatch.getChildFunctionCodeUnitScore();
96  			this.parentScore = fidMatch.getParentFunctionCodeUnitScore();
97  		}
98  
99  		public void print(StringBuilder buf) {
100 			buf.append("score= ");
101 			buf.append(score);
102 			buf.append(" fh=0x");
103 			buf.append(Long.toHexString(fullHash));
104 			buf.append(" child=").append(childScore);
105 			buf.append(" par=").append(parentScore);
106 			buf.append(" address=");
107 			buf.append(Long.toHexString(address));
108 			buf.append(" name=");
109 			buf.append(funcName);
110 			buf.append(" prog=");
111 			buf.append(progName);
112 			if (matchName != null) {
113 				buf.append(" match=");
114 				buf.append(matchName);
115 			}
116 //			if (isFalsePositive)
117 //				buf.append("  FALSE");
118 		}
119 	}

121 	public static class StatRecord {
122 		public int totalFunction;
123 		public int matchUniquely;
124 		public int matchMultiply;
125 		public int hitCount;
126 		public int noMatch;
127 		public int nameMatched;
128 		public int falsePositive;
129 
130 		public StatRecord() {
131 			totalFunction = 0;
132 			matchUniquely = 0;
133 			matchMultiply = 0;
134 			hitCount = 0;
135 			noMatch = 0;
136 			nameMatched = 0;
137 			falsePositive = 0;
138 		}
139 
140 		public static void indent(StringBuilder buf, String last) {
141 			for (int i = last.length(); i < 10; ++i) {
142 				buf.append(' ');
143 			}
144 		}
145 
146 		public void print(StringBuilder buf) {
147 			String str = Integer.toString(totalFunction);
148 			buf.append(str);
149 			indent(buf, str);
150 			str = Integer.toString(noMatch);
151 			buf.append(str);
152 			indent(buf, str);
153 			str = Integer.toString(matchUniquely + matchMultiply);
154 			buf.append(str);
155 			indent(buf, str);
156 			str = Integer.toString(hitCount);
157 			buf.append(str);
158 			indent(buf, str);
159 			str =
160 				'(' + Integer.toString(matchUniquely) + ',' + Integer.toString(matchMultiply) + ')';
161 			buf.append(str);
162 			indent(buf, str);
163 			str = Integer.toString(nameMatched);
164 			buf.append(str);
165 			indent(buf, str);
166 			str = Integer.toString(falsePositive);
167 			buf.append(str);
168 		}
169 
170 		public static String getColumns() {
171 			return "Total     No Match  Possible  Hits      uniq/mult N-Match   False";
172 		}
173 	}

175 	private void addEquivSymbols(String a, String b) {
176 		SymbolPair pair = new SymbolPair(a, b);
177 		equivSymbols.add(pair);
178 		pair = new SymbolPair(b, a);
179 		equivSymbols.add(pair);
180 	}

371 			throws VersionException, CancelledException, IOException {
372 		DomainFile[] files = folder.getFiles();
373 		for (DomainFile domainFile : files) {
374 			monitor.checkCancelled();
375 			// Do not follow folder-links or consider program links.  Using content type
376 			// to filter is best way to control this.  If program links should be considered

384 		for (DomainFolder domainFolder : folders) {
385 			monitor.checkCancelled();
386 			findDomainFiles(programs, domainFolder);
387 		}

397 	private String chooseFunctionName(FidSearchResult result) {
398 		Program program = result.function.getProgram();
399 		SymbolIterator it =
400 			program.getSymbolTable().getSymbolsAsIterator(result.function.getEntryPoint());
401 		for (Symbol symbol : it) {
402 			if (matchAnalysis.containsRawName(symbol.getName())) {
403 				return symbol.getName();
404 			}
405 		}
406 		return result.function.getName();
407 	}

409 	private void processFunctionResult(FidSearchResult result)
410 			throws CancelledException, IOException {
411 		StatRecord record = statRecord;
412 		record.totalFunction += 1;
413 		if (result.matches == null || result.matches.size() == 0) {
414 			record.noMatch += 1;
415 		}
416 		else {
417 			Program program = result.function.getProgram();
418 			matchAnalysis.analyzeNames(result.matches, program, monitor);
419 			String finalMatchName = null;
420 			boolean matchHappened = false;
421 			if (matchAnalysis.getMostOptimisticCount() > 1) {
422 				if (matchAnalysis.getOverallScore() >= FidService.MULTINAME_SCORE_THRESHOLD) {
423 					record.matchMultiply += 1;
424 					matchHappened = true;		// FID will put down a multiple match label
425 				}
426 			}
427 			else {
428 				record.matchUniquely += 1;
429 				matchHappened = true;			// FID will put down a single match
430 				finalMatchName = matchAnalysis.getNameIterator().next();
431 			}
432 			String funcName = chooseFunctionName(result);
433 			NameVersions nameVersions = NameVersions.generate(funcName, program);
434 			boolean exactNameMatch = false;
435 			Iterator<String> iter = matchAnalysis.getRawNameIterator();
436 			while (iter.hasNext()) {
437 				String raw = iter.next();
438 				NameVersions matchNames = NameVersions.generate(raw, program);
439 				if (matchNames.rawName == null) {
440 					continue;
441 				}
442 				if (checkNames(nameVersions.rawName, matchNames.rawName)) {
443 					exactNameMatch = true;
444 					break;
445 				}
446 				if (checkNames(nameVersions.similarName, matchNames.similarName)) {
447 					exactNameMatch = true;
448 					break;
449 				}
450 				if (nameVersions.demangledBaseName != null) {
451 					if (checkNames(nameVersions.demangledBaseName, matchNames.rawName) ||
452 						checkNames(nameVersions.demangledBaseName, matchNames.similarName)) {
453 						exactNameMatch = true;
454 						break;
455 					}
456 				}
457 				if (matchNames.demangledBaseName != null) {
458 					if (checkNames(nameVersions.rawName, matchNames.demangledBaseName) ||
459 						checkNames(nameVersions.similarName, matchNames.demangledBaseName)) {
460 						exactNameMatch = true;
461 						break;
462 					}
463 				}
464 				if (nameVersions.demangledBaseName == null) {
465 					continue;
466 				}
467 				if (matchNames.demangledBaseName == null) {
468 					continue;
469 				}
470 				if (checkNames(nameVersions.demangledBaseName, matchNames.demangledBaseName)) {
471 					exactNameMatch = true;
472 					break;
473 				}
474 				if (nameVersions.demangledNoTemplate != null &&
475 					matchNames.demangledNoTemplate != null) {
476 					if (checkNames(nameVersions.demangledNoTemplate,
477 						matchNames.demangledNoTemplate)) {
478 						exactNameMatch = true;
479 						break;
480 					}
481 				}
482 			}
483 			if (exactNameMatch) {
484 				record.nameMatched += 1;
485 			}
486 
487 			float score = result.matches.get(0).getOverallScore();
488 			if (score >= scoreThreshold && matchHappened) {
489 				record.hitCount += 1;
490 			}
491 			if (exactNameMatch && ((score < scoreThreshold) || !matchHappened)) {
492 				MatchRecord matchRecord = new MatchRecord(result, null, false);
493 				StringBuilder buffer = new StringBuilder();
494 				matchRecord.print(buffer);
495 				lowTruePositive.append(buffer.toString());
496 				lowTruePositive.append('\n');
497 			}
498 			else if ((!exactNameMatch) && (score >= scoreThreshold) && matchHappened) {
499 				record.falsePositive += 1;
500 				MatchRecord matchRecord = new MatchRecord(result, finalMatchName, true);
501 				StringBuilder buffer = new StringBuilder();
502 				matchRecord.print(buffer);
503 				highFalsePositive.append(buffer.toString());
504 				highFalsePositive.append('\n');
505 			}
506 		}
507 
508 	}

510 	private void processProgram(Program program, FidQueryService queryService)
511 			throws MemoryAccessException, CancelledException, VersionException, IOException {
512 		FidProgramSeeker programSeeker = service.getProgramSeeker(program, queryService, 10.0f);
513 		monitor.setMessage("Processing " + program.getName());
514 		monitor.initialize(program.getFunctionManager().getFunctionCount());
515 		FunctionIterator iter = program.getFunctionManager().getFunctionsNoStubs(true);
516 		while (iter.hasNext()) {
517 			Function func = iter.next();
518 			monitor.incrementProgress(1);
519 			if (func.getName().startsWith("FUN_") || func.getName().startsWith("Ordinal_")) {
520 				continue;
521 			}
522 			FidSearchResult searchResult = programSeeker.searchFunction(func, monitor);
523 			if (searchResult == null) {
524 				continue;		// Could not hash function
525 			}
526 			processFunctionResult(searchResult);
527 		}
528 	}

530 	private LinkedList<DomainFile> buildDomainFileList()
531 			throws CancelledException, VersionException, IOException {
532 		ArrayList<DomainFolder> folders = new ArrayList<DomainFolder>();
533 		while (true) {
534 			monitor.checkCancelled();
535 			try {
536 				DomainFolder folder =
537 					askProjectFolder("Add a top-level project folder (cancel to quit)");
538 				folders.add(folder);
539 			}
540 			catch (CancelledException e) {
541 				break;
542 			}
543 		}
544 
545 		LinkedList<DomainFile> domainFiles = new LinkedList<DomainFile>();
546 		monitor.setMessage("Finding domain files...");
547 		for (DomainFolder folder : folders) {
548 			monitor.checkCancelled();
549 			findDomainFiles(domainFiles, folder);
550 		}
551 		return domainFiles;
552 	}

554 	@Override
555 	protected void run() throws Exception {
556 		initialize();
557 
558 		LinkedList<DomainFile> programList = buildDomainFileList();
559 		File lowFile = askFile("Select file to report true matches", "OK");
560 		File highFile = askFile("Select file to report false positives", "OK");
561 		scoreThreshold = (float) askDouble("Choose score threshold", "OK");
562 
563 		lowTruePositive = new FileWriter(lowFile);
564 		highFalsePositive = new FileWriter(highFile);
565 
566 		FidQueryService queryService = null;
567 		Language lastLanguage = null;
568 		int maxPrograms = programList.size();
569 		int counter = 0;
570 		try {
571 			for (DomainFile domainFile : programList) {
572 				Program program = null;
573 				try {
574 					program = (Program) domainFile.getDomainObject(this, false, false, monitor);
575 					if (queryService == null || !lastLanguage.equals(program.getLanguage())) {
576 						if (queryService != null) {
577 							queryService.close();
578 						}
579 						lastLanguage = program.getLanguage();
580 						queryService = service.openFidQueryService(lastLanguage, false);
581 					}
582 					processProgram(program, queryService);
583 					counter += 1;
584 					monitor.setMessage("Processing programs ...");
585 					monitor.initialize(maxPrograms);
586 					monitor.setProgress(counter);
587 				}
588 				finally {
589 					if (program != null) {
590 						program.release(this);
591 					}
592 				}
593 			}
594 		}
595 		catch (CancelledException ex) {
596 			// A cancel in middle of processing still allows results to get printed
597 		}
598 
599 		queryService.close();
600 		lowTruePositive.close();
601 		highFalsePositive.close();
602 
603 		println(StatRecord.getColumns());
604 		StringBuilder buffer = new StringBuilder();
605 		statRecord.print(buffer);
606 		println(buffer.toString());
607 	}
