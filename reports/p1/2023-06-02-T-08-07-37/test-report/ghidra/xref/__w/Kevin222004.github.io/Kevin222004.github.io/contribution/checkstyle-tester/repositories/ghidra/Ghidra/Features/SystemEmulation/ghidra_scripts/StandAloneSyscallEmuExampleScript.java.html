16  //An example emulation script that uses a stand-alone emulator with syscalls.
17  //It provides the set-up code and then demonstrates some use cases.
18  //@author 
19  //@category Emulation
20  //@keybinding
21  //@menupath
22  //@toolbar
23  
24  import java.nio.charset.Charset;
25  
26  import db.Transaction;
27  import ghidra.app.plugin.assembler.Assembler;
28  import ghidra.app.plugin.assembler.Assemblers;
29  import ghidra.app.plugin.processors.sleigh.SleighLanguage;
30  import ghidra.app.script.GhidraScript;
31  import ghidra.pcode.emu.PcodeEmulator;
32  import ghidra.pcode.emu.PcodeThread;
33  import ghidra.pcode.emu.sys.EmuInvalidSystemCallException;
34  import ghidra.pcode.emu.sys.EmuSyscallLibrary;
35  import ghidra.pcode.exec.*;
36  import ghidra.pcode.exec.PcodeExecutorStatePiece.Reason;
37  import ghidra.pcode.utils.Utils;
38  import ghidra.program.database.ProgramDB;
39  import ghidra.program.model.address.*;
40  import ghidra.program.model.data.DataTypeConflictHandler;
41  import ghidra.program.model.data.PointerDataType;
42  import ghidra.program.model.lang.*;
43  import ghidra.program.model.listing.Program;
44  import ghidra.program.model.mem.Memory;
45  import ghidra.program.model.mem.MemoryBlock;
46  import ghidra.program.model.symbol.SourceType;
47  
48  public class StandAloneSyscallEmuExampleScript extends GhidraScript {
49  	private final static Charset UTF8 = Charset.forName("utf8");
50  
51  	Program program = null;
52  
53  	@Override
54  	protected void run() throws Exception {
55  		/*
56  		 * First, get all the services and stuff:
57  		 */
58  		SleighLanguage language = (SleighLanguage) getLanguage(new LanguageID("x86:LE:64:default"));
59  
60  		/*
61  		 * I'll generate a new program, because I don't want to require the user to pick something
62  		 * specific. It won't be displayed, though, so we'll just release it when we're done.
63  		 */
64  		Address entry;
65  		try {
66  			/*
67  			 * "gcc" is the name of the compiler spec, but we're really interested in the Linux
68  			 * syscall calling conventions.
69  			 */
70  			program =
71  				new ProgramDB("syscall_example", language,
72  					language.getCompilerSpecByID(new CompilerSpecID("gcc")), this);
73  			try (Transaction tx = program.openTransaction("Init")) {
74  				AddressSpace space = program.getAddressFactory().getDefaultAddressSpace();
75  				entry = space.getAddress(0x00400000);
76  				Address dataEntry = space.getAddress(0x00600000);
77  				Memory memory = program.getMemory();
78  				memory.createInitializedBlock(".text", entry, 0x1000, (byte) 0, monitor, false);
79  				Assembler asm = Assemblers.getAssembler(program);
80  				asm.assemble(entry,
81  					"MOV RDI, 0x" + dataEntry,
82  					"MOV RAX, 1",
83  					"SYSCALL",
84  					"MOV RAX, 20",
85  					"SYSCALL");
86  				memory.createInitializedBlock(".data", dataEntry, 0x1000, (byte) 0, monitor, false);
87  				memory.setBytes(dataEntry, "Hello, World!\n".getBytes(UTF8));
88  
89  				/*
90  				 * Because "pointer" is a built-in type, and the emulator does not modify the
91  				 * program, we must ensure it has been resolved on the program's data type manager.
92  				 */
93  				program.getDataTypeManager()
94  						.resolve(PointerDataType.dataType, DataTypeConflictHandler.DEFAULT_HANDLER);
95  
96  				/*
97  				 * We must also populate the system call numbering map. Ordinarily, this would be done
98  				 * using the system call analyzer or another script. Here, we'll just fake it out.
99  				 */
100 				AddressSpace other =
101 					program.getAddressFactory().getAddressSpace(SpaceNames.OTHER_SPACE_NAME);
102 				MemoryBlock blockSyscall = program.getMemory()
103 						.createUninitializedBlock(EmuSyscallLibrary.SYSCALL_SPACE_NAME,
104 							other.getAddress(0), 0x1000, true);
105 				blockSyscall.setPermissions(true, false, true);
106 
107 				AddressSpace syscall = program.getAddressFactory()
108 						.getAddressSpace(EmuSyscallLibrary.SYSCALL_SPACE_NAME);
109 				/*
110 				 * The system call names must match those from the EmuSyscall annotations in the
111 				 * system call library, in our case from DemoSyscallLibrary. Because the x64
112 				 * compiler specs define a "syscall" convention, we'll apply it. The syscall
113 				 * dispatcher will use that convention to fetch the parameters out of the machine
114 				 * state, pass them into the system call defintion, and store the result back into
115 				 * the machine.
116 				 */
117 				// Map system call 0 to "write"
118 				program.getFunctionManager()
119 						.createFunction("write", syscall.getAddress(0),
120 							new AddressSet(syscall.getAddress(0)), SourceType.USER_DEFINED)
121 						.setCallingConvention(EmuSyscallLibrary.SYSCALL_CONVENTION_NAME);
122 				// Map system call 1 to "console"
123 				program.getFunctionManager()
124 						.createFunction("console", syscall.getAddress(1),
125 							new AddressSet(syscall.getAddress(1)), SourceType.USER_DEFINED)
126 						.setCallingConvention(EmuSyscallLibrary.SYSCALL_CONVENTION_NAME);
127 			}
128 
129 			/*
130 			 * Create an emulator and start a thread
131 			 */
132 			PcodeEmulator emulator = new PcodeEmulator(language) {
133 				@Override
134 				protected PcodeUseropLibrary<byte[]> createUseropLibrary() {
135 					return new DemoSyscallLibrary(this, program,
136 						StandAloneSyscallEmuExampleScript.this);
137 				}
138 
139 				// Uncomment this to see instructions printed as they are decoded
140 				/*
141 				@Override
142 				protected BytesPcodeThread createThread(String name) {
143 					return new BytesPcodeThread(name, this) {
144 						@Override
145 						protected SleighInstructionDecoder createInstructionDecoder(
146 								PcodeExecutorState<byte[]> sharedState) {
147 							return new SleighInstructionDecoder(language, sharedState) {
148 								@Override
149 								public Instruction decodeInstruction(Address address,
150 										RegisterValue context) {
151 									Instruction instruction = super.decodeInstruction(address, context);
152 									println("Decoded " + address + ": " + instruction);
153 									return instruction;
154 								}
155 							};
156 						}
157 					};
158 				}
159 				*/
160 			};
161 			PcodeThread<byte[]> thread = emulator.newThread();
162 			// The emulator composes the full library for each thread
163 			PcodeUseropLibrary<byte[]> library = thread.getUseropLibrary();
164 
165 			/*
166 			 * The library has a reference to the program and uses it to derive types and the system
167 			 * call numbering. However, the emulator itself does not have access to the program. If we
168 			 * followed the pattern in DebuggerEmuExampleScript, the emulator would have its state bound
169 			 * (indirectly) to the program. We'll need to copy the bytes in. Because we created blocks
170 			 * that were 0x1000 bytes in size, we can be fast and loose with our buffer. Ordinarily, you
171 			 * may want to copy in chunks rather than taking entire memory blocks at a time.
172 			 */
173 			byte[] data = new byte[0x1000];
174 			for (MemoryBlock block : program.getMemory().getBlocks()) {
175 				if (!block.isInitialized()) {
176 					continue; // Skip the syscall/OTHER block
177 				}
178 				Address addr = block.getStart();
179 				block.getBytes(addr, data);
180 				emulator.getSharedState()
181 						.setVar(addr.getAddressSpace(), addr.getOffset(), data.length, true, data);
182 			}
183 
184 			/*
185 			 * Initialize the thread
186 			 */
187 			PcodeProgram init =
188 				SleighProgramCompiler.compileProgram(language, "init", String.format("""
189 						RIP = 0x%s;
190 						RSP = 0x00001000;
191 						""", entry), library);
192 			thread.getExecutor().execute(init, library);
193 			thread.overrideContextWithDefault();
194 			thread.reInitialize();
195 
196 			/*
197 			 * Run the experiment: This should interrupt on the second SYSCALL, because we didn't
198 			 * provide a system call name in OTHER space for 20.
199 			 */
200 			try {
201 				thread.stepInstruction(10);
202 				printerr("We should not have completed 10 steps!");
203 			}
204 			catch (EmuInvalidSystemCallException e) {
205 				println("Terminated via invalid syscall. Good.");
206 			}
207 
208 			/*
209 			 * Inspect the machine. You can always do this by accessing the state directly, but for
210 			 * anything other than simple variables, you may find compiling an expression more
211 			 * convenient.
212 			 */
213 			println("RDI = " +
214 				Utils.bytesToLong(
215 					thread.getState().getVar(language.getRegister("RDI"), Reason.INSPECT), 8,
216 					language.isBigEndian()));
217 
218 			println("RDI = " + Utils.bytesToLong(
219 				SleighProgramCompiler.compileExpression(language, "RDI")
220 						.evaluate(thread.getExecutor()),
221 				8, language.isBigEndian()));
222 
223 			println("RDI+4 = " +
224 				Utils.bytesToLong(SleighProgramCompiler.compileExpression(language, "RDI+4")
225 						.evaluate(thread.getExecutor()),
226 					8, language.isBigEndian()));
227 
228 		}
229 		finally {
230 			if (program != null) {
231 				program.release(this);
232 			}
233 		}
234 	}
235 }
