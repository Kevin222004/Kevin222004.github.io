29  /**
30   * A userop library for the emulator
31   * 
32   * <p>
33   * If you do not need a custom userop library, use {@link PcodeUseropLibrary#NIL}. These libraries
34   * allow you to implement userops, including those declared by the language. Without these, the
35   * emulator must interrupt whenever a userop ({@code CALLOTHER}) is encountered. You can also define
36   * new userops, which can be invoked from Sleigh code injected into the emulator.
37   * 
38   * <p>
39   * These libraries can have both Java-callback and p-code implementations of userops. If only using
40   * p-code implementations, the library can be parameterized with type {@code <T>} and just pass that
41   * over to {@link AnnotatedPcodeUseropLibrary}. Because this will demo a Java callback that assumes
42   * concrete bytes, we will fix the library's type to {@code byte[]}. With careful use of the
43   * {@link PcodeArithmetic}, you can keep the type an abstract {@code <T>} with Java callbacks.
44   * 
45   * <p>
46   * Methods in this class (not including those in its nested classes) are implemented as Java
47   * callbacks.
48   */
49  public class DemoPcodeUseropLibrary extends AnnotatedPcodeUseropLibrary<byte[]> {
50  	private final static Charset UTF8 = Charset.forName("utf8");
51  
52  	private final SleighLanguage language;
53  	private final GhidraScript script;
54  	private final AddressSpace space;
55  
56  	public DemoPcodeUseropLibrary(SleighLanguage language, GhidraScript script) {
57  		this.language = language;
58  		this.script = script;
59  		this.space = language.getDefaultSpace();
60  
61  		new DemoStructuredPart(language.getDefaultCompilerSpec()).generate(ops);
62  	}
63  
64  	/**
65  	 * Treats the input as an offset to a C-style string and prints it to the console
66  	 * 
67  	 * <p>
68  	 * Because we want to dereference start, we will need access to the emulator's state, so we
69  	 * employ the {@link OpState} annotation. {@code start} takes the one input we expect. Because
70  	 * its type is the value type rather than {@link Varnode}, we will get the input's value.
71  	 * Similarly, we can just return the resulting value, and the emulator will place that into the
72  	 * output variable for us.
73  	 * 
74  	 * @param state the calling thread's state
75  	 * @param start the offset of the first character
76  	 * @return the length of the string in bytes
77  	 */
78  	@PcodeUserop
79  	public byte[] print_utf8(@OpExecutor PcodeExecutor<byte[]> executor, byte[] start) {
80  		PcodeExecutorState<byte[]> state = executor.getState();
81  		long offset = Utils.bytesToLong(start, start.length, language.isBigEndian());
82  		long end = offset;
83  		Reason reason = executor.getReason();
84  		while (state.getVar(space, end, 1, true, reason)[0] != 0) {
85  			end++;
86  		}
87  		if (end == offset) {
88  			script.println("");
89  			return Utils.longToBytes(0, Long.BYTES, language.isBigEndian());
90  		}
91  		byte[] bytes = state.getVar(space, offset, (int) (end - offset), true, reason);
92  		String str = new String(bytes, UTF8);
93  		script.println(str);
94  		return Utils.longToBytes(end - offset, Long.BYTES, language.isBigEndian());
95  	}
96  
97  	/**
98  	 * Methods in this class are implemented using p-code compiled from Structured Sleigh
99  	 */
100 	public class DemoStructuredPart extends StructuredSleigh {
101 		final Var RAX = lang("RAX", type("long"));
102 		final Var RCX = lang("RCX", type("byte *"));
103 		final UseropDecl emu_swi = userop(type("void"), "emu_swi", List.of());
104 
105 		protected DemoStructuredPart(CompilerSpec cs) {
106 			super(cs);
107 		}
108 
109 		/**
110 		 * Not really a syscall dispatcher
111 		 * 
112 		 * <p>
113 		 * In cases where the userop expects parameters, you would annotate them with {@link Param}
114 		 * and use them just like other {@link Var}s. See the javadocs.
115 		 * 
116 		 * <p>
117 		 * This is just a cheesy demo: If RAX is 1, then this method computes the number of bytes in
118 		 * the C-style string pointed to by RCX and stores the result in RAX. Otherwise, interrupt
119 		 * the emulator. See {@link DemoSyscallLibrary} for actual system call simulation.
120 		 */
121 		@StructuredUserop
122 		public void syscall() {
123 			_if(RAX.eq(1), () -> {
124 				Var i = local("i", RCX);
125 				_while(i.deref().neq(0), () -> {
126 					i.inc();
127 				});
128 				RAX.set(i.subi(RAX));
129 			})._else(() -> {
130 				emu_swi.call();
131 			});
132 		}
133 	}
