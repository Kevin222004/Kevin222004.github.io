16  //Finds programs containing various audio resources such as WAV's 
17  //@category Resources
18  import ghidra.app.script.GhidraScript;
19  import ghidra.program.model.address.Address;
20  import ghidra.program.model.data.DataType;
21  import ghidra.program.model.data.WAVEDataType;
22  import ghidra.program.model.listing.Data;
23  import ghidra.program.model.mem.Memory;
24  import ghidra.program.model.mem.MemoryBlock;
25  
26  import java.util.ArrayList;
27  import java.util.List;
28  
29  public class FindAudioInProgramScript extends GhidraScript {
30  
31  	@Override
32  	public void run() throws Exception {
33  
34  		int totalFound = 0;
35  		//Look for potential file by checking for format byte patterns
36  
37  		//look for WAV data types
38  		WAVEDataType wdt = new WAVEDataType();
39  
40  		totalFound += findAudioData("WAV", wdt, WAVEDataType.MAGIC, WAVEDataType.MAGIC_MASK);
41  
42  		if (totalFound == 0) {
43  			println("No Audio data found in " + currentProgram.getName());
44  			if (this.isRunningHeadless()) {
45  				currentProgram.setTemporary(true);
46  			}
47  		}
48  
49  	}
50  
51  	private int findAudioData(String dataName, DataType dt, byte[] pattern, byte[] mask) {
52  
53  		println("Looking for " + dataName + "'s in " + currentProgram.getName());
54  
55  		int numDataFound = 0;
56  		List<Address> foundList = scanForAudioData(pattern, mask);
57  		//Loop over all potential found WAVs
58  		for (int i = 0; i < foundList.size(); i++) {
59  			boolean foundData = false;
60  			//See if already applied WAV
61  			Data data = getDataAt(foundList.get(i));
62  			//If not already applied, try to apply WAV data type
63  			if (data == null) {
64  				println("Trying to apply " + dataName + " datatype at " +
65  					foundList.get(i).toString());
66  
67  				try {
68  					Data newData = createData(foundList.get(i), dt);
69  					if (newData != null) {
70  						println("Applied WAV at " + newData.getAddressString(false, true));
71  						foundData = true;
72  					}
73  				}
74  				//If data does not apply correctly then it is not really that kind of data
75  				//Or it is bumping into other data
76  				catch (Exception e) {
77  					println("Invalid " + dataName + " at " + foundList.get(i).toString());
78  				}
79  			}
80  			else if (data.getMnemonicString().equals(dataName)) {
81  				println(dataName + " already applied at " + data.getAddressString(false, true));
82  				foundData = true;
83  			}
84  
85  			//print found message only for those that apply correctly or were already applied
86  			if (foundData) {
87  				println("Found " + dataName + " in program " + currentProgram.getExecutablePath() +
88  					" at address " + foundList.get(i).toString());
89  				numDataFound++;
90  			}
91  
92  		}
93  		return numDataFound;
94  	}
95  
96  	List<Address> scanForAudioData(byte[] imageBytes, byte[] mask) {
97  		Memory memory = currentProgram.getMemory();
98  		MemoryBlock[] blocks = memory.getBlocks();
99  
100 		List<Address> foundImages = new ArrayList<Address>();
101 
102 		for (int i = 0; i < blocks.length; i++) {
103 			if (blocks[i].isInitialized()) {
104 				Address start = blocks[i].getStart();
105 				Address found = null;
106 				while (true) {
107 					if (monitor.isCancelled()) {
108 						break;
109 					}
110 					found =
111 						memory.findBytes(start, blocks[i].getEnd(), imageBytes, mask, true, monitor);
112 					if (found != null) {
113 						foundImages.add(found);
114 						start = found.add(1);
115 					}
116 					else
117 						break;
118 				}
119 			}
120 		}
121 		return foundImages;
122 	}
123 }
