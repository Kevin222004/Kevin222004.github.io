49  /**
50   * This script will resolve system calls for x86 or x64 Linux binaries.
51   * It assumes that in the x64 case, the syscall native instruction is used to make system calls,
52   * and in the x86 case, system calls are made via an indirect call to GS:[0x10].
53   * It should be straightforward to modify this script for other cases.
54   */
55  public class ResolveX86orX64LinuxSyscallsScript extends GhidraScript {
56  
57  	//disassembles to "CALL dword ptr GS:[0x10]"
58  	private static final byte[] x86_bytes = { 0x65, -1, 0x15, 0x10, 0x00, 0x00, 0x00 };
59  
60  	private static final String X86 = "x86";
61  
62  	private static final String SYSCALL_SPACE_NAME = "syscall";
63  
64  	private static final int SYSCALL_SPACE_LENGTH = 0x10000;
65  
66  	//this is the name of the userop (aka CALLOTHER) in the pcode translation of the
67  	//native "syscall" instruction
68  	private static final String SYSCALL_X64_CALLOTHER = "syscall";
69  
70  	//a set of names of all syscalls that do not return
71  	private static final Set<String> noreturnSyscalls = Set.of("exit", "exit_group");
72  
73  	//tests whether an instruction is making a system call
74  	private Predicate<Instruction> tester;
75  
76  	//register holding the syscall number
77  	private String syscallRegister;
78  
79  	//datatype archive containing signature of system calls
80  	private String datatypeArchiveName;
81  
82  	//file containing map from syscall numbers to syscall names
83  	//note that different architectures can have different system call numbers, even
84  	//if they're both Linux...
85  	private String syscallFileName;
86  
87  	//the type of overriding reference to apply 
88  	private RefType overrideType;
89  
90  	//the calling convention to use for system calls (must be defined in the appropriate .cspec file)
91  	private String callingConvention;
92  
93  	@Override
94  	protected void run() throws Exception {
95  
96  		if (!(currentProgram.getExecutableFormat().equals(ElfLoader.ELF_NAME) &&
97  			currentProgram.getLanguage().getProcessor().toString().equals(X86))) {
98  			popup("This script is intended for x86 or x64 Linux files");
99  			return;
100 		}
101 
102 		//determine whether the executable is 32 or 64 bit and set fields appropriately
103 		int size = currentProgram.getLanguage().getLanguageDescription().getSize();
104 		if (size == 64) {
105 			tester = ResolveX86orX64LinuxSyscallsScript::checkX64Instruction;
106 			syscallRegister = "RAX";
107 			datatypeArchiveName = "generic_clib_64";
108 			syscallFileName = "x64_linux_syscall_numbers";
109 			overrideType = RefType.CALLOTHER_OVERRIDE_CALL;
110 			callingConvention = "syscall";
111 		}
112 		else {
113 			tester = ResolveX86orX64LinuxSyscallsScript::checkX86Instruction;
114 			syscallRegister = "EAX";
115 			datatypeArchiveName = "generic_clib";
116 			syscallFileName = "x86_linux_syscall_numbers";
117 			overrideType = RefType.CALL_OVERRIDE_UNCONDITIONAL;
118 			callingConvention = "syscall";
119 		}
120 
121 		//get the space where the system calls live.  
122 		//If it doesn't exist, create it.
123 		AddressSpace syscallSpace =
124 			currentProgram.getAddressFactory().getAddressSpace(SYSCALL_SPACE_NAME);
125 		if (syscallSpace == null) {
126 			//don't muck with address spaces if you don't have exclusive access to the program.
127 			if (!currentProgram.hasExclusiveAccess()) {
128 				popup("Must have exclusive access to " + currentProgram.getName() +
129 					" to run this script");
130 				return;
131 			}
132 			Address startAddr = currentProgram.getAddressFactory()
133 					.getAddressSpace(SpaceNames.OTHER_SPACE_NAME)
134 					.getAddress(0x0L);
135 			AddUninitializedMemoryBlockCmd cmd = new AddUninitializedMemoryBlockCmd(
136 				SYSCALL_SPACE_NAME, null, this.getClass().getName(), startAddr,
137 				SYSCALL_SPACE_LENGTH, true, true, true, false, true);
138 			if (!cmd.applyTo(currentProgram)) {
139 				popup("Failed to create " + SYSCALL_SPACE_NAME);
140 				return;
141 			}
142 			syscallSpace = currentProgram.getAddressFactory().getAddressSpace(SYSCALL_SPACE_NAME);
143 		}
144 		else {
145 			printf("AddressSpace %s found, continuing...\n", SYSCALL_SPACE_NAME);
146 		}
147 
148 		//get all of the functions that contain system calls
149 		//note that this will not find system call instructions that are not in defined functions
150 		Map<Function, Set<Address>> funcsToCalls = getSyscallsInFunctions(currentProgram, monitor);
151 
152 		if (funcsToCalls.isEmpty()) {
153 			popup("No system calls found (within defined functions)");
154 			return;
155 		}
156 
157 		//get the system call number at each callsite of a system call.
158 		//note that this is not guaranteed to succeed at a given system call call site -
159 		//it might be hard (or impossible) to determine a specific constant
160 		Map<Address, Long> addressesToSyscalls =
161 			resolveConstants(funcsToCalls, currentProgram, monitor);
162 
163 		if (addressesToSyscalls.isEmpty()) {
164 			popup("Couldn't resolve any syscall constants");
165 			return;
166 		}
167 
168 		//get the map from system call numbers to system call names
169 		//you might have to create this yourself!
170 		Map<Long, String> syscallNumbersToNames = getSyscallNumberMap();
171 
172 		//at each system call call site where a constant could be determined, create
173 		//the system call (if not already created), then add the appropriate overriding reference
174 		//use syscallNumbersToNames to name the created functions
175 		//if there's not a name corresponding to the constant use a default 
176 		for (Entry<Address, Long> entry : addressesToSyscalls.entrySet()) {
177 			Address callSite = entry.getKey();
178 			Long offset = entry.getValue();
179 			Address callTarget = syscallSpace.getAddress(offset);
180 			Function callee = currentProgram.getFunctionManager().getFunctionAt(callTarget);
181 			if (callee == null) {
182 				String funcName = "syscall_" + String.format("%08X", offset);
183 				if (syscallNumbersToNames.get(offset) != null) {
184 					funcName = syscallNumbersToNames.get(offset);
185 				}
186 				callee = createFunction(callTarget, funcName);
187 				callee.setCallingConvention(callingConvention);
188 
189 				//check if the function name is one of the non-returning syscalls
190 				if (noreturnSyscalls.contains(funcName)) {
191 					callee.setNoReturn(true);
192 				}
193 			}
194 			Reference ref = currentProgram.getReferenceManager()
195 					.addMemoryReference(callSite, callTarget, overrideType, SourceType.USER_DEFINED,
196 						Reference.MNEMONIC);
197 			//overriding references must be primary to be active
198 			currentProgram.getReferenceManager().setPrimary(ref, true);
199 		}
200 
201 		//finally, open the appropriate data type archive and apply its function data types
202 		//to the new system call space, so that the system calls have the correct signatures
203 		AutoAnalysisManager mgr = AutoAnalysisManager.getAnalysisManager(currentProgram);
204 		DataTypeManagerService service = mgr.getDataTypeManagerService();
205 		List<DataTypeManager> dataTypeManagers = new ArrayList<>();
206 		dataTypeManagers.add(service.openDataTypeArchive(datatypeArchiveName));
207 		dataTypeManagers.add(currentProgram.getDataTypeManager());
208 		ApplyFunctionDataTypesCmd cmd = new ApplyFunctionDataTypesCmd(dataTypeManagers,
209 			new AddressSet(syscallSpace.getMinAddress(), syscallSpace.getMaxAddress()),
210 			SourceType.USER_DEFINED, false, false);
211 		cmd.applyTo(currentProgram);
212 	}

214 	//TODO: better error checking!
215 	private Map<Long, String> getSyscallNumberMap() {
216 		Map<Long, String> syscallMap = new HashMap<>();
217 		ResourceFile rFile = Application.findDataFileInAnyModule(syscallFileName);
218 		if (rFile == null) {
219 			popup("Error opening syscall number file, using default names");
220 			return syscallMap;
221 		}
222 		try (FileReader fReader = new FileReader(rFile.getFile(false));
223 				BufferedReader bReader = new BufferedReader(fReader)) {
224 			String line = null;
225 			while ((line = bReader.readLine()) != null) {
226 				//lines starting with # are comments
227 				if (!line.startsWith("#")) {
228 					String[] parts = line.trim().split(" ");
229 					Long number = Long.parseLong(parts[0]);
230 					syscallMap.put(number, parts[1]);
231 				}
232 			}
233 		}
234 		catch (IOException e) {
235 			Msg.showError(this, null, "Error reading syscall map file", e.getMessage(), e);
236 		}
237 		return syscallMap;
238 	}

240 	/**
241 	 * Scans through all of the functions defined in {@code program} and returns
242 	 * a map which takes a function to the set of address in its body which contain
243 	 * system calls
244 	 * @param program program containing functions
245 	 * @param tMonitor monitor
246 	 * @return map function -> addresses in function containing syscalls
247 	 * @throws CancelledException if the user cancels
248 	 */
249 	private Map<Function, Set<Address>> getSyscallsInFunctions(Program program,
250 			TaskMonitor tMonitor) throws CancelledException {
251 		Map<Function, Set<Address>> funcsToCalls = new HashMap<>();
252 		for (Function func : program.getFunctionManager().getFunctionsNoStubs(true)) {
253 			tMonitor.checkCancelled();
254 			for (Instruction inst : program.getListing().getInstructions(func.getBody(), true)) {
255 				if (tester.test(inst)) {
256 					Set<Address> callSites = funcsToCalls.get(func);
257 					if (callSites == null) {
258 						callSites = new HashSet<>();
259 						funcsToCalls.put(func, callSites);
260 					}
261 					callSites.add(inst.getAddress());
262 				}
263 			}
264 		}
265 		return funcsToCalls;
266 	}

268 	/**
269 	 * Uses the symbolic propogator to attempt to determine the constant value in
270 	 * the syscall register at each system call instruction
271 	 * 
272 	 * @param funcsToCalls map from functions containing syscalls to address in each function of 
273 	 * the system call
274 	 * @param program containing the functions
275 	 * @return map from addresses of system calls to system call numbers
276 	 * @throws CancelledException if the user cancels
277 	 */
278 	private Map<Address, Long> resolveConstants(Map<Function, Set<Address>> funcsToCalls,
279 			Program program, TaskMonitor tMonitor) throws CancelledException {
280 		Map<Address, Long> addressesToSyscalls = new HashMap<>();
281 		Register syscallReg = program.getLanguage().getRegister(syscallRegister);
282 		for (Function func : funcsToCalls.keySet()) {
283 			Address start = func.getEntryPoint();
284 			ContextEvaluator eval = new ConstantPropagationContextEvaluator(monitor, true);
285 			SymbolicPropogator symEval = new SymbolicPropogator(program);
286 			symEval.flowConstants(start, func.getBody(), eval, true, tMonitor);
287 			for (Address callSite : funcsToCalls.get(func)) {
288 				Value val = symEval.getRegisterValue(callSite, syscallReg);
289 				if (val == null) {
290 					createBookmark(callSite, "System Call",
291 						"Couldn't resolve value of " + syscallReg);
292 					printf("Couldn't resolve value of " + syscallReg + " at " + callSite + "\n");
293 					continue;
294 				}
295 				addressesToSyscalls.put(callSite, val.getValue());
296 			}
297 		}
298 		return addressesToSyscalls;
299 	}

301 	/**
302 	 * Checks whether an x86 native instruction is a system call
303 	 * @param inst instruction to check
304 	 * @return true precisely when the instruction is a system call
305 	 */
306 	private static boolean checkX86Instruction(Instruction inst) {
307 		try {
308 			return Arrays.equals(x86_bytes, inst.getBytes());
309 		}
310 		catch (MemoryAccessException e) {
311 			Msg.info(ResolveX86orX64LinuxSyscallsScript.class,
312 				"MemoryAccessException at " + inst.getAddress().toString());
313 			return false;
314 		}
315 	}
316 
317 	/**
318 	 * Checks whether an x64 instruction is a system call
319 	 * @param inst instruction to check
320 	 * @return true precisely when the instruction is a system call
321 	 */
322 	private static boolean checkX64Instruction(Instruction inst) {
323 		boolean retVal = false;
324 		for (PcodeOp op : inst.getPcode()) {
325 			if (op.getOpcode() == PcodeOp.CALLOTHER) {
326 				int index = (int) op.getInput(0).getOffset();
327 				if (inst.getProgram()
328 						.getLanguage()
329 						.getUserDefinedOpName(index)
330 						.equals(SYSCALL_X64_CALLOTHER)) {
331 					retVal = true;
332 				}
333 			}
334 		}
335 		return retVal;
336 	}
