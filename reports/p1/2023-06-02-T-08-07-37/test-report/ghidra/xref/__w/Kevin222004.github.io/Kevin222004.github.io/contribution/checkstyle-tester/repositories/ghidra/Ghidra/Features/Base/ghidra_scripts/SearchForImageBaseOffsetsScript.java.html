16  // This script searches for image base offset references of size 32 and 64 to the current cursor
17  // location. This script only works on programs of size 32 and 64. The results are both printed to
18  // the console and are presented in a table with two tabs, one for each size. To apply data types to
19  // undefined ones that are found simply select the desired ones from the table then drag either the
20  // ImageBaseOffset32 or ImageBaseOffset64 data type, whichever is applicable, onto the selection in 
21  // the listing from the Data Type Manager.
22  // 
23  //@category Search
24  
25  import ghidra.app.script.GhidraScript;
26  import ghidra.program.model.address.*;
27  import ghidra.util.exception.CancelledException;
28  
29  public class SearchForImageBaseOffsetsScript extends GhidraScript {
30  
31  	static final int POINTER_BYTE_LEN_64BIT = 8;
32  	static final int POINTER_BYTE_LEN_32BIT = 4;
33  	static final byte BYTE_MASK = (byte) 0xff;
34  	static final int BITS_PER_BYTE = 8;
35  
36  	@Override
37  	public void run() throws Exception {
38  
39  		if (currentProgram == null) {
40  			println("No open program");
41  			return;
42  		}
43  
44  		long defaultPointerSize = currentProgram.getDefaultPointerSize();
45  		if (defaultPointerSize != 4 && defaultPointerSize != 8) {
46  			println("This script only works on 32 or 64 bit programs");
47  			return;
48  		}
49  
50  		Address imageBase = currentProgram.getImageBase();
51  
52  		boolean isBigEndian = currentProgram.getMemory().isBigEndian();
53  
54  		long currentAddressOffset = currentAddress.getOffset();
55  		long imageBaseOffset = imageBase.getOffset();
56  
57  		long currentAddressIbo = currentAddressOffset - imageBaseOffset;
58  
59  		byte searchBytes[] =
60  			createSearchArray(currentAddressIbo, POINTER_BYTE_LEN_64BIT, isBigEndian);
61  		println("searching for possible ibo64 references to " + currentAddress.toString() + " ...");
62  		AddressSet ibo64refs = searchForByteArray(searchBytes);
63  		printAddresses(ibo64refs);
64  
65  		searchBytes = createSearchArray(currentAddressIbo, POINTER_BYTE_LEN_32BIT, isBigEndian);
66  		println("searching for possible ibo32 references to " + currentAddress.toString() + " ...");
67  		AddressSet ibo32refs = searchForByteArray(searchBytes);
68  		printAddresses(ibo32refs);
69  
70  		show("64-bit ImageBaseOffset References", ibo64refs);
71  		show("32-bit ImageBaseOffset References", ibo32refs);
72  
73  	}
74  
75  	/**
76  	 * Method to return an appropriate sized and endian-ordered byte array out of the given long value
77  	 * @param value the given value
78  	 * @param numBytes the number of bytes to copy into the array
79  	 * @return the appropriate sized and endian-ordered byte array for the given value
80  	 * @throws CancelledException if cancelled
81  	 */
82  	private byte[] createSearchArray(long value, int numBytes, boolean isBigEndian)
83  			throws CancelledException {
84  
85  		if (isBigEndian) {
86  			return createBigEndianByteArray(value, numBytes);
87  		}
88  		return createLittleEndianByteArray(value, numBytes);
89  	}
90  
91  	/**
92  	 * Method to create a little endian ordered byte array out of the given long value
93  	 * @param value the given value
94  	 * @param numBytes the number of bytes to copy into the array
95  	 * @return the little endian byte array for the given value
96  	 * @throws CancelledException if cancelled
97  	 */
98  	private byte[] createLittleEndianByteArray(long value, int numBytes)
99  			throws CancelledException {
100 
101 
102 		byte byteArray[] = new byte[numBytes];
103 
104 		for (int i = 0; i < numBytes; i++) {
105 			monitor.checkCancelled();
106 			byteArray[i] = (byte) (value >> (BITS_PER_BYTE * i) & BYTE_MASK);
107 		}
108 
109 		return byteArray;
110 	}
111 
112 	/**
113 	 * Method to create a big endian byte array out of the given long value
114 	 * @param value the given value
115 	 * @param numBytes the number of bytes to copy into the array
116 	 * @return the big endian byte array for the given value
117 	 * @throws CancelledException if cancelled
118 	 */
119 	private byte[] createBigEndianByteArray(long value, int numBytes) throws CancelledException {
120 
121 		byte byteArray[] = new byte[numBytes];
122 
123 		for (int i = 0; i < numBytes; i++) {
124 			monitor.checkCancelled();
125 			byteArray[i] = (byte) (value >> (8 * (numBytes - (i + 1))) & 0xff);
126 		}
127 
128 		return byteArray;
129 	}
130 
131 	/**
132 	 * Method to search for the given byte array and print the address(es) where it is found
133 	 * @param byteArray the given byte array
134 	 * @throws CancelledException if cancelled
135 	 */
136 	private AddressSet searchForByteArray(byte[] byteArray) throws CancelledException {
137 
138 		AddressSet addressSet = new AddressSet();
139 
140 		Address start = currentProgram.getMinAddress();
141 		Address found = find(start, byteArray);
142 		while (found != null) {
143 			monitor.checkCancelled();
144 			addressSet.add(found);
145 			start = found.add(1);
146 			found = find(start, byteArray);
147 		}
148 
149 		return addressSet;
150 	}
151 
152 	private void printAddresses(AddressSet addressSet) throws CancelledException {
153 
154 		AddressIterator addresses = addressSet.getAddresses(true);
155 		while (addresses.hasNext()) {
156 			monitor.checkCancelled();
157 			Address address = addresses.next();
158 			println(address.toString());
159 		}
160 	}
161 
162 }
