16  //
17  // When an array of structures is created at a location, and the
18  // array type contains a pointer type, Ghidra will lay down valid
19  // references as long as the pointer is valid (points to something
20  // in memory).  However, at the first sign of invalidity, the
21  // CodeManager bails and stops laying down references.
22  //
23  // This script, when run at the location of the array of structures,
24  // will update each individual structure element, iterating through
25  // the whole array until the end.
26  //
27  //@category Data Types
28  
29  import ghidra.app.script.GhidraScript;
30  import ghidra.program.database.ProgramDB;
31  import ghidra.program.database.code.CodeManager;
32  import ghidra.program.model.listing.Data;
33  
34  public class FixArrayStructReferencesScript extends GhidraScript {
35  
36  	@Override
37  	protected void run() throws Exception {
38  		Data data = getDataAt(currentAddress);
39  		if (data == null) {
40  			printerr("no data at " + currentAddress);
41  			return;
42  		}
43  		if (!data.isArray()) {
44  			printerr("data at " + currentAddress + " is not an array");
45  			return;
46  		}
47  		CodeManager codeManager = ((ProgramDB) currentProgram).getCodeManager();
48  		int numComponents = data.getNumComponents();
49  		monitor.setMessage("updating data references in array");
50  		monitor.initialize(numComponents);
51  		for (int ii = 0; ii < numComponents; ++ii) {
52  			if (monitor.isCancelled()) {
53  				printerr("cancelled");
54  				return;
55  			}
56  			monitor.incrementProgress(1);
57  			Data component = data.getComponent(ii);
58  			codeManager.updateDataReferences(component);
59  		}
60  	}
61  }
