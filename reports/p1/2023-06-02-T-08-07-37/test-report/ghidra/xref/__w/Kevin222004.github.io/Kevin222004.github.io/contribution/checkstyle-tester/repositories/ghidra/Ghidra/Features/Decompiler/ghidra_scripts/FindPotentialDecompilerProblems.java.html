16  // Finds potential problems in code that will cause trouble for the decompiler.
17  //
18  //  This script runs the decompiler on each currently defined function.
19  //  Function that has potential issues in the decompiler output is flagged in a table with a suggestion.
20  //  For example any references to "in_" variables, or "unaff_" are flagged with a potential solution.
21  //  This script is still a work in progress, but it can help diagnose initial analysis of a binary for problems that
22  //  affect decompiled output and thus the cleanness of code for follow on uses.  Things like unidentified
23  //  epilog functions that have side-effects on the stack will be uncovered.
24  //
25  // @category Analysis
26

41  public class FindPotentialDecompilerProblems extends GhidraScript {
42  
43  	@Override
44  	public void run() throws Exception {
45  
46  		if (isRunningHeadless()) {
47  			printf("This script cannot be run in headless mode.\n");
48  			return;
49  		}
50  
51  		TableChooserDialog tableDialog = createTableChooserDialog(
52  			"Possible Decompiler Problems: " + currentProgram.getName(), null);
53  		configureTableColumns(tableDialog);
54  		tableDialog.show();
55  		IssueEntries entryList = new TableEntryList(tableDialog);
56  
57  		DecompilerCallback<Void> callback =
58  			new DecompilerCallback<Void>(currentProgram, new BasicConfigurer(currentProgram)) {
59  
60  				@Override
61  				public Void process(DecompileResults results, TaskMonitor tMonitor)
62  						throws Exception {
63  					for (ProblemLocation probLoc : processFunc(results)) {
64  						entryList.add(probLoc);
65  					}
66  					return null;
67  				}
68  			};
69  
70  		Set<Function> funcsToDecompile = new HashSet<>();
71  		FunctionIterator fIter = currentProgram.getFunctionManager().getFunctionsNoStubs(true);
72  		fIter.forEach(e -> funcsToDecompile.add(e));
73  
74  		if (funcsToDecompile.isEmpty()) {
75  			popup("No functions to decompile!");
76  			return;
77  		}
78  
79  		ParallelDecompiler.decompileFunctions(callback, funcsToDecompile, monitor);
80  		monitor.checkCancelled();
81  		tableDialog.setMessage("Finished");
82  	}

84  	private List<ProblemLocation> processFunc(DecompileResults decompResult) {
85  
86  		//TODO: skip if func name contains SEH_epilog or SEH_prolog?
87  		//add general way to have architecture-specific messages?
88  
89  		List<ProblemLocation> problems = new ArrayList<>();
90  		Function func = decompResult.getFunction();
91  		HighFunction hf = decompResult.getHighFunction();
92  		if (hf == null) {
93  			problems.add(new ProblemLocation(currentProgram, func.getEntryPoint(),
94  				func.getEntryPoint(), "", "Decompilation Error"));
95  			return problems;
96  		}
97  
98  		Iterator<HighSymbol> symIter = hf.getLocalSymbolMap().getSymbols();
99  		while (symIter.hasNext() && !monitor.isCancelled()) {
100 			HighSymbol sym = symIter.next();
101 			HighVariable highVar = sym.getHighVariable();
102 			if (!(highVar instanceof HighLocal)) {
103 				continue;
104 			}
105 			if (sym.getName().startsWith("in_") && !sym.getName().equals("in_FS_OFFSET")) {
106 				// Has an input variable that is not a parameter
107 				String possible =
108 					"Function signature missing register param, called function passed too many register params, or only a subpiece" +
109 						" of a register actually used.";
110 				if (!(hf.getFunction().getSymbol().isGlobal()) &&
111 					!(hf.getFunction().getCallingConventionName().contains("thiscall"))) {
112 					possible += " Function might need calling convention changed to thiscall";
113 				}
114 				if (sym.getName().startsWith("in_stack_ff")) {
115 					possible =
116 						"Too many stack parameters passed to a called function.  May need to redefine in the called function (could be varargs).";
117 				}
118 				if (sym.getName().startsWith("in_stack_00")) {
119 					possible =
120 						"Too few stack parameters defined for this function.  May need to redefine parameters.";
121 				}
122 
123 				Address funcAddr =
124 					getFirstFuncWithVar(func, sym.getHighVariable().getRepresentative());
125 
126 				//if we didn't find a good location for the cause of the problem, 
127 				//just use the entry point of the function with the problem
128 				if (funcAddr == null || funcAddr.equals(Address.NO_ADDRESS)) {
129 					funcAddr = func.getEntryPoint();
130 				}
131 				problems.add(new ProblemLocation(currentProgram, func.getEntryPoint(), funcAddr,
132 					sym.getName(), possible));
133 			}
134 
135 			if (sym.getName().startsWith("unaff_")) {
136 				Address firstAddr = getFirstCalledFunction(func);
137 				if (sym.getName().equals("unaff_EBP")) {
138 					problems.add(
139 						new ProblemLocation(currentProgram, firstAddr, func.getEntryPoint(),
140 							sym.getName(), "Suspect function is EH_PROLOG/EH_EPILOG"));
141 					continue;
142 				}
143 				// Has a side effect variable
144 				String possible = (firstAddr != null ? "Side effect from a call"
145 						: "Undefined parameter or global register save");
146 				//TODO: sym.getPCAddress() points outside of function bodies in certain cases...
147 				problems.add(new ProblemLocation(currentProgram, func.getEntryPoint(),
148 					sym.getPCAddress(), sym.getName(), possible));
149 			}
150 			//extraout_X: will sym.getHighVariable().getRepresentative return X?
151 			if (sym.getName().startsWith("extraout")) {
152 				// Has a side effect variable
153 				Address funcAddr =
154 					getFirstFuncWithVar(func, sym.getHighVariable().getRepresentative());
155 
156 				if (funcAddr.equals(Address.NO_ADDRESS)) {
157 					funcAddr = func.getEntryPoint();
158 				}
159 
160 				String possible =
161 					"Bad parameter in called function or extra return value/global register/function register side effect";
162 				if (sym.getName().startsWith("extraout_var")) {
163 					possible = "Function containing problem may need return type adjusted.";
164 				}
165 				problems.add(new ProblemLocation(currentProgram, func.getEntryPoint(), funcAddr,
166 					sym.getName(), possible));
167 			}
168 		}
169 		return problems;
170 	}

172 	/**
173 	 * Returns the target of the first (in address order) call in the body of {@code func}
174 	 * which takes {@code vn} as a parameter.
175 	 * @param func {@link Function} whose body to search for calls
176 	 * @param vn {@link Varnode} representing required parameter
177 	 * @return entry point of first function called by {@code func} which uses {@code vn}
178 	 * as a parameter, or {@code Address.NO_ADDRESS} if no such function found.
179 	 */
180 	private Address getFirstFuncWithVar(Function func, Varnode vn) {
181 		Address variableAddr = vn.getAddress();
182 		if (variableAddr == null) {
183 			return Address.NO_ADDRESS;
184 		}
185 
186 		// Note: this handles some cases where functions consist of non-contiguous blocks,
187 		// but since we start at the entry point we might miss things if part of the body of
188 		// the function is before the entry point (in address order)
189 		ReferenceIterator refIter =
190 			func.getProgram().getReferenceManager().getReferenceIterator(func.getEntryPoint());
191 
192 		Address maxAddr = func.getBody().getMaxAddress();
193 
194 		// return the first call to a function which takes vn as an argument
195 		for (Reference ref : CollectionUtils.asIterable(refIter)) {
196 			// check whether we are at an address not in the function
197 			// only necessary in case func consists of non-contiguous blocks
198 			// TODO: handle tail-call elimination
199 			if (!func.getBody().contains(ref.getFromAddress())) {
200 				continue;
201 			}
202 			if (isValidCallReference(ref)) {
203 				Function calledFunc =
204 					currentProgram.getFunctionManager().getFunctionAt(ref.getToAddress());
205 				Parameter[] params = calledFunc.getParameters();
206 				for (Parameter param : params) {
207 					Address addr = param.getMinAddress();
208 					if (addr != null && addr.equals(variableAddr)) {
209 						return ref.getToAddress();
210 					}
211 				}
212 			}
213 			// The references are sorted by their "from" addresses, so if this condition is true, 
214 			// we've searched all the references from the body of func and haven't found anything.
215 			// So, stop looking.
216 			if (ref.getFromAddress().compareTo(maxAddr) > 0) {
217 				return Address.NO_ADDRESS;
218 			}
219 		}
220 		//in case there are no references with "from" addresses after the body of func
221 		return Address.NO_ADDRESS;
222 	}
223 
224 	/**
225 	 * Returns the address of first function called by {@code func}.  That is, the returned {@link Address}
226 	 * is the target of the call instruction with the least address within the body of {@code func}. 
227 	 * @param func the {@link Function} to search for calls
228 	 * @return the {@link Address} of the first called function, or {@code Address.NO_ADDRESS} if
229 	 * no calls are found.
230 	 */
231 	private Address getFirstCalledFunction(Function func) {
232 
233 		//could be issues if func's body has addresses that are before the entry point of
234 		//func - see the comment in getFirstFuncWithVar
235 		ReferenceIterator refIter =
236 			func.getProgram().getReferenceManager().getReferenceIterator(func.getEntryPoint());
237 
238 		Address maxAddr = func.getBody().getMaxAddress();
239 
240 		for (Reference ref : CollectionUtils.asIterable(refIter)) {
241 			// check whether we are at an address not in the function
242 			// only necessary in case func consists of non-contiguous blocks
243 			// TODO: handle tail-call elimination
244 			if (!func.getBody().contains(ref.getFromAddress())) {
245 				continue;
246 			}
247 
248 			// return the first call for the function
249 			if (isValidCallReference(ref)) {
250 				return ref.getToAddress();
251 			}
252 			// The references are sorted by their "from" addresses, so if this condition is true, 
253 			// we've searched all the references from the body of func and haven't found anything.
254 			// So, stop looking.
255 			if (ref.getFromAddress().compareTo(maxAddr) > 0) {
256 				return Address.NO_ADDRESS;
257 			}
258 		}
259 		//in case there are no references with "from" addresses after the body of func
260 		return Address.NO_ADDRESS;
261 	}
262 
263 	//returns true precisely when ref is a call reference to a defined function
264 	private boolean isValidCallReference(Reference ref) {
265 		if (!ref.getReferenceType().isCall()) {
266 			return false;
267 		}
268 		if (ref.getToAddress() == null) {
269 			return false;
270 		}
271 		if (currentProgram.getFunctionManager().getFunctionAt(ref.getToAddress()) != null) {
272 			return true;
273 		}
274 		return false;
275 
276 	}

278 	static class BasicConfigurer implements DecompileConfigurer {
279 		private Program p;
280 
281 		public BasicConfigurer(Program prog) {
282 			p = prog;
283 		}
284 
285 		@Override
286 		public void configure(DecompInterface decompiler) {
287 			decompiler.toggleCCode(true);
288 			decompiler.toggleSyntaxTree(true);
289 			decompiler.setSimplificationStyle("decompile");
290 			DecompileOptions opts = new DecompileOptions();
291 			opts.grabFromProgram(p);
292 			decompiler.setOptions(opts);
293 		}
294 	}

300 	private void configureTableColumns(TableChooserDialog dialog) {
301 		StringColumnDisplay explanationColumn = new StringColumnDisplay() {
302 			@Override
303 			public String getColumnName() {
304 				return "Potential Problem";
305 			}
306 
307 			@Override
308 			public String getColumnValue(AddressableRowObject rowObject) {
309 				ProblemLocation entry = (ProblemLocation) rowObject;
310 				return entry.getExplanation();
311 			}
312 		};
313 
314 		StringColumnDisplay funcColumn = new StringColumnDisplay() {
315 			@Override
316 			public String getColumnName() {
317 				return "Func Name";
318 			}
319 
320 			@Override
321 			public String getColumnValue(AddressableRowObject rowObject) {
322 				ProblemLocation entry = (ProblemLocation) rowObject;
323 				Function func = entry.getProgram().getFunctionManager().getFunctionContaining(
324 					entry.getAddress());
325 				if (func == null) {
326 					return "";
327 				}
328 				return func.getName();
329 			}
330 		};
331 
332 		ColumnDisplay<Address> probLocColumn = new ColumnDisplay<Address>() {
333 			@Override
334 			public String getColumnName() {
335 				return "Location of Possible Cause";
336 			}
337 
338 			@Override
339 			public Address getColumnValue(AddressableRowObject rowObject) {
340 				ProblemLocation probLocation = (ProblemLocation) rowObject;
341 				return probLocation.getWhyAddr();
342 			}
343 
344 			@Override
345 			public int compare(AddressableRowObject o1, AddressableRowObject o2) {
346 				return getColumnValue(o1).compareTo(getColumnValue(o2));
347 			}
348 
349 			@Override
350 			public Class<Address> getColumnClass() {
351 				return Address.class;
352 			}
353 		};
354 
355 		StringColumnDisplay varNameColumn = new StringColumnDisplay() {
356 			@Override
357 			public String getColumnName() {
358 				return "Problematic Variable";
359 			}
360 
361 			@Override
362 			public String getColumnValue(AddressableRowObject rowObject) {
363 				ProblemLocation probLocation = (ProblemLocation) rowObject;
364 				return probLocation.getVarName();
365 			}
366 
367 			@Override
368 			public int compare(AddressableRowObject o1, AddressableRowObject o2) {
369 				return getColumnValue(o1).compareTo(getColumnValue(o2));
370 			}
371 		};
372 
373 		dialog.addCustomColumn(funcColumn);
374 		dialog.addCustomColumn(probLocColumn);
375 		dialog.addCustomColumn(varNameColumn);
376 		dialog.addCustomColumn(explanationColumn);
377 	}

379 	class ProblemLocation implements AddressableRowObject {
380 		private Program program;
381 		private Address problemAddress;
382 		private Address causeAddress;
383 		private String varName;
384 		private String explanation;
385 
386 		ProblemLocation(Program prog, Address problemAddress, Address whyAddr, String varName,
387 				String explanation) {
388 			this.problemAddress = problemAddress;
389 			this.causeAddress = whyAddr;
390 			this.varName = varName;
391 			this.explanation = explanation;
392 			this.program = prog;
393 		}
394 
395 		public Program getProgram() {
396 			return program;
397 		}
398 
399 		@Override
400 		public Address getAddress() {
401 			return getFuncAddr();
402 		}
403 
404 		public Address getFuncAddr() {
405 			if (problemAddress == null) {
406 				return Address.NO_ADDRESS;
407 			}
408 			return problemAddress;
409 		}
410 
411 		public Address getWhyAddr() {
412 			if (causeAddress == null) {
413 				return Address.NO_ADDRESS;
414 			}
415 			return causeAddress;
416 		}
417 
418 		public String getVarName() {
419 			return varName;
420 		}
421 
422 		public String getExplanation() {
423 			return explanation;
424 		}
425 
426 		@Override
427 		public String toString() {
428 			return "Issue at:" + getAddress() + "  found: " + getVarName() + "  " +
429 				getExplanation() + " at " + getWhyAddr();
430 		}
431 	}
