16  // Script to apply any changes the user has made to recovered class virtual function definitions 
17  // edited in the data type manager. To run the script, put the cursor on any member of the 
18  // desired class in the listing then run the script. For each function definition in the given class 
19  // that differs from the associated function signature in the listing, the script will update the 
20  // listing function signatures of any related virtual functions belonging to parent and children 
21  // classes. 
22  // Note: The script will not work if the vftable structures were not originally applied to 
23  // the vftables using the RecoverClassesFromRTTIScript. 
24  // At some point, the Ghidra API will be updated to do this automatically instead of needing the 
25  // script to do so. For now, to make it a bit easier, you can use the below listed key binding
26  // or menupath if you have the "In Tool" checkbox checked for this script in the script manager.
27  //@category C++
28  //@menupath Scripts.ApplyClassFunctionDefinitions
29  //@keybinding shift D
30  
31  import java.util.ArrayList;
32  import java.util.List;
33  
34  import classrecovery.RecoveredClassHelper;
35  import ghidra.app.script.GhidraScript;
36  import ghidra.app.services.DataTypeManagerService;
37  import ghidra.program.model.data.*;
38  import ghidra.program.model.listing.Function;
39  
40  public class ApplyClassFunctionDefinitionUpdatesScript extends GhidraScript {
41  	@Override
42  	public void run() throws Exception {
43  
44  		if (currentProgram == null) {
45  			println("There is no open program");
46  			return;
47  		}
48  
49  		RecoveredClassHelper classHelper = new RecoveredClassHelper(currentProgram, currentLocation,
50  			state.getTool(), this, false, false, false, monitor);
51  
52  		DataTypeManagerService dtms = state.getTool().getService(DataTypeManagerService.class);
53  		List<DataType> selectedDatatypes = dtms.getSelectedDatatypes();
54  		if (selectedDatatypes.size() == 0) {
55  			println("Please select the class function definition(s) you wish to apply.");
56  			return;
57  		}
58  
59  		List<FunctionDefinition> classFunctionDefinitions = new ArrayList<FunctionDefinition>();
60  		for (DataType selectedDataType : selectedDatatypes) {
61  			monitor.checkCancelled();
62  
63  			if (!(selectedDataType instanceof FunctionDefinition)) {
64  				continue;
65  			}
66  
67  			FunctionDefinition functionDefinition = (FunctionDefinition) selectedDataType;
68  			String pathName = functionDefinition.getPathName();
69  			if (!pathName.contains("ClassDataTypes")) {
70  				continue;
71  			}
72  			classFunctionDefinitions.add(functionDefinition);
73  		}
74  
75  		if (classFunctionDefinitions.isEmpty()) {
76  			println(
77  				"Selected function definition(s) must be in a subfolder of the ClassDataTypes folder in the DataTypeManager.");
78  			return;
79  		}
80  
81  		List<Object> changedItems = new ArrayList<Object>();
82  
83  		for (FunctionDefinition functionDef : classFunctionDefinitions) {
84  			monitor.checkCancelled();
85  
86  			List<Object> newChangedItems = classHelper.applyNewFunctionDefinition(functionDef);
87  
88  			changedItems = classHelper.updateList(changedItems, newChangedItems);
89  
90  		}
91  
92  		if (changedItems == null || changedItems.isEmpty()) {
93  			println(
94  				"There were no differences between the selected function definitions and the items that could be updated.");
95  			return;
96  		}
97  
98  		List<Structure> structuresOnList = classHelper.getStructuresOnList(changedItems);
99  		List<Function> functionsOnList = classHelper.getFunctionsOnList(changedItems);
100 
101 		if (!structuresOnList.isEmpty()) {
102 			println();
103 			println("Updated structures:");
104 			for (Structure structure : structuresOnList) {
105 				monitor.checkCancelled();
106 				println(structure.getPathName());
107 			}
108 		}
109 
110 		if (!functionsOnList.isEmpty()) {
111 			println();
112 			println("Updated functions:");
113 			for (Function function : functionsOnList) {
114 				monitor.checkCancelled();
115 				println(function.getEntryPoint().toString());
116 			}
117 		}
118 	}
119 
120 }
