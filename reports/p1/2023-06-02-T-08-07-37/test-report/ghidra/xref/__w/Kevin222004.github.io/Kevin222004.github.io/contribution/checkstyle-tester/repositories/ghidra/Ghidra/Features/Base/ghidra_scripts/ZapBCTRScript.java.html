16  // The script assumes that the cursor is on a 'bctrl' instruction.
17  //
18  // First, a vtbl location is requested. Either the vtbl address or
19  // the name of the owning class can be specified as the vtbl location.
20  // If the owning class name is given, the script searches for a symbol
21  // named "<class>::__vtbl" and uses its address as the vtbl address.
22  //
23  // Second, the offset of the target function within the vtbl is
24  // requested.  That offset is used to extract the target function
25  // pointer from the vtbl.
26  //
27  // Given the target function pointer, the script finds the associated
28  // function name, inserts that function name as an EOL_COMMENT at the
29  // current address, and creates a mnemonic reference from the current
30  // instruction to the function.
31  //
32  // NOTE:  Adresses must be hex values -- without "0x" prefixes.
33  //
34  //@category CustomerSubmission.Analysis
35  //@keybinding alt Z
36  
37  import java.util.List;
38  
39  import ghidra.app.script.GhidraScript;
40  import ghidra.program.model.address.Address;
41  import ghidra.program.model.listing.CodeUnit;
42  import ghidra.program.model.listing.Listing;
43  import ghidra.program.model.mem.Memory;
44  import ghidra.program.model.symbol.*;
45  
46  public class ZapBCTRScript extends GhidraScript {
47  
48  	@Override
49  	public void run() throws Exception {
50  		Address vtblAddr;
51  
52  		// get listing and mem -- used later
53  		Listing listing = currentProgram.getListing();
54  		Memory mem = currentProgram.getMemory();
55  
56  		// get vtbl location (can be class name or address)
57  		String classNameOrAddr = askString("Vtbl location", "Class name or address");
58  		if (classNameOrAddr == null) { // exit, if null input
59  			return;
60  		}
61  
62  		// first try input as class and search for symbol "<class>::__vtbl"
63  		List<Symbol> symbols = currentProgram.getSymbolTable().getSymbols("__vtbl",
64  			getNamespace(null, classNameOrAddr));
65  		// if symbol found, then vtblAddr is the symbol's address
66  		if (symbols.size() == 1) {
67  			vtblAddr = symbols.get(0).getAddress();
68  			// else see if input was address
69  		}
70  		else {
71  			vtblAddr = toAddr(0);
72  			try {
73  				vtblAddr = toAddr(Integer.parseInt(classNameOrAddr, 16));
74  			}
75  			catch (Exception e) {
76  			}
77  			if (vtblAddr.getOffset() == 0) {
78  				println("Invalid class name or address");
79  				println("Note:  addresses must be hex -- with no '0x' prefix");
80  				return;
81  			}
82  		}
83  
84  		// get vtbl offset of tgt function & compute tgt function address
85  		// function address = integer value stored at (vtblAddr +  vtblOffset)
86  		Address funcAddr = toAddr(mem.getInt(
87  			vtblAddr.add(Integer.parseInt(askString("Vtbl Offset", "Hex vtbl offset"), 16))));
88  
89  		// get tgt function name
90  		String funcName = getSymbolAt(funcAddr).getName(true);
91  
92  		// Provide feedback
93  		println("vtblAddr: 0x" + Long.toHexString(vtblAddr.getOffset()));
94  		println("function: " + funcName + " (0x" + Long.toHexString(funcAddr.getOffset()) + ")");
95  
96  		// calculate displacement between instAddr and funcAddr
97  		Address instAddr = currentAddress;
98  		//long displacement = funcAddr.subtract(instAddr);
99  
100 		// insert funcName as EOL comment and
101 		// add a mnemonic ref from instAddr to funcAddr
102 		listing.setComment(instAddr, CodeUnit.EOL_COMMENT, funcName);
103 		listing.getInstructionAt(instAddr).addMnemonicReference(funcAddr, RefType.COMPUTED_CALL,
104 			SourceType.USER_DEFINED);
105 
106 		/*  old code that replaces the 'bctr' with a 'bl'
107 		int code = 0x48000001 | ((int)displacement & 0x3ffffff);
108 		mem.setInt(instAddr, code);
109 		clearListing(instAddr);    // clear/disassemble updates listing
110 		disassemble(instAddr);     // to display bl vice bctr
111 		println("New code word: 0x" + Long.toHexString(code) +
112 		        " -- displacement: 0x" + Long.toHexString(displacement));
113 		*/
114 	}
115 }
