17  //With a user-inputed base address, this script will add a reference on the current switch table entry to corresponding code
18  //Make sure your table entry is defined data (db,dw,etc).
19  //@category ARM
20  
21  import ghidra.app.script.GhidraScript;
22  import ghidra.program.model.address.Address;
23  import ghidra.program.model.listing.Data;
24  import ghidra.program.model.listing.Listing;
25  import ghidra.program.model.listing.Program;
26  import ghidra.program.model.symbol.RefType;
27  import ghidra.util.NumericUtilities;
28  
29  public class AddSingleReferenceInSwitchTable extends GhidraScript {
30  
31      @Override
32      public void run() throws Exception {
33      	
34      	Program program = currentProgram;
35      	Listing listing = program.getListing(); 
36      	
37      	// Ask for base address 
38      	//  (equals the pc when program hits the switch table, 
39      	//   which equals the address of the "add pc, .." instruction + 4)
40      	Address pc = askAddress("Address", "Enter switch base address (hex, don't use 0x)");
41  
42      	// Get current data value
43      	Data data = listing.getDefinedDataAt(currentAddress);
44      	long currVal = NumericUtilities.parseHexLong(data.getValue().toString().substring(2));
45      	
46  		// Calculate referenced addr
47  		Address refAddr = pc.add(2 * currVal);
48  			
49  		// Add reference
50  		println("Adding ref " + refAddr.toString() + " to address " + data.getAddressString(false, true));
51  		data.addValueReference(refAddr, RefType.DATA);
52  
53      }
54  
55  }
