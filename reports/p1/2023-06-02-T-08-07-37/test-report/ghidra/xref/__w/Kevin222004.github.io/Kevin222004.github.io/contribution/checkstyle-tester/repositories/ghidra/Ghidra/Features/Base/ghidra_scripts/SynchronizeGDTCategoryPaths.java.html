16  //
17  // Synchronize the category path name case from the first archive into the second
18  // archive.
19  //
20  //@category Data Types
21  
22  import java.io.File;
23  
24  import ghidra.app.script.GhidraScript;
25  import ghidra.program.model.data.Category;
26  import ghidra.program.model.data.FileDataTypeManager;
27  import ghidra.program.model.data.StandAloneDataTypeManager.ArchiveWarning;
28  import ghidra.util.InvalidNameException;
29  import ghidra.util.exception.DuplicateNameException;
30  
31  public class SynchronizeGDTCategoryPaths extends GhidraScript {
32  
33  	@Override
34  	protected void run() throws Exception {
35  
36  		File firstFile = askFile("Select First GDT File", "Select 1st");
37  		try (FileDataTypeManager firstArchive =
38  			FileDataTypeManager.openFileArchive(firstFile, false)) {
39  			if (hasWarning(firstArchive, firstFile)) {
40  				return;
41  			}
42  
43  			File secondFile = askFile("Select Second GDT File", "Select 2nd");
44  			try (FileDataTypeManager secondArchive =
45  				FileDataTypeManager.openFileArchive(secondFile, true)) {
46  				if (hasWarning(secondArchive, secondFile)) {
47  					return;
48  				}
49  
50  				int transactionID =
51  					secondArchive.startTransaction("Synchronize Category Path Names");
52  				try {
53  					Category firstCategory = firstArchive.getRootCategory();
54  					Category secondCategory = secondArchive.getRootCategory();
55  					synchronizeCategory(firstCategory, secondCategory);
56  				}
57  				finally {
58  					secondArchive.endTransaction(transactionID, true);
59  					secondArchive.save();
60  					secondArchive.close();
61  				}
62  			}
63  		}
64  	}
65  
66  	private boolean hasWarning(FileDataTypeManager archive, File file) {
67  		ArchiveWarning warning = archive.getWarning();
68  		if (warning == ArchiveWarning.NONE) {
69  			return false;
70  		}
71  		if (warning == ArchiveWarning.UPGRADED_LANGUAGE_VERSION) {
72  			return !askYesNo("Archive Upgrade Confirmation",
73  				"A language upgrade has been performed on archive " + file.getName() +
74  					"\nIs it OK to proceed?");
75  		}
76  		popup(
77  			"An architecture language error occured while opening archive (see log for details)\n" +
78  				file.getPath());
79  		return true;
80  	}
81  
82  	private void synchronizeCategory(Category firstCategory, Category secondCategory) {
83  		Category[] firstCategories = firstCategory.getCategories();
84  		for (Category categoryA : firstCategories) {
85  
86  			// loop through categories looking for a case agnostic path match
87  			Category[] secondCategories = secondCategory.getCategories();
88  			boolean foundIt = false;
89  			for (Category categoryB : secondCategories) {
90  				if (categoryA.getName().equalsIgnoreCase(categoryB.getName())) {
91  					// if not the exact same name, rename it
92  					if (!categoryA.getName().equals(categoryB.getName())) {
93  						try {
94  							println(
95  								"Renamed " + categoryB.getName() + " to " + categoryA.getName());
96  							categoryB.setName(categoryA.getName());
97  							foundIt = true;
98  						}
99  						catch (DuplicateNameException | InvalidNameException e) {
100 							// TODO Auto-generated catch block
101 							e.printStackTrace();
102 						}
103 					}
104 					else {
105 						foundIt = true;
106 					}
107 					synchronizeCategory(categoryA, categoryB);
108 				}
109 			}
110 			if (!foundIt) {
111 				println("Couldn't find matching category for " + categoryA.getName());
112 			}
113 		}
114 	}
115 }
