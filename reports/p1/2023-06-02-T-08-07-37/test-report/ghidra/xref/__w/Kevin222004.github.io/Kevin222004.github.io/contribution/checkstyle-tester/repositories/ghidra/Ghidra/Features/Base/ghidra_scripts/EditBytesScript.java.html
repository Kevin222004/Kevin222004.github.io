16  //Convenience script to quickly clear, edit, and recreate the code or data at the current cursor location.
17  //@category Memory
18  //@keybinding
19  //@menupath 
20  //@toolbar 
21  //
22  
23  import java.util.Map.Entry;
24  import java.util.TreeMap;
25  
26  import ghidra.app.cmd.disassemble.DisassembleCommand;
27  import ghidra.app.script.GhidraScript;
28  import ghidra.program.model.address.Address;
29  import ghidra.program.model.address.AddressSet;
30  import ghidra.program.model.data.DataType;
31  import ghidra.program.model.listing.*;
32  import ghidra.program.model.mem.MemoryAccessException;
33  import ghidra.program.model.util.CodeUnitInsertionException;
34  
35  public class EditBytesScript extends GhidraScript {
36  
37  	@Override
38  	public void run() throws Exception {
39  
40  		byte[] newBytes = null;
41  		Address endAddr = null;
42  		Address activeAddr = null;
43  		Address codeEnd = null;
44  		boolean containedInBlock = false;
45  
46  		while (containedInBlock == false) {
47  			monitor.checkCancelled();
48  			newBytes = askBytes("Replace Bytes", "Replace bytes at cursor with:");
49  			endAddr = currentLocation.getByteAddress().add(newBytes.length - 1);
50  			activeAddr = currentLocation.getByteAddress();
51  
52  			containedInBlock = currentProgram.getMemory().getBlock(activeAddr).contains(endAddr);
53  			if (containedInBlock == true) {
54  				break;
55  			}
56  
57  			popup("Bytes entered cannot be contained in current memory block");
58  		}
59  
60  		Address startAddr = currentLocation.getByteAddress();
61  		activeAddr = currentProgram.getListing().getCodeUnitContaining(activeAddr).getAddress();
62  		AddressSet addrSet = new AddressSet(activeAddr, endAddr);
63  		CodeUnitIterator iter = currentProgram.getListing().getCodeUnits(addrSet, true);
64  
65  		AddressSet codeAddrSet = null;
66  
67  		TreeMap<Address, DataType> addrToDataTypeMap = new TreeMap<>();
68  		TreeMap<Address, AddressSet> addrToCodeMap = new TreeMap<>();
69  
70  		while (iter.hasNext()) {
71  
72  			activeAddr = iter.next().getAddress();
73  
74  			Data data = getDataAt(activeAddr);
75  			if (data != null) {
76  				DataType dataType = data.getDataType();
77  				addrToDataTypeMap.put(activeAddr, dataType);
78  				continue;
79  			}
80  
81  			Instruction code = getInstructionContaining(activeAddr);
82  			if (code != null) {
83  				codeEnd = activeAddr.add(code.getLength() - 1);
84  				codeAddrSet = new AddressSet(activeAddr, codeEnd);
85  				addrToCodeMap.put(activeAddr, codeAddrSet);
86  				continue;
87  			}
88  
89  			if (activeAddr.equals(endAddr)) {
90  				break;
91  			}
92  		}
93  
94  		clearListing(startAddr, endAddr);
95  
96  		try {
97  			setBytes(startAddr, newBytes);
98  		}
99  		catch (MemoryAccessException e) {
100 			popup("Bytes cannot be set on uninitialized memory");
101 			return;
102 		}
103 
104 		for (Entry<Address, DataType> entry : addrToDataTypeMap.entrySet()) {
105 			try {
106 				createData(entry.getKey(), entry.getValue());
107 			}
108 			catch (CodeUnitInsertionException e) {
109 				//leaves bytes undefined if there is no 00 byte at the end to
110 				//make a null terminated string 
111 				return;
112 			}
113 		}
114 
115 		for (Entry<Address, AddressSet> entry : addrToCodeMap.entrySet()) {
116 			DisassembleCommand cmd = new DisassembleCommand(entry.getKey(), entry.getValue(), true);
117 			cmd.applyTo(currentProgram, monitor);
118 		}
119 	}
120 }
