16  //Example preScript to configure the PDB symbol server service to use the ~/symbols directory
17  //as the location to store symbol files, and to search Microsoft's public
18  //symbol server.
19  //The ~/symbols directory should already exist and be initialized as a symbol
20  //storage location.
21  //@category PDB
22  import java.util.List;
23  
24  import java.io.File;
25  import java.net.URI;
26  
27  import ghidra.app.plugin.core.analysis.PdbAnalyzer;
28  import ghidra.app.plugin.core.analysis.PdbUniversalAnalyzer;
29  import ghidra.app.script.GhidraScript;
30  import pdb.PdbPlugin;
31  import pdb.symbolserver.*;
32  
33  public class PdbSymbolServerExamplePrescript extends GhidraScript {
34  
35  	@Override
36  	protected void run() throws Exception {
37  		File homeDir = new File(System.getProperty("user.home"));
38  		File symDir = new File(homeDir, "symbols");
39  		LocalSymbolStore localSymbolStore = new LocalSymbolStore(symDir);
40  		HttpSymbolServer msSymbolServer =
41  			new HttpSymbolServer(URI.create("https://msdl.microsoft.com/download/symbols/"));
42  		SymbolServerService symbolServerService =
43  			new SymbolServerService(localSymbolStore, List.of(msSymbolServer));
44  
45  		PdbPlugin.saveSymbolServerServiceConfig(symbolServerService);
46  
47  		// You only need to enable the "allow remote" option on the specific
48  		// analyzer you are using
49  		PdbUniversalAnalyzer.setAllowRemoteOption(currentProgram, true);
50  		PdbAnalyzer.setAllowRemoteOption(currentProgram, true);
51  	}
52  }
