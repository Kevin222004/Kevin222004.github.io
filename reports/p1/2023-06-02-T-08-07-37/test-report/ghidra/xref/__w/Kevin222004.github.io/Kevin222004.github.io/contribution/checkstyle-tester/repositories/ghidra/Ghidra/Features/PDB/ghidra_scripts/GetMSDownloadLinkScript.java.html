16  //Creates a http download URL for a binary that Microsoft has previously published on their symbol server
17  // (ie. windows OS binaries).  Useful for sharing an exact Windows OS binary from your local 
18  // workstation to someone else without requiring the recipient to trust your binary as they
19  // can download it directly from Microsoft's symbol server.
20  // This works by extracting 2 pieces of metadata from the chosen binary's PE header to create
21  // the URL to Microsoft's symbol server and verifying that it exists on Microsoft's symbol server
22  // by connecting to the MS symbol server and probing for the file before declaring success and
23  // showing it to the user.
24  //@category Import
25  
26  import java.io.File;
27  import java.io.IOException;
28  import java.net.URI;
29  import java.nio.file.AccessMode;
30  import java.util.List;
31  
32  import ghidra.app.script.GhidraScript;
33  import ghidra.app.util.bin.FileByteProvider;
34  import ghidra.app.util.bin.format.pe.NTHeader;
35  import ghidra.app.util.bin.format.pe.PortableExecutable;
36  import ghidra.app.util.bin.format.pe.PortableExecutable.SectionLayout;
37  import pdb.symbolserver.*;
38  
39  public class GetMSDownloadLinkScript extends GhidraScript {
40  
41  	private static final String MS_PUBLIC_SYMBOL_SERVER_URL =
42  		"https://msdl.microsoft.com/download/symbols/";
43  
44  	@Override
45  	protected void run() throws Exception {
46  		SymbolServerService symbolService =
47  			new SymbolServerService(new SameDirSymbolStore(null), List.of(
48  				new HttpSymbolServer(URI.create(MS_PUBLIC_SYMBOL_SERVER_URL))));
49  
50  		File f = askFile("File To Scan", "Select");
51  		if (f == null) {
52  			return;
53  		}
54  
55  		try (FileByteProvider bp = new FileByteProvider(f, null, AccessMode.READ)) {
56  			monitor.setMessage("Parsing file " + f.getName());
57  			PortableExecutable pe = new PortableExecutable(bp, SectionLayout.FILE, false, false);
58  			NTHeader ntHeader = pe.getNTHeader();
59  			if (ntHeader != null && ntHeader.getOptionalHeader() != null) {
60  				int timeDateStamp = ntHeader.getFileHeader().getTimeDateStamp();
61  				int sizeOfImage = (int) ntHeader.getOptionalHeader().getSizeOfImage();
62  				println(f + ", timeDateStamp: " + Integer.toHexString(timeDateStamp) +
63  					", sizeOfImage: " + Integer.toHexString(sizeOfImage));
64  				SymbolFileInfo symbolFileInfo = SymbolFileInfo.fromValues(f.getName().toLowerCase(),
65  					Integer.toHexString(timeDateStamp), sizeOfImage);
66  				List<SymbolFileLocation> findResults =
67  					symbolService.find(symbolFileInfo, FindOption.of(FindOption.ALLOW_REMOTE),
68  						monitor);
69  				if (findResults.isEmpty()) {
70  					println("Not found on " + MS_PUBLIC_SYMBOL_SERVER_URL);
71  					return;
72  				}
73  				SymbolFileLocation symLoc = findResults.get(0);
74  				println("Download link: " + symLoc.getLocationStr());
75  			}
76  			else {
77  				printerr("Unrecognized format: " + f);
78  			}
79  		}
80  		catch (IOException e) {
81  			printerr("Failed to parse file: " + f);
82  			printerr(e.getMessage());
83  		}
84  	}
85  }
