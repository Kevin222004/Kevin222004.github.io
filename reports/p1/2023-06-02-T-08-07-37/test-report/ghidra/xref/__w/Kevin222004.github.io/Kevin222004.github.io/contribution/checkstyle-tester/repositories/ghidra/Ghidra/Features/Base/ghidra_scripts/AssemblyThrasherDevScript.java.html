16  //Thoroughly test the assembler by attempting to assemble and match EVERY instruction code unit.
17  //NOTE: I do not de-duplicate, since the address of the instruction may affect the output.
18  //@category Assembly
19  
20  import java.io.ByteArrayOutputStream;
21  import java.io.PrintStream;
22  import java.util.HashSet;
23  import java.util.Set;
24  
25  import javax.swing.Icon;
26  
27  import org.apache.commons.lang3.StringUtils;
28  
29  import generic.theme.GThemeDefaults.Colors.Palette;
30  import ghidra.app.plugin.assembler.*;
31  import ghidra.app.plugin.assembler.sleigh.sem.*;
32  import ghidra.app.plugin.processors.sleigh.SleighDebugLogger;
33  import ghidra.app.plugin.processors.sleigh.SleighDebugLogger.SleighDebugMode;
34  import ghidra.app.script.GhidraScript;
35  import ghidra.app.util.PseudoInstruction;
36  import ghidra.program.model.address.*;
37  import ghidra.program.model.lang.*;
38  import ghidra.program.model.listing.BookmarkManager;
39  import ghidra.program.model.listing.Instruction;
40  import ghidra.program.model.mem.ByteMemBufferImpl;
41  import ghidra.program.model.mem.MemBuffer;
42  import ghidra.util.NumericUtilities;
43  import resources.Icons;
44  
45  public class AssemblyThrasherDevScript extends GhidraScript {
46  	public static final String BOOKMARK_FAIL = "AssemblyFailure";
47  	public static final String BOOKMARK_PASS = "AssemblySuccess";
48  
49  	class Accept extends RuntimeException {
50  		// custom exception
51  	}
52  
53  	class AllMatchByTextSelector extends AssemblySelector {
54  		private Instruction orig;
55  		@SuppressWarnings("unused")
56  		private Address addr;
57  		private String text;
58  
59  		public void setExpected(Instruction ins) {
60  			this.orig = ins;
61  			this.addr = ins.getAddress();
62  			this.text = ins.toString().trim();
63  		}
64  
65  		@Override
66  		public AssemblyResolvedPatterns select(AssemblyResolutionResults rr,
67  				AssemblyPatternBlock ctx) throws AssemblySemanticException {
68  			StringBuilder sb = new StringBuilder();
69  			boolean gotOne = false;
70  			boolean failedOne = false;
71  			for (AssemblyResolution ar : rr) {
72  				if (ar.isError()) {
73  					continue;
74  				}
75  				AssemblyResolvedPatterns can = (AssemblyResolvedPatterns) ar;
76  				if (can.getContext().combine(ctx) == null) {
77  					continue;
78  				}
79  				for (byte[] ins : can.possibleInsVals(ctx)) {
80  					String cantext = disassemble(orig, ins).toString().trim();
81  					if (cantext.equals(text)) {
82  						gotOne = true;
83  					}
84  					else {
85  						sb.append("Mismatch: " + NumericUtilities.convertBytesToString(ins) +
86  							" => " + cantext + ". ");
87  						failedOne = true;
88  					}
89  				}
90  			}
91  			if (!gotOne) {
92  				throw new AssemblySemanticException("No match found");
93  			}
94  			if (failedOne) {
95  				throw new AssemblySemanticException(sb.toString());
96  			}
97  			// Don't return, to avoid modifying the binary.
98  			throw new Accept();
99  		}
100 	}
101 
102 	@Override
103 	protected void run() throws Exception {
104 		clearBackgroundColor(currentProgram.getMemory().getAllInitializedAddressSet());
105 
106 		BookmarkManager bm = currentProgram.getBookmarkManager();
107 		Icon myIcon = Icons.WARNING_ICON;
108 		bm.defineType(BOOKMARK_FAIL, myIcon, Palette.YELLOW, 0);
109 		bm.removeBookmarks(BOOKMARK_FAIL);
110 
111 		monitor.setMessage("Constructing Assembler");
112 		AllMatchByTextSelector checker = new AllMatchByTextSelector();
113 		Assembler asm = Assemblers.getAssembler(currentProgram, checker);
114 		Set<String> done = new HashSet<>();
115 		AddressSet uniques = createAddressSet();
116 		for (Instruction ins : currentProgram.getListing().getInstructions(currentAddress, true)) {
117 			if (monitor.isCancelled()) {
118 				return;
119 			}
120 
121 			SleighDebugLogger debug =
122 				new SleighDebugLogger(currentProgram, ins.getAddress(), SleighDebugMode.MASKS_ONLY);
123 			String linenos = StringUtils.join(debug.getConstructorLineNumbers(), " ");
124 			monitor.setMessage("Assembling " + ins.getAddress());
125 			if (!done.add(linenos)) {
126 				continue;
127 			}
128 			uniques.add(ins.getAddress());
129 			checker.setExpected(ins);
130 			try {
131 				asm.assemble(ins.getAddress(), ins.toString());
132 			}
133 			catch (Accept e) {
134 				bm.setBookmark(ins.getAddress(), BOOKMARK_PASS, "Assembly Succeeded",
135 					e.getMessage());
136 			}
137 			catch (AssemblySemanticException e) {
138 				bm.setBookmark(ins.getAddress(), BOOKMARK_FAIL, "Semantic Error", e.getMessage());
139 			}
140 			catch (AssemblySyntaxException e) {
141 				bm.setBookmark(ins.getAddress(), BOOKMARK_FAIL, "Syntax Error", e.getMessage());
142 			}
143 			catch (UnsupportedOperationException e) {
144 				ByteArrayOutputStream buf = new ByteArrayOutputStream();
145 				PrintStream p = new PrintStream(buf);
146 				e.printStackTrace(p);
147 				bm.setBookmark(ins.getAddress(), BOOKMARK_FAIL, "Unfinished",
148 					new String(buf.toByteArray()));
149 			}
150 			catch (Exception e) {
151 				ByteArrayOutputStream buf = new ByteArrayOutputStream();
152 				PrintStream p = new PrintStream(buf);
153 				e.printStackTrace(p);
154 				bm.setBookmark(ins.getAddress(), BOOKMARK_FAIL, "Severe Error",
155 					new String(buf.toByteArray()));
156 			}
157 		}
158 
159 		println("Unique instructions by constructor: " + done.size());
160 		setBackgroundColor(uniques, Palette.getColor("lightcoral"));
161 	}
162 
163 	protected PseudoInstruction disassemble(Instruction orig, byte[] ins) {
164 		try {
165 			Address at = orig.getAddress();
166 			Language language = currentProgram.getLanguage();
167 			MemBuffer buf = new ByteMemBufferImpl(at, ins, language.isBigEndian());
168 			InstructionPrototype ip = language.parse(buf, orig, false);
169 			return new PseudoInstruction(at, ip, buf, orig);
170 		}
171 		catch (InsufficientBytesException | UnknownInstructionException
172 				| AddressOverflowException e) {
173 			throw new RuntimeException(e);
174 		}
175 	}
176 }
