16  // Given a selection that represents a function re-tag all calls that should be Jumps as a Jump.
17  //   Some compilers for the ARM use a typical (bl <x>)call instruction as a form of long jump.
18  //   The bl instruction is a branch and link, usually associated with a call.
19  //   The ARM language module has the ability to override the call instruction with a CALLOVERRIDE context.
20  //
21  //   There is one degenerate case, where somewhere in a function it jumps to the first instruction in a subroutine
22  //   while other locations really do call the location.
23  //
24  //   This script assumes good flow, that switch stmts are good.
25  //
26  //@category ARM
27  
28  import ghidra.app.cmd.disassemble.SetFlowOverrideCmd;
29  import ghidra.app.cmd.function.CreateFunctionCmd;
30  import ghidra.app.script.GhidraScript;
31  import ghidra.program.model.address.*;
32  import ghidra.program.model.block.*;
33  import ghidra.program.model.listing.*;
34  import ghidra.program.model.symbol.*;
35  
36  import java.util.Iterator;
37  
38  public class Override_ARM_Call_JumpsScript extends GhidraScript {
39  
40  	@Override
41  	public void run() throws Exception {
42  
43  		if (currentSelection == null || currentSelection.isEmpty()) {
44  			this.popup("All instructions in a single subroutine must be selected");
45  			return;
46  		}
47  
48  		AddressSet funcsToClear = new AddressSet();
49  		AddressSet funcsToFix = new AddressSet();
50  
51  		AddressSet locationsFixed = new AddressSet();
52  
53  		Iterator<Function> fiter =
54  			currentProgram.getFunctionManager().getFunctionsOverlapping(currentSelection);
55  		if (!fiter.hasNext()) {
56  			this.popup("A single function must be defined within the selection");
57  			return;
58  		}
59  
60  		Function func = fiter.next();
61  		if (fiter.hasNext()) {
62  			this.popup("Only ONE function allowed to be defined in the selection");
63  			return;
64  		}
65  
66  		// branchSet will contain those addresses that need to be changed to a jump in this block
67  		AddressSet branchSet = new AddressSet();
68  
69  		boolean isBad = false;
70  		SimpleBlockModel basicBlockModel = new SimpleBlockModel(currentProgram);
71  		CodeBlockIterator bbIter =
72  			basicBlockModel.getCodeBlocksContaining(currentSelection, monitor);
73  
74  		while (bbIter.hasNext()) {
75  			CodeBlock bbBlock = bbIter.next();
76  
77  			CodeBlockReferenceIterator bbDestRefIter = bbBlock.getDestinations(monitor);
78  			while (bbDestRefIter.hasNext()) {
79  				CodeBlockReference bbRef = bbDestRefIter.next();
80  
81  				if (bbRef.getFlowType().isCall() &&
82  					currentSelection.contains(bbRef.getDestinationAddress())) {
83  					Instruction instr =
84  						currentProgram.getListing().getInstructionContaining(
85  							bbBlock.getMaxAddress());
86  					// must override at call location
87  					branchSet.addRange(instr.getMinAddress(), instr.getMinAddress());
88  
89  					// don't clear functions that are jumped to if they are really the target of a call
90  					if (!bbRef.getDestinationAddress().equals(currentSelection.getMinAddress())) {
91  						// must clear function at destination
92  						funcsToClear.addRange(bbRef.getDestinationAddress(),
93  							bbRef.getDestinationAddress());
94  					}
95  					isBad = true;
96  				}
97  			}
98  		}
99  
100 		if (!isBad) {
101 			popup("no bad call locations found");
102 			return;
103 		}
104 
105 		// this is one bad one
106 		funcsToFix.addRange(func.getEntryPoint(), func.getEntryPoint());
107 		goTo(func.getEntryPoint());
108 		createSelection(funcsToFix);
109 
110 		locationsFixed.add(branchSet);
111 		createHighlight(locationsFixed);
112 
113 		// get rid of the bad functions
114 		AddressIterator aIter = funcsToClear.getAddresses(true);
115 		while (aIter.hasNext()) {
116 			Address addr = aIter.next();
117 			currentProgram.getFunctionManager().removeFunction(addr);
118 
119 			createBookmark(addr, " Override ARM CALL to Jump fixer", "Removed Bogus function");
120 		}
121 
122 		// for each branch location
123 		//    set CALLoverride
124 		//    clear/disassemble the instruction
125 		//    check if there are flows after that are not reached other than thru the fallthru
126 		//        clear below
127 		AddressIterator addrIter = locationsFixed.getAddresses(true);
128 		while (addrIter.hasNext()) {
129 			Address addr = addrIter.next();
130 
131 			Instruction instruction = currentProgram.getListing().getInstructionAt(addr);
132 			if (instruction == null) {
133 				continue;
134 			}
135 
136 			// data references could have been added because this used to be a call
137 			//   so get rid of data references
138 			Reference[] referencesFrom = instruction.getReferencesFrom();
139 			ReferenceManager refMgr = currentProgram.getReferenceManager();
140 			for (Reference reference : referencesFrom) {
141 				if (reference.getReferenceType().isData()) {
142 					refMgr.delete(reference);
143 				}
144 			}
145 
146 			new SetFlowOverrideCmd(instruction.getMinAddress(), FlowOverride.BRANCH).applyTo(currentProgram);
147 
148 			// make sure reference type got morphed
149 			referencesFrom = instruction.getReferencesFrom();
150 			for (Reference ref : referencesFrom) {
151 				if (ref.getReferenceType().isCall()) {
152 					refMgr.addMemoryReference(ref.getFromAddress(), ref.getToAddress(),
153 						RefType.UNCONDITIONAL_JUMP, ref.getSource(), ref.getOperandIndex());
154 					refMgr.delete(ref);
155 				}
156 			}
157 
158 			createBookmark(addr, "ARM CALL to Jump fixer", "Changed Call to Jump");
159 		}
160 
161 		// fixup functions that were affected
162 		aIter = funcsToFix.getAddresses(true);
163 		while (aIter.hasNext()) {
164 			Address addr = aIter.next();
165 
166 			Instruction start_inst = currentProgram.getListing().getInstructionAt(addr);
167 			CreateFunctionCmd.fixupFunctionBody(currentProgram, start_inst, monitor);
168 
169 			createBookmark(addr, " Override ARM CALL to Jump fixer", "Fixed function");
170 		}
171 	}
172 }
