17  //With cursor on switch's "add pc, .." command, this script will add references on the switch offset table to corresponding code.  
18  //Make sure your table consists of defined data (db,dw,etc).
19  //@category ARM
20  
21  /*
22   * This script has 2 preconditions:
23   *  1.  Your switch table should already be defined data
24   *  2.  Make sure your cursor is on the "add pc, .." command.    	
25   */
26  
27  import ghidra.app.script.GhidraScript;
28  import ghidra.program.model.address.*;
29  import ghidra.program.model.data.*;
30  import ghidra.program.model.listing.*;
31  
32  import ghidra.program.model.symbol.*;
33  import ghidra.util.NumericUtilities;
34  
35  
36  public class AddReferencesInSwitchTable extends GhidraScript {
37  
38      @Override
39      public void run() throws Exception {
40        	println("This script attempts to add references to a switch table to corresponding code.\n"
41      			+ "BEFORE running this script, \n"
42      			+ "     1.  Your switch table should already be defined data.\n"
43      			+ "     2.  Make sure your cursor is on the \"add pc, ..\" command.\n"    			
44      			+ "Note: Adding the same reference twice does not have any impact.\n");
45      	
46      	Program program = currentProgram;
47      	Listing listing = program.getListing(); 
48      	
49      	Address startAddr = currentAddress;
50      	Address pc = startAddr.add(4);
51      	
52      	Address prevAddr = null;
53      	int diff = 0;
54    	    	
55      	// Get data iterator
56      	DataIterator dataIter = listing.getDefinedData(startAddr, true);
57      	
58      	// Find and add reference to first table entry
59      	Data data = dataIter.next();
60      	CalcAndAddReference( data, pc );
61  		prevAddr = data.getMinAddress();
62  		
63  		// Determine address difference between each switch table entry
64  		DataType type = data.getDataType();
65  		String typeName = type.getName();
66  		//println( "type: " + typeName );			
67  		if ( typeName.equalsIgnoreCase("byte") )
68  			diff = 1;
69  		else if ( type.getName().equalsIgnoreCase("word") )
70  			diff = 2;
71  		else if ( type.getName().equalsIgnoreCase("dword") )
72  			diff = 4;
73  		else {
74  			popup( "Sorry, type " + typeName + " is not supported yet.  (Try adding it yourself.)");
75  			return;
76  		}		
77  
78  		// Iterate through rest of table
79  		while (dataIter.hasNext() && !monitor.isCancelled()) {
80  			data = dataIter.next();
81  			Address currAddr = data.getMinAddress();
82  			monitor.setMessage(currAddr.toString());
83  			
84  			// Check if consecutive next entry in switch table
85  			if ( currAddr.subtract(prevAddr) == diff ) {
86  				// Save currAddr as prevAddr
87  				prevAddr = currAddr;
88  
89  				// Add reference
90  				CalcAndAddReference( data, pc );						
91  			}
92  			else {
93  				// Passed end of switch table...Exit script
94  				break;			
95  			}
96  		}
97    
98  	}   // end run()
99  
100 	private void CalcAndAddReference(Data data, Address pc) {
101 		
102 		// Get current data value in switch table
103 		//println("value: " + data.getValue().toString().substring(2));
104 		long currVal = NumericUtilities.parseHexLong(data.getValue().toString().substring(2));
105 		
106 		// Calculate referenced addr
107 		// (using addWrap so that an exception is not thrown when past the switch table)
108 		Address refAddr = pc.addWrap(2 * currVal);
109 		
110 		// Add reference
111 		println("Adding ref " + refAddr.toString() + " to address " + data.getAddressString(false, true));
112 		data.addValueReference(refAddr, RefType.COMPUTED_JUMP);	
113 		
114 	}
115     
116 }
