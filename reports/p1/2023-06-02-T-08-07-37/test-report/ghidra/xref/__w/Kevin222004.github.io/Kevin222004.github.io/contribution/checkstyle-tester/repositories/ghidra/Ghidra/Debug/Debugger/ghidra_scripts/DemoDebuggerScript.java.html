16  //An example debugger script
17  //It launches the current program, places saved breakpoints, and runs it until termination.
18  //This script must be run from the Debugger tool, or another tool with the required plugins.
19  //This script has only been tested with /usr/bin/echo.
20  //@category Debugger
21  //@keybinding
22  //@menupath
23  //@toolbar
24  
25  import java.util.Set;
26  import java.util.concurrent.TimeUnit;
27  
28  import ghidra.app.plugin.core.debug.service.model.launch.DebuggerProgramLaunchOffer.LaunchResult;
29  import ghidra.app.script.GhidraScript;
30  import ghidra.app.services.LogicalBreakpoint;
31  import ghidra.debug.flatapi.FlatDebuggerAPI;
32  import ghidra.program.model.address.Address;
33  import ghidra.trace.model.Trace;
34  
35  public class DemoDebuggerScript extends GhidraScript implements FlatDebuggerAPI {
36  
37  	@Override
38  	protected void run() throws Exception {
39  		/**
40  		 * Here we'll just launch the current program. Note that this is not guaranteed to succeed
41  		 * at all. Launching is subject to an opinion-based service. If no offers are made, this
42  		 * will fail. If the target system is missing required components, this will fail. If the
43  		 * target behaves in an unexpected way, this may fail. One example is targets without an
44  		 * initial break. If Ghidra does not recognize the target platform, this will fail. Etc.,
45  		 * etc., this may fail.
46  		 * 
47  		 * In the event of failure, nothing is cleaned up automatically, since in some cases, the
48  		 * user may be expected to intervene. In our case; however, there's no way to continue this
49  		 * script on a repaired target, so we'll close the connection on failure. An alternative
50  		 * design for this script would expect the user to have already launched a target, and it
51  		 * would just operate on the "current target."
52  		 */
53  		println("Launching " + currentProgram);
54  		LaunchResult result = launch(monitor);
55  		if (result.exception() != null) {
56  			printerr("Failed to launch " + currentProgram + ": " + result.exception());
57  
58  			if (result.model() != null) {
59  				result.model().close();
60  			}
61  
62  			if (result.recorder() != null) {
63  				closeTrace(result.recorder().getTrace());
64  			}
65  			return;
66  		}
67  		Trace trace = result.recorder().getTrace();
68  		println("Successfully launched in trace " + trace);
69  
70  		/**
71  		 * Breakpoints are highly dependent on the module map. To work correctly: 1) The target
72  		 * debugger must provide the module map. 2) Ghidra must have recorded that module map into
73  		 * the trace. 3) Ghidra must recognize the module names and map them to programs open in the
74  		 * tool. These events all occur asynchronously, usually immediately after launch. Most
75  		 * launchers will wait for the target program module to be mapped to its Ghidra program
76  		 * database, but the breakpoint service may still be processing the new mapping.
77  		 */
78  		flushAsyncPipelines(trace);
79  
80  		/**
81  		 * There is also breakpointsEnable(), but that operates on an address-by-address basis,
82  		 * which doesn't quite make sense in this case. We'll instead use getBreakpoints(Program)
83  		 * and enable them only in the new trace. The nested for is to deal with the fact that
84  		 * getBreakpoints(Program) returns a map from address to breakpoint set, i.e., a collection
85  		 * of collections.
86  		 */
87  		println("Enabling breakpoints");
88  		for (Set<LogicalBreakpoint> bs : getBreakpoints(currentProgram).values()) {
89  			for (LogicalBreakpoint lb : bs) {
90  				println("  " + lb);
91  				if (lb.getTraceAddress(trace) == null) {
92  					printerr("    Not mapped!");
93  				}
94  				else {
95  					waitOn(lb.enableForTrace(trace));
96  				}
97  			}
98  		}
99  
100 		/**
101 		 * This runs the target, recording memory around the PC and SP at each break, until it
102 		 * terminates.
103 		 */
104 		while (isTargetAlive()) {
105 			waitForBreak(10, TimeUnit.SECONDS);
106 			/**
107 			 * The recorder is going to schedule some reads upon break, so let's allow them to
108 			 * settle.
109 			 */
110 			flushAsyncPipelines(trace);
111 
112 			println("Reading PC");
113 			Address pc = getProgramCounter();
114 			println("Reading 1024 bytes at PC=" + pc);
115 			readMemory(pc, 1024, monitor);
116 			println("Reading SP");
117 			Address sp = getStackPointer();
118 			println("Reading 8096 bytes at SP=" + sp);
119 			readMemory(sp, 8096, monitor);
120 			/**
121 			 * Allow the commands we just issued to settle.
122 			 */
123 			flushAsyncPipelines(trace);
124 
125 			println("Resuming");
126 			resume();
127 		}
128 		println("Terminated");
129 	}
130 }
