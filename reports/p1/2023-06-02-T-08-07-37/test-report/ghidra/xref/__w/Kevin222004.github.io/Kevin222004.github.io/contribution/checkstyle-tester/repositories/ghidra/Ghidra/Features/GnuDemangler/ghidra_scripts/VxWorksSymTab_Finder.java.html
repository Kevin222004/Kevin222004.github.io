16  // Locates and parses the VxWorks symbol table.  Names the table "vxSymTbl"
17  // and names the table length variable "vxSymTblLen" (if the length variable
18  // appears either directly before or after the symbol table).  Defines the
19  // symbol table as SYMBOL[vxSymTblLen].
20  //
21  // Extracts symbol name, location, and type from each entry.  Disassembles,
22  // creates, and names functions.  Names global variables.
23  //
24  // Any existing Ghidra symbol table entries that collide with VxWorks symbol
25  // table entries are deleted.  Mangled C++ symbol names are demangled.
26  //
27  // The VxWorks symbol table is an array [0..n-1] of (struct SYMBOL) entries.
28  // The table may be immediately followed or preceeded by an (int) vxSymTblLen
29  // value.
30  //
31  // Prerequisites:
32  //
33  //		- Program memory block(s) is(are) aligned with actual load addresses
34  //		  (run something like MemAlignARM_LE.java)
35  //
36  //		- Symbol table cannot be in a memory block with a name that contains
37  //		  the string "text" or "bss"
38  //
39  //		- Modify getVxSymbolClass() to recognize your program's VxWorks
40  //		  symbol table entry structure, if necessary
41  //
42  // @category VxWorks
43

117 		}

149 	// getVxSymbolClass
150 	//
151 	// Creates a SYMBOL structure data type and uses it to create a new
152 	// VxSymbol class instance.
153 	//
154 	// Returns:
155 	//		Success: VxSymbol
156 	//		Failure: null
157 	//------------------------------------------------------------------------
158 	private VxSymbol getVxSymbolClass(int type) {
159 
160 		// Pre-define base data types used to define symbol table entry data type
161 		DataTypeManager builtin = getDataTypeManagerByName("BuiltInTypes");
162 		DataType charType = builtin.getDataType("/char");
163 		DataType charPtrType = PointerDataType.getPointer(charType, 4);
164 		DataType byteType = builtin.getDataType("/byte");
165 		DataType ushortType = builtin.getDataType("/ushort");
166 		DataType intType = builtin.getDataType("/int");
167 		DataType uintType = builtin.getDataType("/uint");
168 		DataType voidType = builtin.getDataType("/void");
169 		DataType voidPtrType = PointerDataType.getPointer(voidType, 4);
170 
171 		// Define a SYMBOL data type (try to put most common first).
172 		// Each SYMBOL data type must include at least 3 fields named
173 		// symNameOff, symLocOff, and symType.
174 		StructureDataType dt = null;
175 		switch (type) {
176 			case 0:
177 
178 				// Version 5.4, 6.4 and 6.8
179 				//
180 				// Total length: 0x14 bytes
181 				//    0x00    uint symHashNode	// NULL
182 				//    0x04    char *symNameOff
183 				//    0x08    void *symLocOff
184 				//    0x0c    int NULL
185 				//    0x10    ushort symGroup
186 				//    0x12    uchar symType
187 				//    0x13    uchar undef
188 				dt = new StructureDataType("SYMBOL", 0x14);
189 				dt.replaceAtOffset(0, uintType, 4, "symHashNode", "");
190 				dt.replaceAtOffset(4, charPtrType, 4, "symNameOff", "");
191 				dt.replaceAtOffset(8, voidPtrType, 4, "symLocOff", "");
192 				dt.replaceAtOffset(0x0c, intType, 4, "", "");
193 				dt.replaceAtOffset(0x10, ushortType, 2, "symGroup", "");
194 				dt.replaceAtOffset(0x12, byteType, 1, "symType", "");
195 				break;
196 
197 			case 1:
198 
199 				// Version 6.1
200 				//
201 				// Total length: 0x18 bytes
202 				//    0x00    uint symHashNode	// NULL
203 				//    0x04    char *symNameOff
204 				//    0x08    void *symLocOff
205 				//    0x0c    int NULL
206 				//    0x10    int NULL
207 				//    0x14    uchar symType
208 				//    0x15    uchar undef[3]
209 				dt = new StructureDataType("SYMBOL", 0x18);
210 				dt.replaceAtOffset(0, uintType, 4, "symHashNode", "");
211 				dt.replaceAtOffset(4, charPtrType, 4, "symNameOff", "");
212 				dt.replaceAtOffset(8, voidPtrType, 4, "symLocOff", "");
213 				dt.replaceAtOffset(0x0c, intType, 4, "", "");
214 				dt.replaceAtOffset(0x10, intType, 4, "", "");
215 				dt.replaceAtOffset(0x14, byteType, 1, "symType", "");
216 				break;
217 
218 			case 2:
219 
220 				// Unknown VxWorks version(s)
221 				//
222 				// Total length: 0x1c bytes
223 				//    0x00    uint symHashNode	// NULL
224 				//    0x04    char *symNameOff
225 				//    0x08    void *symLocOff
226 				//    0x0c    int unk;				// no clear pattern to values
227 				//    0x10    int NULL
228 				//    0x14    int NULL
229 				//    0x18    uchar symType
230 				//    0x19    uchar undef[3]
231 				dt = new StructureDataType("SYMBOL", 0x1c);
232 				dt.replaceAtOffset(0, uintType, 4, "symHashNode", "");
233 				dt.replaceAtOffset(4, charPtrType, 4, "symNameOff", "");
234 				dt.replaceAtOffset(8, voidPtrType, 4, "symLocOff", "");
235 				dt.replaceAtOffset(0x0c, intType, 4, "", "");
236 				dt.replaceAtOffset(0x10, intType, 4, "", "");
237 				dt.replaceAtOffset(0x14, intType, 4, "", "");
238 				dt.replaceAtOffset(0x18, byteType, 1, "symType", "");
239 				break;
240 
241 			case 3:
242 
243 				// Version 5.5, possibly 5.3 as well
244 				//
245 				// Total length: 0x10 bytes
246 				//    0x00    uint symHashNode	// NULL
247 				//    0x04    char *symNameOff
248 				//    0x08    void *symLocOff
249 				//    0x0c    ushort symGroup		// NULL
250 				//    0x0e    uchar symType
251 				//    0x0f    uchar undef
252 				dt = new StructureDataType("SYMBOL", 0x10);
253 				dt.replaceAtOffset(0, uintType, 4, "symHashNode", "");
254 				dt.replaceAtOffset(4, charPtrType, 4, "symNameOff", "");
255 				dt.replaceAtOffset(8, voidPtrType, 4, "symLocOff", "");
256 				dt.replaceAtOffset(0x0c, ushortType, 2, "symGroup", "");
257 				dt.replaceAtOffset(0x0e, byteType, 1, "symType", "");
258 				break;
259 			case 4:
260 				// version 5.0
261 				// sizeof = 0x0e bytes
262 				//    0x00    uint symHashNode	// NULL
263 				//    0x04    char *symNameOff
264 				//    0x08    void *symLocOff
265 				//    0x0c    uchar symType
266 				//    0x0d    uchar undef
267 				dt = new StructureDataType("SYMBOL", 0x0e);
268 				dt.replaceAtOffset(0, uintType, 4, "symHashNode", "");
269 				dt.replaceAtOffset(4, charPtrType, 4, "symNameOff", "");
270 				dt.replaceAtOffset(8, voidPtrType, 4, "symLocOff", "");
271 				dt.replaceAtOffset(0x0c, byteType, 1, "symType", "");
272 				break;
273 			default:
274 
275 				return null;
276 		}
277 
278 		// Return a VxSymbol class for this SYMBOL data type
279 		return new VxSymbol(dt);
280 	}

286 	//------------------------------------------------------------------------
287 	private boolean isExecute(Address addr) {
288 
289 		// Search all program memory blocks
290 		for (MemoryBlock block : getMemoryBlocks()) {
291 			if (block.contains(addr)) {
292 				return block.isExecute();
293 			}
294 		}
295 
296 		return false;

301 	//
302 	// Is offset in an existing memory block?
303 	//------------------------------------------------------------------------
304 	private boolean isAddress(long offset) {
305 
306 		// Search all program memory blocks
307 		for (MemoryBlock block : getMemoryBlocks()) {
308 
309 			if (block.getStart().getOffset() <= offset && block.getEnd().getOffset() >= offset) {
310 				return true;
311 			}
312 		}
313 		return false;    // no match
314 	}

329 	//------------------------------------------------------------------------
330 	// isString
331 	//
332 	// Are the bytes starting at addr a C string?
333 	//
334 	// Algorithm:  Scan bytes until finding either an invalid char or null.
335 	//             If scan stops at null, return true -- else false.
336 	//------------------------------------------------------------------------
337 	private boolean isString(Address addr) {
338 		byte _byte;
339 
340 		try {
341 			_byte = getByte(addr);
342 		}
343 		catch (Exception except) {
344 			return false;
345 		}
346 
347 		while (	// May need to add valid character examples here.
348 		(_byte == 0x09 || _byte == 0x0a || _byte == 0x0d || (_byte > 0x19 && _byte < 0x80)) &&
349 			_byte != 0x00) {
350 
351 			if (monitor.isCancelled()) {
352 				return false;
353 			}
354 
355 			addr = addr.add(1);
356 			try {
357 				_byte = getByte(addr);
358 			}
359 			catch (Exception except) {
360 				return false;
361 			}
362 		}
363 
364 		if (_byte == 0x00) {
365 			return true;  // Scan stopped at null.
366 		}
367 		return false; // Scan stopped at invalid char.
368 	}

370 	//------------------------------------------------------------------------
371 	// clearString
372 	//
373 	// Remove data or instructions that overlap the null-terminated
374 	// string at addr (may happen if disassembly creates invalid references
375 	// or compiler optimization creates shared strings).
376 	//
377 	// Use get*Containing() in case a string that ends with the string
378 	// at addr has already been defined (e.g., the string at addr is
379 	// "CoolFunc" and the string "g_pfCoolFunc" overlaps it).
380 	//------------------------------------------------------------------------
381 	private void clearString(Address addr) throws Exception {
382 		Data data;
383 		Instruction inst;
384 
385 		// Clear the string, breaking on the terminating null character
386 		while (getByte(addr) != 0) {
387 			data = getDataContaining(addr);
388 			if (data != null) {
389 				removeDataAt(data.getAddress());
390 			}
391 			inst = getInstructionContaining(addr);
392 			if (inst != null) {
393 				removeInstructionAt(inst.getAddress());
394 			}
395 
396 			addr = addr.add(1);
397 		}
398 
399 		// Now clear at string's terminating null character
400 		data = getDataContaining(addr);
401 		if (data != null) {
402 			removeDataAt(data.getAddress());
403 		}
404 		inst = getInstructionContaining(addr);
405 		if (inst != null) {
406 			removeInstructionAt(inst.getAddress());
407 		}
408 	}

411 	// isSymTblEntry
412 	//
413 	// Does data pointed to by entry look like a VxWorks symbol table entry?
414 	// Test is weak.
415 	//------------------------------------------------------------------------
416 	private boolean isSymTblEntry(Address entry, VxSymbol vxSymbol) throws Exception {
417 
418 		// First dword must be null or a valid ptr (typically into the sym table)
419 		long value = getInt(entry) & 0xffffffffL;
420 		if ((value != 0) && !isAddress(value)) {
421 			if (debug) {
422 				println("1: " + entry + " --> " + Long.toHexString(value));
423 			}
424 			return false;
425 		}
426 
427 		// symNameOff field must point to a non-null C string
428 		value = getInt(entry.add(vxSymbol.nameOffset())) & 0xffffffffL;
429 		if (!isAddress(value)) {
430 			if (debug) {
431 				println("2: " + entry + " --> " + Long.toHexString(value));
432 			}
433 			return false;
434 		}
435 		Address symNameAddr = toAddr(value);
436 		if (!isString(symNameAddr)) {
437 			if (debug) {
438 				println("3: " + entry + " --> " + Long.toHexString(value));
439 			}
440 			return false;
441 		}
442 		if (getByte(symNameAddr) == 0) {
443 			return false;
444 		}
445 
446 		// symLocOff field can be almost anything (e.g., external mem ref)
447 		//value = (long)getInt(entry.add(vxSymbol.locOffset())) & 0xffffffffL;
448 		//if (value == 0) {
449 		//	if (debug) println("4: " + entry);
450 		//	return false;
451 		//}
452 
453 		// symType field must be recognized type code (this test is weak)
454 		byte symType = getByte(entry.add(vxSymbol.typeOffset()));
455 		if (!isValidSymType(symType)) {
456 			if (debug) {
457 				println("5: " + entry + " --> " + symType);
458 			}
459 			return false;
460 		}
461 
462 		return true;
463 	}

465 	private boolean isValidSymType(byte symType) {
466 		switch (symType) {
467 			case 0: // Undefined Symbol
468 				return false;
469 			case 2: // Local Absolute
470 			case 3: // Global Absolute
471 			case 6: // Local Data
472 			case 7: // Global Data
473 			case 8: // Local BSS
474 			case 9: // Global BSS
475 			case 4: // Local .text
476 			case 5: // Global .text
477 			case 0x11: // External ref -- ignore
478 				return true;
479 			default:
480 				return false;
481 		}
482 	}

484 	//------------------------------------------------------------------------
485 	// findSymTbl
486 	//
487 	// Searches all memory blocks for data that looks like a run of testLen
488 	// VxWorks symbol table entries.
489 	//
490 	// Returns:
491 	//		Success: table address
492 	//		Failure: null
493 	//------------------------------------------------------------------------
494 	private Address findSymTbl(VxSymbol vxSymbol) throws Exception {
495 
496 		int testLen = 100;		// number of symbol tbl entries to look for
497 
498 		boolean hasNonExecute = checkNonExecute();
499 
500 		// Iterate through all memory blocks
501 		for (MemoryBlock block : currentProgram.getMemory().getBlocks()) {
502 
503 			// Skip code/execute blocks if there are non-execute blocks,
504 			//  otherwise search everything.
505 			if (hasNonExecute && block.isExecute()) {
506 				continue;
507 			}
508 
509 			// skip uninit
510 			if (!block.isInitialized()) {
511 				continue;
512 			}
513 
514 			// Search current block for run of testLen symbol table entries
515 			int testBlkSize = vxSymbol.length * testLen;
516 			printf("   block: " + block.getName() + " (" + block.getStart() + ", " +
517 				block.getEnd() + ") ");
518 			printf("testBlkSize = " + Integer.toHexString(testBlkSize) + "  ");
519 			System.out.flush();
520 			long prevOffset = 0;
521 			Address cursor = block.getStart();
522 			while ((cursor != null) && isAddress(cursor.getOffset() + testBlkSize, block)) {
523 
524 				// Script cancel check and visual feedback
525 				if (monitor.isCancelled()) {
526 					return null;
527 				}
528 				if ((cursor.getOffset() - prevOffset) >= 0x100000) {
529 					printf(".");
530 					System.out.flush();
531 					prevOffset = cursor.getOffset();
532 				}
533 
534 				// Determine whether cursor now points to a symbol table
535 				int i = 0;
536 				for (Address entry = cursor; isSymTblEntry(entry, vxSymbol) &&
537 					(i < testLen); entry = entry.add(vxSymbol.length()), i++) {
538 				}
539 				if (i == testLen) {
540 					// May have symbol table -- verify length
541 					if (getSymTblLen(cursor, vxSymbol) != 0) {
542 						printf("\n");
543 						System.out.flush();
544 						return cursor;	// found  table -- stop searching
545 					}
546 					if (debug) {
547 						printf("Possible symbol table at " + cursor + " has length error\n");
548 					}
549 				}
550 
551 				cursor = cursor.add(4);
552 			}
553 			printf("\n");
554 			printf("   search terminated at:  " + cursor + "\n");
555 			System.out.flush();
556 		}
557 		return null;
558 	}
559 
560 	private boolean checkNonExecute() {
561 		for (MemoryBlock block : currentProgram.getMemory().getBlocks()) {
562 			if (!block.isExecute()) {
563 				return true;
564 			}
565 		}
566 		return false;
567 	}

569 	//------------------------------------------------------------------------
570 	// getSymTblLen
571 	//
572 	// Counts number of entries in VxWorks table at address symTbl.
573 	//
574 	// Returns:
575 	//		Success: number of table entries (> 0)
576 	//		Failure: 0
577 	//------------------------------------------------------------------------
578 	private int getSymTblLen(Address symTbl, VxSymbol vxSymbol) throws Exception {
579 
580 		Address entry = symTbl;
581 		int j = 0;
582 		while (isSymTblEntry(entry, vxSymbol)) {
583 			entry = entry.add(vxSymbol.length());
584 			j++;
585 		}
586 
587 		return j;
588 
589 		// NOTE: Found an example of a VxWorks symbol table that was not
590 		//	      directly adjacent to the symbol table length variable...
591 		//	      so removed the following constraint.
592 
593 		/*
594 		// Symbol table length entry may be at beginning or end of the symbol
595 		// table...so compare computed length with both values.  If either
596 		// matches, table length is verified.
597 		if ((j == getInt(entry)) || (j == getInt(symTbl.subtract(4)))) {
598 			return j;
599 		} else {
600 			return 0;
601 		}
602 		*/
603 	}

605 	/**
606 	 * Look before/after the table to see if there is a size value there and mark it if it agrees with TableLen
607 	 * 
608 	 * @param symTbl
609 	 * @param vxSymbol
610 	 * @param tableLen
611 	 * @throws Exception
612 	 */
613 	private void markSymbolTableLen(Address symTbl, VxSymbol vxSymbol, int symTblLen)
614 			throws Exception {
615 		// NOTE: Found an example of a VxWorks symbol table that was not
616 		//	      directly adjacent to the symbol table length variable...
617 
618 		// Name the VxWorks symbol length variable
619 		// (if it appears either directly before or after the symbol table)
620 		Address symTblLenPtr = null;
621 		long foreOff = symTbl.getOffset() - 4;
622 		long aftOff = symTbl.getOffset() + symTblLen * vxSymbol.length();
623 		if (isAddress(foreOff) && getInt(toAddr(foreOff)) == symTblLen) {
624 			symTblLenPtr = toAddr(foreOff);
625 		}
626 		else if (isAddress(aftOff) && getInt(toAddr(aftOff)) == symTblLen) {
627 			symTblLenPtr = toAddr(aftOff);
628 		}
629 		if (symTblLenPtr != null) {
630 			removeConflictingSymbols("vxSymTblLen", symTblLenPtr);
631 			createLabel(symTblLenPtr, "vxSymTblLen", true);
632 			createDWord(symTblLenPtr);
633 		}
634 		else {
635 			println("Warning: Symbol Table Size not found before of after table");
636 		}
637 	}

639 	//------------------------------------------------------------------------
640 	// removeConflictingSymbols
641 	//
642 	// Deletes all symbols with the same name and the primary symbol at addr.
643 	//------------------------------------------------------------------------
644 	private void removeConflictingSymbols(String name, Address addr) {
645 
646 		// Delete any existing symbols with the same name
647 		for (Symbol sym : currentProgram.getSymbolTable().getSymbols(name)) {
648 			sym.delete();
649 		}
650 
651 		// Delete primary Ghidra symbol at the same address
652 		Symbol sym = getSymbolAt(addr);
653 		if (sym != null) {
654 			sym.delete();
655 		}
656 
657 		return;
658 	}
659 
660 	//------------------------------------------------------------------------
661 	// applyDemangled
662 	//
663 	// Apply demangled symbol name to symbol.
664 	//------------------------------------------------------------------------
665 	private void applyDemangled(Address addr, String mangled, String demangled) {
666 
667 		if (demangled != null) {
668 			new DemanglerCmd(addr, mangled).applyTo(currentProgram, monitor);
669 			List<Symbol> symbols =
670 				getSymbols(mangled, currentProgram.getGlobalNamespace());
671 			if (!symbols.isEmpty()) {
672 				currentProgram.getSymbolTable().removeSymbolSpecial(symbols.get(0));
673 			}
674 		}
675 
676 		return;
677 	}
678 
679 	//------------------------------------------------------------------------
680 	// doLocalDisassemble
681 	//
682 	// Do our own disassembly, and don't let auto-analysis start until after
683 	// this script finishes.  This speeds up the script substantially and
684 	// allows auto-analysis to operate with more information (and code/data
685 	// that isn't rapidly changing).
686 	//------------------------------------------------------------------------
687 	private void doLocalDisassemble(Address addr) {
688 
689 		// Only disassemble in memory blocks marked executable
690 		if (!isExecute(addr)) {
691 			return;
692 		}
693 
694 		DisassembleCommand cmd = new DisassembleCommand(addr, null, true);
695 		cmd.enableCodeAnalysis(false);	// Queues changes up for later analysis
696 		cmd.applyTo(currentProgram, monitor);
697 
698 		AddressSet set = cmd.getDisassembledAddressSet();
699 		AutoAnalysisManager.getAnalysisManager(currentProgram).codeDefined(set);
700 
701 		return;
702 	}

704 	//------------------------------------------------------------------------
705 	// getScriptAnalysisMode
706 	//
707 	// Force auto-analysis to wait until script completes.
708 	//------------------------------------------------------------------------
709 	@Override
710 	public AnalysisMode getScriptAnalysisMode() {
711 		return AnalysisMode.SUSPENDED;
712 	}

714 	//========================================================================
715 	// Main
716 	//========================================================================
717 	@Override
718 	public void run() throws Exception {
719 
720 		// Find VxWorks symbol table
721 		Address symTbl = null;
722 		VxSymbol vxSymbol = getVxSymbolClass(0);
723 		for (int i = 0; ((vxSymbol != null) && !monitor.isCancelled()); i++, vxSymbol =
724 			getVxSymbolClass(i)) {
725 
726 			println("Searching for symbol table variant " + i);
727 			if ((symTbl = findSymTbl(vxSymbol)) != null) {
728 				break;
729 			}
730 		}
731 		if (vxSymbol == null) {
732 			return;
733 		}
734 		int symTblLen = getSymTblLen(symTbl, vxSymbol);
735 		println("Symbol table at " + symTbl + " (" + symTblLen + " entries)");
736 
737 		// Name the VxWorks symbol table
738 		removeConflictingSymbols("vxSymTbl", symTbl);
739 		createLabel(symTbl, "vxSymTbl", true);
740 		markSymbolTableLen(symTbl, vxSymbol, symTblLen);
741 
742 		// Create symbol data type and symbol table structure
743 		println("Creating SYMBOL data type and symbol table structure...");
744 		vxSymbol.createGhidraType();
745 		clearListing(symTbl, symTbl.add(symTblLen * vxSymbol.length() - 1));
746 		createData(symTbl, new ArrayDataType(vxSymbol.dataType(), symTblLen, vxSymbol.length()));
747 
748 		// Create a GNU demangler instance
749 		GnuDemangler demangler = new GnuDemangler();
750 		if (!demangler.canDemangle(currentProgram)) {
751 			println("Unable to create demangler.");
752 			return;
753 		}
754 
755 		// Process VxWorks symbol table entries
756 		println("Processing symbol table entries.");
757 		Address symEntry = symTbl;
758 		for (int i = 0; (i < symTblLen) && !monitor.isCancelled(); i++, symEntry =
759 			symEntry.add(vxSymbol.length())) {
760 
761 			// Extract symbol table entry values
762 			Address symNameAddr = toAddr(getInt(symEntry.add(vxSymbol.nameOffset())) & 0xffffffffL);
763 			Address symLoc = toAddr(getInt(symEntry.add(vxSymbol.locOffset())) & 0xffffffffL);
764 			byte symType = getByte(symEntry.add(vxSymbol.typeOffset()));
765 
766 			// Remove any data or instructions that overlap string at *symNameAddr
767 			clearString(symNameAddr);
768 
769 			// Turn *symNameAddr into a string and store it in symName
770 			String symName;
771 			try {
772 				symName = (String) createAsciiString(symNameAddr).getValue();
773 			}
774 			catch (Exception e) {
775 				println("createAsciiString: caught exception...");
776 				println(e.getMessage());
777 				println(e.toString());
778 				return;
779 			}
780 			if (symName.length() > 2000) {
781 				symName = symName.substring(0, 2000);
782 			}
783 
784 			// Demangle symName
785 			String symDemangledName = null;
786 			try {
787 				symDemangledName = demangler.demangle(symName).getSignature(false);
788 			}
789 			catch (DemangledException e) {		// report demangling error
790 				if (!e.isInvalidMangledName()) {
791 					println("demangle: Demangling error");
792 				}
793 			}
794 			catch (RuntimeException e) {		// ignore unmangled symNames
795 			}
796 
797 			// Status update
798 			if (symDemangledName != null) {
799 				println("i=" + i + ", nameAddr: " + symNameAddr + ", loc: " + symLoc + ", type: " +
800 					symType + ", name: " + symName + ", demangled: " + symDemangledName);
801 			}
802 			else {
803 				println("i=" + i + ", nameAddr: " + symNameAddr + ", loc: " + symLoc + ", type: " +
804 					symType + ", name: " + symName);
805 			}
806 
807 			// Clear any conflicting symbols from the Ghidra symbol table
808 			removeConflictingSymbols(symName, symLoc);
809 
810 			// If entry type is data, simply create a Ghidra symbol for it.
811 			// If entry type is code, disassemble it and create function.
812 			switch (symType) {
813 
814 				case 0: // Undefined Symbol
815 					println("NULL symType!");
816 					break;
817 
818 				case 2: // Local Absolute 
819 				case 3: // Global Absolute
820 				case 6: // Local Data
821 				case 7: // Global Data
822 				case 8: // Local BSS
823 				case 9: // Global BSS
824 					createLabel(symLoc, symName, true);
825 					applyDemangled(symLoc, symName, symDemangledName);
826 					break;
827 
828 				case 4: // Local .text
829 				case 5: // Global .text  
830 					doLocalDisassemble(symLoc);
831 					createFunction(symLoc, symName);
832 					if (getFunctionAt(symLoc) != null) {
833 						getFunctionAt(symLoc).setName(symName, SourceType.USER_DEFINED);
834 						applyDemangled(symLoc, symName, symDemangledName);
835 					}
836 					else {
837 						println("createFunction: Failed to create function");
838 						createLabel(symLoc, symName, true);
839 						applyDemangled(symLoc, symName, symDemangledName);
840 					}
841 					break;
842 
843 				case 0x11: // External ref -- ignore
844 					break;
845 
846 				default:
847 					println("Invalid symType!");
848 					break;
849 			}
850 		}
851 	}
