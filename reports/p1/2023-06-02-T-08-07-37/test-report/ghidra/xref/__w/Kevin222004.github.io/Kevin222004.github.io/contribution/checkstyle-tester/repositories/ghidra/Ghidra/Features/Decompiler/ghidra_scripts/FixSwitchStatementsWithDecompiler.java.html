16  // Fix any unknown switch instructions with the decompiler.
17  //
18  // Your mileage may vary!  This should only be run after existing code has been found.
19  // The results should be checked for validity.
20  //
21  // @category Analysis
22  
23  import java.util.*;
24  
25  import ghidra.app.cmd.function.DecompilerSwitchAnalysisCmd;
26  import ghidra.app.decompiler.DecompileResults;
27  import ghidra.app.decompiler.parallel.DecompilerCallback;
28  import ghidra.app.decompiler.parallel.ParallelDecompiler;
29  import ghidra.app.plugin.core.analysis.SwitchAnalysisDecompileConfigurer;
30  import ghidra.app.plugin.core.bookmark.BookmarkEditCmd;
31  import ghidra.app.script.GhidraScript;
32  import ghidra.program.model.listing.*;
33  import ghidra.program.model.symbol.FlowType;
34  import ghidra.program.model.symbol.Reference;
35  import ghidra.util.task.TaskMonitor;
36  
37  public class FixSwitchStatementsWithDecompiler extends GhidraScript {
38  
39  	@Override
40  	public void run() throws Exception {
41  
42  		Map<Function, Instruction> instructionsByFunction = filterFunctions();
43  
44  		DecompilerCallback<Void> callback = new DecompilerCallback<Void>(currentProgram,
45  			new SwitchAnalysisDecompileConfigurer(currentProgram)) {
46  
47  			@Override
48  			public Void process(DecompileResults results, TaskMonitor m) throws Exception {
49  
50  				Function func = results.getFunction();
51  				DecompilerSwitchAnalysisCmd cmd = new DecompilerSwitchAnalysisCmd(results);
52  				cmd.applyTo(currentProgram);
53  				Instruction instr = instructionsByFunction.get(func);
54  				BookmarkEditCmd bmcmd = new BookmarkEditCmd(instr.getMinAddress(),
55  					BookmarkType.INFO, "FixSwitchStatementsWithDecompiler", "Fixed switch stmt");
56  				bmcmd.applyTo(currentProgram);
57  				return null;
58  			}
59  		};
60  
61  		Set<Function> functions = instructionsByFunction.keySet();
62  		try {
63  			ParallelDecompiler.decompileFunctions(callback, functions, monitor);
64  		}
65  		finally {
66  			callback.dispose();
67  		}
68  	}
69  
70  	private Map<Function, Instruction> filterFunctions() {
71  
72  		// search for all dynamic jump locations that have no target
73  
74  		Map<Function, Instruction> results = new HashMap<>();
75  		Listing list = currentProgram.getListing();
76  		FunctionManager functionManager = currentProgram.getFunctionManager();
77  		InstructionIterator it = list.getInstructions(true);
78  		while (it.hasNext()) {
79  
80  			if (monitor.isCancelled()) {
81  				return Collections.emptyMap();
82  			}
83  
84  			Instruction instr = it.next();
85  			FlowType flowType = instr.getFlowType();
86  			if (!flowType.isJump() || !flowType.isComputed()) {
87  				continue;
88  			}
89  
90  			Reference[] refsFrom = instr.getReferencesFrom();
91  			if (refsFrom.length == 0 || refsFrom.length > 2) {
92  				continue;
93  			}
94  
95  			Function func = functionManager.getFunctionContaining(instr.getMinAddress());
96  			if (func == null) {
97  				println("No function at " + instr.getMinAddress());
98  				continue;
99  			}
100 			results.put(func, instr);
101 		}
102 		return results;
103 	}
104 }
