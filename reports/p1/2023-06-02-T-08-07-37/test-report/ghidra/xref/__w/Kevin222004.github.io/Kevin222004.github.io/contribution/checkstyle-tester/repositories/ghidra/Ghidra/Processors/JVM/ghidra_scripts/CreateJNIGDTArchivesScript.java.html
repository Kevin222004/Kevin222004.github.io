17  // Parses sample linux and windows JNI header files into .gdt data type archive.
18  //
19  // To replace existing header files and have the data type ID's synchronized
20  //
21  // Must run SynchronizeGDTCategoryPaths.java script with old and replacement GDT
22  //  archive to synchronize upper/lower case paths
23  ///   (only on windows archives)
24  //
25  // Then Run DataTypeArchiveTransformer in eclipse to synchronize old data types ID's
26  //
27  //@category Data Types
28  
29  import java.io.File;
30  import java.io.IOException;
31  
32  import generic.jar.ResourceFile;
33  import ghidra.app.plugin.core.datamgr.util.DataTypeArchiveUtility;
34  import ghidra.app.script.GhidraScript;
35  import ghidra.app.util.cparser.C.CParserUtils;
36  import ghidra.app.util.cparser.C.CParserUtils.CParseResults;
37  import ghidra.app.util.cparser.C.ParseException;
38  import ghidra.program.model.data.DataTypeManager;
39  import ghidra.program.model.data.FileDataTypeManager;
40  import ghidra.util.Msg;
41  
42  public class CreateJNIGDTArchivesScript extends GhidraScript {
43  
44  	private File outputDirectory;
45  	
46  	// location of header files base directory
47  	private static String headerFilePath = "/data/HeaderFiles";
48  
49  	@Override
50  	protected void run() throws Exception {
51  		outputDirectory = askDirectory("Select Directory for GDT files", "Select GDT Output Dir");
52  		
53  		parseGDT_Linux_JNI();
54  
55  		parseGDT_Windows_JNI();
56  
57  	}
58  
59  	private void parseHeaderFilesToGDT(DataTypeManager openTypes[], File outputDir, String gdtName, String languageID, String compiler, String[] filenames, String[] args)
60  			throws ParseException, ghidra.app.util.cparser.CPP.ParseException, IOException {
61  		
62  		String dataTypeFile = outputDir + File.separator + gdtName + ".gdt";
63  		
64  		File f = getArchiveFile(dataTypeFile);
65  		
66          FileDataTypeManager dtMgr = FileDataTypeManager.createFileArchive(f);
67          
68  		CParseResults results = CParserUtils.parseHeaderFiles(openTypes, filenames, args, dtMgr, languageID, compiler, monitor);
69  		
70  		Msg.info(this, results.getFormattedParseMessage(null));
71  
72  		dtMgr.save();
73  		dtMgr.close();
74  	}
75  	
76  	/**
77  	 * Turn string into a file, delete old archive/lock file if it exists
78  	 * 
79  	 * @param dataTypeFile
80  	 * 
81  	 * @return file
82  	 */
83  	private File getArchiveFile(String dataTypeFile) {
84  		File f = new File(dataTypeFile);
85  		if (f.exists()) {
86  			f.delete();
87  		}
88  		String lockFile = dataTypeFile + ".ulock";
89  		File lf = new File(lockFile);
90  		if (lf.exists()) {
91  			lf.delete();
92  		}
93  		return f;
94  	}
95  
96  
97  	public void parseGDT_Linux_JNI() throws Exception {	
98  		String filenames[] = {
99  				"jni.h",
100 				"jawt.h",
101 				"jdwpTransport.h",
102 				"jvmti.h",
103 				"jvmticmlr.h",
104 				"classfile_constants.h",
105 		};
106 		
107 		String args[] = {
108 				"-I"+headerFilePath+"/jni/linux",
109 				"-I"+headerFilePath+"/jni/linux/linux",
110 				"-D_X86_",
111 				"-D__STDC__",
112 				"-D_GNU_SOURCE",
113 				"-D__WORDSIZE=64",
114 				"-D__builtin_va_list=void *",
115 				"-D__DO_NOT_DEFINE_COMPILE",
116 				"-D_Complex",
117 				"-D__NO_STRING_INLINES",
118 				"-D__signed__",
119 				"-D__extension__=",
120 				"-D__GLIBC_HAVE_LONG_LONG=1",
121 				"-D__need_sigset_t",
122 				"-Daligned_u64=uint64_t",
123 		};
124 
125 		
126 		// Using another archive while parsing will cause:
127 		//  - a dependence on the other archive
128 		//  - any missing data types while parsing are supplied if present from existingDTMgr
129 		//  - after parsing all data types parsed that have an equivalent data type will be
130 		//    replaced by the data type from the existingDTMgr
131 		//
132 		// NOTE: This will only occur if the data type from the exisitngDTMgr is equivalent.
133 		//
134 		ResourceFile clib64ArchiveFile = DataTypeArchiveUtility.findArchiveFile("generic_clib_64.gdt");
135 		File file = new File(clib64ArchiveFile.getAbsolutePath());
136 		DataTypeManager existingDTMgr = FileDataTypeManager.openFileArchive(file, false);
137 		DataTypeManager openTypes[] = { existingDTMgr };
138 		
139 		parseHeaderFilesToGDT(openTypes, outputDirectory, "jni_linux", "x86:LE:64:default", "gcc", filenames, args);
140 	}
