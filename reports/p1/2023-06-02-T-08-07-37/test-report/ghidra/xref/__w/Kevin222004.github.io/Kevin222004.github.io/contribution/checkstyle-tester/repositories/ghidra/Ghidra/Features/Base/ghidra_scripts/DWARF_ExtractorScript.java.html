16  //This script extracts DWARF 2, 3, or 4 debug information
17  //and applies it to the program.
18  //
19  // This script reads DWARF information from the debug sections of a binary and
20  // imports the information into Ghidra.
21  // This information includes: data type definitions, namespace and class information,
22  // function signatures, and function source file locations.
23  // Currently, the script is based off of DWARF2, DWARF3, and DWARF4 specification. However,
24  // not all features or possibilities have been implemented.
25  // The default max of 512MB RAM allocated for Ghidra is not enough for running the
26  // script on larger binaries. Setting -Xmx2g should be sufficient.
27  //
28  // Features that still require implementing:
29  // - All TAG data types have not yet been implemented. This script will not complete if
30  //     it encounters an unknown data type and that data type is used.
31  // - Location description processing is currently hardcoded for the most common expressions.
32  //     This should be updated to support any valid location description.
33  // - Expression description processing is incomplete. All description opcodes should be supported.
34  // - Location description and Expression description should probably be set up similiar
35  //     to frysk (see frysk-core/frysk/debuginfo/LocationExpression.java)
36  // - Handle all errors correctly, setup warning/error system to print out or log
37  //     warnings/errors while processing
38  // - Program tree currently does not allow any duplicate names
39  // - Match all datatypes found by the demangler with datatypes processed in DWARF
40  // - Various testing and fixes on a variety of different compiled binaries
41  //
42  //
43  // Possible Future Improvements:
44  // - Create a custom data type manager to handle adding datatypes temporarily
45  // - Only add datatypes to Ghidra's data type manager after all data types have been processed
46  // - Increase speed of the script as most of the waiting involves waiting on adding
47  //     data types to Ghidra's data type manager
48  // - Remove any hacks that are currently necessary for the script to add and use data types correctly
49  // - Possibility to add data types to a subfolder based on namespace or class during data
50  //     type processing
51  // - Check source language type and look for certain language constructs while processing
52  //
53  //
54  //@author User Submitted - based on DWARF_script.java
55  //@category Binary
56  //
57  
58  import ghidra.app.script.GhidraScript;
59  import ghidra.app.util.bin.format.dwarf4.next.*;
60  
61  public class DWARF_ExtractorScript extends GhidraScript {
62  
63  	@Override
64  	public void run() throws Exception {
65  		if (!DWARFProgram.isDWARF(currentProgram)) {
66  			popup("Unable to find DWARF information, aborting");
67  			return;
68  		}
69  		DWARFImportOptions importOptions = new DWARFImportOptions();
70  		importOptions.setImportLimitDIECount(Integer.MAX_VALUE);
71  		try (DWARFProgram dwarfProg = new DWARFProgram(currentProgram, importOptions, monitor)) {
72  			DWARFParser dp = new DWARFParser(dwarfProg, monitor);
73  			DWARFImportSummary importSummary = dp.parse();
74  			importSummary.logSummaryResults();
75  		}
76  	}
77  }
