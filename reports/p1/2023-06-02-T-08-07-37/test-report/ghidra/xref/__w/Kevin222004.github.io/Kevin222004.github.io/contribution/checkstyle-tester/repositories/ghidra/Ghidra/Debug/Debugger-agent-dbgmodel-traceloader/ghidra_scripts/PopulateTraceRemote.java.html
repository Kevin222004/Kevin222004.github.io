33  /**
34   * This script populates a trace database for demonstrations purposes and opens it in the current
35   * tool.
36   * 
37   * <p>
38   * Your current tool had better be the "TraceBrowser"! The demonstration serves two purposes. 1) It
39   * puts interesting data into the TraceBrowser and leaves some annotations as an exercise. 2) It
40   * demonstrates how a decent portion the Trace API works.
41   * 
42   * <p>
43   * A Trace is basically a collection of observations of memory and registers over the lifetime of an
44   * application or computer system. In Ghidra, the Trace object also supports many of the same
45   * annotations as does Program. In the same way that Program brings knowledge markup to an image of
46   * bytes, Trace brings knowledge markup to bytes observed over time.
47   * 
48   * <p>
49   * Effectively, if you take the cross-product of Program with time and add Threads, Breakpoints,
50   * etc., you get Trace. It's a lot. In order to use all the UI components which take a Program,
51   * Trace can present itself as a Program at a particular point in time.
52   * 
53   * <p>
54   * Each particular component will be introduced as its used in the script below, but for now some
55   * core concepts:
56   * 
57   * <ul>
58   * <li>A point in time is called a "tick." These don't necessarily correspond to any real unit of
59   * time, though they may. The only requirement is that they are numbered in chronological
60   * order.</li>
61   * <li>Every annotation has a "lifespan," which is the range of ticks for which the annotation is
62   * effective. Some annotations may overlap, others may not. In general, if the corresponding concept
63   * in Program permits address overlap, then Trace permits both address and time overlap. If not,
64   * then neither is permitted. In essense, Trace defines overlap as the intersection of rectangles,
65   * where an annotation's X dimension is it's address range, and its Y dimension is its lifespan.
66   * </li>
67   * <li>Observations in memory happen at a particular tick and are assumed in effect until another
68   * observation changes that. To record the "freshness" of observations, the memory manager tags
69   * regions as KNOWN, UNKNOWN, or ERROR. An observation implicitly marks the affected region as
70   * KNOWN. The intent is to grey the background for regions where memory is UNKNOWN for the current
71   * tick.</li>
72   * <li>Observations of registers behave exactly the same as observations for memory, by leveraging
73   * Ghidra's "register space." The only difference is that those observations must be recorded with
74   * respect to a given thread. Each thread is effectively allocated its own copy of the register
75   * space. Most the the API components require you to obtain a special "register space" for a given
76   * thread before recording observations of or applying annotations to that thread.</li>
77   * </ul>
78   * 
79   * <p>
80   * After you've run this script, a trace should appear in the UI. Note that there is not yet a way
81   * to save a trace in the UI. As an exercise, try adding data units to analyze the threads' stacks.
82   * It may take some getting accustomed to, but the rules for laying down units should be very
83   * similar to those in a Program. However, the Trace must take the applied units and decide how far
84   * into the future they are effective. In general, it defaults to "from here on out." However, two
85   * conditions may cause the trace to choose an ending tick: 1) The underlying bytes change sometime
86   * in the future, and 2) There is an overlapping code unit sometime in the future.
87   * 
88   * <p>
89   * The trace chooses the latest tick possible preceding any byte change or existing code unit, so
90   * that the unit's underlying bytes remain constant for its lifespan, and the unit does not overlap
91   * any existing unit. This rule causes some odd behavior for null-terminated strings. I intend to
92   * adjust this rule slightly for static data types wrt/ byte changes. For those, the placed unit
93   * should be truncated as described above, however, another data unit of the same type can be placed
94   * at the change. The same rule is then applied iteratively into the future until an overlapping
95   * unit is encountered, or there are no remaining byte changes.
96   */
97  public class PopulateTraceRemote extends GhidraScript {
98  
99  	private Language lang;
100 	private CompilerSpec cspec;
101 	private Trace trace;
102 	private TraceTimeManager timeManager;
103 
104 	private AddressSpace defaultSpace;
105 
106 	private DebuggerTraceManagerService manager;
107 	private DebuggerModelService targets;
108 
109 	/**
110 	 * Create an address in the processor's (x86_64) default space.
111 	 * 
112 	 * @param offset the byte offset
113 	 * @return the address
114 	 */
115 	protected Address addr(long offset) {
116 		return defaultSpace.getAddress(offset);
117 	}
118 
119 	/**
120 	 * Create an address range in the processor's default space.
121 	 * 
122 	 * @param min the minimum byte offset
123 	 * @param max the maximum (inclusive) byte offset
124 	 * @return the range
125 	 */
126 	protected AddressRange rng(long min, long max) {
127 		return new AddressRangeImpl(addr(min), addr(max));
128 	}
129 
130 	/**
131 	 * Get a register by name
132 	 * 
133 	 * @param name the name
134 	 * @return the register
135 	 */
136 	protected Register reg(String name) {
137 		return lang.getRegister(name);
138 	}
139 
140 	@Override
141 	protected void run() throws Exception {
142 
143 		File f = askFile("Trace", "Load");
144 
145 		cspec = currentProgram.getCompilerSpec();
146 		lang = currentProgram.getLanguage();
147 		defaultSpace = lang.getAddressFactory().getDefaultAddressSpace();
148 		trace = new DBTrace(f.getName(), cspec, this);
149 
150 		PluginTool tool = state.getTool();
151 		manager = tool.getService(DebuggerTraceManagerService.class);
152 		targets = tool.getService(DebuggerModelService.class);
153 
154 		try (Transaction tx = trace.openTransaction("Populate Events")) {
155 			timeManager = trace.getTimeManager();
156 			timeManager.createSnapshot("init");
157 		}
158 
159 		manager.openTrace(trace);
160 		manager.activateTrace(trace);
161 
162 		Set<DebuggerObjectModel> models = targets.getModels();
163 		DebuggerObjectModel model = (DebuggerObjectModel) models.toArray()[0];
164 		TargetInterpreter interpreter =
165 			DebugModelConventions.findSuitable(TargetInterpreter.class, model.getModelRoot()).get();
166 		interpreter.execute(".opendump " + f.getAbsolutePath()).get();
167 		interpreter.execute("g");
168 		TargetAttacher attacher =
169 			DebugModelConventions.findSuitable(TargetAttacher.class, model.getModelRoot()).get();
170 		// TODO: Is "Available" the correct path?
171 		attacher.attach(model.getModelObject("Available", "[0]").as(TargetAttachable.class)).get();
172 	}
173 }
