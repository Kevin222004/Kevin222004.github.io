16  // An example script that will print to the console, for a given function, all other functions
17  // that call it and all functions that it calls.
18  //import ghidra.app.plugin.core.navigation.locationreferences.ReferenceUtils;
19  import java.util.*;
20  
21  import ghidra.app.plugin.core.navigation.locationreferences.ReferenceUtils;
22  import ghidra.app.script.GhidraScript;
23  import ghidra.program.model.address.*;
24  import ghidra.program.model.listing.*;
25  import ghidra.program.model.symbol.*;
26  import ghidra.program.util.FunctionSignatureFieldLocation;
27  import ghidra.util.exception.CancelledException;
28  
29  public class PrintFunctionCallTreesScript extends GhidraScript {
30  
31  	@Override
32  	protected void run() throws Exception {
33  
34  		Function function = getCurrentFunction();
35  		if (function == null) {
36  			println("Cursor is not in or on a function");
37  			return;
38  		}
39  
40  		printIncomingCalls(function);
41  		println("\n");
42  		printOutgoingCalls(function);
43  	}
44  
45  	private void printIncomingCalls(Function function) throws CancelledException {
46  		Address functionAddress = function.getEntryPoint();
47  		FunctionSignatureFieldLocation location =
48  			new FunctionSignatureFieldLocation(function.getProgram(), functionAddress);
49  		Set<Address> addresses = ReferenceUtils.getReferenceAddresses(location, monitor);
50  		FunctionManager functionManager = currentProgram.getFunctionManager();
51  		Set<Function> callingFunctions = new HashSet<>();
52  		for (Address fromAddress : addresses) {
53  			Function callerFunction = functionManager.getFunctionContaining(fromAddress);
54  			if (callerFunction != null) {
55  				callingFunctions.add(callerFunction);
56  			}
57  		}
58  
59  		// sort them by address
60  		List<Function> list = new ArrayList<>(callingFunctions);
61  		Collections.sort(list, (f1, f2) -> f1.getEntryPoint().compareTo(f2.getEntryPoint()));
62  
63  		for (Function f : list) {
64  			println("Incoming Function Call: " + f.getName() + " @ " + f.getEntryPoint());
65  		}
66  	}
67  
68  	private void printOutgoingCalls(Function function) {
69  		AddressSetView functionBody = function.getBody();
70  		Set<Reference> references = getReferencesFrom(currentProgram, functionBody);
71  		Set<Function> outgoingFunctions = new HashSet<>();
72  		FunctionManager functionManager = currentProgram.getFunctionManager();
73  		for (Reference reference : references) {
74  			Address toAddress = reference.getToAddress();
75  			Function calledFunction = functionManager.getFunctionAt(toAddress);
76  			maybeAddIncomingFunction(outgoingFunctions, reference, calledFunction);
77  		}
78  
79  		// sort them by address
80  		List<Function> list = new ArrayList<>(outgoingFunctions);
81  		Collections.sort(list, (f1, f2) -> f1.getEntryPoint().compareTo(f2.getEntryPoint()));
82  
83  		for (Function f : list) {
84  			println("Outgoing Function Call: " + f.getName() + " @ " + f.getEntryPoint());
85  		}
86  	}
87  
88  	private void maybeAddIncomingFunction(Set<Function> incomingFunctions, Reference reference,
89  			Function calledFunction) {
90  		if (calledFunction != null) {
91  			incomingFunctions.add(calledFunction);
92  		}
93  		else if (isCallReference(reference)) {
94  			// we have a call reference, but no function
95  			println("Outgoing function call with no function from " + reference.getFromAddress() +
96  				" to " + reference.getToAddress());
97  		}
98  	}
99  
100 	private boolean isCallReference(Reference reference) {
101 		RefType type = reference.getReferenceType();
102 		if (type.isCall()) {
103 			return true;
104 		}
105 
106 		if (type.isIndirect()) {
107 			Listing listing = currentProgram.getListing();
108 			Instruction instruction = listing.getInstructionAt(reference.getFromAddress());
109 			if (instruction != null) {
110 				FlowType flowType = instruction.getFlowType();
111 				return flowType.isCall();
112 			}
113 		}
114 
115 		return false;
116 	}
117 
118 	private Set<Reference> getReferencesFrom(Program program, AddressSetView addresses) {
119 		Set<Reference> set = new HashSet<>();
120 		ReferenceManager referenceManager = program.getReferenceManager();
121 		AddressIterator addressIterator = addresses.getAddresses(true);
122 		while (addressIterator.hasNext()) {
123 			Address address = addressIterator.next();
124 			Reference[] referencesFrom = referenceManager.getReferencesFrom(address);
125 			if (referencesFrom != null) {
126 				for (Reference reference : referencesFrom) {
127 					set.add(reference);
128 				}
129 			}
130 		}
131 		return set;
132 	}
133 
134 	private Function getCurrentFunction() {
135 		FunctionManager functionManager = currentProgram.getFunctionManager();
136 		return functionManager.getFunctionContaining(currentAddress);
137 	}
