16  //Adds all files under a folder to version control.
17  //@category    Version Control
18  //@menupath    Tools.Version Control.Add All
19  
20  import ghidra.app.script.GhidraScript;
21  import ghidra.framework.model.*;
22  import ghidra.program.database.ProgramContentHandler;
23  
24  public class VersionControl_AddAll extends GhidraScript {
25  
26  	public VersionControl_AddAll() {
27  	}
28  
29  	@Override
30  	public void run() throws Exception {
31  
32  		if (currentProgram != null) {
33  			popup("This script should be run from a tool with no open programs");
34  			return;
35  		}
36  
37  		DomainFolder rootFolder =
38  			askProjectFolder("Choose root folder to recursively 'add to version control'");
39  		String checkInComment =
40  			askString("Check-in comment", "Enter the comment that will be used", "Initial import");
41  
42  		long start_ts = System.currentTimeMillis();
43  		monitor.initialize(0);
44  		monitor.setIndeterminate(true);
45  
46  		int filesProcessed = 0;
47  		for (DomainFile file : ProjectDataUtils.descendantFiles(rootFolder)) {
48  			if (monitor.isCancelled()) {
49  				break;
50  			}
51  			// Do not follow folder-links or consider program links.  Using content type
52  			// to filter is best way to control this.  If program links should be considered
53  			// "Program.class.isAssignableFrom(domainFile.getDomainObjectClass())"
54  			// should be used.  It may also be appropriate to handle other content types.
55  			if (!ProgramContentHandler.PROGRAM_CONTENT_TYPE.equals(file.getContentType()) ||
56  				file.isVersioned()) {
57  				continue;// skip
58  			}
59  			filesProcessed++;
60  			monitor.setMessage("Adding file " + file.getName() + " (" + filesProcessed + ")");
61  			file.addToVersionControl(checkInComment, false, monitor);
62  
63  		}
64  		long end_ts = System.currentTimeMillis();
65  
66  		println("Finished adding all programs to version control for folder: " +
67  			rootFolder.getPathname());
68  		println("Total files: " + filesProcessed);
69  		println("Total time: " + (end_ts - start_ts));
70  	}
71  
72  }
