16  // Given a variable in the decompiler, walk backward through function calls to find any constants
17  //   that find their way directly into the variable.  Very useful for getting a list of all the
18  //   constants passed to a parameter, or to a parameter at a given location in the program.
19  //
20  //   The guts of this script past the main could be used to analyze
21  //   constants passed to any function on any processor.
22  //   It is not restricted to windows.
23  //
24  //@category Search

52  	@Override
53  	public void run() throws Exception {
54  		TableChooserExecutor executor = null;
55  		// executor = createTableExecutor();
56  
57  		tableDialog = createTableChooserDialog("Constant Values", executor);
58  		configureTableColumns(tableDialog);
59  		tableDialog.show();
60  		tableDialog.setMessage("Searching...");
61  
62  		try {
63  			// get the decompiler context
64  			// setup the decompiler
65  			decomplib = setUpDecompiler(currentProgram);
66  
67  			// find out what we are on (param, variable, etc...
68  			@SuppressWarnings("unused")
69  			HashMap<Address, Long> constLocs;
70  
71  			Varnode var = getVarnodeLocation();
72  
73  			if (var != null && !var.isConstant()) {
74  				constLocs = backtrackToConstant(var, tableDialog);
75  			}
76  			else {
77  				// if couldn't find a varnode location, then must be on a
78  				// function parameter with no body
79  				// just backtrack the parameter from the function parameter
80  				if (currentLocation instanceof FunctionParameterFieldLocation) {
81  					FunctionParameterFieldLocation funcPFL =
82  						(FunctionParameterFieldLocation) currentLocation;
83  					Function f = funcPFL.getParameter().getFunction();
84  					int paramIndex = funcPFL.getParameter().getOrdinal();
85  					constLocs = backtrackParamToConstant(f, paramIndex, tableDialog);
86  				}
87  				else if (currentLocation instanceof VariableNameFieldLocation) {
88  					VariableNameFieldLocation varNFL = (VariableNameFieldLocation) currentLocation;
89  					Variable funcvar = varNFL.getVariable();
90  					Function f = funcvar.getFunction();
91  					if (funcvar instanceof Parameter) {
92  						int paramIndex = ((Parameter) funcvar).getOrdinal();
93  						constLocs = backtrackParamToConstant(f, paramIndex, tableDialog);
94  					}
95  				}
96  				else if (currentLocation instanceof DecompilerLocation) {
97  					// must be inside the decompiler, locate the parameter we are on within a function.
98  					DecompilerLocation decL = (DecompilerLocation) currentLocation;
99  					ClangToken token = decL.getToken();
100 					if (token instanceof ClangVariableToken) {
101 						ClangVariableToken clangVar = (ClangVariableToken) token;
102 						ClangNode parent = token.Parent();
103 						if (parent instanceof ClangStatement) {
104 							ClangStatement clangStmt = (ClangStatement) parent;
105 							PcodeOp pcodeOp = clangStmt.getPcodeOp();
106 							if (pcodeOp.getOpcode() == PcodeOp.CALL) {
107 								Varnode input = pcodeOp.getInput(0);
108 								if (input.isAddress()) {
109 									Address faddr = input.getAddress();
110 									Function f = getReferencedFunction(faddr);
111 									int paramIndex = 0;
112 									for (int i = 0; i < clangStmt.numChildren(); i++) {
113 										ClangNode child = clangStmt.Child(i);
114 										if (child.equals(clangVar)) {
115 											constLocs = backtrackParamToConstant(f, paramIndex,
116 												tableDialog);
117 											break;
118 										}
119 										if (child instanceof ClangVariableToken) {
120 											paramIndex++;
121 										}
122 									}
123 								}
124 							}
125 						}
126 					}
127 				}
128 				else {
129 					tableDialog.setMessage("****   please put the cursor on a variable!   ****");
130 				}
131 			}
132 		}
133 		finally {
134 			decomplib.dispose();
135 		}
136 		tableDialog.setMessage("Finished!");
137 	}

139 	private Function getReferencedFunction(Address functionAddress) {
140 		Function f = currentProgram.getFunctionManager().getFunctionAt(functionAddress);
141 		// couldn't find the function, see if there is an external ref there.
142 		if (f == null) {
143 			Reference[] referencesFrom =
144 				currentProgram.getReferenceManager().getReferencesFrom(functionAddress);
145 			for (Reference reference : referencesFrom) {
146 				if (reference.isExternalReference()) {
147 					functionAddress = reference.getToAddress();
148 					f = currentProgram.getFunctionManager().getFunctionAt(functionAddress);
149 					if (f != null) {
150 						break;
151 					}
152 				}
153 			}
154 		}
155 		return f;
156 	}

163 	private void configureTableColumns(TableChooserDialog tableChooserDialog) {
164 		// First column added is the Constant value that is found.
165 		// Note the special compare method that must compare the constant
166 		// values not as a default string rendering, but by actual value.
167 		StringColumnDisplay constColumn = new StringColumnDisplay() {
168 			@Override
169 			public int compare(AddressableRowObject o1, AddressableRowObject o2) {
170 				ConstUseLocation e1 = (ConstUseLocation) o1;
171 				ConstUseLocation e2 = (ConstUseLocation) o2;
172 				Long v1 = e1.getConstValue();
173 				Long v2 = e2.getConstValue();
174 				if (SystemUtilities.isEqual(v1, v2)) {
175 					return 0;
176 				}
177 				if (v1 == null) {
178 					return -1;
179 				}
180 				if (v2 == null) {
181 					return 1;
182 				}
183 
184 				return (int) (v1 - v2);
185 			}
186 
187 			@Override
188 			public String getColumnName() {
189 				return "Constant Value";
190 			}
191 
192 			@Override
193 			public String getColumnValue(AddressableRowObject rowObject) {
194 				ConstUseLocation entry = (ConstUseLocation) rowObject;
195 				Long val = entry.getConstValue();
196 				if (val == null) {
197 					return "";
198 				}
199 				return Long.toHexString(val);
200 			}
201 		};
202 
203 		// Displays a preview of anything defined at an address if the constant
204 		// discovered in the backtrack search could be treated as an address
205 		//
206 		StringColumnDisplay addrPreviewColumn = new StringColumnDisplay() {
207 			@Override
208 			public String getColumnName() {
209 				return "Addr Preview";
210 			}
211 
212 			@Override
213 			public String getColumnValue(AddressableRowObject rowObject) {
214 				ConstUseLocation entry = (ConstUseLocation) rowObject;
215 				Long val = entry.getConstValue();
216 				if (val == null) {
217 					return "";
218 				}
219 				Address addr = entry.getAddress();
220 				if (addr == null) {
221 					return "";
222 				}
223 				Address potAddr = addr.getNewAddress(val);
224 				if (addr.getAddressSpace().isOverlaySpace()) {
225 					potAddr = addr.getAddressSpace().getOverlayAddress(potAddr);
226 				}
227 				if (!currentProgram.getMemory().contains(potAddr)) {
228 					return "";
229 				}
230 				Listing listing = currentProgram.getListing();
231 				Data data = listing.getDefinedDataAt(potAddr);
232 				if (data != null) {
233 					return data.toString();
234 				}
235 				Function f = currentProgram.getFunctionManager().getFunctionAt(potAddr);
236 				if (f != null) {
237 					return f.getPrototypeString(false, false);
238 				}
239 				return "";
240 			}
241 		};

244 		// where a

257 						.getFunctionContaining(
258 							entry.getAddress());
259 				if (func == null) {
260 					return "";
261 				}
262 				return func.getName();
263 			}
264 		};
265 
266 		// column to display the note at the address where a
267 		// this is usually an error condition
268 		StringColumnDisplay noteColumn = new StringColumnDisplay() {
269 			@Override
270 			public String getColumnName() {
271 				return "Note";
272 			}
273 
274 			@Override
275 			public String getColumnValue(AddressableRowObject rowObject) {
276 				ConstUseLocation entry = (ConstUseLocation) rowObject;
277 				String note = entry.getNote();
278 				if (note == null) {
279 					return "";
280 				}
281 				return note;
282 			}
283 		};
284 
285 		tableChooserDialog.addCustomColumn(constColumn);
286 		tableChooserDialog.addCustomColumn(addrPreviewColumn);

329 				GhidraState scriptState = new GhidraState(state.getTool(), state.getProject(), prog,

331 				try {
332 					ResourceFile scriptSource = GhidraScriptUtil.findScriptByName(name);

334 						GhidraScriptProvider provider = GhidraScriptUtil.getProvider(scriptSource);
335 						GhidraScript script = provider.getScriptInstance(scriptSource, writer);
336 						script.execute(scriptState, monitor, writer);
337 						return;
338 					}

342 				}
343 				throw new IllegalArgumentException("Script does not exist: " + name);
344 			}

383 	 * Try to locate the Varnode that represents the variable in the listing or
384 	 * decompiler. In the decompiler this could be a local/parameter at any
385 	 * point in the decompiler. In the listing, it must be a parameter variable.
386 	 * 
387 	 * @return the varnode
388 	 */
389 	private Varnode getVarnodeLocation() {
390 		Varnode var = null;
391 
392 		if (currentLocation instanceof DecompilerLocation) {
393 			DecompilerLocation dloc;
394 
395 			// get the Varnode under the cursor
396 			dloc = (DecompilerLocation) currentLocation;
397 			ClangToken tokenAtCursor = dloc.getToken();
398 			var = DecompilerUtils.getVarnodeRef(tokenAtCursor);
399 			// fixupParams(dloc.getDecompile(), currentLocation.getAddress());
400 			if (tokenAtCursor == null) {
401 				println("****   please put the cursor on a variable in the decompiler!");
402 				return null;
403 			}
404 			lastResults = dloc.getDecompile();
405 		}
406 		else {
407 			// if we don't have one, make one, and map variable to a varnode
408 			HighSymbol highVar = computeVariableLocation(currentProgram, currentLocation);
409 			if (highVar != null) {
410 				var = highVar.getHighVariable().getRepresentative();
411 			}
412 			else {
413 				return null;
414 			}
415 		}
416 		return var;
417 	}

421 		Set<Address> keys;

427 			tableChooserDialog.add(new ConstUseLocation(currentProgram, loc, constant, null));
428 		}

437 	private HighSymbol computeVariableLocation(Program currProgram, ProgramLocation location) {
438 		HighSymbol highVar = null;
439 		Address storageAddress = null;
440 
441 		// make sure what we are over can be mapped to decompiler
442 		// param, local, etc...
443 
444 		if (location instanceof VariableLocation) {
445 			VariableLocation varLoc = (VariableLocation) location;
446 			storageAddress = varLoc.getVariable().getMinAddress();
447 		}
448 		else if (location instanceof FunctionParameterFieldLocation) {
449 			FunctionParameterFieldLocation funcPFL = (FunctionParameterFieldLocation) location;
450 			storageAddress = funcPFL.getParameter().getMinAddress();
451 		}
452 		else if (location instanceof OperandFieldLocation) {
453 			OperandFieldLocation opLoc = (OperandFieldLocation) location;
454 			int opindex = opLoc.getOperandIndex();
455 			if (opindex >= 0) {
456 				Instruction instr = currProgram.getListing().getInstructionAt(opLoc.getAddress());
457 				if (instr != null) {
458 					Register reg = instr.getRegister(opindex);
459 					if (reg != null) {
460 						storageAddress = reg.getAddress();
461 					}
462 				}
463 			}
464 		}
465 
466 		if (storageAddress == null) {
467 			return null;
468 		}
469 
470 		Address addr = currentLocation.getAddress();
471 		if (addr == null) {
472 			return null;
473 		}
474 
475 		Function f = currProgram.getFunctionManager().getFunctionContaining(addr);
476 		if (f == null) {
477 			return null;
478 		}
479 
480 		DecompileResults results = decompileFunction(f, decomplib);
481 
482 		HighFunction hf = results.getHighFunction();
483 		if (hf == null) {
484 			return null;
485 		}
486 
487 		// try to map the variable
488 		highVar = hf.getMappedSymbol(storageAddress, f.getEntryPoint().subtractWrap(1L));
489 		if (highVar == null) {
490 			highVar = hf.getMappedSymbol(storageAddress, null);
491 		}
492 		if (highVar == null) {
493 			highVar = hf.getMappedSymbol(storageAddress, f.getEntryPoint());
494 		}
495 
496 		if (highVar != null) {
497 			// fixupParams(results, location.getAddress());
498 		}
499 
500 		return highVar;
501 	}

544 	 * Backtrack to a constant given a varnode within a decompiled function This
545 	 * isn't useful for functions that can't be decompiled
546 	 * 
547 	 * @param var
548 	 *            varnode that represents a variable in a decompilation
549 	 * @param tableChooserDialog
550 	 *            - accumulate entries. Don't like passing it, but this way the
551 	 *            user gets immediate feedback as locations are found
552 	 * @return a map of Addresses->constants (constants could be NULL)
553 	 * @throws CancelledException if cancelled
554 	 */
555 	private HashMap<Address, Long> backtrackToConstant(Varnode var,
556 			TableChooserDialog tableChooserDialog) throws CancelledException {
557 
558 		HashMap<Address, Long> constUse = new HashMap<Address, Long>();
559 
560 		if (!decomplib.openProgram(currentProgram)) {
561 			println("Decompile Error: " + decomplib.getLastMessage());
562 			return constUse;
563 		}
564 
565 		// follow varnode back to any constants, accumulating back to function
566 		// param
567 		ArrayList<FunctionParamUse> funcList = new ArrayList<FunctionParamUse>();
568 
569 		ArrayList<PcodeOp> defUseList = new ArrayList<PcodeOp>();
570 		followToParam(constUse, defUseList, lastResults.getHighFunction(), var, funcList, null);
571 
572 		addConstants(tableChooserDialog, constUse);
573 
574 		return followFunctionParamToConstant(funcList, constUse, tableChooserDialog);
575 	}

577 	/**
578 	 * Backtrack to a constant given a start position of a parameter of a given
579 	 * function Useful if you want to start from a function paramter.
580 	 * 
581 	 * @param f function to start in
582 	 * @param paramIndex parameter index to backtrack from
583 	 * @param tableChooserDialog accumulate entries. Don't like passing it, but this way the
584 	 *         user gets immediate feedback as locations are found
585 	 * @return a map of Addresses to constants (constants could be NULL)
586 	 * @throws CancelledException if cancelled
587 	 */
588 	private HashMap<Address, Long> backtrackParamToConstant(Function f, int paramIndex,
589 			TableChooserDialog tableChooserDialog) throws CancelledException {
590 		HashMap<Address, Long> constUse = new HashMap<Address, Long>();
591 
592 		ArrayList<FunctionParamUse> funcList = new ArrayList<FunctionParamUse>();
593 
594 		if (f == null) {
595 			return constUse;
596 		}
597 		Varnode pvnode = null;
598 		Parameter parm = f.getParameter(paramIndex);
599 		if (parm == null) {
600 			this.popup(
601 				"Please put the cursor on a function parameter variable\nIf the function has not had it's parameters identified\nplease do so and try again");
602 			return constUse;
603 		}
604 
605 		// TODO: Parameter storage could consist of more than one varnode
606 		pvnode = parm.getFirstStorageVarnode();
607 
608 		if (pvnode == null) {
609 			return constUse;
610 		}
611 		funcList.add(new FunctionParamUse(f.getName(), f.getEntryPoint(), paramIndex, pvnode,
612 			new ArrayList<PcodeOp>()));
613 
614 		addConstants(tableChooserDialog, constUse);
615 
616 		return followFunctionParamToConstant(funcList, constUse, tableChooserDialog);
617 	}

619 	private HashMap<Address, Long> followFunctionParamToConstant(
620 			ArrayList<FunctionParamUse> funcList, HashMap<Address, Long> constUse,
621 			TableChooserDialog tableChooserDialog) throws CancelledException {
622 
623 		// any routines we bumped into, process back up the chain
624 		HashSet<Address> doneRoutines = new HashSet<Address>();
625 		while (funcList.size() > 0) {
626 			// get the next function the variable has been traced back to
627 			FunctionParamUse funcVarUse = funcList.remove(0);
628 			Address addr = funcVarUse.getAddress();
629 			int paramIndex = funcVarUse.getParamIndex();
630 			ArrayList<PcodeOp> defUseList = funcVarUse.getDefUseList();
631 
632 			// will do this at another time.
633 			if (doneRoutines.contains(addr)) {
634 				continue;
635 			}
636 			doneRoutines.add(addr);
637 
638 			// find all functions referring to that place
639 			ReferenceIterator referencesTo =
640 				currentProgram.getReferenceManager().getReferencesTo(addr);
641 			for (Reference reference : referencesTo) {
642 				monitor.checkCancelled();
643 
644 				// get function containing.
645 				Address refAddr = reference.getFromAddress();
646 				if (refAddr.getAddressSpace().getType() == AddressSpace.TYPE_NONE ||
647 					refAddr.isExternalAddress()) {
648 					continue;
649 				}
650 
651 				Function refFunc =
652 					currentProgram.getFunctionManager().getFunctionContaining(refAddr);
653 
654 				HashMap<Address, Long> localConstUse = new HashMap<Address, Long>();
655 
656 				if (refFunc == null) {
657 					localConstUse.put(refAddr, null);
658 					String problem = "*** No function at " + refAddr +
659 						".\nCould not analyze constant use past this undefined function!";
660 					addConstantProblem(tableChooserDialog, refAddr, problem);
661 					refFunc = UndefinedFunction.findFunction(currentProgram, refAddr, monitor);
662 				}
663 
664 				if (refFunc != null) {
665 					// decompile function
666 					// look for call to this function
667 					// display call
668 					ArrayList<PcodeOp> localDefUseList = new ArrayList<PcodeOp>(defUseList);
669 
670 					this.monitor.setMessage("Analyzing : " + refFunc.getName() + " for refs to " +
671 						addr + ":" + paramIndex);
672 
673 					analyzeFunction(localConstUse, decomplib, currentProgram, refFunc, refAddr,
674 						funcVarUse, paramIndex, localDefUseList, funcList);
675 					addConstants(tableChooserDialog, localConstUse);
676 				}
677 
678 				constUse.putAll(localConstUse);
679 			}
680 		}
681 
682 		return constUse;
683 	}

689 	private void analyzeFunction(HashMap<Address, Long> constUse, DecompInterface decompInterface,
690 			Program prog, Function f, Address refAddr, FunctionParamUse funcVarUse, int paramIndex,
691 			ArrayList<PcodeOp> defUseList, ArrayList<FunctionParamUse> funcList) {
692 		if (f == null) {
693 			return;
694 		}
695 
696 		decompileFunction(f, decompInterface);
697 
698 		Instruction instr = prog.getListing().getInstructionAt(refAddr);
699 		if (hfunction == null) {
700 			return;
701 		}
702 		Iterator<PcodeOpAST> ops = hfunction.getPcodeOps(refAddr.getPhysicalAddress());
703 		while (ops.hasNext() && !monitor.isCancelled()) {
704 			PcodeOpAST pcodeOpAST = ops.next();
705 			if (pcodeOpAST.getOpcode() == PcodeOp.CALL) {
706 				// get the second parameter
707 				Varnode parm = pcodeOpAST.getInput(paramIndex + 1); // 1st param is the call dest
708 				if (parm == null) {
709 					constUse.put(instr.getAddress(), null);
710 					String problem = "  *** Warning, it appears that function '" +
711 						funcVarUse.getName() + "' at " + funcVarUse.getAddress() +
712 						" does not have it's parameters recovered!\n" +
713 						"        Use Commit Params/Return in the decompiler on this function.";
714 					addErrorNote(instr.getAddress(), problem);
715 					break;
716 				}
717 				// see if it is a constant
718 				if (parm.isConstant() || parm.isAddress()) {
719 					// then this is a resource id
720 					// lookup the resource and create a reference
721 					long value = parm.getOffset();
722 					// TODO: not so fast, if there is a defUseList, must apply
723 					// it to get the real constant USED!
724 					try {
725 						value = applyDefUseList(value, defUseList);
726 						constUse.put(instr.getAddress(), value);
727 						println("   " + f.getName() + "    " + instr.getAddress() + " : 0x" +
728 							Long.toHexString(value));
729 					}
730 					catch (InvalidInputException exc) {
731 						// do nothing
732 					}
733 				}
734 				else {
735 					followToParam(constUse, defUseList, hfunction, parm, funcList, null);
736 				}
737 				break;
738 				// if this is anything else, get the high variable to see if it
739 				// can be traced back to a param
740 				// then repeat with any calls to this function at whatever the
741 				// param is
742 			}
743 			else if (pcodeOpAST.getOpcode() == PcodeOp.BRANCH) {
744 				Address faddr = pcodeOpAST.getInput(0).getAddress();
745 				if (funcVarUse == null || faddr == null) {
746 					continue;
747 				}
748 				if (!faddr.equals(funcVarUse.getAddress())) {
749 					continue;
750 				}
751 				Varnode rep = funcVarUse.getRepresentative();
752 				if (rep == null) {
753 					continue;
754 				}
755 				followToParam(constUse, defUseList, hfunction, rep, funcList, null);
756 			}
757 		}
758 	}

760 	private long applyDefUseList(long value, ArrayList<PcodeOp> defUseList)
761 			throws InvalidInputException {
762 		if (defUseList == null || defUseList.size() <= 0) {
763 			return value;
764 		}
765 		Iterator<PcodeOp> iterator = defUseList.iterator();
766 		while (iterator.hasNext()) {
767 			PcodeOp pcodeOp = iterator.next();
768 			int opcode = pcodeOp.getOpcode();
769 			switch (opcode) {
770 				case PcodeOp.INT_AND:
771 					if (pcodeOp.getInput(0).isConstant()) {
772 						value = value & pcodeOp.getInput(0).getOffset();
773 					}
774 					else if (pcodeOp.getInput(1).isConstant()) {
775 						value = value & pcodeOp.getInput(1).getOffset();
776 					}
777 					else {
778 						throw new InvalidInputException(
779 							" Unhandled Pcode OP " + pcodeOp.toString());
780 					}
781 					break;
782 				default:
783 					throw new InvalidInputException(" Unhandled Pcode OP " + pcodeOp.toString());
784 			}
785 		}
786 		return value;
787 	}

789 	private void followToParam(HashMap<Address, Long> constUse, ArrayList<PcodeOp> defUseList,
790 			HighFunction highFunction, Varnode vnode, ArrayList<FunctionParamUse> funcList,
791 			HashSet<SequenceNumber> doneSet) {
792 
793 		// follow back up through
794 		PcodeOp def = vnode.getDef();
795 		Function function = highFunction.getFunction();
796 		if (def == null) {
797 			HighVariable hvar = vnode.getHigh();
798 			if (hvar instanceof HighParam) {
799 				funcList.add(new FunctionParamUse(function.getName(), function.getEntryPoint(),
800 					((HighParam) hvar).getSlot(), hvar.getRepresentative(), defUseList));
801 				return;
802 			}
803 			if (hvar instanceof HighGlobal) {
804 				followThroughGlobal(constUse, defUseList, hvar, funcList, doneSet);
805 			}
806 			return;
807 		}
808 
809 		// have we done this vnode source already?
810 		Address funcEntry = function.getEntryPoint();
811 		SequenceNumber seqNum = def.getSeqnum();
812 		remapAddress(funcEntry, def.getOutput().getPCAddress());
813 
814 		if (doneSet == null) {
815 			doneSet = new HashSet<SequenceNumber>();
816 		}
817 		if (seqNum != null && doneSet.contains(seqNum)) {
818 			return;
819 		}
820 		doneSet.add(seqNum);
821 
822 		int opcode = def.getOpcode();
823 		switch (opcode) {
824 			case PcodeOp.COPY:
825 				if (def.getInput(0).isConstant()) {
826 					long value = def.getInput(0).getOffset();
827 					try {
828 						value = applyDefUseList(value, defUseList);
829 						constUse.put(remapAddress(funcEntry, def.getOutput().getPCAddress()),
830 							value);
831 						println("   " + function.getName() + "    " +
832 							def.getOutput().getPCAddress() + " : 0x" + Long.toHexString(value));
833 					}
834 					catch (InvalidInputException exc) {
835 						// Do nothing
836 					}
837 					return;
838 				}
839 				followToParam(constUse, defUseList, highFunction, def.getInput(0), funcList,
840 					doneSet);
841 				return;
842 			case PcodeOp.LOAD:
843 				if (def.getInput(0).isConstant() && def.getInput(1).isConstant()) {
844 					long space = def.getInput(0).getOffset();
845 					long offset = def.getInput(1).getOffset();
846 					if (space != funcEntry.getAddressSpace().getSpaceID()) {
847 						break;
848 					}
849 					try {
850 						offset = applyDefUseList(offset, defUseList);
851 						constUse.put(remapAddress(funcEntry, def.getOutput().getPCAddress()),
852 							offset);
853 						println("   " + function.getName() + "    " +
854 							def.getOutput().getPCAddress() + " : 0x" + Long.toHexString(offset));
855 					}
856 					catch (InvalidInputException exc) {
857 						// Do nothing
858 					}
859 					return;
860 				}
861 				followToParam(constUse, defUseList, highFunction, def.getInput(0), funcList,
862 					doneSet);
863 				return;
864 			case PcodeOp.INT_ZEXT:
865 				followToParam(constUse, defUseList, highFunction, def.getInput(0), funcList,
866 					doneSet);
867 				return;
868 			case PcodeOp.MULTIEQUAL:
869 				for (int i = 0; i < def.getNumInputs(); i++) {
870 					ArrayList<PcodeOp> splitUseList = new ArrayList<>(defUseList);
871 					followToParam(constUse, splitUseList, highFunction, def.getInput(i), funcList,
872 						doneSet);
873 				}
874 				return;
875 			case PcodeOp.CAST:
876 				followToParam(constUse, defUseList, highFunction, def.getInput(0), funcList,
877 					doneSet);
878 				return;
879 			case PcodeOp.INDIRECT:
880 				Varnode output = def.getOutput();
881 				if (output.getAddress().equals(def.getInput(0).getAddress())) {
882 					followToParam(constUse, defUseList, highFunction, def.getInput(0), funcList,
883 						doneSet);
884 					return;
885 				}
886 				if (def.getInput(0).isUnique()) {
887 					followToParam(constUse, defUseList, highFunction, def.getInput(0), funcList,
888 						doneSet);
889 					return;
890 				}
891 				break;
892 			case PcodeOp.INT_AND:
893 				if (def.getInput(1).isConstant()) {
894 					defUseList.add(0, def);
895 					followToParam(constUse, defUseList, highFunction, def.getInput(0), funcList,
896 						doneSet);
897 					return;
898 				}
899 				break;
900 			case PcodeOp.INT_ADD:
901 				if (def.getInput(1).isConstant()) {
902 					defUseList.add(0, def);
903 					followToParam(constUse, defUseList, highFunction, def.getInput(0), funcList,
904 						doneSet);
905 					return;
906 				}
907 				if (vnode.getHigh() instanceof HighParam) {
908 					// Do nothing
909 				}
910 				break;
911 			case PcodeOp.PTRADD: // Pointer + some offset (usually an array access)
912 				// if (!def.getInput(1).isConstant() ||
913 				// !def.getInput(2).isConstant()) {
914 				// break;
915 				// }
916 				// DataType outDt = def.getOutput().getHigh().getDataType();
917 				// println ("        type = " + outDt.getName());
918 				// long value = getSigned(def.getInput(1)) *
919 				// def.getInput(2).getOffset();
920 				//
921 				// try {
922 				// value = applyDefUseList(value, defUseList);
923 				// constUse.put(def.getOutput().getPCAddress(), value);
924 				// } catch (InvalidInputException exc) {}
925 				break;
926 
927 			case PcodeOp.PTRSUB: // Pointer + some sub element access (usually a
928 								// structure ref)
929 				Varnode offsetVal = def.getInput(1);
930 				if (!offsetVal.isConstant()) {
931 					break;
932 				}
933 				Varnode baseVal = def.getInput(0);
934 				if (baseVal.isConstant()) {
935 					// both constant, just use it and return the address
936 					long value = baseVal.getOffset() + offsetVal.getOffset();
937 					try {
938 						value = applyDefUseList(value, defUseList);
939 						constUse.put(remapAddress(funcEntry, def.getOutput().getPCAddress()),
940 							value);
941 						println("   " + function.getName() + "    " +
942 							def.getOutput().getPCAddress() + " : 0x" + Long.toHexString(value));
943 					}
944 					catch (InvalidInputException exc) {
945 						// Do nothing
946 					}
947 					return;
948 				}
949 				// TODO: handle access into data structure
950 				// DataType inDt = def.getInput(0).getHigh().getDataType();
951 				// DataType subDt = def.getOutput().getHigh().getDataType();
952 				// println ("        type = " + subDt.getName());
953 				// long subOff = getSigned(def.getInput(1));
954 				// outDt = subDt;
955 				// if (subDt instanceof Pointer) {
956 				// outDt = ((Pointer) subDt).getDataType();
957 				// if (outDt == null) {
958 				// }
959 				// }
960 				// try {
961 				// subOff = applyDefUseList(subOff, defUseList);
962 				// constUse.put(def.getOutput().getPCAddress(), subOff);
963 				// } catch (InvalidInputException exc) {}
964 				break;
965 		}
966 
967 		constUse.put(remapAddress(funcEntry, vnode.getPCAddress()), null);
968 		println("   " + function.getName() + "    " + vnode.getPCAddress() + " : Lost");
969 		// println("     Lost IT! " + vnode.getPCAddress());
970 	}

972 	private void followThroughGlobal(HashMap<Address, Long> constUse, ArrayList<PcodeOp> defUseList,
973 			HighVariable hvar,
974 			ArrayList<FunctionParamUse> funcList,
975 			HashSet<SequenceNumber> doneSet) {
976 		Address loc = hvar.getRepresentative().getAddress();
977 		PcodeOp def = hvar.getRepresentative().getDef();
978 		SequenceNumber seqnum = null;
979 		if (def != null) {
980 			seqnum = def.getSeqnum();
981 			if (doneSet.contains(seqnum)) {
982 				return;
983 			}
984 		}
985 		else {
986 			seqnum = new SequenceNumber(loc, 0);
987 		}
988 
989 		ReferenceIterator referencesTo = currentProgram.getReferenceManager().getReferencesTo(loc);
990 		while (referencesTo.hasNext()) {
991 			Reference reference = referencesTo.next();
992 			if (!reference.getReferenceType().isWrite()) {
993 				continue;
994 			}
995 			currentProgram.getFunctionManager().getFunctionContaining(reference.getFromAddress());
996 			Address refAddr = reference.getFromAddress();
997 			addErrorNote(refAddr, "Write to global variable");
998 		}
999 
1000 		doneSet.add(seqnum);
1001 	}

1005 			address = funcEntry.getAddressSpace().getOverlayAddress(address);
1006 		}
1007 		return address;
