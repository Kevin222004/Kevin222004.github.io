17  // Script requests target and replacement strings.
18  // It then iterates through most comments in the program, replacing
19  // the target string with the specified replacement string.
20  // Comments scanned include:
21  //    - EOL
22  //    - Plate
23  //    - Post
24  //    - Pre
25  //    - Function (plate)
26  //    - Function repeatable
27  //    - Local variable (stack and register)
28  //
29  // Notes:
30  //    - Script scans every address within the program, so it is slow.
31  //    - Script doesn't scan param comments.
32  //
33  //@category CustomerSubmission.Search
34  
35  import ghidra.app.script.GhidraScript;
36  import ghidra.program.model.address.Address;
37  import ghidra.program.model.listing.Function;
38  import ghidra.program.model.listing.FunctionIterator;
39  import ghidra.program.model.listing.Variable;
40  import ghidra.program.model.mem.MemoryBlock;
41  
42  public class ReplaceInComments extends GhidraScript {
43  
44  	@Override
45  	public void run() throws Exception {
46  
47  		// get target string
48  		String tgtStr = askString("Target string", "Target String");
49  		if (tgtStr == null)
50  			return;
51  
52  		// get replacement string
53  		String replStr = askString("Replacement string", "Replacement String");
54  		if (replStr == null)
55  			return;
56  
57  		// initialize count and get memory blocks
58  		int count = 0;
59  		MemoryBlock[] blocks = getMemoryBlocks();
60  
61  		// iterate through all memory blocks in current program
62  		for (int i = 0; i < blocks.length; i++) {
63  
64  			// iterate through all addresses for current block
65  			MemoryBlock m = blocks[i];
66  			println("Scanning block beginning at 0x" + m.getStart().toString());
67  			Address a;
68  			for (a = m.getStart(); !a.equals(m.getEnd().add(1)); a = a.add(1)) {
69  
70  				if (monitor.isCancelled())
71  					return;
72  				String curComment, newComment;
73  
74  				// replace target with replacement within each comment at address
75  
76  				curComment = getEOLComment(a);
77  				if (curComment != null) {
78  					newComment = curComment.replaceAll(tgtStr, replStr);
79  					if (!curComment.equals(newComment)) {
80  						println("   0x" + a.toString() + ":  " + newComment);
81  						setEOLComment(a, newComment);
82  						count = count + 1;
83  					}
84  				}
85  
86  				curComment = getPlateComment(a);
87  				if (curComment != null) {
88  					newComment = curComment.replaceAll(tgtStr, replStr);
89  					if (!curComment.equals(newComment)) {
90  						println("   0x" + a.toString() + ":  " + newComment);
91  						setPlateComment(a, newComment);
92  						count = count + 1;
93  					}
94  				}
95  
96  				curComment = getPostComment(a);
97  				if (curComment != null) {
98  					newComment = curComment.replaceAll(tgtStr, replStr);
99  					if (!curComment.equals(newComment)) {
100 						println("   0x" + a.toString() + ":  " + newComment);
101 						setPostComment(a, newComment);
102 						count = count + 1;
103 					}
104 				}
105 
106 				curComment = getPreComment(a);
107 				if (curComment != null) {
108 					newComment = curComment.replaceAll(tgtStr, replStr);
109 					if (!curComment.equals(newComment)) {
110 						println("   0x" + a.toString() + ":  " + newComment);
111 						setPreComment(a, newComment);
112 						count = count + 1;
113 					}
114 				}
115 			}
116 		}
117 
118 		// iterate through all functions in current program's listing
119 		FunctionIterator funcs = currentProgram.getListing().getFunctions(true);
120 		println("Scanning function and local variable comments...");
121 		while (funcs.hasNext() && !monitor.isCancelled()) {
122 
123 			// get current function and scan its comments
124 			Function f = funcs.next();
125 			String curComment, newComment;
126 
127 			// function plate comment
128 			curComment = f.getComment();
129 			if (curComment != null) {
130 				newComment = curComment.replaceAll(tgtStr, replStr);
131 				if (!curComment.equals(newComment)) {
132 					println("   " + f.getName() + ".comment");
133 					f.setComment(newComment);
134 					count = count + 1;
135 				}
136 			}
137 
138 			// function repeatable comment
139 			curComment = f.getRepeatableComment();
140 			if (curComment != null) {
141 				newComment = curComment.replaceAll(tgtStr, replStr);
142 				if (!curComment.equals(newComment)) {
143 					println("   " + f.getName() + ".repeatableComment");
144 					f.setRepeatableComment(newComment);
145 					count = count + 1;
146 				}
147 			}
148 
149 			// iterate through all variable comments for current function
150 			Variable[] vars = f.getLocalVariables();
151 			for (int i = 0; i < vars.length; i++) {
152 				Variable v = vars[i];
153 				curComment = v.getComment();
154 				if (curComment != null) {
155 					newComment = curComment.replaceAll(tgtStr, replStr);
156 					if (!curComment.equals(newComment)) {
157 						println("   " + f.getName() + "::" + v.getName() + ":  " + newComment);
158 						v.setComment(newComment);
159 						count = count + 1;
160 					}
161 				}
162 			}
163 		}
164 
165 		println("Comments changed:  " + count);
166 	}
167 }
