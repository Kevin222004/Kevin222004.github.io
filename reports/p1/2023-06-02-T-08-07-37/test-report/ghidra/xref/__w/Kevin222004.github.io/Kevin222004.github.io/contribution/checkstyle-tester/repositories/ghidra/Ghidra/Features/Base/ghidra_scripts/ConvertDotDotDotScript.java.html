18  // Sets function to have varArgs wherever a function's entry point address has a "decompiler_tags"
19  // property containing "<dotdotdot/>". It then removes the dotdotdot tag from the decompiler_tags.
20  // If the decompiler tags is then empty, it is also removed.
21  // Currently this doesn't remove the dotdotdot tag if there isn't a function at that address in the program.
22  //@category Update
23  
24  import ghidra.app.script.GhidraScript;
25  import ghidra.program.model.address.Address;
26  import ghidra.program.model.address.AddressIterator;
27  import ghidra.program.model.listing.Function;
28  import ghidra.program.model.listing.FunctionManager;
29  import ghidra.program.model.pcode.HighFunction;
30  import ghidra.program.model.util.PropertyMapManager;
31  import ghidra.program.model.util.StringPropertyMap;
32  
33  public class ConvertDotDotDotScript extends GhidraScript {
34  	
35  	long numDots;
36  	long numEmpty;
37  
38      @Override
39      public void run() throws Exception {
40          FunctionManager fm = currentProgram.getFunctionManager();
41  		PropertyMapManager pm = currentProgram.getUsrPropertyManager();
42  		StringPropertyMap stringmap = pm.getStringPropertyMap(HighFunction.DECOMPILER_TAG_MAP);
43  		if (stringmap == null) {
44      		println("Your program doesn't have any DotDotDot tags to convert.");
45  			return;
46  		}
47  		numDots = 0;
48  		numEmpty = 0;
49  		AddressIterator iter = stringmap.getPropertyIterator();
50  		while (iter.hasNext()) {
51  			Address addr = iter.next();
52  			Function f = fm.getFunctionAt(addr);
53  			if (f == null) {
54  				continue;
55  			}
56  			if (hasDotDotDotTag(stringmap, addr)) {
57  				if (!f.hasVarArgs()) {
58  					f.setVarArgs(true);
59  					println("Function "+f.getName()+" at "+addr.toString()+" now has varargs.");
60  				}
61  				removeDotDotDotTag(stringmap, addr);
62  			}
63  			else {
64  				if (removeEmptyTag(stringmap, addr)) {
65  				}
66  			}
67  		}
68  		println("Removed total of "+numDots+" dotdotdot tag"+((numDots!=1)?"s":"")+".");
69  		println("Removed total of "+numEmpty+" empty "+HighFunction.DECOMPILER_TAG_MAP+
70  				" propert"+((numEmpty!=1)?"ies":"y")+".");
71      }
72  
73  	private boolean hasDotDotDotTag(StringPropertyMap stringmap, Address addr) {
74  		String funcstring = stringmap.getString(addr);
75  		return funcstring.indexOf("<dotdotdot/>") != -1;
76  	}
77  
78  	private void removeDotDotDotTag(StringPropertyMap stringmap, Address addr) {
79  		String funcString = stringmap.getString(addr);
80  		if (funcString == null) {
81  			return;
82  		}
83  		String dotString = "<dotdotdot/>";
84  		int start = funcString.indexOf(dotString);
85  		if (start >= 0) {
86  			String prefix = funcString.substring(0, start);
87  			String suffix = "";
88  			try {
89  				suffix = funcString.substring(start+dotString.length());
90  			} catch (IndexOutOfBoundsException e) {
91  			}
92  			String newString = prefix + suffix;
93  			if (newString.length() != 0) {
94  				stringmap.add(addr, newString);
95  				println("Removed dotdotdot tag " + " from " +
96  						HighFunction.DECOMPILER_TAG_MAP + " property at " + addr.toString() + ".");
97  			}
98  			else {
99  				stringmap.remove(addr);
100 				println("Removed " + HighFunction.DECOMPILER_TAG_MAP +
101 						" property at " + addr.toString() + " since it is now empty after removing dotdotdot tag.");
102 			}
103 			numDots++;
104 		}
105 	}
106 
107 	private boolean removeEmptyTag(StringPropertyMap stringmap, Address addr) {
108 		String currentString = stringmap.getString(addr);
109 		if (currentString != null && currentString.length() == 0) {
110 			stringmap.remove(addr);
111 			println("Removed " + HighFunction.DECOMPILER_TAG_MAP +
112 					" property at " + addr.toString() + " since it is empty.");
113 			numEmpty++;
114 			return true;
115 		}
116 		return false;
117 	}
118 
119 }
