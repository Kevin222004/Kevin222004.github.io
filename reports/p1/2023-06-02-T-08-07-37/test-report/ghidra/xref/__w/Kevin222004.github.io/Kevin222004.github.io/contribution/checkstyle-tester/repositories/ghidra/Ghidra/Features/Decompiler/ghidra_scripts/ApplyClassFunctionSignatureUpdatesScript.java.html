16  // Script to apply any changes the user has made to recovered class virtual function signatures 
17  // edited in the listing. To run the script, put the cursor on a changed virtual function in 
18  // the listing then run the script. If the function signature in the given class differs from
19  // the associated function definition in the data type manager, the script will update the associated 
20  // function definition and any other related function signatures in the listing.
21  // Note: The script will not work if the vftable structures were not originally applied to 
22  // the vftables using the RecoverClassesFromRTTIScript.
23  // At some point, the Ghidra API will be updated to do this automatically instead of needing the 
24  // script to do so. For now, to make it a bit easier, you can use the below listed key binding
25  // or menupath if you have the "In Tool" checkbox checked for this script in the script manager.
26  //@category C++
27  //@menupath Scripts.ApplyClassFunctionSignatures
28  //@keybinding shift S
29  
30  import java.util.List;
31  
32  import classrecovery.RecoveredClassHelper;
33  import ghidra.app.script.GhidraScript;
34  import ghidra.program.model.address.Address;
35  import ghidra.program.model.data.*;
36  import ghidra.program.model.listing.Data;
37  import ghidra.program.model.listing.Function;
38  import ghidra.program.model.symbol.Namespace;
39  
40  public class ApplyClassFunctionSignatureUpdatesScript extends GhidraScript {
41  	@Override
42  	public void run() throws Exception {
43  
44  		if (currentProgram == null) {
45  			println("There is no open program");
46  			return;
47  		}
48  
49  		RecoveredClassHelper classHelper = new RecoveredClassHelper(currentProgram, currentLocation,
50  			state.getTool(), this, false, false, false, monitor);
51  		
52  		if(currentAddress == null) {
53  			println("Cursor must be in a class function.");
54  			return;
55  		}
56  		Function function = getFunctionContaining(currentAddress);
57  		if(function == null) {
58  			println("Cursor must be in a class function.");
59  			return;
60  		}
61  		
62  		if(function.isThunk()) {
63  			println("User should not edit thunks as they are auto-updated from thunked function. " +
64  				"Please undo changes to thunk then edit thunked function and rerun script");
65  			return;
66  		}
67  
68  		if (function.getName().contains("purecall")) {
69  			println("Function definitions are not affected by purecall changes.");
70  			return;
71  		}
72  	
73  
74  		Namespace classNamespace = classHelper.getClassNamespace(currentAddress);
75  		if (classNamespace == null) {
76  			println("Cursor must be in a class function.");
77  			return;
78  		}
79  		
80  		// get a vftable that points to this function - doesn't matter which since it will
81  		// be used to get the underlying function definition which will then be used to update
82  		// all related function signatures
83  		List<Address> vftablesContainingFunction = classHelper.getVftablesContaining(function);
84  
85  		// get all vftables that point to given function
86  		if (vftablesContainingFunction.isEmpty()) {
87  			println(
88  				"Function is not a virtual function so has no function definition or related " +
89  					"function signatures to update");
90  			return;
91  		}
92  
93  		// get one that has a class vftableStructure applied there
94  		Address vftableWithAppliedStructure = null;
95  		for (Address vftableAddress : vftablesContainingFunction) {
96  			monitor.checkCancelled();
97  
98  			Data dataAt = getDataAt(vftableAddress);
99  			if (dataAt == null) {
100 				continue;
101 			}
102 
103 			DataType baseDataType = dataAt.getBaseDataType();
104 
105 			if (baseDataType.getCategoryPath().getPath().contains("ClassDataTypes")) {
106 				vftableWithAppliedStructure = vftableAddress;
107 				break;
108 			}
109 		}
110 	
111 		if (vftableWithAppliedStructure == null) {
112 			println(
113 				"The vftable(s) containing this function do not have a valid vftable structure " +
114 					"applied. Please run the RecoverClassesFromRTTIScript.java on this program before " +
115 					"using this script to update virtual functions.");
116 			return;
117 		}
118 		List<Object> changedItems =
119 			classHelper.applyNewFunctionSignature(function, vftableWithAppliedStructure);
120 
121 		if (changedItems.isEmpty()) {
122 			println("No differences found between function signature at " +
123 				function.getEntryPoint().toString() +
124 				" and its associated function definition in the data type manager.");
125 			return;
126 		}
127 
128 		List<Structure> structuresOnList = classHelper.getStructuresOnList(changedItems);
129 		List<FunctionDefinition> functionDefinitionsOnList =
130 			classHelper.getFunctionDefinitionsOnList(changedItems);
131 		List<Function> functionsOnList = classHelper.getFunctionsOnList(changedItems);
132 
133 		if (!structuresOnList.isEmpty()) {
134 			println();
135 			println("Updated structures:");
136 			for (Structure structure : structuresOnList) {
137 				monitor.checkCancelled();
138 				println(structure.getPathName());
139 			}
140 		}
141 
142 		if (!functionDefinitionsOnList.isEmpty()) {
143 			println();
144 			println("Updated function definition:");
145 			for (FunctionDefinition functionDef : functionDefinitionsOnList) {
146 				monitor.checkCancelled();
147 				println(functionDef.getPathName());
148 			}
149 		}
150 
151 		if (!functionsOnList.isEmpty()) {
152 			println();
153 			println("Updated functions:");
154 			for (Function functionOnList : functionsOnList) {
155 				monitor.checkCancelled();
156 				println(functionOnList.getEntryPoint().toString());
157 			}
158 		}
159 
160 	}
161 
162 
163 }
