16  //For an 8051 program, changes source of labels in SFR, BITS, and SFR-BITS address spaces to be 
17  // "imported."
18  //@category Update
19  
20  import ghidra.app.script.GhidraScript;
21  import ghidra.program.model.symbol.*;
22  
23  public class Update8051 extends GhidraScript {
24  
25  	private final static String SFR = "SFR";
26  	private final static String BITS = "BITS";
27  	private final static String SFR_BITS = "SFR-BITS";
28  	
29      @Override
30      public void run() throws Exception {
31      	if (currentProgram.getAddressFactory().getNumAddressSpaces() == 1) {
32      		println("Program is not an 8051");
33      		return;
34      	}
35  		SymbolTable st = currentProgram.getSymbolTable();
36  		SymbolIterator iter = st.getDefinedSymbols();
37  		int count = 0;
38  		while (iter.hasNext()) {
39  			Symbol symbol = iter.next();
40  			String spaceName = symbol.getAddress().getAddressSpace().getName();
41  			if (spaceName.equals(SFR) || spaceName.equals(BITS) || spaceName.equals(SFR_BITS)) {
42  				symbol.setSource(SourceType.IMPORTED);
43  				println("Changed source on " + symbol.getName());
44  				++count;
45  			}
46  		}
47  		if (count == 0) {
48  			println("No address spaces found for " +SFR + ", "+ BITS + ", " + SFR_BITS + ".");
49  		}
50  		else {
51  			String str = (count > 1 ? " symbols to update." : " symbol to update."); 
52  			println("Found " + count + str);
53  		}
54      }
55  
56  }
