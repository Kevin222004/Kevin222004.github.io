16  //This script applies labels and comments to the WallaceSrc.exe program for use with GhidraClass exercises 
17  //@category Training.GhidraClass
18  
19  
20  import ghidra.app.script.GhidraScript;
21  import ghidra.program.model.address.Address;
22  import ghidra.program.model.data.ArrayDataType;
23  import ghidra.program.model.data.BooleanDataType;
24  import ghidra.program.model.data.CharDataType;
25  import ghidra.program.model.data.IntegerDataType;
26  import ghidra.program.model.data.PointerDataType;
27  import ghidra.program.model.data.Structure;
28  import ghidra.program.model.data.StructureDataType;
29  import ghidra.program.model.listing.Function;
30  import ghidra.program.model.listing.Function.FunctionUpdateType;
31  import ghidra.program.model.listing.Parameter;
32  import ghidra.program.model.listing.ParameterImpl;
33  import ghidra.program.model.symbol.Namespace;
34  import ghidra.program.model.symbol.SourceType;
35  import ghidra.program.model.symbol.SymbolTable;
36  import ghidra.util.exception.InvalidInputException;
37  
38  public class MarkupWallaceSrcScript extends GhidraScript {
39  	
40  
41  	@Override
42  	public void run() throws Exception {
43  		
44  		if(!currentProgram.getName().contains("WallaceSrc") || (!currentProgram.getExecutableMD5().equals("2527c463a079c81af7b3bc1d26bd3b5d"))) {
45  			println("This script is only meant to work on the WallaceSrc executable with md5 hash 2527c463a079c81af7b3bc1d26bd3b5d.");
46  			return;
47  		}																							
48  		
49  		//Create Person structure
50  		Structure personStruct = new StructureDataType("Person", 0);
51  		personStruct.add(new IntegerDataType(), "id", "");		
52  		ArrayDataType adt = new ArrayDataType(new CharDataType(), 32, 1);
53  		personStruct.add(adt, "name", "");
54  		personStruct.add(new BooleanDataType(), "likesCheese", "");
55  		PointerDataType ptrPersonStruct = new PointerDataType(personStruct);
56  		personStruct.add(ptrPersonStruct, "next", "");
57  		
58  		//Create Gadget structure	
59  		Structure gadgetStruct = new StructureDataType("Gadget", 0);
60  		PointerDataType charPtr = new PointerDataType(new CharDataType());
61  		gadgetStruct.add(charPtr,"name","");
62  		gadgetStruct.add(new IntegerDataType(),"type", "");
63  		gadgetStruct.add(new BooleanDataType(), "deployed","");
64  		gadgetStruct.add(ptrPersonStruct, "workingOn","");
65  		
66  		//apply data types to function parameters, locals, and returns
67  		
68  		//Gadget::Gadget(Gadget * this, undefined4 param_1)
69  		Function gadgetFunction = getFunctionAt(toAddr(0x00411440));
70  		Parameter[] parameters = gadgetFunction.getParameters();
71  		parameters[0] = new ParameterImpl("this", new PointerDataType(gadgetStruct), currentProgram);		
72  		gadgetFunction.replaceParameters(FunctionUpdateType.DYNAMIC_STORAGE_ALL_PARAMS,true, SourceType.USER_DEFINED, parameters);
73  					
74  		//deployGadget - return type = Gadget * 
75  		Function deployGadgetFunction = getFunctionAt(toAddr(0x004118f0));
76  		deployGadgetFunction.setReturnType(new PointerDataType(gadgetStruct), SourceType.USER_DEFINED);
77  				
78  		//initializePeople(Person *) 
79  		Function initPeopleFunction = getFunctionAt(toAddr(0x004117c0));
80  		parameters = initPeopleFunction.getParameters();
81  		parameters[0] = new ParameterImpl("people", new PointerDataType(personStruct), currentProgram);
82  		initPeopleFunction.replaceParameters(FunctionUpdateType.DYNAMIC_STORAGE_ALL_PARAMS,true, SourceType.USER_DEFINED, parameters);		
83  		
84  		//use(Gadget *this, Person *person)
85  		Function useFunction = getFunctionAt(toAddr(0x00411570));
86  		parameters = useFunction.getParameters();
87  		parameters[0] = new ParameterImpl("this", new PointerDataType(gadgetStruct), currentProgram);	
88  		parameters[1] = new ParameterImpl("person", new PointerDataType(personStruct), currentProgram);	
89  		useFunction.replaceParameters(FunctionUpdateType.DYNAMIC_STORAGE_ALL_PARAMS,true, SourceType.USER_DEFINED, parameters);
90  		
91  		//addPerson(Person ** list, char * name)
92  		Function addPersonFunction = getFunctionAt(toAddr(0x00411860));
93  		parameters = addPersonFunction.getParameters();
94  		parameters[0] = new ParameterImpl("list", new PointerDataType(new PointerDataType(personStruct)), currentProgram);	
95  		parameters[1] = new ParameterImpl("name", new PointerDataType(new CharDataType()), currentProgram);	
96  		addPersonFunction.replaceParameters(FunctionUpdateType.DYNAMIC_STORAGE_ALL_PARAMS,true, SourceType.USER_DEFINED, parameters);
97  		
98  		//addPeople(Person ** list)
99  		Function addPeopleFunction = getFunctionAt(toAddr(0x00411700));
100 		parameters = addPeopleFunction.getParameters();
101 		parameters[0] = new ParameterImpl("list", new PointerDataType(new PointerDataType(personStruct)), currentProgram);	
102 		addPeopleFunction.replaceParameters(FunctionUpdateType.DYNAMIC_STORAGE_ALL_PARAMS,true, SourceType.USER_DEFINED, parameters);
103 		
104 		//print(Gadget * pGadget)
105 		Function printFunction = getFunctionAt(toAddr(0x004115d0));
106 		parameters = printFunction.getParameters();
107 		parameters[0] = new ParameterImpl("this", new PointerDataType(gadgetStruct), currentProgram);		
108 		printFunction.replaceParameters(FunctionUpdateType.DYNAMIC_STORAGE_ALL_PARAMS,true, SourceType.USER_DEFINED, parameters);
109 		
110 		// Create labels for some of the functions
111 		SymbolTable symbolTable = currentProgram.getSymbolTable();
112 
113 		
114 		//create the Class "Gadget" to put most function symbols in
115 		Namespace namespace = null;
116 		namespace = symbolTable.getNamespace("Gadget", null);
117 		if(namespace == null) {
118 			 namespace = symbolTable.createClass(null, "Gadget", SourceType.USER_DEFINED);
119 		}
120 				
121 		//Functions in Gadget class
122 		createNewLabel(toAddr(0x00411440), "Gadget", namespace, SourceType.USER_DEFINED);
123 		createNewLabel(toAddr(0x004115d0), "print", namespace, SourceType.USER_DEFINED);		
124 		createNewLabel(toAddr(0x00411570), "use", namespace, SourceType.USER_DEFINED);	
125 		
126 		//Functions not in class
127 		createNewLabel(toAddr(0x004117c0), "initializePeople", namespace, SourceType.USER_DEFINED);
128 		createNewLabel(toAddr(0x004118f0), "deployGadget", namespace, SourceType.USER_DEFINED);		
129 		createNewLabel(toAddr(0x00411700), "addPeople", namespace, SourceType.USER_DEFINED);
130 		createNewLabel(toAddr(0x00411860), "addPerson", namespace, SourceType.USER_DEFINED);
131 		createNewLabel(toAddr(0x00418138), "personList", namespace, SourceType.USER_DEFINED);
132 		createNewLabel(toAddr(0x00411a30), "main", null, SourceType.USER_DEFINED);		
133 		
134 		// Add other labels
135 		Function function = currentProgram.getFunctionManager().getFunctionAt(toAddr(0x004117c0));	
136 		createNewLabel(toAddr(0x004117e5), "LoopOverPeople", function, SourceType.USER_DEFINED);
137 		if(getSymbolAt(toAddr(0x00418138)).getSource().equals(SourceType.DEFAULT)){
138 			createLabel(toAddr(0x00418138),"personList", true);
139 		}
140 		
141 		// Add comments
142 	    setPlateComment(toAddr(0x00411440), "This is the init method for the Gadget class");
143 	    setPlateComment(toAddr(0x004115d0), "This method prints the status of a Person -- whether they are deployed or not and who they are deployed on. ");
144 	    setPlateComment(toAddr(0x00411700), "This function adds all the people to the Person list.");
145 	    setPlateComment(toAddr(0x004117c0), "This function initializes each person's record with whether or not they like cheese, their id, and a pointer to the next person.");
146 	    setPlateComment(toAddr(0x00411860), "This function adds a person to the Person list.");
147 	    setPlateComment(toAddr(0x004118f0), "This function checks to see if the person on the list is Wallace and if so, it deploys the Infrared Garden Gnome.");
148 	    setEOLComment(toAddr(0x004117e7), "Randomly assign whether each person likes cheese or not.");	
149 	}
150 	
151 	void createNewLabel(Address address, String name, Namespace namespace, SourceType sourceType) {
152 		SymbolTable symbolTable = currentProgram.getSymbolTable();
153 		if(getSymbolAt(address).getSource().equals(SourceType.DEFAULT)){
154 			try {
155 				symbolTable.createLabel(address, name, namespace, sourceType);
156 			} catch (InvalidInputException e) {
157 				println("Invalid input to create label.");
158 			}
159 		}
160 	}
161 }
