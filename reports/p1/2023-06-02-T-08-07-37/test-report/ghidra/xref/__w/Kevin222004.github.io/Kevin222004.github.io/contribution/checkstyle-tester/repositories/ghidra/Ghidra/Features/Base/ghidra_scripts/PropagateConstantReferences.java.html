16  //This script propagates constants in a function creating references wherever a store or load is
17  //found.  If a register has a value at the beginning of a function, that register value is assumed
18  //to be a constant.
19  //Any values loaded from memory are assumed to be constant.
20  //If a reference does not make sense on an operand, then it is added to the mnemonic.
21  //
22  //@category Analysis
23  
24  import ghidra.app.plugin.core.analysis.ConstantPropagationContextEvaluator;
25  import ghidra.app.script.GhidraScript;
26  import ghidra.program.model.address.Address;
27  import ghidra.program.model.address.AddressSet;
28  import ghidra.program.model.listing.Function;
29  import ghidra.program.model.listing.FunctionIterator;
30  import ghidra.program.util.ContextEvaluator;
31  import ghidra.program.util.SymbolicPropogator;
32  
33  public class PropagateConstantReferences extends GhidraScript {
34  
35  	@Override
36  	public void run() throws Exception {
37  		long numInstructions = currentProgram.getListing().getNumInstructions();
38  		monitor.initialize((int) (numInstructions));
39  		monitor.setMessage("Constant Propagation Markup");
40  
41  		// set up the address set to restrict processing
42  		AddressSet restrictedSet =
43  			new AddressSet(currentSelection);
44  		if (restrictedSet.isEmpty()) {
45  			Function curFunc =
46  				currentProgram.getFunctionManager().getFunctionContaining(
47  					currentLocation.getAddress());
48  			if (curFunc != null) {
49  				restrictedSet =
50  					new AddressSet(curFunc.getEntryPoint());
51  			}
52  			else {
53  				restrictedSet =
54  					new AddressSet(currentLocation.getAddress());
55  			}
56  
57  		}
58  
59  		// iterate over all functions within the restricted set
60  		FunctionIterator fiter =
61  			currentProgram.getFunctionManager().getFunctions(restrictedSet, true);
62  		while (fiter.hasNext()) {
63  			if (monitor.isCancelled()) {
64  				break;
65  			}
66  
67  			// get the function body
68  			Function func = fiter.next();
69  			Address start = func.getEntryPoint();
70  
71  			// follow all flows building up context
72  			// use context to fill out addresses on certain instructions 
73  			ContextEvaluator eval = new ConstantPropagationContextEvaluator(monitor, true);
74  
75  			SymbolicPropogator symEval = new SymbolicPropogator(currentProgram);
76  
77  			symEval.flowConstants(start, func.getBody(), eval, true, monitor);
78  		}
79  	}
80  
81  }
