16  // Shows how to find data types by name in data type managers other than the current program's, 
17  // which is how the <tt>GhidraScript.getDataTypes(String)</tt> works.
18  //@category Examples
19  
20  import ghidra.app.script.GhidraScript;
21  import ghidra.app.services.DataTypeManagerService;
22  import ghidra.framework.plugintool.PluginTool;
23  import ghidra.program.model.data.DataType;
24  import ghidra.program.model.data.DataTypeManager;
25  
26  import java.util.Iterator;
27  
28  public class FindDataTypeScript extends GhidraScript {
29  
30  	@Override
31  	protected void run() throws Exception {
32  		DataTypeManager manager = getDataTypeManagerByName("generic_C_lib");
33  		if (manager == null) {
34  			println("Archive must not be open in the Data Type Manager");
35  			return;
36  		}
37  
38  		DataType specificDataType = manager.getDataType("/complex.h/defines/define__COMPLEX_H");
39  		println("Data type: " + specificDataType);
40  
41  		//
42  		// without the category path
43  		//
44  		println("Now searching for any matching type in generic_C_lib:");
45  		Iterator<DataType> allDataTypes = manager.getAllDataTypes();
46  		while (allDataTypes.hasNext()) {
47  			DataType dataType = allDataTypes.next();
48  			String dataTypeName = dataType.getName();
49  			if (dataTypeName.indexOf("COMPLEX") != -1) {
50  				println("\tFound match: " + dataType);
51  			}
52  		}
53  		println("Done searching");
54  
55  		//
56  		// In the built-in DTM		
57  		//
58  		manager = getDataTypeManagerByName("BuiltInTypes");
59  		println("Now searching for any matching type in BuiltInTypes:");
60  		allDataTypes = manager.getAllDataTypes();
61  		while (allDataTypes.hasNext()) {
62  			DataType dataType = allDataTypes.next();
63  			String dataTypeName = dataType.getName();
64  			if (dataTypeName.indexOf("sdword") != -1) {
65  				println("\tFound match: " + dataType);
66  			}
67  		}
68  		println("Done searching");
69  	}
70  
71  	private DataTypeManager getDataTypeManagerByName(String name) {
72  		PluginTool tool = state.getTool();
73  		DataTypeManagerService service = tool.getService(DataTypeManagerService.class);
74  		DataTypeManager[] dataTypeManagers = service.getDataTypeManagers();
75  		for (DataTypeManager manager : dataTypeManagers) {
76  			String managerName = manager.getName();
77  			if (name.equals(managerName)) {
78  				return manager;
79  			}
80  		}
81  		return null;
82  	}
83  }
