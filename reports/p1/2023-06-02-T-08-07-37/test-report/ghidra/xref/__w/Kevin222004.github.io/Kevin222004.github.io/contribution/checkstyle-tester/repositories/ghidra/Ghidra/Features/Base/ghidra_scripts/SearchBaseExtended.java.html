87  	public void loadSelectedInstructions() {
88  		if (currentProgram == null || currentSelection == null) {
89  			return;
90  		}
91  
92  		try {
93  			//Builds object that is used to extract all the instructions masks and values
94  			SleighDebugLogger logger = null;
95  
96  			//Grabs the AddressRange for the first continuous selection of instructions
97  			AddressRange addrRange = currentSelection.getFirstRange();
98  			if (addrRange == null) {
99  				return;
100 			}//makes sure that something was returned
101 
102 			//Creates a list that is used to determine the location of the instructions starting byte
103 			Listing list = currentProgram.getListing();
104 
105 			Address tempAddr = addrRange.getMinAddress();//sets the beginning scan address
106 
107 			/*
108 			 * The purpose of this while loop is to iterate through the memory that is currently selected by user. 
109 			 * All the instructions within this selection range are extracted and corresponding masks made from them.
110 			 * These masks will then be used to search through memory to find instructions of the same type.
111 			 */
112 			while (tempAddr.compareTo(addrRange.getMaxAddress()) <= 0) {
113 
114 				//Determines if current location is start of instruction, tempIns will be null if not beginning of instruction
115 				Instruction tempIns = list.getInstructionAt(tempAddr);
116 
117 				if (tempIns != null) {//means that current address is start of an instruction
118 
119 					logger =
120 						new SleighDebugLogger(currentProgram, tempAddr, SleighDebugMode.VERBOSE);
121 					if (logger.parseFailed()) {
122 						break;
123 					}
124 
125 					//This takes care of the headers
126 					byte[] mask = logger.getInstructionMask();
127 					byte[] value = logger.getMaskedBytes(mask);
128 					if (mask == null || value == null) {
129 						break;
130 					}//need to move on to next byte, not sure if this is right command to use
131 
132 					//Builds a structure representing the recently found instruction mask and value
133 					Case tCase = new Case();
134 					tCase.mask = mask;
135 					tCase.value = value;
136 					tCase.textRep = tempIns.getMnemonicString();
137 					mnemonics.add(tCase); //adds the mnemonic mask and value to the arraylist
138 
139 					//Gets a code unit which can be used to determine if the operands are constants.
140 					CodeUnit cu = list.getCodeUnitAt(tempAddr);
141 
142 					//Iterates through all the operands for the currently selected instruction and stores them accordingly
143 					for (int x = 1; x <= logger.getNumOperands(); x++) {
144 						mask = logger.getOperandValueMask(x - 1);
145 						value = logger.getMaskedBytes(mask);
146 
147 						if (mask == null || value == null) {
148 							break;
149 						}//move on to next instruction
150 
151 						//Builds case to store the operands mask and value
152 						OperandCase otCase = new OperandCase();
153 						otCase.mask = mask;
154 						otCase.value = value;
155 						//Object hey = tempIns.getDefaultOperandRepresentationList(x-1);
156 						otCase.textRep = tempIns.getDefaultOperandRepresentation(x - 1);
157 
158 						//Determines if the given operand is a constant value. If it is a constant then proper flag is set.
159 						if (cu.getScalar(x - 1) != null) {
160 							otCase.constant = true;
161 						}
162 
163 						//Determines if current structure is large enough to hold new operand, if it isn't increases structure size
164 						if (ops.size() < x && ops.size() > -1) {
165 							ops.add(new LinkedHashMap<Case, OperandCase>());
166 						}
167 
168 						//Adds the operand to the data-structure with a mapping to the instruction mnemonic extracted earlier in line 87
169 						ops.get(x - 1).put(tCase, otCase);
170 					}
171 					//Increments the address pointer to point to the beginning of next instruction.
172 					tempAddr = tempAddr.add(tempIns.getLength());
173 				}
174 				else {
175 
176 					//Increments the address pointer by 1
177 					//This would be hit if the tempAddr didn't get offset correctly.
178 					tempAddr.addWrap(1);
179 				}
180 			}
181 		}
182 		catch (Exception e) {
183 			println(e.getMessage());
184 		}
185 	}

187 	//performs the application of filters and search instructions for matches. Doesn't load instructions or fill the controlList object.
188 	//Must have mnemonics, ops and controlList populated prior to running this method.
189 	public void executeSearch() {
190 
191 		//Applies the filters provided through the controlList object to the instructions provided through the mnemonics and ops structures.
192 		MaskValueCase finalSearchString = getFinalMaskAndValue(mnemonics, ops, controlList);
193 
194 		String valueString = new String();
195 		String maskString = new String();
196 
197 		for (byte element : finalSearchString.value) {
198 			valueString = valueString.concat(toHexString(element, true, false)) + " ";
199 		}
200 
201 		for (byte element : finalSearchString.mask) {
202 			maskString = maskString.concat(toHexString(element, true, false)) + " ";
203 		}
204 
205 		printf("Final Search Bytes: ");
206 		println(valueString);
207 		printf("Final Search Mask: ");
208 		println(maskString);
209 
210 		//Searches memory for matches to provided instructions with given filters. db structure is populated with the results.
211 		findLocations(finalSearchString, db);
212 
213 		//Displays results in a table.
214 		Address[] tableArray = new Address[db.size()];
215 		for (int x = 0; x < db.size(); x++) {
216 			tableArray[x] = db.get(x).addr;
217 		}
218 
219 		try {
220 			show(tableArray);
221 		}
222 		catch (ImproperUseException iue) {
223 			// Do nothing; this code should only be run headed, anyway
224 		}
225 	}

227 	/*
228 	 * Builds the mask and value byte streams for a single instruction that is represented by the mnemonic entered into the first parameter.
229 	 * 
230 	 * That mnemonic is used as a key to withdrawal the operands from the data-structure and the SLMaskControl contains the filter information to be
231 	 * applied to the instruction.
232 	 */
233 	private MaskValueCase buildSingleInstructionMask(Case mnemonic,
234 			ArrayList<LinkedHashMap<Case, OperandCase>> localOperands, SLMaskControl localState) {
235 
236 		byte[] tempMask = new byte[mnemonic.mask.length];
237 		byte[] tempValue = new byte[mnemonic.value.length];
238 
239 		//Applies the mnemonic's value and mask if needed
240 		if (localState.useMnemonic) {
241 			tempValue = byteArrayOr(tempValue, mnemonic.value);
242 			tempMask = byteArrayOr(tempMask, mnemonic.mask);
243 		}
244 		//Applies the first operands mask and value to running stream if needed
245 		if (localState.useOp1 && localOperands.size() >= 1 &&
246 			localOperands.get(0).get(mnemonic) != null) {
247 
248 			OperandCase temp = localOperands.get(0).get(mnemonic);
249 
250 			if (localState.useConst || !temp.constant) {
251 
252 				if (temp.value != null && temp.mask != null && tempValue != null &&
253 					tempMask != null) {
254 					tempValue = byteArrayOr(tempValue, temp.value);
255 					tempMask = byteArrayOr(tempMask, temp.mask);
256 				}
257 			}
258 		}
259 		//Applies the second operands mask and value to running stream if needed
260 		if (localState.useOp2 && localOperands.size() >= 2 &&
261 			localOperands.get(1).get(mnemonic) != null) {
262 
263 			OperandCase temp = localOperands.get(1).get(mnemonic);
264 
265 			if (localState.useConst || !temp.constant) {
266 
267 				if (temp.value != null && temp.mask != null && tempValue != null &&
268 					tempMask != null) {
269 					tempValue = byteArrayOr(tempValue, temp.value);
270 					tempMask = byteArrayOr(tempMask, temp.mask);
271 				}
272 			}
273 		}
274 
275 		MaskValueCase tempCase = new MaskValueCase();
276 		tempCase.mask = tempMask;
277 		tempCase.value = tempValue;
278 
279 		return tempCase;
280 	}

282 	/*
283 	 * Combines all the masks in the data-structure together into a single byte stream. This stream is used to search memory.
284 	 */
285 	private MaskValueCase combineInstructionMasks(ArrayList<byte[]> masks, ArrayList<byte[]> values,
286 			int length) {
287 
288 		if (masks.size() != values.size()) {
289 			throw new IllegalArgumentException();
290 		}
291 
292 		int index = 0;
293 
294 		/*
295 		 * This portion of code takes the masks and values from each of the different commands and concatenates them together to form
296 		 * a single mask and value byte array. This byte array is then used to scan memory to look for other sequences of commands.
297 		 */
298 		byte[] finalValueArray = new byte[length];
299 		byte[] finalMaskArray = new byte[length];
300 		for (int x = 0; x < values.size(); x++) { //iterates through each mnemonic
301 
302 			for (int i = 0; i < values.get(x).length && index < length; i++) { //iterate through the mnemonics bytes
303 				finalValueArray[index] = values.get(x)[i];
304 				finalMaskArray[index] = masks.get(x)[i];
305 				index++;
306 			}
307 		}
308 
309 		MaskValueCase tempFinalParams = new MaskValueCase();
310 		tempFinalParams.mask = finalMaskArray;
311 		tempFinalParams.value = finalValueArray;
312 
313 		return tempFinalParams;
314 	}

316 	/*
317 	 * Calculates the final mask and value byte streams. These streams are then used to search through memory to look for similar 
318 	 * assembly instructions.
319 	 */
320 	private MaskValueCase getFinalMaskAndValue(ArrayList<Case> privateMnemonics,
321 			ArrayList<LinkedHashMap<Case, OperandCase>> localOperands,
322 			ArrayList<SLMaskControl> control) {
323 
324 		ArrayList<byte[]> masks = new ArrayList<>();
325 		ArrayList<byte[]> values = new ArrayList<>();
326 
327 		//used for storing the byte stream currently being work on prior to being added to final data structure
328 		int totalLength = 0;
329 
330 		boolean continueFilter = true;//controls whether the last mask given will be applied on the rest of the mnemonics if the controlList structure is smaller than mnemonics
331 
332 		if (privateMnemonics == null || localOperands == null || control == null) {
333 			throw new IllegalArgumentException("Null Structure");
334 		}
335 
336 		//TODO Need to do some thorough checking on values entered into the control and mnemonics
337 		//TODO Need to make sure that all possible cases of unequal amounts are considered.
338 
339 		/*
340 		 * This loop will scan through the mnemonics and apply the provided MaskControls.
341 		 * 
342 		 * The masks will be applied from the arraylist until there aren't any remaining in the list. At this point
343 		 * a mask of all falses will be applied. This will in effect ignore rest of the mnemonics
344 		 * 
345 		 * TODO Apply a switch that will allow the last mask in the array to be applied to rest of the mnemonics within selection.
346 		 */
347 		for (int x = 0; x < privateMnemonics.size(); x++) {
348 
349 			MaskValueCase result = new MaskValueCase();
350 
351 			if (x < control.size()) {
352 				//need to apply the filter from control		
353 				result = buildSingleInstructionMask(privateMnemonics.get(x), localOperands,
354 					control.get(x));
355 			}
356 			else if (continueFilter) {
357 				//apply the last available mask in the control structure
358 				result = buildSingleInstructionMask(privateMnemonics.get(x), localOperands,
359 					control.get(control.size() - 1));
360 			}
361 			else {
362 				//apply an empty filter basically ignoring
363 				result = buildSingleInstructionMask(privateMnemonics.get(x), localOperands,
364 					new SLMaskControl(false, false, false, false));
365 			}
366 
367 			if (result.value.length == result.mask.length) {
368 				masks.add(result.mask);
369 				values.add(result.value);
370 				totalLength += result.value.length;
371 			}
372 		}
373 
374 		return combineInstructionMasks(masks, values, totalLength);
375 	}

377 	/*
378 	 * Populates the database with the locations where the specified byte arrays are found.
379 	 * The first parameter should only have two elements. Element 0 should contain mask and element 1 
380 	 * contain the value.
381 	 * 
382 	 * The second parameter gets populated with the results of the search.
383 	 */
384 	private void findLocations(MaskValueCase searchArrays, ArrayList<Case> localDatabase) {
385 
386 		if (currentProgram == null || localDatabase == null || searchArrays == null) {
387 			throw new IllegalArgumentException("Null Data-Structure");
388 		}
389 		if (searchArrays.mask.length != searchArrays.value.length) {
390 			throw new IllegalArgumentException("Mask and value lengths are different.");
391 		}
392 
393 		if (containsOnBit(searchArrays.mask)) {
394 			Memory mem = currentProgram.getMemory();
395 
396 			//Gets the start and end address to search through
397 			Address endAddress = currentProgram.getMaxAddress();
398 
399 			Address currentPosition = currentProgram.getMinAddress();
400 			while (currentPosition.compareTo(endAddress) < 0) {
401 
402 				//Searches memory for the given mask and value.
403 				currentPosition = mem.findBytes(currentPosition, endAddress, searchArrays.value,
404 					searchArrays.mask, true, monitor);
405 
406 				//Determines if a new location was found.
407 				if (currentPosition == null) {
408 					break;
409 				}
410 
411 				Case temp = new Case();
412 				temp.mask = searchArrays.mask;
413 				temp.value = searchArrays.value;
414 				temp.addr = currentPosition;
415 				localDatabase.add(temp);
416 
417 				currentPosition = currentPosition.add(1);
418 			}
419 		}
420 		else {
421 			return;
422 		}
423 	}

425 	/*
426 	 * Used for determining if there is a "On" bit in a byte stream.
427 	 * 
428 	 * This is necessary because if you do a memory search for a mask with no "on" bit then it will return every memory address.
429 	 */
430 	private boolean containsOnBit(byte[] array) {
431 
432 		for (byte element : array) {
433 			Byte temp = Byte.valueOf(element);
434 			int value = temp.intValue();
435 			if (value != 0) {
436 				return true;
437 			}
438 		}
439 
440 		return false;
441 	}

443 	/*
444 	 * Takes two arrays of bytes and performs a bitwise or operation and returns the result. Returns null if the arguments aren't of same length.
445 	 */
446 	private byte[] byteArrayOr(byte[] arr1, byte[] arr2) {
447 		byte[] result = new byte[arr1.length];
448 
449 		if (arr1.length != arr2.length) {
450 			return null;
451 		}
452 
453 		for (int x = 0; x < arr1.length; x++) {
454 			result[x] = (byte) (arr1[x] | arr2[x]);
455 		}
456 
457 		return result;
458 	}
