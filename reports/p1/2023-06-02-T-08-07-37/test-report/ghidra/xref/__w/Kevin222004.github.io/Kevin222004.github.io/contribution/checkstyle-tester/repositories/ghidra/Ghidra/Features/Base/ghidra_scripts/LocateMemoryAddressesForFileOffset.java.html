17  //This example script locates a memory address for a file offset.  
18  //Prompt user for a file offset.
19  //Print the associated memory address to the Ghidra console
20  //Print the file offset as a Ghidra comment at the memory address in the Ghidra Listing
21  //If multiple addresses are located, then print the addresses to the console (do not set a 
22  //Ghidra comment)
23  //@category Examples
24  
25  import java.util.List;
26  
27  import ghidra.app.script.GhidraScript;
28  import ghidra.program.model.address.Address;
29  import ghidra.program.model.listing.CodeUnit;
30  import ghidra.program.model.mem.Memory;
31  import ghidra.util.exception.CancelledException;
32  
33  public class LocateMemoryAddressesForFileOffset extends GhidraScript {
34  
35  	@Override
36  	public void run() throws Exception {
37  		long myFileOffset = getFileOffset();
38  		Memory mem = currentProgram.getMemory();
39  		List<Address> addressList = mem.locateAddressesForFileOffset(myFileOffset);
40  		if (addressList.isEmpty()) {
41  			println("No memory address found for: " + Long.toHexString(myFileOffset));
42  		}
43  		else if (addressList.size() == 1) {
44  			Address address = addressList.get(0);
45  			processAddress(address, mem.getBlock(address).getName(), myFileOffset);
46  
47  		}
48  		//address set size is > 1, file offset matches to multiple addresses.  
49  		//Let the user decide which address they want.
50  		else {
51  			println("Possible memory block:address are:");
52  			for (Address addr : addressList) {
53  				println(mem.getBlock(addr).getName() + ":" + addr.toString());
54  			}
55  		}
56  	}
57  
58  	public long getFileOffset()
59  			throws CancelledException, NumberFormatException, IllegalArgumentException {
60  		String userFileOffset =
61  			askString("File offset", "Please provide a hexadecimal file offset");
62  		long myFileOffset = 0;
63  		myFileOffset = Long.parseLong(userFileOffset, 16);
64  		if (myFileOffset < 0) {
65  			throw new IllegalArgumentException(
66  				"Offset cannot be a negative value." + userFileOffset);
67  		}
68  		return myFileOffset;
69  	}
70  
71  	public void processAddress(Address addr, String memBlockName, long fileOffset) {
72  		println("File offset " + Long.toHexString(fileOffset) +
73  			" is associated with memory block:address " + memBlockName + ":" + addr.toString());
74  		CodeUnit myCodeUnit = currentProgram.getListing().getCodeUnitContaining(addr);
75  		String comment = myCodeUnit.getComment(0);
76  		if (comment == null) {
77  			myCodeUnit.setComment(0,
78  				this.getScriptName() + ": File offset: " + Long.toHexString(fileOffset) +
79  					", Memory block:address " + memBlockName + ":" + addr.toString());
80  		}
81  		else {
82  			myCodeUnit.setComment(0,
83  				comment + ", " + this.getScriptName() + ": File offset: " +
84  					Long.toHexString(fileOffset) + ", Memory block:address " + memBlockName + ":" +
85  					addr.toString());
86  		}
87  	}
88  }
