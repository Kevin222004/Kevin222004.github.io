16  //
17  // Compare 2 file data type archives (.gdt). Output where data types exist only in one archive.
18  // Also indicate where the same named data type has a different type. This doesn't indicate
19  // where the internal contents of the same named data types differ.
20  //
21  //@category Data Types
22  
23  import java.io.File;
24  import java.io.PrintWriter;
25  import java.util.Iterator;
26  
27  import ghidra.app.script.GhidraScript;
28  import ghidra.program.model.data.*;
29  import ghidra.program.model.data.StandAloneDataTypeManager.ArchiveWarning;
30  import ghidra.util.UniversalID;
31  
32  public class CompareGDTs extends GhidraScript {
33  
34  	private File firstFile;
35  	private File secondFile;
36  	private File outputFile;
37  	private FileDataTypeManager firstArchive;
38  	private FileDataTypeManager secondArchive;
39  	private PrintWriter printWriter;
40  	boolean matchByName;
41  	boolean checkPointers;
42  	boolean checkArrays;
43  
44  	@Override
45  	protected void run() throws Exception {
46  		firstFile = askFile("Select First GDT File", "Select 1st");
47  		secondFile = askFile("Select Second GDT File", "Select 2nd");
48  		outputFile = askFile("Select Output File", "Select output file");
49  		if (outputFile.exists()) {
50  			boolean overwrite = askYesNo("Overwrite existing output file?",
51  				"The specified output file already exists. \nDo you want to overwrite it?");
52  			if (!overwrite) {
53  				println("Output file " + outputFile.getAbsolutePath() +
54  					" already exists. User aborted...");
55  				return;
56  			}
57  		}
58  
59  		firstArchive = openDataTypeArchive(firstFile, false);
60  		if (firstArchive.getWarning() != ArchiveWarning.NONE) {
61  			popup(
62  				"An architecture language error occured while opening archive (see log for details)\n" +
63  					firstFile.getPath());
64  			return;
65  		}
66  
67  		secondArchive = openDataTypeArchive(secondFile, false);
68  		if (secondArchive.getWarning() != ArchiveWarning.NONE) {
69  			popup(
70  				"An architecture language error occured while opening archive (see log for details)\n" +
71  					secondFile.getPath());
72  			firstArchive.close();
73  			return;
74  		}
75  
76  		matchByName = askYesNo("Match Data Types By Path Name?",
77  			"Do you want to match data types by their path names (rather than by Universal ID)?");
78  		checkPointers = askYesNo("Check Pointers?", "Do you want to check Pointers?");
79  		checkArrays = askYesNo("Check Arrays?", "Do you want to check Arrays?");
80  
81  		printWriter = new PrintWriter(outputFile);
82  		try {
83  			compareDataTypes();
84  		}
85  		finally {
86  			printWriter.close();
87  			firstArchive.close();
88  			secondArchive.close();
89  		}
90  	}
91  
92  	private void output(String message) {
93  		printWriter.println(message);
94  	}
95  
96  	private void compareDataTypes() {
97  
98  		output("\nComparing " + firstFile.getAbsolutePath() + "\n        & " +
99  			secondFile.getAbsolutePath() + ".");
100 
101 		output("\nThe following data types are only in " + firstFile.getAbsolutePath() + ".");
102 		long onlyInFirst = outputEachDataTypeOnlyInFirst(firstArchive, secondArchive);
103 		output(onlyInFirst + " data types that were only in first archive.");
104 
105 		output("\nThe following data types are only in " + secondFile.getAbsolutePath() + ".");
106 		long onlyInSecond = outputEachDataTypeOnlyInFirst(secondArchive, firstArchive);
107 		output(onlyInSecond + " data types that were only in second archive.");
108 
109 		output("\nThe following are different kinds of data types.");
110 		long differentKinds = outputWhereTypesDiffer(firstArchive, secondArchive);
111 		output(differentKinds + " data types had different types.");
112 
113 		output("\nThe following data types are defined differently.");
114 		long differentDefinitions = outputWhereDefinitionsDiffer(firstArchive, secondArchive);
115 		output(differentDefinitions + " data types had different definitions.");
116 
117 		output("\nThe following data types are different sizes");
118 		long differentSizes = outputWhereSizesDiffer(firstArchive, secondArchive);
119 		output(differentSizes + " data types had different sizes.");
120 
121 		output("\n");
122 	}
123 
124 	private long outputEachDataTypeOnlyInFirst(FileDataTypeManager dtmArchive1,
125 			FileDataTypeManager dtmArchive2) {
126 
127 		long missingCount = 0;
128 		Iterator<DataType> allDataTypes = dtmArchive1.getAllDataTypes();
129 		while (allDataTypes.hasNext()) {
130 			DataType dataType = allDataTypes.next();
131 			if (outputIfMissingDataType(dataType, dtmArchive2)) {
132 				missingCount++;
133 			}
134 		}
135 		return missingCount;
136 	}
137 
138 	private boolean outputIfMissingDataType(DataType dataType, FileDataTypeManager dtmArchive) {
139 
140 		if (!checkPointers && dataType instanceof Pointer) {
141 			return false;
142 		}
143 
144 		if (!checkArrays && dataType instanceof Array) {
145 			return false;
146 		}
147 
148 		DataType matchingDataType = getMatchingDataType(dataType, dtmArchive);
149 		if (matchingDataType == null) {
150 			String pathName = dataType.getPathName();
151 			output(pathName);
152 			return true;
153 		}
154 		return false;
155 	}
156 
157 	private DataType getMatchingDataType(DataType dataType, FileDataTypeManager dtmArchive) {
158 
159 		if (!matchByName) {
160 			UniversalID universalID = dataType.getUniversalID();
161 			if (universalID != null) {
162 				return dtmArchive.findDataTypeForID(universalID);
163 			}
164 		}
165 		return dtmArchive.getDataType(dataType.getCategoryPath(), dataType.getName());
166 	}
167 
168 	private long outputWhereTypesDiffer(FileDataTypeManager dtmArchive1,
169 			FileDataTypeManager dtmArchive2) {
170 
171 		long differCount = 0;
172 		Iterator<DataType> allDataTypes = dtmArchive1.getAllDataTypes();
173 		while (allDataTypes.hasNext()) {
174 			DataType dataType = allDataTypes.next();
175 			if (outputIfDifferentTypes(dataType, dtmArchive2)) {
176 				differCount++;
177 			}
178 		}
179 		return differCount;
180 	}
181 
182 	private long outputWhereSizesDiffer(FileDataTypeManager dtmArchive1,
183 			FileDataTypeManager dtmArchive2) {
184 
185 		long differCount = 0;
186 		Iterator<DataType> allDataTypes = dtmArchive1.getAllDataTypes();
187 		while (allDataTypes.hasNext()) {
188 			DataType dataType = allDataTypes.next();
189 			if (outputIfDifferentSizes(dataType, dtmArchive2)) {
190 				differCount++;
191 			}
192 		}
193 		return differCount;
194 	}
195 
196 	private boolean outputIfDifferentTypes(DataType dataType, FileDataTypeManager dtmArchive) {
197 
198 		if (!checkPointers && dataType instanceof Pointer) {
199 			return false;
200 		}
201 
202 		if (!checkArrays && dataType instanceof Array) {
203 			return false;
204 		}
205 
206 		DataType matchingDataType = getMatchingDataType(dataType, dtmArchive);
207 		if (matchingDataType != null) {
208 			Class<?> dtClass = dataType.getClass();
209 			Class<?> sameNamedDtClass = matchingDataType.getClass();
210 			if (dtClass != sameNamedDtClass) {
211 				String message = dataType.getPathName() + "   (" + dtClass.getSimpleName() + ") " +
212 					"   vs   (" + sameNamedDtClass.getClass() + ")";
213 				output(message);
214 				return true;
215 			}
216 		}
217 		return false;
218 	}
219 
220 	private long outputWhereDefinitionsDiffer(FileDataTypeManager dtmArchive1,
221 			FileDataTypeManager dtmArchive2) {
222 
223 		long differCount = 0;
224 		Iterator<DataType> allDataTypes = dtmArchive1.getAllDataTypes();
225 		while (allDataTypes.hasNext()) {
226 			DataType dataType = allDataTypes.next();
227 			if (outputIfDifferentDefinitions(dataType, dtmArchive2)) {
228 				differCount++;
229 			}
230 		}
231 		return differCount;
232 	}
233 
234 	private boolean outputIfDifferentDefinitions(DataType dataType,
235 			FileDataTypeManager dtmArchive) {
236 
237 		if (!checkPointers && dataType instanceof Pointer) {
238 			return false;
239 		}
240 
241 		if (!checkArrays && dataType instanceof Array) {
242 			return false;
243 		}
244 
245 		DataType matchingDataType = getMatchingDataType(dataType, dtmArchive);
246 		if (matchingDataType != null) {
247 			Class<?> dtClass = dataType.getClass();
248 			Class<?> sameNamedDtClass = matchingDataType.getClass();
249 			if (dtClass == sameNamedDtClass) {
250 				if (!dataType.isEquivalent(matchingDataType)) {
251 					String message =
252 						dataType.getPathName() + "   (" + dtClass.getSimpleName() + ")";
253 					output(message);
254 					return true;
255 				}
256 			}
257 		}
258 		return false;
259 	}
260 
261 	private boolean outputIfDifferentSizes(DataType dataType, FileDataTypeManager dtmArchive) {
262 
263 		if (!checkPointers && dataType instanceof Pointer) {
264 			return false;
265 		}
266 
267 		if (!checkArrays && dataType instanceof Array) {
268 			return false;
269 		}
270 
271 		DataType matchingDataType = getMatchingDataType(dataType, dtmArchive);
272 		if (matchingDataType != null) {
273 			Class<?> dtClass = dataType.getClass();
274 			Class<?> sameNamedDtClass = matchingDataType.getClass();
275 			if (dtClass == sameNamedDtClass) {
276 				if (dataType.getLength() != matchingDataType.getLength()) {
277 					String message = dataType.getPathName() + "   (" + dtClass.getSimpleName() +
278 						") " + dataType.getLength() + " !=  " + matchingDataType.getLength();
279 					output(message);
280 					return true;
281 				}
282 			}
283 		}
284 		return false;
285 	}
