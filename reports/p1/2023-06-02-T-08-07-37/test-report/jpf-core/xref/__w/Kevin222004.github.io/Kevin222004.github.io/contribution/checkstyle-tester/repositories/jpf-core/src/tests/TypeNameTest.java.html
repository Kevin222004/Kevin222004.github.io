18  /**
19   * unfortunately this has to be in the unnamed package, since we need
20   * B to be in the unnamed package (so that it has a builtin type name), and as
21   * of Java 1.4 you can't import unnamed package classes into named packages
22   */
23  import gov.nasa.jpf.util.test.TestJPF;
24  
25  import org.junit.Test;
26  
27  /*
28   * just a helper class for the JavaLangObject raw test that happens to be
29   * named like a builtin typecode (byte)
30   */
31  class B {
32  
33    int data;
34  
35    public B(int d) {
36      data = d;
37    }
38  
39    @Override
40    public boolean equals(Object other) {
41      if (other == null || !(other instanceof B)) {
42        return false;
43      }
44  
45      return ((B) other).data == data;
46    }
47  
48    @Override
49    public String toString() {
50      return "B {data=" + data + "}";
51    }
52  }
53  
54  public class TypeNameTest extends TestJPF {
55  
56    @Test
57    public void testArrayCloning() {
58      if (verifyNoPropertyViolation()) {
59        // test for collisions between typecodes of builtin types
60        // and user defined classes (e.g. "B" for byte)
61        B[] b = new B[10];
62        b[3] = new B(42);
63  
64        Object o = b.clone();
65        B[] bb = (B[]) o;
66        assert b[3].equals(bb[3]);
67  
68        byte[] a = new byte[10];
69        a[3] = 42;
70        o = a.clone();
71        byte[] aa = (byte[]) o;
72        assert a[3] == aa[3];
73      }
74    }
75  }
