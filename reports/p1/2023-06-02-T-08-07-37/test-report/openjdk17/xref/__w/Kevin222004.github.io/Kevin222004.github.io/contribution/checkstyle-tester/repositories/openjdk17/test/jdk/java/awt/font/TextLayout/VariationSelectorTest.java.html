1   /* @test
2    * @summary Verify two identical 'a's are rendered
3    * @bug 8187100
4    * @ignore Requires a special font installed.
5    */
6   import javax.swing.JFrame;
7   import javax.swing.JComponent;
8   import javax.swing.SwingUtilities;
9   import javax.swing.WindowConstants;
10  import java.awt.Font;
11  import java.awt.Graphics;
12  import java.awt.Graphics2D;
13  import java.awt.font.FontRenderContext;
14  import java.awt.font.GlyphVector;
15  
16  public class VariationSelectorTest {
17      // A font supporting Unicode variation selectors is required
18      // At least DejaVu 2.20 from 2007
19      private static final Font FONT = new Font("DejaVu Sans", Font.PLAIN, 12);
20  
21      public static void main(String[] args) {
22          final String fontName = FONT.getFontName();
23          if (!fontName.equals("DejaVuSans")) {
24              System.err.println("*** Warning: Font DejaVuSans not installed.");
25              System.err.println("*** Using font: " + fontName);
26          }
27          SwingUtilities.invokeLater(() -> {
28              JFrame frame = new JFrame();
29              frame.setDefaultCloseOperation(WindowConstants.DISPOSE_ON_CLOSE);
30              frame.add(new MyComponent());
31              frame.setSize(200, 200);
32              frame.setVisible(true);
33              frame.setLocationRelativeTo(null);
34          });
35      }
36  
37      private static class MyComponent extends JComponent {
38          @Override
39          protected void paintComponent(Graphics g) {
40              Graphics2D g2d = (Graphics2D) g;
41              FontRenderContext frc = g2d.getFontRenderContext();
42              String text = "a";
43              GlyphVector gv = FONT.layoutGlyphVector(
44                                   frc, text.toCharArray(), 0, text.length(),
45                                   Font.LAYOUT_LEFT_TO_RIGHT);
46              System.out.println("'a'=" + gv.getNumGlyphs());
47              g2d.drawString("=" + gv.getNumGlyphs() + " ('a')", 100, 50);
48              g2d.drawGlyphVector(gv, 80, 50);
49              String text2 = "a\ufe00";
50              GlyphVector gv2 = FONT.layoutGlyphVector(
51                                   frc, text2.toCharArray(), 0, text2.length(),
52                                   Font.LAYOUT_LEFT_TO_RIGHT);
53              g2d.drawGlyphVector(gv2, 80, 100);
54              System.out.println("'a'+VS=" + gv2.getNumGlyphs());
55              g2d.drawString("=" + gv2.getNumGlyphs() + " ('a'+VS)", 100, 100);
56              if ((gv.getNumGlyphs() == 1) && (gv2.getNumGlyphs() == 1)) {
57                  System.out.println("PASS");
58                  g2d.drawString("PASS", 10, 15);
59              } else {
60                  System.err.println("FAIL");
61                  g2d.drawString("FAIL", 10, 15);
62              }
63          }
64      }
65  }
