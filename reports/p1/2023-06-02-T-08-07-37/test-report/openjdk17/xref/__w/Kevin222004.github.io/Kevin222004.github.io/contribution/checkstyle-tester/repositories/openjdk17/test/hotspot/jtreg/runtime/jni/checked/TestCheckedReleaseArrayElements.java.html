24  /* @test
25   * @bug 8258077
26   * @summary verify multiple release calls on a copied array work when checked
27   * @library /test/lib
28   * @run main/native TestCheckedReleaseArrayElements launch
29   */
30  
31  import java.util.Arrays;
32  import jdk.test.lib.process.ProcessTools;
33  import jdk.test.lib.process.OutputAnalyzer;
34  import jdk.test.lib.Utils;
35  import jtreg.SkippedException;
36  
37  public class TestCheckedReleaseArrayElements {
38  
39      static {
40          System.loadLibrary("TestCheckedReleaseArrayElements");
41      }
42  
43      public static void main(String[] args) throws Throwable {
44          if (args == null || args.length == 0) {
45              test();
46          } else {
47              // Uses executeProcess() instead of executeTestJvm() to avoid passing options
48              // that might generate output on stderr (which should be empty for this test).
49              ProcessBuilder pb =
50                  ProcessTools.createJavaProcessBuilder("-Xcheck:jni",
51                                                        "-Djava.library.path=" + Utils.TEST_NATIVE_PATH,
52                                                        "TestCheckedReleaseArrayElements");
53              OutputAnalyzer output = ProcessTools.executeProcess(pb);
54              output.shouldHaveExitValue(0);
55              output.stderrShouldBeEmpty();
56              output.stdoutShouldNotBeEmpty();
57          }
58      }
59  
60      /*
61       * If GetIntArrayElements returns a copy, we update the array in slices
62       * calling ReleaseIntArrayElements with JNI_COMMIT to write-back the
63       * updates, which are then checked. Finally we use JNI_ABORT to free
64       * the copy.
65       */
66      public static void test() {
67          final int slices = 3;
68          final int sliceLength = 3;
69          int[] arr = new int[slices * sliceLength];
70  
71          if (!init(arr)) {
72              throw new SkippedException("Test skipped as GetIntArrayElements did not make a copy");
73          }
74  
75          System.out.println("Array before: " + Arrays.toString(arr));
76  
77          // We fill the array in slices so that arr[i] = i
78          for (int i = 0; i < slices; i++) {
79              int start = i * sliceLength;
80              fill(arr, start, sliceLength);
81              System.out.println("Array during: " + Arrays.toString(arr));
82              check(arr, (i + 1) * sliceLength);
83          }
84          System.out.println("Array after: " + Arrays.toString(arr));
85          cleanup(arr);
86      }
87  
88      /*
89       * Calls GetIntArrayElements and stashes the native pointer for
90       * use by fill() if a copy was made.
91       * Returns true if a copy was made else false.
92      */
93      static native boolean init(int[] arr);
94  
95      /*
96       * Fills in target[start] to target[start+count-1], so that
97       * target[i] == i. The update is done natively using the raw
98       * pointer into the array.
99       */
100     static native void fill(int[] target, int start, int count);
101 
102     /*
103      * Properly release the copied array
104      */
105     static native void cleanup(int[] target);
106 
107 
108     static void check(int[] source, int count) {
109         for (int i = 0; i < count; i++) {
110             if (source[i] != i) {
111                 System.out.println("Failing source array: " + Arrays.toString(source));
112                 throw new RuntimeException("Expected source[" + i + "] == " +
113                                            i + " but got " + source[i]);
114             }
115         }
116         for (int i = count; i < source.length; i++) {
117             if (source[i] != 0) {
118                 System.out.println("Failing source array: " + Arrays.toString(source));
119                 throw new RuntimeException("Expected source[" + i +
120                                            "] == 0 but got " + source[i]);
121             }
122         }
123 
124     }
125 }
