49  import jdk.test.lib.process.OutputAnalyzer;
50  import lib.jdb.JdbCommand;
51  import lib.jdb.JdbTest;
52  
53  class RedefinePopTarg {
54      static public void main(String[] args) {
55          RedefinePopTarg mine = new RedefinePopTarg();
56          mine.a1(44);   // @1 delete the call that we are in when the pop occurs
57          mine.a4();
58      }
59  
60      public void a1(int p1) {
61          System.out.println("a1: @1 breakpoint here");
62      }
63  
64      public void a4() {
65          System.out.println("a4: The next line should not say Ni!");
66          System.out.println("a4: Ni!");   // @1 delete
67      }
68  }
69  
70  public class RedefinePop extends JdbTest {
71  
72      public static void main(String argv[]) {
73          new RedefinePop().run();
74      }
75  
76      private RedefinePop() {
77          super(RedefinePopTarg.class.getName(), "RedefinePop.java");
78      }
79  
80      @Override
81      protected void runCases() {
82          setBreakpoints(1);
83          jdb.command(JdbCommand.run());
84          redefineClass(1, "-g");
85          jdb.command(JdbCommand.pop());
86          jdb.contToExit(1);
87  
88          new OutputAnalyzer(getDebuggeeOutput())
89                  .shouldNotContain("Internal exception:");
90      }
91  }
