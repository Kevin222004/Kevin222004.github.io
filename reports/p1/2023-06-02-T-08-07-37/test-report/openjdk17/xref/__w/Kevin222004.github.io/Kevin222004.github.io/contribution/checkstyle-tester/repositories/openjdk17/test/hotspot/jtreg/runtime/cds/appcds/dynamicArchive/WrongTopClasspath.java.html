25  import java.io.File;
26  
27  /*
28   * @test
29   * @summary correct classpath for bottom archive, but bad classpath for top archive
30   * @requires vm.cds
31   * @library /test/lib /test/hotspot/jtreg/runtime/cds/appcds /test/hotspot/jtreg/runtime/cds/appcds/test-classes
32   * @build GenericTestApp sun.hotspot.WhiteBox
33   * @run driver jdk.test.lib.helpers.ClassFileInstaller -jar WhiteBox.jar sun.hotspot.WhiteBox
34   * @run driver jdk.test.lib.helpers.ClassFileInstaller -jar GenericTestApp.jar GenericTestApp
35   * @run driver jdk.test.lib.helpers.ClassFileInstaller -jar WrongJar.jar GenericTestApp
36   * @run main/othervm -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI -Xbootclasspath/a:./WhiteBox.jar WrongTopClasspath
37   */
38  
39  import jdk.test.lib.helpers.ClassFileInstaller;
40  
41  public class WrongTopClasspath extends DynamicArchiveTestBase {
42  
43      public static void main(String[] args) throws Exception {
44          runTest(WrongTopClasspath::test);
45      }
46  
47      static void test(String args[]) throws Exception {
48          String topArchiveName = getNewArchiveName("top");
49          String baseArchiveName = getNewArchiveName("base");
50          TestCommon.dumpBaseArchive(baseArchiveName);
51  
52          String appJar    = ClassFileInstaller.getJarPath("GenericTestApp.jar");
53          String wrongJar  = ClassFileInstaller.getJarPath("WrongJar.jar");
54          String mainClass = "GenericTestApp";
55  
56          // Dump the top archive using "-cp GenericTestApp.jar" ...
57          dump2_WB(baseArchiveName, topArchiveName,
58                   "-Xlog:cds*",
59                   "-Xlog:cds+dynamic=debug",
60                   "-cp", appJar, mainClass)
61              .assertNormalExit();
62  
63          // ... but try to load the top archive using "-cp WrongJar.jar".
64          // Use -Xshare:auto so top archive can fail after base archive has succeeded,
65          // but the app will continue to run.
66          run2_WB(baseArchiveName, topArchiveName,
67                  "-Xlog:cds*",
68                  "-Xlog:cds+dynamic=debug",
69                  "-Xlog:class+path=info",
70                  "-Xshare:auto",
71                  "-cp", wrongJar, mainClass,
72                  "assertShared:java.lang.Object",  // base archive still useable
73                  "assertNotShared:GenericTestApp") // but top archive is not useable
74            .assertNormalExit("The top archive failed to load");
75      }
76  }
