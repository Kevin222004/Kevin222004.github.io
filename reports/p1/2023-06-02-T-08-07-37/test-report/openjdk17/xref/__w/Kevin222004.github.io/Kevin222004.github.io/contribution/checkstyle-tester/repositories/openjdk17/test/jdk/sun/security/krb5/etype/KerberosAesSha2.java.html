159     private static byte[] stringToKey(AesSha2DkCrypto dk,
160             char[] pass, byte[] salt, byte[] params) throws Exception {
161         Method m = AesSha2DkCrypto.class.getDeclaredMethod("stringToKey",
162                 char[].class, byte[].class, byte[].class);
163         m.setAccessible(true);
164         return (byte[])m.invoke(dk, pass, salt, params);
165     }
166 
167     private static byte[] deriveKey(AesSha2DkCrypto dk, byte[] baseKey,
168             int usage, byte type) throws Exception {
169         Method m = AesSha2DkCrypto.class.getDeclaredMethod("deriveKey",
170                 byte[].class, int.class, byte.class);
171         m.setAccessible(true);
172         return (byte[]) m.invoke(dk, baseKey, usage, type);
173     }
174 
175     private static byte[] cat(byte[] b1, byte[] b2) {
176         byte[] result = Arrays.copyOf(b1, b1.length + b2.length);
177         System.arraycopy(b2, 0, result, b1.length, b2.length);
178         return result;
179     }
180 
181     private static byte[] enc(AesSha2DkCrypto dk, byte[] confounder,
182             byte[] bk, byte[] text) throws Exception {
183         return dk.encryptRaw(bk, 2, new byte[16], cat(confounder, text),
184                 0, confounder.length + text.length);
185     }
186 
187     private static byte[] checksum(AesSha2DkCrypto dk, byte[] baseKey, byte[] text)
188             throws Exception {
189         return dk.calculateChecksum(baseKey, 2, text, 0, text.length);
190     }
191 
192     private static byte[] hex(String var) {
193         var = var.replaceAll("\\s", "");
194         return HexFormat.of().parseHex(var);
195     }
196 
197     private static void check(byte[] b1, byte[] b2) throws Exception {
198         if (!Arrays.equals(b1, b2)) {
199             dump(b1); dump(b2);
200             throw new Exception("Failure");
201         }
202     }
203 
204     private static void dump(byte[] data) throws Exception {
205         HexPrinter.simple().dest(System.err).format(data);
206     }
