24  /*
25   * @test
26   * @bug 8159262
27   * @summary Test differing scenarios where a module's readability list and a package's exportability list should be walked
28   * @modules java.base/jdk.internal.misc
29   * @library /test/lib
30   * @compile ../AccessCheck/ModuleLibrary.java
31   * @compile ModuleSameCLMain.java
32   * @compile ModuleNonBuiltinCLMain.java
33   * @compile CustomSystemClassLoader.java
34   * @run driver ModuleStress
35   */
36  
37  import jdk.test.lib.process.ProcessTools;
38  import jdk.test.lib.process.OutputAnalyzer;
39  import jdk.test.lib.compiler.InMemoryJavaCompiler;
40  import jdk.test.lib.helpers.ClassFileInstaller;
41  
42  public class ModuleStress {
43  
44      public static void main(String[] args) throws Exception {
45  
46          // Test #1: java -version
47          //   All modules' readability lists and packages' exportability
48          //   lists should contain only modules defined to the 3 builtin
49          //   loaders (boot, application, platform).  Thus there is
50          //   not a need to walk those lists at a GC safepoint since
51          //   those loaders never die.
52          ProcessBuilder pb = ProcessTools.createJavaProcessBuilder(
53               "-Xbootclasspath/a:.",
54               "-Xlog:module=trace",
55               "-version");
56  
57          OutputAnalyzer oa = new OutputAnalyzer(pb.start());
58          oa.shouldNotContain("must be walked")
59            .shouldNotContain("being walked")
60            .shouldHaveExitValue(0);
61  
62          // Next 2 tests involve the use of class p1.c1 and p2.c2
63          String source1 = "package p1;"   +
64                           "import p2.c2;" +
65                           "public class c1 {" +
66                           "    public c1() {" +
67                           "       p2.c2 c2_obj = new p2.c2();" +
68                           "       c2_obj.method2();" +
69                           "   }" +
70                           "}";
71  
72          String source2 = "package p2;" +
73                           "public class c2 {" +
74                           "    public void method2() { }" +
75                           "}";
76  
77          ClassFileInstaller.writeClassToDisk("p2/c2",
78               InMemoryJavaCompiler.compile("p2.c2", source2),  System.getProperty("test.classes"));
79  
80          ClassFileInstaller.writeClassToDisk("p1/c1",
81               InMemoryJavaCompiler.compile("p1.c1", source1), System.getProperty("test.classes"));
82  
83          // Test #2: Load two modules defined to the same customer class loader.
84          //   m1x's module readability list and package p2's exportability should
85          //   not be walked at a GC safepoint since both modules are defined to
86          //   the same loader and thus have the exact same life cycle.
87          pb = ProcessTools.createJavaProcessBuilder(
88               "-Xbootclasspath/a:.",
89               "-Xlog:module=trace",
90               "ModuleSameCLMain");
91  
92          oa = new OutputAnalyzer(pb.start());
93          oa.shouldNotContain("must be walked")
94            .shouldNotContain("being walked")
95            .shouldHaveExitValue(0);
96  
97          // Test #3: Load two modules in differing custom class loaders.
98          //   m1x's module readability list and package p2's exportability list must
99          //   be walked at a GC safepoint since both modules are defined to non-builtin
100         //   class loaders which could die and thus be unloaded.
101         pb = ProcessTools.createJavaProcessBuilder(
102              "-Xbootclasspath/a:.",
103              "-Xlog:module=trace",
104              "ModuleNonBuiltinCLMain");
105 
106         oa = new OutputAnalyzer(pb.start());
107         oa.shouldContain("module m1x reads list must be walked")
108           .shouldContain("package p2 defined in module m2x, exports list must be walked")
109           .shouldNotContain("module m2x reads list must be walked")
110           .shouldHaveExitValue(0);
111 
112         // Test #4: Load two modules in differing custom class loaders,
113         //   of which one has been designated as the custom system class loader
114         //   via -Djava.system.class.loader=CustomSystemClassLoader. Since
115         //   m3x is defined to the system class loader, m2x's module readability
116         //   list does not have to be walked at a GC safepoint, but package p2's
117         //   exportability list does.
118         pb = ProcessTools.createJavaProcessBuilder(
119              "-Djava.system.class.loader=CustomSystemClassLoader",
120              "-Xbootclasspath/a:.",
121              "-Xlog:module=trace",
122              "ModuleNonBuiltinCLMain");
123 
124         oa = new OutputAnalyzer(pb.start());
125         oa.shouldContain("package p2 defined in module m2x, exports list must be walked")
126           .shouldNotContain("module m2x reads list must be walked")
127           .shouldHaveExitValue(0);
128 
129     }
130 }
