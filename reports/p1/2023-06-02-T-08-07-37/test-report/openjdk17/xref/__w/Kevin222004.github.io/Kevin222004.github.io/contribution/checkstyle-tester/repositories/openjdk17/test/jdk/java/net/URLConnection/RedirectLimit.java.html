31  /*
32   * Simulate a server that redirects ( to a different URL) 9 times
33   * and see if the client correctly follows the trail
34   */
35  
36  import java.io.*;
37  import java.net.*;
38  import java.util.concurrent.CountDownLatch;
39  
40  import jdk.test.lib.net.URIBuilder;
41  
42  class RedirLimitServer extends Thread {
43      static final int TIMEOUT = 20 * 1000;
44      static final int NUM_REDIRECTS = 9;
45  
46      static final String reply1 = "HTTP/1.1 307 Temporary Redirect\r\n" +
47          "Date: Mon, 15 Jan 2001 12:18:21 GMT\r\n" +
48          "Server: Apache/1.3.14 (Unix)\r\n" +
49          "Location: http://localhost:";
50      static final String reply2 = ".html\r\n" +
51          "Connection: close\r\n" +
52          "Content-Type: text/html; charset=iso-8859-1\r\n\r\n" +
53          "<html>Hello</html>";
54      static final String reply3 = "HTTP/1.1 200 Ok\r\n" +
55          "Date: Mon, 15 Jan 2001 12:18:21 GMT\r\n" +
56          "Server: Apache/1.3.14 (Unix)\r\n" +
57          "Connection: close\r\n" +
58          "Content-Type: text/html; charset=iso-8859-1\r\n\r\n" +
59          "World";
60  
61      final ServerSocket ss;
62      final int port;
63      final CountDownLatch readyToStart = new CountDownLatch(1);
64  
65      RedirLimitServer(ServerSocket ss) throws IOException {
66          this.ss = ss;
67          port = this.ss.getLocalPort();
68          this.ss.setSoTimeout(TIMEOUT);
69      }
70  
71      static final byte[] requestEnd = new byte[] {'\r', '\n', '\r', '\n' };
72  
73      // Read until the end of a HTTP request
74      void readOneRequest(InputStream is) throws IOException {
75          int requestEndCount = 0, r;
76          StringBuilder sb = new StringBuilder();
77          while ((r = is.read()) != -1) {
78              sb.append((char)r);
79              if (r == requestEnd[requestEndCount]) {
80                  requestEndCount++;
81                  if (requestEndCount == 4) {
82                      break;
83                  }
84              } else {
85                  requestEndCount = 0;
86              }
87          }
88          System.out.println("Server got request: " + sb.toString());
89      }
90  
91      public void run() {
92          try {
93              readyToStart.countDown();
94              for (int i=0; i<NUM_REDIRECTS; i++) {
95                  try (Socket s = ss.accept()) {
96                      System.out.println("Server accepted socket: " + s);
97                      s.setSoTimeout(TIMEOUT);
98                      readOneRequest(s.getInputStream());
99                      System.out.println("Redirecting to: /redirect" + i);
100                     String reply = reply1 + port + "/redirect" + i + reply2;
101                     s.getOutputStream().write(reply.getBytes());
102                 }
103             }
104             try (Socket s = ss.accept()) {
105                 System.out.println("Server accepted socket: " + s);
106                 s.setSoTimeout(TIMEOUT);
107                 readOneRequest(s.getInputStream());
108                 System.out.println("Replying...");
109                 s.getOutputStream().write(reply3.getBytes());
110             }
111         } catch (Exception e) {
112             e.printStackTrace();
113         }
114     }
115 };
116 
117 public class RedirectLimit {
118     public static void main(String[] args) throws Exception {
119         try (ServerSocket ss = new ServerSocket(0, 0, InetAddress.getLoopbackAddress())) {
120             int port = ss.getLocalPort();
121             RedirLimitServer server = new RedirLimitServer(ss);
122             server.start();
123             server.readyToStart.await();
124             URL url = URIBuilder.newBuilder()
125                     .scheme("http")
126                     .loopback()
127                     .port(port)
128                     .toURL();
129             URLConnection conURL = url.openConnection(Proxy.NO_PROXY);
130 
131             conURL.setDoInput(true);
132             conURL.setAllowUserInteraction(false);
133             conURL.setUseCaches(false);
134 
135             try (InputStream in = conURL.getInputStream()) {
136                 if ((in.read() != (int) 'W') || (in.read() != (int) 'o')) {
137                     throw new RuntimeException("Unexpected string read");
138                 }
139             }
140         }
141     }
