411     @Test
412     public void testAnnotations(Path base) throws IOException {
413         ElementType[] types = {
414                 ElementType.FIELD,
415                 ElementType.METHOD,
416                 ElementType.PARAMETER,
417                 ElementType.RECORD_COMPONENT
418         };
419         for (int i = 0; i < (1 << types.length); i++) {
420             Set<ElementType> set = EnumSet.noneOf(ElementType.class);
421             for (int b = 0; b < types.length; b++) {
422                 if ((i & (1 << b)) != 0) {
423                     set.add(types[b]);
424                 }
425             }
426             testAnnotations(base, set);
427         }
428     }

430     void testAnnotations(Path base, Set<ElementType> types) throws IOException {
431         out.println("test " + types);
432         String name = types.isEmpty() ? "none" : types.stream()
433                 .map(k -> k.name().toLowerCase(Locale.US))
434                 .collect(Collectors.joining("-"));
435         Path dir = base.resolve(name);
436         Path src = dir.resolve("src");
437         String target = types.isEmpty() ? "" : types.stream()
438                 .map(s -> "ElementType." + s)
439                 .collect(Collectors.joining(", ", "@Target({", "})"));
440         tb.writeJavaFiles(src,
441                 """
442                     package p;
443                     import java.lang.annotation.*;
444                     @Documented
445                     """
446                     + target + "\n"
447                     + " public @interface Anno { }\n",
448                 "package p; public @interface UndocAnno { }",
449                 """
450                     package p; public record R(@Anno int i) { }
451                     """);
452 
453         javadoc("-d", dir.resolve("out").toString(),
454                 "-quiet", "-noindex",
455                 "--no-platform-links",
456                 "-sourcepath", src.toString(),
457                 "-private",
458                 "p");
459         checkExit(Exit.OK);
460 
461         checkOutput("p/R.html", false,
462                 "UndocAnno");
463 
464         Set<ElementType> t = types.isEmpty() ? EnumSet.allOf(ElementType.class) : types;
465         String anno = """
466             <a href="Anno.html" title="annotation interface in p">@Anno</a>""";
467         String rcAnno = t.contains(ElementType.RECORD_COMPONENT) ? anno + " " : "";
468         String fAnno = t.contains(ElementType.FIELD) ? "<span class=\"annotations\">" + anno + "\n</span>" : "";
469         String pAnno = t.contains(ElementType.PARAMETER) ? anno + "\n " : "";
470         String mAnno= t.contains(ElementType.METHOD) ? "<span class=\"annotations\">" + anno + "\n</span>" : "";
471 
472         checkOutput("p/R.html", true,
473                 """
474                     <div class="type-signature"><span class="modifiers">public record </span><span c\
475                     lass="element-name type-name-label">R</span>("""
476                         + rcAnno
477                         + """
478                             int&nbsp;i)
479                             <span class="extends-implements">extends java.lang.Record</span></div>""",
480                 "<div class=\"member-signature\">"
481                         + fAnno
482                         + """
483                             <span class="modifiers">private final</span>&nbsp;<span class="return-type">int<\
484                             /span>&nbsp;<span class="element-name">i</span></div>""",
485                 """
486                     <div class="member-signature"><span class="modifiers">public</span>&nbsp;<span c\
487                     lass="element-name">R</span><wbr><span class="parameters">("""
488                         + pAnno
489                         + "int&nbsp;i)</span></div>",
490                 "<div class=\"member-signature\">"
491                         + mAnno
492                         + """
493                             <span class="modifiers">public</span>&nbsp;<span class="return-type">int</span>&\
494                             nbsp;<span class="element-name">i</span>()</div>""");
495 
496     }
