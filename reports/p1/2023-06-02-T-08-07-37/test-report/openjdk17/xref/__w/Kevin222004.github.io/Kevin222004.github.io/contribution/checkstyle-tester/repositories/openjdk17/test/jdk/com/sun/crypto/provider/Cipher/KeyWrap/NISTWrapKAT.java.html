80      private static byte[] toBytes(String hex, int hexLen) {
81          if (hexLen < hex.length()) {
82              hex = hex.substring(0, hexLen);
83          } else {
84              hexLen = hex.length();
85          }
86          int outLen = hexLen >> 1;
87          BigInteger temp = new BigInteger(hex, 16);
88          byte[] val = temp.toByteArray();
89          if (val.length == outLen) {
90              return val;
91          } else {
92              byte[] out = new byte[outLen];
93              if (val.length < outLen) {
94                  // enlarge
95                  System.arraycopy(val, 0, out, outLen - val.length, val.length);
96              } else {
97                  // truncate
98                  System.arraycopy(val, val.length - outLen, out, 0, outLen);
99              }
100             return out;
101         }

252     @Test(dataProvider = "testData")
253     public void testKeyWrap(String algo, String key, int keyLen,
254             String data, int dataLen, String expected) throws Exception {
255         System.out.println("Testing " +  algo + " Cipher with wrapping " +
256             dataLen + "-byte key with " + 8*keyLen + "-bit KEK");
257         int allowed = Cipher.getMaxAllowedKeyLength("AES");
258         if (keyLen > allowed) {
259             System.out.println("=> skip, exceeds max allowed size " + allowed);
260             return;
261         }
262         Cipher c1 = Cipher.getInstance(algo, "SunJCE");
263         Cipher c2 = Cipher.getInstance(algo, "SunJCE");
264         Cipher c3 = Cipher.getInstance(algo, "SunJCE");
265 
266         byte[] keyVal = toBytes(key, keyLen << 1);
267         byte[] dataVal = toBytes(data, dataLen << 1);
268 
269         SecretKey cipherKey = new SecretKeySpec(keyVal, "AES");
270         SecretKey toBeWrappedKey = new SecretKeySpec(dataVal, "AES");
271 
272         c1.init(Cipher.WRAP_MODE, cipherKey);
273         IvParameterSpec ivSpec = new IvParameterSpec(c1.getIV());
274         c2.init(Cipher.WRAP_MODE, cipherKey, ivSpec);
275         AlgorithmParameters params = AlgorithmParameters.getInstance("AES");
276         params.init(ivSpec);
277         c3.init(Cipher.WRAP_MODE, cipherKey, params);
278 
279         // first test WRAP with known values
280         byte[] wrapped = c1.wrap(toBeWrappedKey);
281         byte[] wrapped2 = c2.wrap(toBeWrappedKey);
282         byte[] wrapped3 = c3.wrap(toBeWrappedKey);
283 
284         byte[] expectedVal = toBytes(expected, expected.length());
285 
286         if (!Arrays.equals(wrapped, expectedVal) ||
287                 !Arrays.equals(wrapped2, expectedVal) ||
288                 !Arrays.equals(wrapped3, expectedVal)) {
289             throw new Exception("Wrap test failed; got different result");
290         }
291 
292         // then test UNWRAP and compare with the initial values
293         c1.init(Cipher.UNWRAP_MODE, cipherKey);
294         ivSpec = new IvParameterSpec(c1.getIV());
295         c2.init(Cipher.UNWRAP_MODE, cipherKey, ivSpec);
296         params = AlgorithmParameters.getInstance("AES");
297         params.init(ivSpec);
298         c3.init(Cipher.UNWRAP_MODE, cipherKey, params);
299 
300         Key unwrapped = c1.unwrap(wrapped, "AES", Cipher.SECRET_KEY);
301         Key unwrapped2 = c2.unwrap(wrapped, "AES", Cipher.SECRET_KEY);
302         Key unwrapped3 = c3.unwrap(wrapped, "AES", Cipher.SECRET_KEY);
303 
304         if (!Arrays.equals(unwrapped.getEncoded(), dataVal) ||
305                 !Arrays.equals(unwrapped2.getEncoded(), dataVal) ||
306                 !Arrays.equals(unwrapped3.getEncoded(), dataVal)) {
307             throw new Exception("Unwrap failed; got different result");
308         }
309     }
