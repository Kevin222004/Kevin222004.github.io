24  /*
25   * @test
26   * @summary Test archived module graph with custom runtime image
27   * @requires vm.cds.archived.java.heap
28   * @library /test/jdk/lib/testlibrary /test/lib /test/hotspot/jtreg/runtime/cds/appcds
29   * @build sun.hotspot.WhiteBox
30   * @compile CheckArchivedModuleApp.java
31   * @run driver jdk.test.lib.helpers.ClassFileInstaller -jar app.jar CheckArchivedModuleApp
32   * @run driver jdk.test.lib.helpers.ClassFileInstaller -jar WhiteBox.jar sun.hotspot.WhiteBox
33   * @run driver ArchivedModuleWithCustomImageTest
34   */
35  
36  import java.io.File;
37  import java.nio.file.Files;
38  import java.nio.file.Path;
39  import java.nio.file.Paths;
40  
41  import jdk.test.lib.compiler.CompilerUtils;
42  import jdk.test.lib.process.OutputAnalyzer;
43  import jdk.test.lib.process.ProcessTools;
44  import jdk.test.lib.helpers.ClassFileInstaller;
45  
46  public class ArchivedModuleWithCustomImageTest {
47      private static final String JAVA_HOME = System.getProperty("java.home");
48      private static final String TEST_MODULE = "test";
49      private static final Path jdkHome = Paths.get(System.getProperty("test.jdk"));
50      private static final Path jdkMods = jdkHome.resolve("jmods");
51      private static final Path testSrc = Paths.get(System.getProperty("test.src"));
52      private static final Path src = testSrc.resolve("src").resolve(TEST_MODULE);
53      private static final Path classes = Paths.get("classes");
54      private static final Path jmods = Paths.get("jmods");
55  
56      public static void main(String[] args) throws Throwable {
57          if (Files.notExists(jdkMods)) {
58              System.out.println("No jmods/ in test JDK, not supported.");
59              return;
60          }
61  
62          // compile test module class
63          if (!CompilerUtils.compile(src, classes)) {
64              throw new RuntimeException("Compilation failure.");
65          }
66  
67          // create custom runtime image named 'myimage'
68          Files.createDirectories(jmods);
69          Path image = Paths.get("myimage");
70          runJmod(classes.toString(), TEST_MODULE);
71          runJlink(image, TEST_MODULE);
72  
73          // test using 'myimage'
74          testArchivedModuleUsingImage(image);
75  
76          Files.delete(jmods.resolve(TEST_MODULE + ".jmod"));
77      }
78  
79      private static void runJlink(Path image, String modName) throws Throwable {
80          Path jlink = Paths.get(JAVA_HOME, "bin", "jlink");
81          OutputAnalyzer output = ProcessTools.executeProcess(jlink.toString(),
82                          "--output", image.toString(),
83                          "--add-modules", modName,
84                          "--module-path", jdkMods + File.pathSeparator + jmods);
85          output.shouldHaveExitValue(0);
86      }
87  
88      private static void runJmod(String cp, String modName) throws Throwable {
89          Path jmod = Paths.get(JAVA_HOME, "bin", "jmod");
90          OutputAnalyzer output = ProcessTools.executeProcess(jmod.toString(),
91                         "create",
92                         "--class-path", cp,
93                         "--module-version", "1.0",
94                         "--main-class", "jdk.test.Test",
95                         jmods.resolve(modName + ".jmod").toString());
96          output.shouldHaveExitValue(0);
97      }
98  
99      private static void testArchivedModuleUsingImage(Path image)
100                             throws Throwable {
101         String wbJar = ClassFileInstaller.getJarPath("WhiteBox.jar");
102         String use_whitebox_jar = "-Xbootclasspath/a:" + wbJar;
103         String appJar = ClassFileInstaller.getJarPath("app.jar");
104         Path customJava = Paths.get(image.toString(), "bin", "java");
105 
106         // -Xshare:dump with custom runtime image
107         String[] dumpCmd = {
108             customJava.toString(),
109             "-XX:SharedArchiveFile=./ArchivedModuleWithCustomImageTest.jsa",
110             "-Xshare:dump", "-Xlog:cds"};
111         printCommand(dumpCmd);
112         ProcessBuilder pbDump = new ProcessBuilder();
113         pbDump.command(dumpCmd);
114         OutputAnalyzer output = TestCommon.executeAndLog(
115             pbDump, "custom.runtime.image.dump");
116         TestCommon.checkDump(output);
117 
118         // Test case 1):
119         // test archived module graph objects are used with custome runtime image
120         System.out.println("------------------- Test case 1 -------------------");
121         String[] runCmd = {customJava.toString(),
122                            use_whitebox_jar,
123                            "-XX:SharedArchiveFile=./ArchivedModuleWithCustomImageTest.jsa",
124                            "-cp",
125                            appJar,
126                            "-Xshare:on",
127                            "-XX:+UnlockDiagnosticVMOptions",
128                            "-XX:+WhiteBoxAPI",
129                            "CheckArchivedModuleApp",
130                            "yes",
131                            "yes"};
132         printCommand(runCmd);
133         ProcessBuilder pbRun = new ProcessBuilder();
134         pbRun.command(runCmd);
135         output = TestCommon.executeAndLog(pbRun, "custom.runtime.image.run");
136         output.shouldHaveExitValue(0);
137 
138 
139         // Test case 2):
140         // verify --show-module-resolution output
141         System.out.println("------------------- Test case 2 -------------------");
142 
143         // myimage/bin/java -Xshare:off --show-module-resolution -version
144         String[] showModuleCmd1 = {customJava.toString(),
145                                    "-Xshare:off",
146                                    "--show-module-resolution",
147                                    "-version"};
148         printCommand(showModuleCmd1);
149         pbRun = new ProcessBuilder();
150         pbRun.command(showModuleCmd1);
151         output = TestCommon.executeAndLog(
152             pbRun, "custom.runtime.image.showModuleResolution.nocds");
153         output.shouldHaveExitValue(0);
154         String moduleResolutionOut1 = output.getStdout();
155 
156         // myimage/bin/java -Xshare:on --show-module-resolution -version
157         //    -XX:SharedArchiveFile=./ArchivedModuleWithCustomImageTest.jsa
158         String[] showModuleCmd2 = {
159             customJava.toString(),
160             "-XX:SharedArchiveFile=./ArchivedModuleWithCustomImageTest.jsa",
161             "-Xshare:on",
162             "--show-module-resolution",
163             "-version"};
164         printCommand(showModuleCmd2);
165         pbRun = new ProcessBuilder();
166         pbRun.command(showModuleCmd2);
167         output = TestCommon.executeAndLog(
168             pbRun, "custom.runtime.image.showModuleResolution.cds");
169         if (output.getStderr().contains("sharing")) {
170             String moduleResolutionOut2 = output.getStdout();
171             TestCommon.checkOutputStrings(
172                 moduleResolutionOut1, moduleResolutionOut2, "\n");
173         }
174     }
175 
176     private static void printCommand(String opts[]) {
177         StringBuilder cmdLine = new StringBuilder();
178         for (String cmd : opts)
179             cmdLine.append(cmd).append(' ');
180         System.out.println("Command line: [" + cmdLine.toString() + "]");
181     }
182 }
