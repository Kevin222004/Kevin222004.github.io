22  
23          derexact = baseexact;   // <<fail>> <Der> = <Base>
24          baseexact = derexact;   // <<fail>> <Base> = <Der>

27          derext = basesuper;     // <<fail>> <? extends Der> = <? super Base>
28          baseext = dersuper;     // <<fail>> <? extends Base> = <? super Der>
29          basesuper = dersuper;   // <<fail>> <? super Base> = <? super Der>
30          basesuper = derexact;   // <<fail>> <? super Base> = <Der>
31      }
