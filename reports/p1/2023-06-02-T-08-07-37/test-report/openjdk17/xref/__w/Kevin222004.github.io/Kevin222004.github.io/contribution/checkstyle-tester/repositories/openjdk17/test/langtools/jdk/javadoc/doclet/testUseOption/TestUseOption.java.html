44      @Test
45      public void test1() {
46          javadoc("-d", "out-1",
47                  "-sourcepath", testSrc,
48                  "-use",
49                  "pkg1", "pkg2");
50          checkExit(Exit.OK);
51  
52          // Eight tests for class use.
53          for (int i = 1; i <= 8; i++) {
54              checkOutput("pkg1/class-use/C1.html", true,
55                      "Test " + i + " passes");
56          }
57  
58          // Three more tests for package use.
59          for (int i = 1; i <= 3; i++) {
60              checkOutput("pkg1/package-use.html", true,
61                      "Test " + i + " passes");
62          }
63  
64          checkOrder("pkg1/class-use/UsedClass.html",
65                  "Field in C1.",
66                  "Field in C2.",
67                  "Field in C4.",
68                  "Field in C5.",
69                  "Field in C6.",
70                  "Field in C7.",
71                  "Field in C8.",
72                  "Method in C1.",
73                  "Method in C2.",
74                  "Method in C4.",
75                  "Method in C5.",
76                  "Method in C6.",
77                  "Method in C7.",
78                  "Method in C8."
79          );
80  
81          checkOutput("pkg1/class-use/UsedClass.html", true,
82            "that return types with arguments of type"
83          );
84          checkOutput("pkg1/class-use/UsedClass.html", true,
85            """
86                <a href="../C1.html#methodInC1ReturningType()" class="member-name-link">methodInC1ReturningType</a>"""
87          );
88          checkOutput("pkg1/class-use/UsedInterface.html", true,
89            """
90                Classes in <a href="../package-summary.html">pkg1</a> that implement <a href="..\
91                /UsedInterface.html" title="interface in pkg1">UsedInterface</a>"""
92          );
93          checkOutput("pkg1/class-use/UsedInterfaceA.html", true,
94            """
95                Classes in <a href="../package-summary.html">pkg1</a> that implement <a href="..\
96                /UsedInterfaceA.html" title="interface in pkg1">UsedInterfaceA</a>"""
97          );
98          checkOutput("pkg1/class-use/UsedClass.html", false,
99             "methodInC1Protected"
100         );
101         checkOutput("pkg1/class-use/UsedInterface.html", true,
102            """
103                <a href="../AnAbstract.html" class="type-name-link" title="class in pkg1">AnAbstract</a>"""
104         );
105         checkOutput("pkg1/class-use/UsedInterface.html", true,
106             "../C10.html#withReturningTypeParameters()"
107         );
108         checkOutput("pkg1/class-use/UsedInterface.html", true,
109             "../C10.html#withTypeParametersOfType(java.lang.Class)"
110         );
111         checkOutput("pkg1/class-use/UsedInterface.html", true,
112             """
113                 "../package-summary.html">pkg1</a> that return <a href="../UsedInterface.html" title="interface in pkg1\""""
114         );
115         checkOutput("pkg1/class-use/UsedInterface.html", true,
116             """
117                 <a href="../C10.html#addAll(pkg1.UsedInterface...)" class="member-name-link">addAll</a>"""
118         );
119         checkOutput("pkg1/class-use/UsedInterface.html", true,
120             """
121                 <a href="../C10.html#create(pkg1.UsedInterfaceA,pkg1.UsedInterface,java.lang.String)" class="member-name-link">"""
122         );
123         checkOutput("pkg1/class-use/UsedInterface.html", true,
124             """
125                 <a href="../C10.html#withTypeParametersOfType(java.lang.Class)" class="member-name-link">withTypeParametersOfType</a>"""
126         );
127         checkOutput("pkg1/class-use/UsedInterface.html", true,
128             """
129                 Subinterfaces of <a href="../UsedInterface.html" title="interface in pkg1">UsedI\
130                 nterface</a> in <a href="../package-summary.html">pkg1""",
131             """
132                 <div class="col-first even-row-color"><code>interface&nbsp;</code></div>
133                 <div class="col-second even-row-color"><code><a href="\
134                 ../SubInterface.html" class="type-name-link" title="interface in pkg1">SubInterface</a>&lt;T&gt;\
135                 </code></div>"""
136         );
137         checkOutput("pkg1/class-use/UsedThrowable.html", true,
138             """
139                 Methods in <a href="../package-summary.html">pkg1</a> that throw <a href="../Use\
140                 dThrowable.html" title="class in pkg1">UsedThrowable</a>""",
141             """
142                 <div class="col-first even-row-color"><code>void</code></div>
143                 <div class="col-second even-row-color"><span class="type-name-label">C1.</span><code>\
144                 <a href="../C1.html#methodInC1ThrowsThrowable()" class="member-name-link">\
145                 methodInC1ThrowsThrowable</a>()</code></div>"""
146         );
147     }

149     @Test
150     public void test2() {
151         javadoc("-d", "out-2",
152                 "-sourcepath", testSrc,
153                 "-use",
154                 testSrc("C.java"), testSrc("UsedInC.java"), "pkg3");
155         checkExit(Exit.OK);
156 
157         checkOutput("class-use/UsedInC.html", true,
158                 """
159                     Uses of <a href="../UsedInC.html" title="class in Unnamed Package">UsedInC</a> i\
160                     n <a href="../package-summary.html">Unnamed Package</a>"""
161         );
162         checkOutput("class-use/UsedInC.html", true,
163                 """
164                     <li>
165                     <section class="detail" id="unnamed-package">
166                     """
167         );
168         checkOutput("package-use.html", true,
169                 """
170                     <div class="col-first even-row-color"><a href="class-use/UsedInC.html#unnamed-package">UsedInC</a></div>""",
171                 """
172                     <div class="col-first even-row-color"><a href="#unnamed-package">Unnamed Package</a></div>
173                     <div class="col-last even-row-color">&nbsp;</div>"""
174         );
175     }
