72       * Return a byte array with a simple stream containing an Dummy object.
73       * @return  a byte with a simple serialization object
74       */
75      private static byte[] simpleStream() {
76          ByteArrayOutputStream boas = new ByteArrayOutputStream();
77          try (ObjectOutputStream ois = new ObjectOutputStream(boas)) {
78              ois.writeObject(new Dummy("Here"));
79              return boas.toByteArray();
80          } catch (IOException ioe) {
81              Assert.fail("unexpected IOE", ioe);
82          }
83          throw new RuntimeException("should not reach here");
84      }

86      /**
87       * Initialize the filter factory, supplying one if not already set.
88       * Does not/can not replace any MyFilterFactory.
89       *
90       * @param dynFilterFactory a filter factory to use if not already set
91       * @return the filter factory in effect
92       */
93      private static MyFilterFactory setupFilterFactory(MyFilterFactory dynFilterFactory) {
94          if ((Config.getSerialFilterFactory() instanceof MyFilterFactory ff))
95              return ff;
96          Config.setSerialFilterFactory(dynFilterFactory);
97          return dynFilterFactory;
98      }

100     // If the configured filter has not been set, set it
101     // It can only be set once for the process, so avoid setting it again
102     private static ObjectInputFilter setupFilter(ObjectInputFilter serialFilter) {
103 
104         var configFilter = Config.getSerialFilter();
105         if (configFilter == serialFilter || configFilter instanceof Validator)
106             return configFilter;        // if already set or a type we can use, no change
107 
108         if (configFilter == null) {
109             Config.setSerialFilter(serialFilter);
110             return serialFilter;        // none set already, set it
111         }
112 
113         return configFilter;
114     }

116     private static boolean isValidFilterFactory() {
117         return !(ObjectInputFilter.Config.getSerialFilterFactory() instanceof NotMyFilterFactory);
118     }

120     /**
121      * Returns true if serialFilter actions are ok, either no SM or SM has serialFilter Permission
122      */
123     @SuppressWarnings("removal")
124     private static boolean hasFilterPerm() {
125         boolean hasSerialPerm = true;
126         SecurityManager sm = System.getSecurityManager();
127         if (sm != null) {
128             try {
129                 Permission p = new SerializablePermission("serialFilter");
130                 sm.checkPermission(p);
131                 hasSerialPerm = true;
132             } catch (AccessControlException ace2) {
133                 hasSerialPerm = false;      // SM and serialFilter not allowed
134             }
135         }
136         return hasSerialPerm;
137     }

139     @DataProvider(name="FilterCases")
140     static Object[][] filterCases() {
141         if (isValidFilterFactory()) {
142             return new Object[][]{
143                     {contextFilterFactory, null, null},  // no overrides
144                     {contextFilterFactory, v1, null},    // context filter
145                     {contextFilterFactory, v1, v2},      // per stream filter
146             };
147         } else {
148             // There are zero cases to run with an unknown filter factory. (NotMyFilterFactory)
149             return new Object[0][0];
150         }
151     }

159     /**
160      * Setting and resetting the filter factory is not allowed.
161      * The filter factory may have been on the command line (depending on which @run this is).
162      * If the jdk.SerialFilterFactory is the built-in filter factory, set it once.
163      * Try to set it again, the second should throw.
164      */
165     @Test
166     @SuppressWarnings("removal")
167     void testSecondSetShouldThrow() {
168         if (System.getSecurityManager() != null) {
169             // Skip test when running with SM
170             return;
171         }
172         var currFF = Config.getSerialFilterFactory();
173         if (currFF.getClass().getClassLoader() == null) {
174             try {
175                 // Not already set, set it
176                 Config.setSerialFilterFactory(contextFilterFactory);
177                 currFF = contextFilterFactory;
178             } catch (IllegalStateException ise) {
179                 Assert.fail("First setSerialFilterFactory should not throw");
180             }
181         }
182         // Setting it again will throw
183         Assert.expectThrows(IllegalStateException.class,
184                 () -> Config.setSerialFilterFactory(new MyFilterFactory("f11")));
185         var resetFF = Config.getSerialFilterFactory();
186         Assert.assertEquals(resetFF, currFF, "Setting again should not change filter factory");
187     }

189     /**
190      * Test that the filter factory is set when expected and is called when expected.
191      * This test only covers the cases when a filter factory is supplied
192      * either via a command line property or via the API.
193      * The cases where the builtin filter factory applies are tested in SerialFilterTest.
194      *
195      * @param dynFilterFactory a FilterFactory to set
196      * @param dynFilter a serial filter to be used for the configured filter
197      * @param streamFilter a serial filter to be used for the stream filter
198      * @throws IOException if an I/O error occurs (should not occur)
199      * @throws ClassNotFoundException for class not found (should not occur)
200      */
201     @Test(dataProvider="FilterCases")
202     @SuppressWarnings("removal")
203     void testCase(MyFilterFactory dynFilterFactory, Validator dynFilter, Validator streamFilter)
204                 throws IOException, ClassNotFoundException {
205 
206         // Set the Filter Factory and System-wide filter
207         ObjectInputFilter configFilter;
208         MyFilterFactory factory;
209         try {
210             configFilter = setupFilter(dynFilter);
211             factory = setupFilterFactory(dynFilterFactory);
212             Assert.assertTrue(hasFilterPerm(),
213                     "setSerialFilterFactory and setFilterFactory succeeded without serialFilter permission");
214         } catch (AccessControlException ace) {
215             Assert.assertFalse(hasFilterPerm(),
216                     "setSerialFilterFactory failed even with serialFilter permission");
217             return;         // test complete
218         }
219         factory.reset();
220 
221         InputStream is = new ByteArrayInputStream(simpleStream);
222         ObjectInputStream ois = new ObjectInputStream(is);
223 
224         Assert.assertNull(factory.current(), "initially current should be null");
225         Assert.assertEquals(factory.next(), configFilter, "initially next should be the configured filter");
226         var currFilter = ois.getObjectInputFilter();
227         if (currFilter != null && currFilter.getClass().getClassLoader() == null) {
228             // Builtin loader;  defaults to configured filter
229             Assert.assertEquals(currFilter, configFilter, "getObjectInputFilter should be configured filter");
230         } else {
231             Assert.assertEquals(currFilter, configFilter, "getObjectInputFilter should be null");
232         }
233         if (streamFilter != null) {
234             ois.setObjectInputFilter(streamFilter);
235             // MyFilterFactory is called when the stream filter is changed; verify values passed it
236             Assert.assertEquals(factory.current(), currFilter, "when setObjectInputFilter, current should be current filter");
237             Assert.assertEquals(factory.next(), streamFilter, "next should be stream specific filter");
238 
239             // Check the OIS filter after the factory has updated it.
240             currFilter = ois.getObjectInputFilter();
241             Assert.assertEquals(currFilter, streamFilter, "getObjectInputFilter should be set");
242 
243             // Verify that it can not be set again
244             Assert.assertThrows(IllegalStateException.class, () -> ois.setObjectInputFilter(streamFilter));
245         }
246         if (currFilter instanceof Validator validator) {
247             validator.reset();
248             Object o = ois.readObject();       // Invoke only for the side effect of calling the Filter
249             Assert.assertEquals(validator.count, 1, "Wrong number of calls to the stream filter");
250         } else {
251             Object o = ois.readObject();       // Invoke only for the side effect of calling the filter
252         }
253     }

264     // Test that setting the filter factory after any deserialization (any testCase)
265     // throws IllegalStateException with the specific message
266     @Test(dependsOnMethods="testCase")
267     void testSetFactoryAfterDeserialization() {
268         if (hasFilterPerm()) {
269             // Only test if is allowed by SM.
270             BinaryOperator<ObjectInputFilter> factory = Config.getSerialFilterFactory();
271             IllegalStateException ise = Assert.expectThrows(IllegalStateException.class, () -> Config.setSerialFilterFactory(factory));
272             Assert.assertTrue(ise.getMessage().startsWith("Cannot replace filter factory: "));
273         }
274     }

277     // Test that OIS.setObjectInputFilter does not allow a null filter to replace
278     // a non-null filter.  And does allow a null filter to replace a null filter
279     @Test
280     void testDisableFailFilter() throws IOException {
281         if (hasFilterPerm()) {
282             // Only test if is allowed by SM.
283             ObjectInputFilter curr = null;
284             try (ObjectInputStream ois = new ObjectInputStream(new ByteArrayInputStream(simpleStream))) {
285                 curr = ois.getObjectInputFilter();
286                 // Try to set the filter to null
287                 ois.setObjectInputFilter(null);
288                 if (curr != null) {
289                     Assert.fail("setting filter to null after a non-null filter should throw");
290                 }
291             } catch (IllegalStateException ise) {
292                 if (curr == null) {
293                     Assert.fail("setting filter to null after a null filter should not throw");
294                 }
295             }
296         }
297     }

299     /**
300      * A simple filter factory that retains its arguments.
301      */
302     private static class MyFilterFactory
303             implements BinaryOperator<ObjectInputFilter> {
304         private final String name;
305         private ObjectInputFilter current;
306         private ObjectInputFilter next;
307 
308         MyFilterFactory(String name) {
309             this.name = name;
310             current = new Validator("UnsetCurrent");
311             next = new Validator("UnsetNext");
312         }
313 
314         public ObjectInputFilter apply(ObjectInputFilter curr, ObjectInputFilter next) {
315             this.current = curr;
316             this.next = next;
317             if (curr == null & next == null)
318                 return Config.getSerialFilter();    // Default to the configured filter
319             return next;
320         }
321 
322         public void  reset() {
323             current = new Validator("UnsetCurrent");
324             next = new Validator("UnsetNext");
325         }
326 
327         public ObjectInputFilter current() {
328             return current;
329         }
330 
331         public ObjectInputFilter next() {
332             return next;
333         }
334 
335         public String toString() {
336             return name + ":: curr: " + current + ", next: " + next;
337         }
338     }

371     /**
372      * A filter that accumulates information about the checkInput callbacks
373      * that can be checked after readObject completes.
374      */
375     static class Validator implements ObjectInputFilter {
376         private final String name;
377         long count;          // Count of calls to checkInput
378 
379         Validator(String name) {
380             this.name = name;
381             count = 0;
382         }
383 
384         void reset() {
385             count = 0;
386         }
387 
388         @Override
389         public Status checkInput(FilterInfo filter) {
390             count++;
391             return Status.ALLOWED;
392         }
393 
394         public String toString(){
395             return name + ": count: " + count;
396         }
397     }
