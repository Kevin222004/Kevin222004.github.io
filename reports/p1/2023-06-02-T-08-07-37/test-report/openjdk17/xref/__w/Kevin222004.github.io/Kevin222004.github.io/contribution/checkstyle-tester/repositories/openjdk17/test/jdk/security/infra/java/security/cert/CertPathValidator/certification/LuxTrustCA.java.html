175 
176         System.setProperty("jdk.security.certpath.ocspNonce", "true");
177         ValidatePathWithParams pathValidator = new ValidatePathWithParams(null);
178 
179         if (args.length >= 1 && "CRL".equalsIgnoreCase(args[0])) {
180             pathValidator.enableCRLCheck();
181         } else {
182             // OCSP check by default
183             pathValidator.enableOCSPCheck();
184         }
185 
186         // Validate valid
187         pathValidator.validate(new String[]{VALID, INT},
188                 ValidatePathWithParams.Status.GOOD, null, System.out);
189 
190         // Validate Revoked
191         pathValidator.validate(new String[]{REVOKED, INT},
192                 ValidatePathWithParams.Status.REVOKED,
193                 "Wed Jul 10 04:48:49 PDT 2019", System.out);
194     }
