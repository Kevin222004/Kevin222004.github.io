43  public class TestDefaultMethodsSyntax {
44  
45      static int checkCount = 0;
46  
47      enum VersionKind {
48          PRE_LAMBDA("7"),
49          LAMBDA("8"),
50          POST_LAMBDA("9");
51  
52          String versionString;
53  
54          VersionKind(String versionString) {
55              this.versionString = versionString;
56          }
57  
58          List<String> getOptions() {
59              return Arrays.asList("-XDallowStaticInterfaceMethods", "-source", versionString);
60          }
61      }
62  
63      enum ModifierKind {
64          NONE(""),
65          PUBLIC("public"),
66          PROTECTED("protected"),
67          PRIVATE("private"),
68          ABSTRACT("abstract"),
69          STATIC("static"),
70          NATIVE("native"),
71          SYNCHRONIZED("synchronized"),
72          FINAL("final"),
73          STRICTFP("strictfp"),
74          DEFAULT("default");
75  
76          String modStr;
77  
78          private ModifierKind(String modStr) {
79              this.modStr = modStr;
80          }
81  
82          static boolean intersect(ModifierKind mk, ModifierKind... mks) {
83              for (ModifierKind mk2 : mks) {
84                  if (mk == mk2) return true;
85              }
86              return false;
87          }
88  
89          static boolean compatible(MethodKind mk, ModifierKind mod1, ModifierKind mod2, EnclosingKind ek) {
90              if (intersect(ABSTRACT, mod1, mod2) || intersect(NATIVE, mod1, mod2)) {
91                  return mk == MethodKind.NO_BODY;
92              } else if (intersect(DEFAULT, mod1, mod2) || intersect(STATIC, mod1, mod2)
93                      || intersect(PRIVATE, mod1, mod2)) {
94                  return mk == MethodKind.BODY;
95              } else {
96                  return ek == EnclosingKind.INTERFACE ?
97                          mk == MethodKind.NO_BODY : mk == MethodKind.BODY;
98              }
99          }
100 
101         boolean compatible(EnclosingKind ek) {
102             switch (this) {
103                 case PROTECTED:
104                     return ek != EnclosingKind.INTERFACE;
105                 default:
106                     return true;
107             }
108         }
109 
110         static boolean compatible(ModifierKind m1, ModifierKind m2, EnclosingKind ek) {
111             Result res1 = allowedModifierPairs[m1.ordinal()][m2.ordinal()];
112             Result res2 = allowedModifierPairs[m2.ordinal()][m1.ordinal()];
113             if (res1 != res2) {
114                 throw new AssertionError(String.format("Ill-formed table: [%s,%s] != [%s,%s]", m1, m2, m2, m1));
115             } else {
116                 return res1.compatible(ek, m1, m2);
117             }
118         }
119 
120         interface Result {
121             boolean compatible(EnclosingKind ek, ModifierKind m1, ModifierKind m2);
122         }
123 
124         static final Result T = new Result() {
125             @Override
126             public boolean compatible(EnclosingKind ek, ModifierKind m1, ModifierKind m2) {
127                 return true;
128             }
129         };
130 
131         static final Result F = new Result() {
132             @Override
133             public boolean compatible(EnclosingKind ek, ModifierKind m1, ModifierKind m2) {
134                 return false;
135             }
136         };
137 
138         static final Result C = new Result() {
139             @Override
140             public boolean compatible(EnclosingKind ek, ModifierKind m1, ModifierKind m2) {
141                 return ek != EnclosingKind.INTERFACE;
142             }
143         };
144 
145         static final Result I = new Result() {
146             @Override
147             public boolean compatible(EnclosingKind ek, ModifierKind m1, ModifierKind m2) {
148                 return ek == EnclosingKind.INTERFACE;
149             }
150         };
151 
152         static Result[][] allowedModifierPairs = {
153             /*                     NONE  PUBLIC  PROTECTED  PRIVATE  ABSTRACT  STATIC  NATIVE  SYNCHRONIZED  FINAL  STRICTFP  DEFAULT */
154             /* NONE */           { T   , T    , C        , T       , T       , T     , C     , C           , C    , C       , I   },
155             /* PUBLIC */         { T   , F    , F        , F       , T       , T     , C     , C           , C    , C       , I   },
156             /* PROTECTED */      { C   , F    , F        , F       , C       , C     , C     , C           , C    , C       , F   },
157             /* PRIVATE */        { T   , F    , F        , F       , F       , T     , C     , C           , C    , T       , F   },
158             /* ABSTRACT */       { T   , T    , C        , F       , F       , F     , F     , F           , F    , F       , F   },
159             /* STATIC */         { T   , T    , C        , T       , F       , F     , C     , C           , C    , T       , F   },
160             /* NATIVE */         { C   , C    , C        , C       , F       , C     , F     , C           , C    , F       , F   },
161             /* SYNCHRONIZED */   { C   , C    , C        , C       , F       , C     , C     , F           , C    , C       , F   },
162             /* FINAL */          { C   , C    , C        , C       , F       , C     , C     , C           , F    , C       , F   },
163             /* STRICTFP */       { C   , C    , C        , T       , F       , T     , F     , C           , C    , F       , I   },
164             /* DEFAULT */        { I   , I    , F        , F       , F       , F     , F     , F           , F    , I       , F   }};
165     }
166 
167     enum MethodKind {
168         NO_BODY("void m();"),
169         BODY("void m() { }");
170 
171         String methStr;
172 
173         private MethodKind(String methStr) {
174             this.methStr = methStr;
175         }
176     }
177 
178     enum EnclosingKind {
179         ABSTRACT_CLASS("abstract class Test "),
180         INTERFACE("interface Test ");
181 
182         String enclStr;
183 
184         EnclosingKind(String enclStr) {
185             this.enclStr = enclStr;
186         }
187     }

189     public static void main(String... args) throws Exception {
190 
191         //create default shared JavaCompiler - reused across multiple compilations
192         JavaCompiler comp = ToolProvider.getSystemJavaCompiler();
193         try (StandardJavaFileManager fm = comp.getStandardFileManager(null, null, null)) {
194 
195             for (VersionKind vk : VersionKind.values()) {
196                 for (EnclosingKind ek : EnclosingKind.values()) {
197                     for (MethodKind mk : MethodKind.values()) {
198                         for (ModifierKind modk1 : ModifierKind.values()) {
199                             for (ModifierKind modk2 : ModifierKind.values()) {
200                                 new TestDefaultMethodsSyntax(vk, ek, mk, modk1, modk2).run(comp, fm);
201                             }
202                         }
203                     }
204                 }
205             }
206             System.out.println("Total check executed: " + checkCount);
207         }
208     }
209 
210     VersionKind vk;
211     EnclosingKind ek;
212     MethodKind mk;
213     ModifierKind modk1, modk2;
214     JavaSource source;
215     DiagnosticChecker diagChecker;
216 
217     TestDefaultMethodsSyntax(VersionKind vk, EnclosingKind ek, MethodKind mk, ModifierKind modk1, ModifierKind modk2) {
218         this.vk = vk;
219         this.ek = ek;
220         this.mk = mk;
221         this.modk1 = modk1;
222         this.modk2 = modk2;
223         this.source = new JavaSource();
224         this.diagChecker = new DiagnosticChecker();
225     }
226 
227     class JavaSource extends SimpleJavaFileObject {
228 
229         String template = "#EK {\n" +
230                           "   #MOD1 #MOD2 #METH\n" +
231                           "}\n";
232 
233         String source;
234 
235         public JavaSource() {
236             super(URI.create("myfo:/Test.java"), JavaFileObject.Kind.SOURCE);
237             source = template.replaceAll("#EK", ek.enclStr)
238                     .replaceAll("#MOD1", modk1.modStr)
239                     .replaceAll("#MOD2", modk2.modStr)
240                     .replaceAll("#METH", mk.methStr);
241         }
242 
243         @Override
244         public CharSequence getCharContent(boolean ignoreEncodingErrors) {
245             return source;
246         }
247     }
248 
249     void run(JavaCompiler tool, StandardJavaFileManager fm) throws Exception {
250         JavacTask ct = (JavacTask)tool.getTask(null, fm, diagChecker,
251                 vk.getOptions(), null, Arrays.asList(source));
252         try {
253             ct.analyze();
254         } catch (Throwable ex) {
255             throw new AssertionError("Error thrown when analyzing the following source:\n" + source.getCharContent(true));
256         }
257         check();
258     }
259 
260     void check() {
261         boolean errorExpected = !ModifierKind.compatible(modk1, modk2, ek);
262 
263         errorExpected |= !ModifierKind.compatible(mk, modk1, modk2, ek);
264 
265         errorExpected |= !modk1.compatible(ek) || !modk2.compatible(ek);
266 
267         errorExpected |= ModifierKind.intersect(ModifierKind.DEFAULT, modk1, modk2) &&
268                 vk == VersionKind.PRE_LAMBDA;
269 
270         errorExpected |= ModifierKind.intersect(ModifierKind.STATIC, modk1, modk2) &&
271                 ek == EnclosingKind.INTERFACE && vk == VersionKind.PRE_LAMBDA;
272 
273         errorExpected |= ModifierKind.intersect(ModifierKind.PRIVATE, modk1, modk2) &&
274                 ek == EnclosingKind.INTERFACE && (vk == VersionKind.LAMBDA || vk == VersionKind.PRE_LAMBDA);
275 
276         checkCount++;
277         if (diagChecker.errorFound != errorExpected) {
278             throw new AssertionError("Problem when compiling source:\n" + source.getCharContent(true) +
279                     "\nfound error: " + diagChecker.errorFound);
280         }
281     }
282 
283     static class DiagnosticChecker implements javax.tools.DiagnosticListener<JavaFileObject> {
284 
285         boolean errorFound;
286 
287         public void report(Diagnostic<? extends JavaFileObject> diagnostic) {
288             if (diagnostic.getKind() == Diagnostic.Kind.ERROR) {
289                 errorFound = true;
290             }
291         }
292     }
