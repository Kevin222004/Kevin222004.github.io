7   
8   import com.sun.tools.classfile.ClassFile;
9   import com.sun.tools.classfile.ConstantPoolException;
10  import com.sun.tools.classfile.Method;
11  import com.sun.tools.classfile.Attribute;
12  import com.sun.tools.classfile.Code_attribute;
13  import com.sun.tools.classfile.LineNumberTable_attribute;
14  import com.sun.tools.classfile.LineNumberTable_attribute.Entry;
15  
16  import java.io.IOException;
17  
18  public class NestedLineNumberTest {
19  
20      public static void main(String[] args) throws Exception {
21          Entry[] lines = findEntries();
22          if (lines == null || lines.length != 1) {
23              int found = lines == null ? 0 : lines.length;
24              error(String.format("LineNumberTable contains wrong number of entries - expected %d, found %d", 1, found));
25          }
26  
27          int line = lines[0].line_number;
28          if (line != 54) {
29              error(String.format("LineNumberTable contains wrong line number - expected %d, found %d", 54, line));
30          }
31      }
32  
33      static Entry[] findEntries() throws IOException, ConstantPoolException {
34          ClassFile self = ClassFile.read(NestedLineNumberTest.Test.class.getResourceAsStream("NestedLineNumberTest$Test.class"));
35          for (Method m : self.methods) {
36              if ("<init>".equals(m.getName(self.constant_pool))) {
37                  Code_attribute code_attribute = (Code_attribute)m.attributes.get(Attribute.Code);
38                  for (Attribute at : code_attribute.attributes) {
39                      if (Attribute.LineNumberTable.equals(at.getName(self.constant_pool))) {
40                          return ((LineNumberTable_attribute)at).line_number_table;
41                      }
42                  }
43              }
44          }
45          return null;
46      }
47  
48      static void error(String msg) {
49          throw new AssertionError(msg);
50      }
51  
52      // The default constructor in this class should get only one LineNumberTable entry,
53      // pointing to the first line of the declaration of class Test.
54      static class Test {
55          static class Empty { }
56      }
57  }
