50  public class ThrowingTasks {
51      static final Random rnd = new Random();
52  
53      @SuppressWarnings("serial")
54      static class UncaughtExceptions
55          extends ConcurrentHashMap<Class<?>, Integer> {
56  
57          void inc(Class<?> key) {
58              compute(key, (k, v) -> (v == null) ? 1 : v + 1);
59          }
60      }
61  
62      /** Double-check that HashTable and ConcurrentHashMap are work-alikes. */
63      @SuppressWarnings("serial")
64      static class UncaughtExceptionsTable
65          extends Hashtable<Class<?>, Integer> {
66  
67          synchronized void inc(Class<?> key) {
68              Integer v = get(key);
69              put(key, (v == null) ? 1 : v + 1);
70          }
71      }
72  
73      static final UncaughtExceptions uncaughtExceptions
74          = new UncaughtExceptions();
75      static final UncaughtExceptionsTable uncaughtExceptionsTable
76          = new UncaughtExceptionsTable();
77      static final AtomicInteger totalUncaughtExceptions
78          = new AtomicInteger(0);
79      static final CountDownLatch uncaughtExceptionsLatch
80          = new CountDownLatch(24);
81  
82      static final ThreadGroup tg = new ThreadGroup("Flaky");
83  
84      static final RuntimeException rte = new RuntimeException();
85      static final Error error = new Error();
86      static final Throwable weird = new Throwable();
87      static final Exception checkedException = new Exception();
88  
89      static class Thrower implements Runnable {
90          final Throwable t;
91          Thrower(Throwable t) { this.t = t; }
92          public void run() {
93              if (t != null)
94                  ThrowingTasks.<RuntimeException>uncheckedThrow(t);
95          }
96      }
97  
98      static final List<Thrower> throwers = Arrays.asList(
99          new Thrower(null),
100         new Thrower(rte),
101         new Thrower(error),
102         new Thrower(weird),
103         new Thrower(checkedException));
104 
105     static class Flaky implements Runnable {
106         final Runnable beforeExecute;
107         final Runnable execute;
108         Flaky(Runnable beforeExecute,
109               Runnable execute) {
110             this.beforeExecute = beforeExecute;
111             this.execute = execute;
112         }
113         public void run() { execute.run(); }
114     }
115 
116     static final List<Flaky> flakes = new ArrayList<>();
117     static {
118         for (Thrower x : throwers)
119             for (Thrower y : throwers)
120                 flakes.add(new Flaky(x, y));
121         Collections.shuffle(flakes);
122     }
123 
124     static final CountDownLatch allStarted = new CountDownLatch(flakes.size());
125     static final CountDownLatch allContinue = new CountDownLatch(1);
126 
127     static class PermissiveSecurityManger extends SecurityManager {
128         public void checkPermission(Permission p) { /* bien sur, Monsieur */ }
129     }
130 
131     static void checkTerminated(ThreadPoolExecutor tpe) {
132         try {
133             check(tpe.getQueue().isEmpty());
134             check(tpe.isShutdown());
135             check(tpe.isTerminated());
136             check(! tpe.isTerminating());
137             equal(tpe.getActiveCount(), 0);
138             equal(tpe.getPoolSize(), 0);
139             equal(tpe.getTaskCount(), tpe.getCompletedTaskCount());
140             check(tpe.awaitTermination(0L, TimeUnit.SECONDS));
141         } catch (Throwable t) { unexpected(t); }
142     }
143 
144     /**
145      * Waits for condition to become true, first spin-polling, then sleep-polling.
146      */
147     static void spinAwait(Supplier<Boolean> waitingForGodot) {
148         for (int spins = 0; !waitingForGodot.get(); ) {
149             if ((spins = (spins + 1) & 3) > 0) {
150                 Thread.yield();
151             } else {
152                 try { Thread.sleep(4); }
153                 catch (InterruptedException unexpected) {
154                     throw new AssertionError(unexpected);
155                 }
156             }
157         }
158     }
159 
160     static class CheckingExecutor extends ThreadPoolExecutor {
161         private final ReentrantLock lock = new ReentrantLock();
162         CheckingExecutor() {
163             super(10, 10,
164                   1L, TimeUnit.HOURS,
165                   new LinkedBlockingQueue<Runnable>(),
166                   (ThreadFactory) (runnable) -> {
167                       Thread thread = new Thread(tg, runnable);
168                       Thread.UncaughtExceptionHandler handler = (t, e) -> {
169                           check(! t.isInterrupted());
170                           totalUncaughtExceptions.getAndIncrement();
171                           uncaughtExceptions.inc(e.getClass());
172                           uncaughtExceptionsTable.inc(e.getClass());
173                           uncaughtExceptionsLatch.countDown();
174                       };
175                       thread.setUncaughtExceptionHandler(handler);
176                       return thread;
177                   });
178         }
179         @Override protected void beforeExecute(Thread t, Runnable r) {
180             final boolean lessThanCorePoolSize;
181             // Add a lock to sync allStarted.countDown() and
182             // allStarted.getCount() < getCorePoolSize()
183             lock.lock();
184             try {
185                 allStarted.countDown();
186                 lessThanCorePoolSize = allStarted.getCount() < getCorePoolSize();
187             } finally {
188                 lock.unlock();
189             }
190             if (lessThanCorePoolSize) {
191                 try { allContinue.await(); }
192                 catch (InterruptedException x) { unexpected(x); }
193             }
194             beforeExecuteCount.getAndIncrement();
195             check(! isTerminated());
196             ((Flaky)r).beforeExecute.run();
197         }
198         @Override protected void afterExecute(Runnable r, Throwable t) {
199             //System.out.println(tg.activeCount());
200             afterExecuteCount.getAndIncrement();
201             check(((Thrower)((Flaky)r).execute).t == t);
202             check(! isTerminated());
203         }
204         @Override protected void terminated() {
205             try {
206                 terminatedCount.getAndIncrement();
207                 if (rnd.nextBoolean()) {
208                     check(isShutdown());
209                     check(isTerminating());
210                     check(! isTerminated());
211                     check(! awaitTermination(0L, TimeUnit.MINUTES));
212                 }
213             } catch (Throwable t) { unexpected(t); }
214         }
215     }
216 
217     static final AtomicInteger beforeExecuteCount = new AtomicInteger(0);
218     static final AtomicInteger afterExecuteCount  = new AtomicInteger(0);
219     static final AtomicInteger terminatedCount    = new AtomicInteger(0);
220 
221     private static void realMain(String[] args) throws Throwable {
222         if (rnd.nextBoolean())
223             System.setSecurityManager(new PermissiveSecurityManger());
224 
225         CheckingExecutor tpe = new CheckingExecutor();
226 
227         for (Runnable task : flakes)
228             tpe.execute(task);
229 
230         if (rnd.nextBoolean()) {
231             allStarted.await();
232             equal(tpe.getTaskCount(),
233                   (long) flakes.size());
234             equal(tpe.getCompletedTaskCount(),
235                   (long) flakes.size() - tpe.getCorePoolSize());
236         }
237         allContinue.countDown();
238 
239         //System.out.printf("thread count = %d%n", tg.activeCount());
240         uncaughtExceptionsLatch.await();
241 
242         spinAwait(() -> tg.activeCount() == tpe.getCorePoolSize());
243 
244         tpe.shutdown();
245 
246         check(tpe.awaitTermination(10L, TimeUnit.MINUTES));
247         checkTerminated(tpe);
248 
249         List<Map<Class<?>, Integer>> maps = new ArrayList<>();
250         maps.add(uncaughtExceptions);
251         maps.add(uncaughtExceptionsTable);
252         for (Map<Class<?>, Integer> map : maps) {
253             equal(map.get(Exception.class), throwers.size() + 1);
254             equal(map.get(weird.getClass()), throwers.size() + 1);
255             equal(map.get(Error.class), throwers.size() + 1);
256             equal(map.get(RuntimeException.class), throwers.size() + 1);
257             equal(map.size(), 4);
258         }
259         equal(totalUncaughtExceptions.get(), 4*throwers.size() + 4);
260 
261         equal(beforeExecuteCount.get(), flakes.size());
262         equal(afterExecuteCount.get(), throwers.size());
263         equal(tpe.getCompletedTaskCount(), (long) flakes.size());
264         equal(terminatedCount.get(), 1);
265 
266         // check for termination operation idempotence
267         tpe.shutdown();
268         tpe.shutdownNow();
269         check(tpe.awaitTermination(10L, TimeUnit.MINUTES));
270         checkTerminated(tpe);
271         equal(terminatedCount.get(), 1);
272     }

274     //--------------------- Infrastructure ---------------------------
275     static volatile int passed = 0, failed = 0;
276     static void pass() {passed++;}
277     static void fail() {failed++; Thread.dumpStack();}
278     static void fail(String msg) {System.out.println(msg); fail();}
279     static void unexpected(Throwable t) {failed++; t.printStackTrace();}
280     static void check(boolean cond) {if (cond) pass(); else fail();}
281     static void equal(Object x, Object y) {
282         if (x == null ? y == null : x.equals(y)) pass();
283         else fail(x + " not equal to " + y);}
284     public static void main(String[] args) throws Throwable {
285         try {realMain(args);} catch (Throwable t) {unexpected(t);}
286         System.out.printf("%nPassed = %d, failed = %d%n%n", passed, failed);
287         if (failed > 0) throw new AssertionError("Some tests failed");}
288     @SuppressWarnings("unchecked")
289     static <T extends Throwable> void uncheckedThrow(Throwable t) throws T {
290         throw (T)t; // rely on vacuous cast
291     }
