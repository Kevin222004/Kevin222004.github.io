7   class MostSpecific08 {
8   
9       static class C {
10          int getInt() { return -1; }
11          Integer getInteger() { return -1; }
12      }
13  
14      interface IntResult { }
15      interface ReferenceResult<X> { }
16  
17      interface PrimitiveFunction {
18          int f(C c);
19      }
20  
21      interface ReferenceFunction<X> {
22          X f(C c);
23      }
24  
25      interface Tester {
26          IntResult apply(PrimitiveFunction p);
27          <Z> ReferenceResult<Z> apply(ReferenceFunction<Z> p);
28      }
29  
30      void testMref(Tester t) {
31          IntResult pr = t.apply(C::getInt); //ok - unoverloaded mref
32          ReferenceResult<Integer> rr = t.apply(C::getInteger); //ok - unoverloaded mref
33      }
34  
35      void testLambda(Tester t) {
36          IntResult pr1 = t.apply(c->c.getInt()); //ambiguous - implicit
37          IntResult pr2 = t.apply((C c)->c.getInt()); //ok
38          ReferenceResult<Integer> rr1 = t.apply(c->c.getInteger()); //ambiguous - implicit
39          ReferenceResult<Integer> rr2 = t.apply((C c)->c.getInteger()); //ok
40      }
