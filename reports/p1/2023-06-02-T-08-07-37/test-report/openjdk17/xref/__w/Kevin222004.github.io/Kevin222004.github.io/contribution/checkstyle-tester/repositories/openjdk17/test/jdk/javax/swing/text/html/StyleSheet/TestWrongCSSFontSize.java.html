24  import java.awt.Component;
25  import java.awt.Graphics;
26  import java.awt.image.BufferedImage;
27  import java.io.File;
28  import java.io.IOException;
29  import java.util.Arrays;
30  import java.util.concurrent.atomic.AtomicReference;
31  import java.util.function.Consumer;
32  import javax.imageio.ImageIO;
33  import javax.swing.JEditorPane;
34  import javax.swing.SwingUtilities;
35  import javax.swing.text.GlyphView;
36  import javax.swing.text.View;
37  
38  /*
39   * @test
40   * @bug 8257664
41   * @summary  Tests inherited font-size with parent percentage specification.
42   * @run main TestWrongCSSFontSize
43   * @run main TestWrongCSSFontSize -w3cUnits
44   */
45  public class TestWrongCSSFontSize {
46  
47      private static final String TEXT =
48              "<html><head><style>" +
49              "body { font-size: 14 }" +
50              "div span { font-size: 150% }" +
51              "span { font-size: 200% }" +
52              "h2, .h2 { font-size: 150% }" +
53              "</style></head><body>" +
54  
55              "<h2>Foo</h2>" +
56              "<div class=h2>Bar</div>" +
57              "<ol class=h2><li>Baz</li></ol>" +
58              "<table class=h2><tr><td>Qux</td></tr></table>" +
59              "<table><thead class=h2><tr><th>Qux</th></tr></thead></table>" +
60              "<table><tr class=h2><td>Qux</td></tr></table>" +
61              "<table><tr><td class=h2>Qux</td></tr></table>" +
62              "<div><span>Quux</span></div>" +
63  
64              "</body></html>";
65  
66      private static final int expectedFontSize = 21;
67      private static final int expectedAssertions = 8;
68  
69      private final boolean w3cUnits;
70  
71      private JEditorPane editor;
72  
73      public TestWrongCSSFontSize(boolean w3cUnits) {
74          this.w3cUnits = w3cUnits;
75      }
76  
77      public void setUp() {
78          editor = new JEditorPane();
79          editor.setContentType("text/html");
80          if (w3cUnits) {
81              editor.putClientProperty(JEditorPane.W3C_LENGTH_UNITS, Boolean.TRUE);
82          }
83          editor.setText(TEXT);
84          editor.setSize(editor.getPreferredSize()); // layout
85      }
86  
87      public void run() {
88          int count = forEachTextRun(editor.getUI()
89                  .getRootView(editor), this::assertFontSize);
90          if (count != expectedAssertions) {
91              throw new AssertionError((w3cUnits ? "w3cUnits - " : "")
92                      + "assertion count expected ["
93                      + expectedAssertions + "] but found [" + count + "]");
94          }
95      }
96  
97      private int forEachTextRun(View view, Consumer<GlyphView> action) {
98          int tested = 0;
99          for (int i = 0; i < view.getViewCount(); i++) {
100             View child = view.getView(i);
101             if (child instanceof GlyphView) {
102                 if (child.getElement()
103                         .getAttributes().getAttribute("CR") == Boolean.TRUE) {
104                     continue;
105                 }
106                 action.accept((GlyphView) child);
107                 tested += 1;
108             } else {
109                 tested += forEachTextRun(child, action);
110             }
111         }
112         return tested;
113     }
114 
115     private void assertFontSize(GlyphView child) {
116         printSource(child);
117         int actualFontSize = child.getFont().getSize();
118         if (actualFontSize != expectedFontSize) {
119             throw new AssertionError((w3cUnits ? "w3cUnits - " : "")
120                     + "font size expected ["
121                     + expectedFontSize + "] but found [" + actualFontSize +"]");
122         }
123     }
124 
125     private void printSource(View textRun) {
126         try {
127             editor.getEditorKit().write(System.out,
128                     editor.getDocument(), textRun.getStartOffset(),
129                     textRun.getEndOffset() - textRun.getStartOffset());
130         } catch (Exception e) {
131             e.printStackTrace();
132         }
133     }
134 
135     private static void captureImage(Component comp, String suffix) {
136         try {
137             BufferedImage capture = new BufferedImage(comp.getWidth(),
138                     comp.getHeight(), BufferedImage.TYPE_INT_ARGB);
139             Graphics g = capture.getGraphics();
140             comp.paint(g);
141             g.dispose();
142 
143             ImageIO.write(capture, "png",
144                     new File(TestWrongCSSFontSize.class
145                                 .getSimpleName() + suffix + ".png"));
146         } catch (IOException e) {
147             e.printStackTrace();
148         }
149     }
150 
151     public static void main(String[] args) throws Throwable {
152         TestWrongCSSFontSize test = new TestWrongCSSFontSize(argW3CUnits(args));
153         AtomicReference<Throwable> failure = new AtomicReference<>();
154         SwingUtilities.invokeAndWait(() -> {
155             try {
156                 test.setUp();
157                 test.run();
158             } catch (Throwable e) {
159                 failure.set(e);
160             } finally {
161                 String suffix = test.w3cUnits ? "-w3cUnits" : "";
162                 if (failure.get() != null) {
163                     captureImage(test.editor, suffix + "-failure");
164                 } else if (argCapture(args)) {
165                     captureImage(test.editor, suffix + "-success");
166                 }
167             }
168         });
169         if (failure.get() != null) {
170             throw failure.get();
171         }
172     }
173 
174     private static boolean argW3CUnits(String[] args) {
175         return Arrays.asList(args).contains("-w3cUnits");
176     }
177 
178     private static boolean argCapture(String[] args) {
179         return Arrays.asList(args).contains("-capture");
180     }
