31  import jdk.test.lib.process.Proc;
32  
33  import java.util.Random;
34  import java.util.List;
35  
36  public class ProcTest {
37      public static void main(String[] args) throws Exception {
38          if (args.length == 0) {
39              // Test launcher
40              Proc p1 = Proc.create("ProcTest")
41                      .args("1")
42                      .debug("p1")
43                      .start();
44              Proc p2 = Proc.create("ProcTest")
45                      .args("2")
46                      .debug("p2")
47                      .start();
48              while (true) {
49                  String s1 = p1.readData(); // p1 shows to p2
50                  if (s1 != null) p2.println(s1);
51                  String s2 = p2.readData(); // p2 shows to p1
52                  if (s2 != null) p1.println(s2);
53                  if (s1 == null && s2 == null) break;
54              }
55              p1.waitFor();
56              p2.waitFor();
57          } else {
58              // Sub process, args[0] is random seed.
59              List<String> gestures = List.of("Rock", "Paper", "Scissors");
60              int wins = 0;
61              Random r = new Random(Long.parseLong(args[0]));
62              while (true) {
63                  String my = gestures.get(r.nextInt(3));
64                  Proc.textOut(my); // show first, next line might block
65                  String peer = Proc.textIn();
66                  if (!my.equals(peer)) {
67                      if (my.equals("Paper") && peer.equals("Rock")
68                              || my.equals("Rock") && peer.equals("Scissors")
69                              || my.equals("Scissors") && peer.equals("Paper")) {
70                          wins++;
71                      } else {
72                          wins--;
73                      }
74                  }
75                  // Message not from textOut() will be ignored by readData().
76                  System.out.println(my + " vs " + peer + ", I win " + wins + " times");
77                  if (wins > 2 || wins < -2) {
78                      break;
79                  }
80              }
81          }
82      }
83  }
