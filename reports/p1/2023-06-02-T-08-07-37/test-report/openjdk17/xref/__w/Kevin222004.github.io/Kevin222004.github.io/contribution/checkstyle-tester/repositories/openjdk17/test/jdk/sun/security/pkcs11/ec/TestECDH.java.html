149     private final static void test(Provider p, String pub1s, String priv1s,
150             String pub2s, String priv2s, String secrets) throws Exception {
151         KeyFactory kf = KeyFactory.getInstance("EC", p);
152         PublicKey pub1 = kf.generatePublic(new X509EncodedKeySpec(parse(pub1s)));
153         System.out.println("Testing using parameters "
154                 + ((ECPublicKey)pub1).getParams() + "...");
155 
156         PrivateKey priv1 = kf.generatePrivate(new PKCS8EncodedKeySpec(parse(priv1s)));
157         PublicKey pub2 = kf.generatePublic(new X509EncodedKeySpec(parse(pub2s)));
158         PrivateKey priv2 = kf.generatePrivate(new PKCS8EncodedKeySpec(parse(priv2s)));
159         byte[] secret = parse(secrets);
160 
161         KeyAgreement ka1 = KeyAgreement.getInstance("ECDH", p);
162         ka1.init(priv1);
163         ka1.doPhase(pub2, true);
164         byte[] s1 = ka1.generateSecret();
165         if (Arrays.equals(secret, s1) == false) {
166             System.out.println("expected: " + toString(secret));
167             System.out.println("actual:   " + toString(s1));
168             throw new Exception("Secret 1 does not match");
169         }
170 
171         KeyAgreement ka2 = KeyAgreement.getInstance("ECDH", p);
172         ka2.init(priv2);
173         ka2.doPhase(pub1, true);
174         byte[] s2 = ka2.generateSecret();
175         if (Arrays.equals(secret, s2) == false) {
176             System.out.println("expected: " + toString(secret));
177             System.out.println("actual:   " + toString(s2));
178             throw new Exception("Secret 2 does not match");
179         }
180     }
