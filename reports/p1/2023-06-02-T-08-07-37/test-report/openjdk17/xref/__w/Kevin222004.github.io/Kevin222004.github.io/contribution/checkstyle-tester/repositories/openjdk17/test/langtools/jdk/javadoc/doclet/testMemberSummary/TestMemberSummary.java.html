24  /*
25   * @test
26   * @bug      4951228 6290760 8025633 8026567 8081854 8162363 8175200 8177417 8186332 8182765
27   *           8258602 8266044
28   * @summary  Test the case where the overridden method returns a different
29   *           type than the method in the child class.  Make sure the
30   *           documentation is inherited but the return type isn't.
31   * @library  ../../lib
32   * @modules jdk.javadoc/jdk.javadoc.internal.tool
33   * @build    javadoc.tester.*
34   * @run main TestMemberSummary
35   */
36  
37  import javadoc.tester.JavadocTester;
38  
39  public class TestMemberSummary extends JavadocTester {
40  
41      public static void main(String... args) throws Exception {
42          TestMemberSummary tester = new TestMemberSummary();
43          tester.runTests();
44      }
45  
46      @Test
47      public void test() {
48          javadoc("-d", "out",
49                  "-private", "--no-platform-links",
50                  "-sourcepath", testSrc,
51                  "pkg","pkg2");
52          checkExit(Exit.OK);
53  
54          checkOutput("pkg/PublicChild.html", true,
55                  // Check return type in member summary.
56                  """
57                      <code><a href="PublicChild.html" title="class in pkg">PublicChild</a></code></div>
58                      <div class="col-second even-row-color method-summary-table method-summary-table-\
59                      tab2 method-summary-table-tab4"><code><a href="#returnTypeTest()" class="member-\
60                      name-link">returnTypeTest</a>()</code></div>""",
61                  // Check return type in member detail.
62                  """
63                      <div class="member-signature"><span class="modifiers">public</span>&nbsp;<span c\
64                      lass="return-type"><a href="PublicChild.html" title="class in pkg">PublicChild</\
65                      a></span>&nbsp;<span class="element-name">returnTypeTest</span>()</div>""",
66                  """
67                      <div class="col-constructor-name even-row-color"><code><a href="#%3Cinit%3E()" c\
68                      lass="member-name-link">PublicChild</a>()</code></div>
69                      <div class="col-last even-row-color">&nbsp;</div>""");
70  
71          checkOutput("pkg/PrivateParent.html", true,
72                  """
73                      <div class="col-first even-row-color"><code>private </code></div>
74                      <div class="col-constructor-name even-row-color"><code><a href="#%3Cinit%3E(int)\
75                      " class="member-name-link">PrivateParent</a><wbr>(int&nbsp;i)</code></div>""");
76  
77          // Legacy anchor dimensions (6290760)
78          checkOutput("pkg2/A.html", true,
79                  """
80                      <section class="detail" id="f(T[])">
81                      <h3 id="f(java.lang.Object[])">f</h3>
82                      """);
83      }
