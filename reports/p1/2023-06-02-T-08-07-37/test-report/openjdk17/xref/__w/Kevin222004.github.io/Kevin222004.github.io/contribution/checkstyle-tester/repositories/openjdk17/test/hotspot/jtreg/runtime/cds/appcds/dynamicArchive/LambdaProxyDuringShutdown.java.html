25  /*
26   * @test
27   * @bug 8261860
28   * @summary VM should not crash if a lambda proxy class is created during
29   *          shutdown and its nest host is not linked.
30   * @requires vm.cds
31   * @library /test/lib /test/hotspot/jtreg/runtime/cds/appcds
32   *          /test/hotspot/jtreg/runtime/cds/appcds/dynamicArchive/test-classes
33   * @build LambdaProxyDuringShutdownApp sun.hotspot.WhiteBox LambdaVerification
34   * @run driver jdk.test.lib.helpers.ClassFileInstaller -jar lambda_proxy_shutdown.jar LambdaVerification
35   *             LambdaProxyDuringShutdownApp MyShutdown Outer Outer$Inner
36   * @run driver jdk.test.lib.helpers.ClassFileInstaller -jar WhiteBox.jar sun.hotspot.WhiteBox
37   * @run driver jdk.test.lib.helpers.ClassFileInstaller sun.hotspot.WhiteBox
38   * @run main/othervm -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI -Xbootclasspath/a:. LambdaProxyDuringShutdown
39   */
40  
41  import jdk.test.lib.helpers.ClassFileInstaller;
42  
43  public class LambdaProxyDuringShutdown extends DynamicArchiveTestBase {
44      public static void main(String[] args) throws Exception {
45          runTest(LambdaProxyDuringShutdown::test);
46      }
47  
48      static void test() throws Exception {
49          String topArchiveName = getNewArchiveName();
50          String appJar = ClassFileInstaller.getJarPath("lambda_proxy_shutdown.jar");
51          String mainClass = "LambdaProxyDuringShutdownApp";
52          String wbJar = ClassFileInstaller.getJarPath("WhiteBox.jar");
53          String use_whitebox_jar = "-Xbootclasspath/a:" + wbJar;
54          String appOutput = "Hello from Inner";
55  
56          dump(topArchiveName,
57              "-XX:+UnlockDiagnosticVMOptions",
58              "-XX:+WhiteBoxAPI",
59              "-Xlog:class+load=debug,cds=debug,cds+dynamic=info",
60              use_whitebox_jar,
61              "-cp", appJar, mainClass)
62              .assertNormalExit(output -> {
63                  // Nest host should be skipped since it is not in the linked state.
64                  output.shouldContain("Skipping Outer: Not linked")
65                  // Lambda proxy is loaded normally.
66                        .shouldMatch("class.load.*Outer[$]Inner[$][$]Lambda[$].*0x.*source:.Outer")
67                        .shouldContain(appOutput)
68                        .shouldHaveExitValue(0);
69              });
70  
71          run(topArchiveName,
72              "-XX:+UnlockDiagnosticVMOptions",
73              "-XX:+WhiteBoxAPI",
74              use_whitebox_jar,
75              "-Xlog:class+load=debug",
76              "-cp", appJar, mainClass, "run")
77              .assertNormalExit(output -> {
78                  // Only the Inner class is loaded from the dynamic archive.
79                  // The nest host (Outer) and its lambda proxy are not loaded
80                  // from the dynamic archive.
81                  output.shouldMatch("class.load.*Outer.source:.*lambda_proxy_shutdown.jar")
82                        .shouldMatch("class.load.*Outer[$]Inner[$][$]Lambda[$].*0x.*source:.Outer")
83                        .shouldMatch("class.load. Outer[$]Inner.source:.*shared.*objects.*file.*(top)")
84                        .shouldContain(appOutput)
85                        .shouldHaveExitValue(0);
86              });
87      }
88  }
