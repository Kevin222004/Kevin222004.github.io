172     public static void main(String[] args) throws Exception {
173 
174         ValidatePathWithParams pathValidator = new ValidatePathWithParams(null);
175 
176         if (args.length >= 1 && "CRL".equalsIgnoreCase(args[0])) {
177             pathValidator.enableCRLCheck();
178         } else {
179             // OCSP check by default
180             pathValidator.enableOCSPCheck();
181         }
182 
183         // Validate valid
184         pathValidator.validate(new String[]{VALID, INT},
185                 ValidatePathWithParams.Status.GOOD, null, System.out);
186 
187         // Validate Revoked
188         pathValidator.validate(new String[]{REVOKED, INT},
189                 ValidatePathWithParams.Status.REVOKED,
190                 "Thu Dec 22 23:14:55 PST 2016", System.out);
191 
192         // reset validation date back to current date
193         pathValidator.resetValidationDate();
194     }
