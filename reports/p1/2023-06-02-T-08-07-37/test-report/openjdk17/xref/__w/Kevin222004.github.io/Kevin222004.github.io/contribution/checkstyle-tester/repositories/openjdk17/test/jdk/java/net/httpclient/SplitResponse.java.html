61  /**
62   * Similar test to QuickResponses except that each byte of the response
63   * is sent in a separate packet, which tests the stability of the implementation
64   * for receiving unusual packet sizes. Additionally, tests scenarios there
65   * connections that are retrieved from the connection pool may reach EOF before
66   * being reused.
67   */
68  public class SplitResponse {
69  
70      static String response(String body, boolean serverKeepalive) {
71          StringBuilder sb = new StringBuilder();
72          sb.append("HTTP/1.1 200 OK\r\n");
73          if (!serverKeepalive)
74              sb.append("Connection: Close\r\n");
75  
76          sb.append("Content-length: ")
77                  .append(body.getBytes(ISO_8859_1).length)
78                  .append("\r\n");
79          sb.append("\r\n");
80          sb.append(body);
81          return sb.toString();
82      }

100     SplitResponse(boolean useSSL) throws IOException {
101         this.useSSL = useSSL;
102         context = new SimpleSSLContext().get();
103         SSLContext.setDefault(context);
104         factory = useSSL ? SSLServerSocketFactory.getDefault()
105                          : ServerSocketFactory.getDefault();
106     }

108     public HttpClient newHttpClient() {
109         HttpClient client;
110         if (useSSL) {
111             client = HttpClient.newBuilder()
112                                .sslContext(context)
113                                .build();
114         } else {
115             client = HttpClient.newHttpClient();
116         }
117         return client;
118     }

132     public static void main(String[] args) throws Exception {
133         boolean useSSL = false;
134         if (args != null && args.length >= 1) {
135             useSSL = Protocol.valueOf(args[0]).equals(Protocol.HTTPS);
136         } else {
137             args = new String[] {"HTTP", "connection:KEEP_ALIVE:CLOSE", "mode:SYNC:ASYNC"};
138         }
139 
140         LinkedHashSet<Mode> modes = new LinkedHashSet<>();
141         LinkedHashSet<Connection> keepAlive = new LinkedHashSet<>();
142         Stream.of(args).skip(1).forEach(s -> {
143             if (s.startsWith("connection:")) {
144                 Stream.of(s.split(":")).skip(1).forEach(c -> {
145                     keepAlive.add(Connection.valueOf(c));
146                 });
147             } else if (s.startsWith("mode:")) {
148                 Stream.of(s.split(":")).skip(1).forEach(m -> {
149                     modes.add(Mode.valueOf(m));
150                 });
151             } else {
152                 System.err.println("Illegal argument: " + s);
153                 System.err.println("Allowed syntax is: HTTP|HTTPS [connection:KEEP_ALIVE[:CLOSE]] [mode:SYNC[:ASYNC]");
154                 throw new IllegalArgumentException(s);
155             }
156         });
157 
158         if (keepAlive.isEmpty()) keepAlive.addAll(EnumSet.allOf(Connection.class));
159         if (modes.isEmpty()) modes.addAll(EnumSet.allOf(Mode.class));
160 
161         SplitResponse sp = new SplitResponse(useSSL);
162 
163         for (Version version : Version.values()) {
164             for (Connection serverKeepalive : keepAlive) {
165                 // Note: the mock server doesn't support Keep-Alive, but
166                 // pretending that it might exercises code paths in and out of
167                 // the connection pool, and retry logic
168                 for (Mode mode : modes) {
169                     sp.test(version,serverKeepalive == Connection.KEEP_ALIVE,mode == Mode.ASYNC);
170                 }
171             }
172         }
173     }

256     // Sends the response, mostly, one byte at a time with a small delay
257     // between bytes, to encourage that each byte is read in a separate read
258     Thread sendSplitResponse(String s, MockServer server) {
259         System.out.println("Server: creating new thread to send ... ");
260         Thread t = new Thread(() -> {
261             System.out.println("Server: waiting for server to receive headers");
262             conn = server.activity();
263             System.out.println("Server: Start sending response");
264 
265             try {
266                 int len = s.length();
267                 out.println("Server: going to send [" + s + "]");
268                 for (int i = 0; i < len; i++) {
269                     String onechar = s.substring(i, i + 1);
270                     try {
271                         conn.send(onechar);
272                     } catch(SocketException | SSLException x) {
273                         if (!useSSL || i != len - 1) throw x;
274                         if (x.getMessage().contains("closed by remote host")) {
275                             String osname = System.getProperty("os.name", "unknown");
276                             // On Solaris we can receive an exception when
277                             // the client closes the connection after receiving
278                             // the last expected char.
279                             if (osname.contains("SunO")) {
280                                 System.out.println(osname + " detected");
281                                 System.out.println("WARNING: ignoring " + x);
282                                 System.err.println(osname + " detected");
283                                 System.err.println("WARNING: ignoring " + x);
284                             }
285                         }
286                     }
287                     Thread.sleep(10);
288                 }
289                 out.println("Server: sent [" + s + "]");
290             } catch (IOException | InterruptedException e) {
291                 throw new RuntimeException(e);
292             }
293         });
294         t.setDaemon(true);
295         t.start();
296         return t;
297     }
