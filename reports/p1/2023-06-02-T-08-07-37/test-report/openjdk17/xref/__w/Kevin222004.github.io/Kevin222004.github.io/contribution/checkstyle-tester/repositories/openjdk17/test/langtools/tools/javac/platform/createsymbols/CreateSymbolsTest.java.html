38  import java.io.IOException;
39  import java.net.URL;
40  import java.net.URLClassLoader;
41  import java.nio.file.FileVisitResult;
42  import java.nio.file.FileVisitor;
43  import java.nio.file.Files;
44  import java.nio.file.Path;
45  import java.nio.file.Paths;
46  import java.nio.file.attribute.BasicFileAttributes;
47  import java.util.ArrayList;
48  import java.util.List;
49  import javax.tools.JavaCompiler;
50  import javax.tools.StandardJavaFileManager;
51  import javax.tools.ToolProvider;
52  
53  public class CreateSymbolsTest {
54  
55      static final String CREATE_SYMBOLS_NAME = "symbolgenerator.CreateSymbols";
56  
57      public static void main(String... args) throws Exception {
58          new CreateSymbolsTest().doRun();
59      }
60  
61      void doRun() throws Exception {
62          Path testClasses = Paths.get(System.getProperty("test.classes"));
63          Path compileDir = testClasses.resolve("data");
64          deleteRecursively(compileDir);
65          Files.createDirectories(compileDir);
66          Path createSymbols = findFile("../../make/langtools/src/classes/build/tools/symbolgenerator/CreateSymbols.java");
67  
68          if (createSymbols == null) {
69              System.err.println("Warning: cannot find CreateSymbols, skipping.");
70              return ;
71          }
72  
73          Path createTestImpl = findFile("tools/javac/platform/createsymbols/CreateSymbolsTestImpl.java");
74  
75          if (createTestImpl == null) {
76              throw new AssertionError("Warning: cannot find CreateSymbolsTestImpl, skipping.");
77          }
78  
79          Path toolBox = findFile("tools/lib/toolbox/");
80  
81          if (toolBox == null) {
82              throw new AssertionError("Warning: cannot find ToolBox, skipping.");
83          }
84  
85          JavaCompiler compiler = ToolProvider.getSystemJavaCompiler();
86  
87          try (StandardJavaFileManager fm = compiler.getStandardFileManager(null, null, null)) {
88              List<Path> files = new ArrayList<>();
89  
90              files.add(createSymbols);
91              files.add(createTestImpl);
92  
93              files.add(toolBox.resolve("AbstractTask.java"));
94              files.add(toolBox.resolve("JavacTask.java"));
95              files.add(toolBox.resolve("Task.java"));
96              files.add(toolBox.resolve("ToolBox.java"));
97  
98              Boolean res =
99                      compiler.getTask(null,
100                                       null,
101                                       null,
102                                       List.of("-d",
103                                               compileDir.toAbsolutePath().toString(),
104                                               "-g",
105                                               "--add-modules", "jdk.jdeps",
106                                               "--add-exports", "jdk.compiler/com.sun.tools.javac.api=ALL-UNNAMED",
107                                               "--add-exports", "jdk.compiler/com.sun.tools.javac.file=ALL-UNNAMED",
108                                               "--add-exports", "jdk.compiler/com.sun.tools.javac.jvm=ALL-UNNAMED",
109                                               "--add-exports", "jdk.compiler/com.sun.tools.javac.main=ALL-UNNAMED",
110                                               "--add-exports", "jdk.compiler/com.sun.tools.javac.util=ALL-UNNAMED",
111                                               "--add-exports", "jdk.jdeps/com.sun.tools.classfile=ALL-UNNAMED"),
112                                       null,
113                                       fm.getJavaFileObjectsFromPaths(files)
114                                     ).call();
115             if (!res) {
116                 throw new IllegalStateException("Cannot compile test.");
117             }
118         }
119 
120         URLClassLoader cl = new URLClassLoader(new URL[] {testClasses.toUri().toURL(), compileDir.toUri().toURL()});
121         Class<?> createSymbolTest = cl.loadClass("CreateSymbolsTestImpl");
122 
123         createSymbolTest.getMethod("main", String[].class).invoke(null, (Object) new String[0]);
124     }
125 
126     Path findFile(String path) {
127         Path testSrc = Paths.get(System.getProperty("test.src", "."));
128 
129         for (Path d = testSrc; d != null; d = d.getParent()) {
130             if (Files.exists(d.resolve("TEST.ROOT"))) {
131                 Path file = d.resolve(path);
132                 if (Files.exists(file)) {
133                     return file;
134                 }
135             }
136         }
137 
138         return null;
139     }
140 
141     void deleteRecursively(Path dir) throws IOException {
142         Files.walkFileTree(dir, new FileVisitor<Path>() {
143             @Override
144             public FileVisitResult preVisitDirectory(Path dir, BasicFileAttributes attrs) throws IOException {
145                 return FileVisitResult.CONTINUE;
146             }
147             @Override public FileVisitResult visitFile(Path file, BasicFileAttributes attrs) throws IOException {
148                 Files.delete(file);
149                 return FileVisitResult.CONTINUE;
150             }
151             @Override public FileVisitResult visitFileFailed(Path file, IOException exc) throws IOException {
152                 return FileVisitResult.CONTINUE;
153             }
154             @Override public FileVisitResult postVisitDirectory(Path dir, IOException exc) throws IOException {
155                 Files.delete(dir);
156                 return FileVisitResult.CONTINUE;
157             }
158         });
159     }
