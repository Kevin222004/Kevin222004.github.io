221     public static void main(String[] args) throws Exception {
222 
223         ValidatePathWithParams pathValidator = new ValidatePathWithParams(null);
224         boolean ocspEnabled = false;
225 
226         if (args.length >= 1 && "CRL".equalsIgnoreCase(args[0])) {
227             pathValidator.enableCRLCheck();
228         } else {
229             // OCSP check by default
230             pathValidator.enableOCSPCheck();
231             ocspEnabled = true;
232         }
233 
234         // Validate valid
235         pathValidator.validate(new String[]{VALID, INT_VALID},
236                 ValidatePathWithParams.Status.GOOD, null, System.out);
237 
238         // Revoked certificate is using SHA1 signature
239         if (ocspEnabled) {
240             // Revoked test certificate is expired
241             // and backdated revocation check is only possible with OCSP
242             pathValidator.setValidationDate("July 01, 2016");
243         }
244 
245         // Validate Revoked
246         pathValidator.validate(new String[]{REVOKED, INT_REVOKED},
247                 ValidatePathWithParams.Status.REVOKED,
248                 "Fri Jan 29 01:06:42 PST 2016", System.out);
249 
250         // reset validation date back to current date
251         pathValidator.resetValidationDate();
252     }
