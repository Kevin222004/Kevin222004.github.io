25  /*
26   * @test
27   * @summary When dumping the CDS archive, try to load VM anonymous classes to make sure they
28   *          are handled properly. Note: these are not "anonymous inner classes" in the Java source code,
29   *          but rather classes that are not recorded in any ClassLoaderData::dictionary(),
30   *          such as classes that are generated for Lambda expressions.
31   *          See https://blogs.oracle.com/jrose/anonymous-classes-in-the-vm.
32   * @library /test/lib /test/hotspot/jtreg/runtime/cds/appcds /test/hotspot/jtreg/runtime/cds/appcds/test-classes
33   * @requires vm.cds
34   * @requires vm.jvmti
35   * @run driver AnonVmClassesDuringDump
36   */
37  
38  import jdk.test.lib.helpers.ClassFileInstaller;
39  
40  public class AnonVmClassesDuringDump {
41      public static String appClasses[] = {
42          Hello.class.getName(),
43      };
44      public static String agentClasses[] = {
45          AnonVmClassesDuringDumpTransformer.class.getName(),
46      };
47  
48      public static String cdsDiagnosticOption = "-XX:+AllowArchivingWithJavaAgent";
49  
50      public static final boolean dynamicMode =
51          Boolean.getBoolean(System.getProperty("test.dynamic.cds.archive", "false"));
52  
53      public static void main(String[] args) throws Throwable {
54          String agentJar =
55              ClassFileInstaller.writeJar("AnonVmClassesDuringDumpTransformer.jar",
56                                          ClassFileInstaller.Manifest.fromSourceFile("AnonVmClassesDuringDumpTransformer.mf"),
57                                          agentClasses);
58  
59          String appJar =
60              ClassFileInstaller.writeJar("AnonVmClassesDuringDumpApp.jar", appClasses);
61  
62          TestCommon.testDump(appJar, TestCommon.list(Hello.class.getName()),
63                              "-javaagent:" + agentJar,
64                              "-XX:+UnlockDiagnosticVMOptions", cdsDiagnosticOption,
65                              // Set the following property to see logs for dynamically generated classes
66                              // in STDOUT
67                              "-Djava.lang.invoke.MethodHandle.DUMP_CLASS_FILES=true");
68  
69          String prefix = ".class.load. ";
70          // class name pattern like the following:
71          // jdk.internal.loader.BuiltinClassLoader$$Lambda$1/1816757085
72          // java.lang.invoke.LambdaForm$MH/1585787493
73          String class_pattern = ".*Lambda([a-z0-9$]+)/([0-9]+).*";
74          String suffix = ".*source: shared objects file.*";
75          String pattern = prefix + class_pattern + suffix;
76          // during run time, anonymous classes shouldn't be loaded from the archive
77          TestCommon.run("-cp", appJar,
78              "-XX:+UnlockDiagnosticVMOptions", cdsDiagnosticOption, Hello.class.getName())
79              .assertNormalExit(dynamicMode ?
80                  output -> output.shouldMatch(pattern) :
81                  output -> output.shouldNotMatch(pattern));
82  
83          // inspect the archive and make sure no anonymous class is in there
84          TestCommon.run("-cp", appJar,
85              "-XX:+UnlockDiagnosticVMOptions", cdsDiagnosticOption,
86              "-XX:+PrintSharedArchiveAndExit", "-XX:+PrintSharedDictionary", Hello.class.getName())
87              .assertNormalExit(dynamicMode ?
88                  output -> output.shouldMatch(pattern) :
89                  output -> output.shouldNotMatch(pattern));
90      }
91  }
