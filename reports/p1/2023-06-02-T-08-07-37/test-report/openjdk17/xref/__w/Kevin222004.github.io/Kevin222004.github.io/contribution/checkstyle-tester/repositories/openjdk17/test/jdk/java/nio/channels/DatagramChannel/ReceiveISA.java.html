24  /*
25   * @test
26   * @bug 4503641 8130394 8249773
27   * @summary Check that DatagramChannel.receive returns a new SocketAddress
28   *          when it receives a packet from the same source address but
29   *          different endpoint.
30   */
31  import java.nio.*;
32  import java.nio.channels.*;
33  import java.net.*;
34  import static java.lang.System.out;
35  
36  public class ReceiveISA {
37  
38      public static void main(String args[]) throws Exception {
39  
40          String regex = "Dia duit![0-2]";
41  
42          try (DatagramChannel dc1 = DatagramChannel.open();        // client
43               DatagramChannel dc2 = DatagramChannel.open();        // client
44               DatagramChannel dc3 = DatagramChannel.open();
45               DatagramChannel dc4 = DatagramChannel.open()) {      // client
46  
47              dc3.socket().bind((SocketAddress) null); // bind server to any port
48  
49              // get server address
50              InetAddress lh = InetAddress.getLocalHost();
51              InetSocketAddress isa = new InetSocketAddress(lh, dc3.socket().getLocalPort());
52  
53              ByteBuffer bb = ByteBuffer.allocateDirect(100);
54              bb.put("Dia duit!0".getBytes());
55              bb.flip();
56  
57              ByteBuffer bb1 = ByteBuffer.allocateDirect(100);
58              bb1.put("Dia duit!1".getBytes());
59              bb1.flip();
60  
61              ByteBuffer bb2 = ByteBuffer.allocateDirect(100);
62              bb2.put("Dia duit!2".getBytes());
63              bb2.flip();
64  
65              ByteBuffer bb3 = ByteBuffer.allocateDirect(100);
66              bb3.put("garbage".getBytes());
67              bb3.flip();
68  
69              dc1.send(bb, isa);      // packet 1 from dc1
70              dc4.send(bb3, isa);     // interference, packet 4 from dc4
71              dc1.send(bb1, isa);     // packet 2 from dc1
72              dc2.send(bb2, isa);     // packet 3 from dc2
73  
74  
75              // receive 4 packets
76              dc3.socket().setSoTimeout(1000);
77              ByteBuffer rb = ByteBuffer.allocateDirect(100);
78              SocketAddress sa[] = new SocketAddress[3];
79  
80              for (int i = 0; i < 3;) {
81                  SocketAddress receiver = dc3.receive(rb);
82                  rb.flip();
83                  byte[] bytes = new byte[rb.limit()];
84                  rb.get(bytes, 0, rb.limit());
85                  String msg = new String(bytes);
86  
87                  if (msg.matches("Dia duit![0-2]")) {
88                      if (msg.equals("Dia duit!0")) {
89                          sa[0] = receiver;
90                          i++;
91                      }
92                      if (msg.equals("Dia duit!1")) {
93                          sa[1] = receiver;
94                          i++;
95                      }
96                      if (msg.equals("Dia duit!2")) {
97                          sa[2] = receiver;
98                          i++;
99                      }
100                 } else {
101                     out.println("Interfered packet sender address is : " + receiver);
102                     out.println("random interfered packet is : " + msg);
103                 }
104                 rb.clear();
105             }
106 
107             /*
108              * Check that sa[0] equals sa[1] (both from dc1)
109              * Check that sa[1] not equal to sa[2] (one from dc1, one from dc2)
110              */
111 
112             if (!sa[0].equals(sa[1])) {
113                 throw new Exception("Source address for packets 1 & 2 should be equal");
114             }
115 
116             if (sa[1].equals(sa[2])) {
117                 throw new Exception("Source address for packets 2 & 3 should be different");
118             }
119         }
120     }
121 }
