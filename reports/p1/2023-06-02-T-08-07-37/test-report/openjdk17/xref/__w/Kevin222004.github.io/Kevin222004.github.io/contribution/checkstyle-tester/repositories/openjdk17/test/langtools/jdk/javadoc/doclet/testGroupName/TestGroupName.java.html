24  /*
25   * @test
26   * @bug 8190003 8196201 8196202 8184205
27   * @summary Special characters in group names should be escaped
28   * @library /tools/lib ../../lib
29   * @modules jdk.javadoc/jdk.javadoc.internal.tool
30   * @build toolbox.ToolBox javadoc.tester.*
31   * @run main TestGroupName
32   */
33  
34  import java.io.IOException;
35  import java.nio.file.Path;
36  import java.nio.file.Paths;
37  
38  import javadoc.tester.JavadocTester;
39  import toolbox.ToolBox;
40  
41  public class TestGroupName extends JavadocTester {
42  
43      public final ToolBox tb;
44      public static void main(String... args) throws Exception {
45          TestGroupName tester = new TestGroupName();
46          tester.runTests(m -> new Object[] { Paths.get(m.getName()) });
47      }
48  
49      public TestGroupName() {
50          tb = new ToolBox();
51      }
52  
53      @Test
54      public void testPackageGroups(Path base) throws IOException {
55          Path src = base.resolve("src");
56          tb.writeJavaFiles(src,
57                  "package p1; public class C1 { }",
58                  "package p2; public class C2 { }",
59                  "package p3; public class C3 { }");
60  
61          javadoc("-d", base.resolve("out").toString(),
62                  "-sourcepath", src.toString(),
63                  "-group", "abc < & > def", "p1",
64                  "p1", "p2", "p3");
65          checkExit(Exit.OK);
66  
67          checkOutput("index.html", true,
68                  """
69                      <button id="all-packages-table-tab1" role="tab" aria-selected="false" aria-contr\
70                      ols="all-packages-table.tabpanel" tabindex="-1" onkeydown="switchTab(event)" onc\
71                      lick="show('all-packages-table', 'all-packages-table-tab1', 2)" class="table-tab\
72                      ">abc &lt; &amp; &gt; def</button>""");
73      }
74  
75      @Test
76      public void testModuleGroups(Path base) throws IOException {
77          Path src = base.resolve("src");
78          tb.writeJavaFiles(src.resolve("ma"),
79                  "module ma { exports pa1; }",
80                  "package pa1; public class CA1 { }",
81                  "package pa2; public class CA2 { }",
82                  "package pa3; public class CA3 { }");
83  
84          tb.writeJavaFiles(src.resolve("mb"),
85                  "module mb { exports pb1; }",
86                  "package pb1; public class CB1 { }",
87                  "package pb2; public class CB2 { }",
88                  "package pb3; public class CB3 { }");
89  
90          tb.writeJavaFiles(src.resolve("mc"),
91                  "module mc { exports pc1; }",
92                  "package pc1; public class CC1 { }",
93                  "package pc2; public class CC2 { }",
94                  "package pc3; public class CC3 { }");
95  
96          javadoc("-d", base.resolve("out").toString(),
97                  "--module-source-path", src.toString(),
98                  "-group", "abc < & > def", "ma",
99                  "--module", "ma,mb,mc");
100 
101         checkExit(Exit.OK);
102 
103         checkOutput("index.html", true,
104                 """
105                     <button id="all-modules-table-tab2" role="tab" aria-selected="false" aria-contro\
106                     ls="all-modules-table.tabpanel" tabindex="-1" onkeydown="switchTab(event)" oncli\
107                     ck="show('all-modules-table', 'all-modules-table-tab2', 2)" class="table-tab">Ot\
108                     her Modules</button>""");
109     }
110 }
