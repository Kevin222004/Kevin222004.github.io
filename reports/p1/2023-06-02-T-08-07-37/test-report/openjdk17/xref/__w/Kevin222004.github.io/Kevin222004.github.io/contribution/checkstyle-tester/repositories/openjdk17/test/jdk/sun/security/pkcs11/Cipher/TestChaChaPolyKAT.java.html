122     @Override
123     public void main(Provider p) throws Exception {
124         System.out.println("Testing " + p.getName());
125 
126         try {
127             Cipher.getInstance(ALGO, p);
128         } catch (NoSuchAlgorithmException nsae) {
129             System.out.println("Skip; no support for " + ALGO);
130             return;
131         }
132 
133         int testsPassed = 0;
134         int testNumber = 0;
135 
136         System.out.println("----- AEAD Tests -----");
137         for (TestData test : aeadTestList) {
138             System.out.println("*** Test " + ++testNumber + ": " +
139                     test.testName);
140             if (runAEADTest(p, test)) {
141                 testsPassed++;
142             }
143         }
144         System.out.println();
145 
146         System.out.println("Total tests: " + testNumber +
147                 ", Passed: " + testsPassed + ", Failed: " +
148                 (testNumber - testsPassed));
149         if (testsPassed != testNumber) {
150             throw new RuntimeException("One or more tests failed.  " +
151                     "Check output for details");
152         }
153     }

155     private static boolean runAEADTest(Provider p, TestData testData)
156             throws GeneralSecurityException {
157         boolean result = false;
158 
159         Cipher mambo = Cipher.getInstance(ALGO, p);
160         SecretKeySpec mamboKey = new SecretKeySpec(testData.key, "ChaCha20");
161         IvParameterSpec mamboSpec = new IvParameterSpec(testData.nonce);
162 
163         mambo.init(testData.direction, mamboKey, mamboSpec);
164 
165         byte[] out = new byte[mambo.getOutputSize(testData.input.length)];
166         int outOff = 0;
167         try {
168             mambo.updateAAD(testData.aad);
169             outOff += mambo.update(testData.input, 0, testData.input.length,
170                     out, outOff);
171             outOff += mambo.doFinal(out, outOff);
172         } catch (AEADBadTagException abte) {
173             // If we get a bad tag or derive a tag mismatch, log it
174             // and register it as a failure
175             System.out.println("FAIL: " + abte);
176             return false;
177         }
178 
179         if (!Arrays.equals(out, 0, outOff, testData.expOutput, 0, outOff)) {
180             System.out.println("ERROR - Output Mismatch!");
181             System.out.println("Expected:\n" +
182                     dumpHexBytes(testData.expOutput, 16, "\n", " "));
183             System.out.println("Actual:\n" +
184                     dumpHexBytes(out, 16, "\n", " "));
185             System.out.println();
186         } else {
187             result = true;
188         }
189 
190         return result;
191     }

193     /**
194      * Dump the hex bytes of a buffer into string form.
195      *
196      * @param data The array of bytes to dump to stdout.
197      * @param itemsPerLine The number of bytes to display per line
198      *      if the {@code lineDelim} character is blank then all bytes
199      *      will be printed on a single line.
200      * @param lineDelim The delimiter between lines
201      * @param itemDelim The delimiter between bytes
202      *
203      * @return The hexdump of the byte array
204      */
205     private static String dumpHexBytes(byte[] data, int itemsPerLine,
206             String lineDelim, String itemDelim) {
207         return dumpHexBytes(ByteBuffer.wrap(data), itemsPerLine, lineDelim,
208                 itemDelim);
209     }
210 
211     private static String dumpHexBytes(ByteBuffer data, int itemsPerLine,
212             String lineDelim, String itemDelim) {
213         StringBuilder sb = new StringBuilder();
214         if (data != null) {
215             data.mark();
216             int i = 0;
217             while (data.remaining() > 0) {
218                 if (i % itemsPerLine == 0 && i != 0) {
219                     sb.append(lineDelim);
220                 }
221                 sb.append(String.format("%02X", data.get())).append(itemDelim);
222                 i++;
223             }
224             data.reset();
225         }
226 
227         return sb.toString();
228     }
