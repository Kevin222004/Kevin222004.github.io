25   * @test
26   * @bug 8006263
27   * @summary Supplementary test cases needed for doclint
28   * @modules jdk.javadoc/jdk.javadoc.internal.doclint

33  import java.io.ByteArrayOutputStream;
34  import java.io.IOException;
35  import java.io.PrintStream;
36  import java.io.PrintWriter;
37  import java.io.StringWriter;
38  import java.net.URI;
39  import java.security.Permission;
40  import java.util.Arrays;
41  import java.util.List;
42  import java.util.Objects;
43  import javax.tools.JavaFileObject;
44  import javax.tools.SimpleJavaFileObject;
45  
46  import com.sun.source.util.JavacTask;
47  import com.sun.tools.javac.api.JavacTool;
48  
49  import jdk.javadoc.internal.doclint.DocLint;
50  import jdk.javadoc.internal.doclint.DocLint.BadArgs;
51  
52  public class RunTest {
53      static class SimpleSecurityManager extends SecurityManager {
54          boolean allowExit = false;
55  
56          @Override
57          public void checkExit(int status) {
58              if (!allowExit)
59                  throw new SecurityException("System.exit(" + status + ")");
60          }
61          @Override
62          public void checkPermission(Permission perm) { }
63  
64      }
65  
66      public static void main(String... args) throws Exception {
67          // if no security manager already installed, install one to
68          // prevent System.exit
69          SimpleSecurityManager secmgr = null;
70          if (System.getSecurityManager() == null) {
71              System.setSecurityManager(secmgr = new SimpleSecurityManager() { });
72          }
73  
74          try {
75              new RunTest().run();
76          } finally {
77              if (secmgr != null)
78                  secmgr.allowExit = true;
79          }
80      }
81  
82      void run() throws Exception {
83          testMain();
84          testRun();
85          testInit();
86          testArgsNoFiles();
87  
88          if (errors > 0)
89              throw new Exception(errors + " errors found");
90      }
91  
92      void testMain() {
93          System.err.println("test main(String[])");
94          testMain(true, "-help");
95          testMain(false, "-unknownOption");
96      }
97  
98      void testMain(boolean expectOK, String... args) {
99          try {
100             DocLint.main(args);
101             if (!expectOK)
102                 error("expected SecurityException (from System.exit) not thrown");
103         } catch (SecurityException e) {
104             System.err.println(e);
105             if (expectOK)
106                 error("unexpected SecurityException caught");
107         }
108     }
109 
110     void testRun() throws BadArgs, IOException {
111         System.err.println("test run(String[])");
112         DocLint dl = new DocLint();
113         String[] args = { "-help" };
114         ByteArrayOutputStream baos = new ByteArrayOutputStream();
115         PrintStream ps = new PrintStream(baos);
116         PrintStream prev = System.out;
117         try {
118             System.setOut(ps);
119             dl.run(args);
120         } finally {
121             System.setOut(prev);
122         }
123         ps.close();
124         String stdout = baos.toString();
125 
126         StringWriter sw = new StringWriter();
127         PrintWriter pw = new PrintWriter(sw);
128         dl.run(pw, args);
129         pw.close();
130         String direct = sw.toString();
131 
132         if (!stdout.equals(direct)) {
133             error("unexpected output");
134             System.err.println("EXPECT>>" + direct + "<<");
135             System.err.println("FOUND>>" + stdout + "<<");
136         }
137     }
138 
139     void testInit() {
140         System.err.println("test init");
141         DocLint dl = new DocLint();
142         String name = dl.getName();
143         if (!Objects.equals(name, "doclint"))
144             error("unexpected result for DocLint.getName()");
145 
146         List<? extends JavaFileObject> files =
147                 Arrays.asList(createFile("Test.java", "/** &0; */ class Test{ }"));
148         String[] goodArgs = { "-Xmsgs" };
149         testInit(true, goodArgs, files);
150 
151         String[] badArgs = { "-unknown" };
152         testInit(false, badArgs, files);
153     }
154 
155     void testInit(boolean expectOK, String[] args, List<? extends JavaFileObject> files) {
156         JavacTool javac = JavacTool.create();
157         JavacTask task = javac.getTask(null, null, null, null, null, files);
158         try {
159             DocLint dl = new DocLint();
160             dl.init(task, args, true);
161             if (!expectOK)
162                 error("expected IllegalArgumentException not thrown");
163             task.call();
164         } catch (IllegalArgumentException e) {
165             System.err.println(e);
166             if (expectOK)
167                 error("unexpected IllegalArgumentException caught");
168         }
169     }
170 
171     void testArgsNoFiles() throws BadArgs, IOException {
172         System.err.println("test args, no files");
173         DocLint dl = new DocLint();
174 
175         StringWriter sw = new StringWriter();
176         PrintWriter pw = new PrintWriter(sw);
177         dl.run(pw, "-Xmsgs");
178         pw.close();
179         String out = sw.toString();
180 
181         String expect = "No files given";
182         if (!Objects.equals(out.trim(), expect)) {
183             error("unexpected output");
184             System.err.println("EXPECT>>" + expect + "<<");
185             System.err.println("FOUND>>" + out + "<<");
186         }
187 
188     }
189 
190     JavaFileObject createFile(String name, final String body) {
191         return new SimpleJavaFileObject(URI.create(name), JavaFileObject.Kind.SOURCE) {
192             @Override
193             public CharSequence getCharContent(boolean ignoreEncodingErrors) throws IOException {
194                 return body;
195             }
196         };
197     }
198 
199     void error(String msg) {
200         System.err.println("Error: " + msg);
201         errors++;
202     }
203 
204     int errors;
205 }
