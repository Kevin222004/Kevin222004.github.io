24  /*
25   * @test
26   * @bug 8044767 8139067 8210408 8263202
27   * @summary Basic tests for ResourceBundle with modules:
28   *          1) Named module "test" contains resource bundles for root and en,
29   *          and separate named modules "eubundles" and "asiabundles" contain
30   *          other resource bundles.
31   *          2) ResourceBundle.getBundle caller is in named module "test",
32   *          resource bundles are grouped in main (module "mainbundles"),
33   *          EU (module "eubundles"), and Asia (module "asiabundles").
34   *          3) ResourceBundle.getBundle caller is in named module "test" and all
35   *          resource bundles are in single named module "bundles".
36   *          4) ResourceBundle.getBundle caller is in named module "test" and all
37   *          resource bundles in xml format are in single named module "bundles".
38   *          5) Resource bundles in a local named module with no ResourceBundleProviders.
39   * @library /test/lib
40   *          ..
41   * @build jdk.test.lib.JDKToolLauncher
42   *        jdk.test.lib.Utils
43   *        jdk.test.lib.compiler.CompilerUtils
44   *        jdk.test.lib.process.ProcessTools
45   *        ModuleTestUtil
46   * @run testng BasicTest

69      private static final String SRC_DIR_XML = "srcXml";

73      private static final String MODS_DIR_APPBASIC2 = "modsAppbasic2";
74      private static final String MODS_DIR_BASIC = "modsBasic";
75      private static final String MODS_DIR_SIMPLE = "modsSimple";
76      private static final String MODS_DIR_XML = "modsXml";
77      private static final String MODS_DIR_MODLOCAL = "modsModlocal";

89      private static final List<String> MODULE_LIST_BASIC = List.of("mainbundles",

95      @DataProvider(name = "basicTestData")
96      Object[][] basicTestData() {
97          return new Object[][] {
98                  // Named module "test" contains resource bundles for root and en,
99                  // and separate named modules "eubundles" and "asiabundles"
100                 // contain other resource bundles.
101                 {SRC_DIR_APPBASIC, MODS_DIR_APPBASIC, MODULE_LIST, LOCALE_LIST,
102                         ".properties"},
103                 {SRC_DIR_APPBASIC2, MODS_DIR_APPBASIC2, MODULE_LIST, LOCALE_LIST,
104                         ".properties"},
105 
106                 // Resource bundles are grouped in main (module "mainbundles"),
107                 // EU (module "eubundles"), and Asia (module "asiabundles").
108                 {SRC_DIR_BASIC, MODS_DIR_BASIC, MODULE_LIST_BASIC, LOCALE_LIST_BASIC,
109                         ".properties"},
110 
111                 // All resource bundles are in single named module "bundles".
112                 {SRC_DIR_SIMPLE, MODS_DIR_SIMPLE, MODULE_LIST_SIMPLE, LOCALE_LIST,
113                         ".properties"},
114 
115                 // All resource bundles in xml format are in single named
116                 // module "bundles".
117                 {SRC_DIR_XML, MODS_DIR_XML, MODULE_LIST_SIMPLE, LOCALE_LIST, ".xml"},
118 
119                 // Resource bundles local in named module "test".
120                 {SRC_DIR_MODLOCAL, MODS_DIR_MODLOCAL, List.of("test"), LOCALE_LIST,
121                         ".properties"},
122         };
123     }

125     @Test(dataProvider = "basicTestData")
126     public void runBasicTest(String src, String mod, List<String> moduleList,
127             List<String> localeList, String resFormat) throws Throwable {
128         Path srcPath = Paths.get(Utils.TEST_SRC, src);
129         Path modPath = Paths.get(Utils.TEST_CLASSES, mod);
130         moduleList.forEach(mn -> ModuleTestUtil.prepareModule(srcPath, modPath,
131                 mn, resFormat));
132         ModuleTestUtil.runModule(modPath.toString(), MAIN, localeList);
133         ModuleTestUtil.runModuleWithLegacyCode(modPath.toString(), MAIN, localeList);
134     }
135 
136     @Test
137     public void RunBasicTestWithCp() throws Throwable {
138         Path jarPath = Paths.get(Utils.TEST_CLASSES, EXTRA_JAR_BASIC);
139         Path srcPath = Paths.get(Utils.TEST_SRC, SRC_DIR_BASIC);
140         Path modPath = Paths.get(Utils.TEST_CLASSES, MODS_DIR_BASIC);
141         Path classPath = Paths.get(Utils.TEST_CLASSES).resolve("classes")
142                 .resolve("basic");
143 
144         jarBasic(srcPath, classPath, jarPath);
145         // jdk.test.Main should NOT load bundles from the jar file specified
146         // by the class-path.
147         ModuleTestUtil.runModuleWithCp(jarPath.toString(), modPath.toString(),
148                 MAIN, List.of("es", "vi"), false);
149     }
150 
151     @Test
152     public void runModLocalTestWithCp() throws Throwable {
153         Path jarPath = Paths.get(Utils.TEST_CLASSES, EXTRA_JAR_MODLOCAL);
154         Path srcPath = Paths.get(Utils.TEST_SRC, SRC_DIR_MODLOCAL);
155         Path modPath = Paths.get(Utils.TEST_CLASSES, MODS_DIR_MODLOCAL);
156 
157         jarModLocal(srcPath, jarPath);
158         // jdk.test.Main should load bundles from the jar file specified by
159         // the class-path.
160         ModuleTestUtil.runModuleWithCp(jarPath.toString(), modPath.toString(),
161                 MAIN, List.of("vi"), true);
162     }

164     /**
165      * Create extra_basic.jar to be added to the class path. It contains .class
166      * and .properties resource bundles.
167      */
168     private static void jarBasic(Path srcPath, Path classPath, Path jarPath)
169             throws Throwable {
170         boolean compiled = CompilerUtils.compile(srcPath.resolve("extra"),
171                 classPath);
172         assertTrue(compiled, "Compile Java files for extra_basic.jar failed.");
173 
174         JDKToolLauncher launcher = JDKToolLauncher.createUsingTestJDK("jar");
175         launcher.addToolArg("-cf")
176                 .addToolArg(jarPath.toString())
177                 .addToolArg("-C")
178                 .addToolArg(classPath.toString())
179                 .addToolArg("jdk/test/resources/eu")
180                 .addToolArg("-C")
181                 .addToolArg(srcPath.resolve("extra").toString())
182                 .addToolArg("jdk/test/resources/asia");
183 
184         int exitCode = ProcessTools.executeCommand(launcher.getCommand())
185                 .getExitValue();
186         assertEquals(exitCode, 0, "Create extra_basic.jar failed. "
187                 + "Unexpected exit code: " + exitCode);
188     }
189 
190     /**
191      * Create extra_modlocal.jar to be added to the class path. Expected
192      * properties files are picked up from the class path.
193      */
194     private static void jarModLocal(Path srcPath, Path jarPath) throws Throwable {
195         JDKToolLauncher launcher = JDKToolLauncher.createUsingTestJDK("jar");
196         launcher.addToolArg("-cf")
197                 .addToolArg(jarPath.toString())
198                 .addToolArg("-C")
199                 .addToolArg(srcPath.resolve("extra").toString())
200                 .addToolArg("jdk/test/resources");
201 
202         int exitCode = ProcessTools.executeCommand(launcher.getCommand())
203                 .getExitValue();
204         assertEquals(exitCode, 0, "Create extra_modlocal.jar failed. "
205                 + "Unexpected exit code: " + exitCode);
206     }
