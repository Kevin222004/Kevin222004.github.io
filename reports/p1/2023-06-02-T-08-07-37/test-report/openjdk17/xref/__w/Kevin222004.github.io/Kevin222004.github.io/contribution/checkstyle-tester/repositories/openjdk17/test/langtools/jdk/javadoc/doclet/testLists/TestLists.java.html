24  /*

26   * @bug 8241625 8241895 8242326 8261976
27   * @summary test the lists generated by the doclet
28   * @library  /tools/lib ../../lib

33  
34  import java.io.IOException;
35  import java.nio.file.Path;
36  
37  import javadoc.tester.JavadocTester;
38  import toolbox.ToolBox;
39  
40  public class TestLists extends JavadocTester {
41  
42      public static void main(String... args) throws Exception {
43          TestLists tester = new TestLists();
44          tester.runTests(m -> new Object[]{Path.of(m.getName())});
45      }
46  
47      private final ToolBox tb = new ToolBox();
48  
49      @Test
50      public void testMemberLists(Path base) throws IOException {
51          Path src = base.resolve("src");
52          tb.writeJavaFiles(src,
53                  """
54                      package p; public class C {
55                        public C() { }
56                        public C(int i) { }
57                        public int f1;
58                        public int f2;
59                        public void m1() { }
60                        public void m2() { }
61                      }
62                      """,
63                  """
64                      package p; public enum E { E1, E2 }
65                      """,
66                  """
67                      package p; public @interface A { int value(); }
68                      """
69          );
70  
71          javadoc("-d", base.resolve("out").toString(),
72                  "-sourcepath", src.toString(),
73                  "p");
74          checkExit(Exit.OK);
75  
76          checkOutput("p/C.html", true,
77                  """
78                      <h2>Field Details</h2>
79                      <ul class="member-list">
80                      <li>
81                      <section class="detail" id="f1">
82                      <h3>f1</h3>
83                      """,
84                  """
85                      <h2>Constructor Details</h2>
86                      <ul class="member-list">
87                      <li>
88                      <section class="detail" id="&lt;init&gt;()">
89                      <h3>C</h3>""",
90                  """
91                      <section class="method-details" id="method-detail">
92                      <h2>Method Details</h2>
93                      <ul class="member-list">
94                      <li>
95                      <section class="detail" id="m1()">
96                      <h3>m1</h3>
97                      """);
98  
99          checkOutput("p/E.html", true,
100                 """
101                     <h2>Enum Constant Details</h2>
102                     <ul class="member-list">
103                     <li>
104                     <section class="detail" id="E1">
105                     """);
106 
107         checkOutput("p/A.html", true,
108                 """
109                     <h2>Element Details</h2>
110                     <ul class="member-list">
111                     <li>
112                     <section class="detail" id="value()">""");
113     }
