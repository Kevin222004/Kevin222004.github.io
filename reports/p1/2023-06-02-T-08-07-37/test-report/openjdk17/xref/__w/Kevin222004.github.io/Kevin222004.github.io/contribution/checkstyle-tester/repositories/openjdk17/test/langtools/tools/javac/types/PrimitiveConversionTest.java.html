54      enum Result {
55          OK_BOTH(true),
56          FAIL_BOTH(false),
57          OK_ASSIGN_ONLY(true);
58  
59          boolean value;
60  
61          Result(boolean value) {
62              this.value = value;
63          }
64      }
65  
66      enum ConversionKind {
67          ASSIGNMENT_CONVERSION(EnumSet.of(Result.OK_BOTH, Result.OK_ASSIGN_ONLY)) {
68              @Override
69              void check(TypeHarness harness, Type from, Type to, Result expected) {
70                  harness.assertAssignable(from, to, resSet.contains(expected));
71              }
72          },
73          METHOD_CONVERSION(EnumSet.of(Result.OK_BOTH)) {
74              @Override
75              void check(TypeHarness harness, Type from, Type to, Result expected) {
76                  harness.assertConvertible(from, to, resSet.contains(expected));
77              }
78          };
79  
80          EnumSet<Result> resSet;
81  
82          private ConversionKind(EnumSet<Result> resSet) {
83              this.resSet = resSet;
84          }
85  
86          abstract void check(TypeHarness harness, Type from, Type to, Result expected);
87      }

184     PrimitiveConversionTest() {
185         Type[] primitiveTypes = new Type[] {
186             predef.byteType,
187             predef.shortType,
188             predef.intType,
189             predef.longType,
190             predef.floatType,
191             predef.doubleType,
192             predef.charType,
193             predef.booleanType };
194 
195         ClassType c1 = fac.Class(fac.TypeVariable());
196         ClassType c2 = fac.Class();
197         c2.supertype_field = subst(c1,
198                 Mapping(c1.getTypeArguments().head, predef.stringType));
199         Type c3 = erasure(c1);
200 
201         Type[] referenceTypes = {
202             subst(c1,
203                     Mapping(c1.getTypeArguments().head, predef.stringType)),
204             c2,
205             c3,
206             fac.TypeVariable() };
207 
208         Type[] arrayTypes = new Type[primitiveTypes.length + referenceTypes.length];
209         int idx = 0;
210         for (Type t : join(Type.class, primitiveTypes, referenceTypes)) {
211             arrayTypes[idx++] = fac.Array(t);
212         }
213 
214         types1 = join(Type.class, primitiveTypes, referenceTypes, arrayTypes);
215 
216         types2 = new Type[] {
217             predef.byteType,
218             predef.shortType,
219             fac.Constant((short)0x0001),

248     void testConversion(ConversionKind convKind, TestKind testKind) {
249         Type[] rows = testKind.getFromTypes(this);
250         Type[] cols = testKind.getToTypes(this);
251         for (int i = 0; i < rows.length ; i++) {
252             for (int j = 0; j < cols.length ; j++) {
253                 convKind.check(this, rows[i], cols[j], testKind.getResults(this)[i][j]);
254             }
255         }
256     }

258     @SuppressWarnings("unchecked")
259     <T> T[] join(Class<T> type, T[]... args) {
260         int totalLength = 0;
261         for (T[] arr : args) {
262             totalLength += arr.length;
263         }
264         T[] new_arr = (T[])Array.newInstance(type, totalLength);
265         int idx = 0;
266         for (T[] arr : args) {
267             System.arraycopy(arr, 0, new_arr, idx, arr.length);
268             idx += arr.length;
269         }
270         return new_arr;
271     }
