102     void checkCommon(boolean checkC5) {
103         // Test common to all runs of javadoc. The class signature should print
104         // properly enclosed definition list tags and the Annotation Type
105         // Optional Element should print properly nested definition list tags
106         // for default value.
107         checkOutput("pkg1/C1.html", true,
108                 """
109                     <div class="type-signature"><span class="modifiers">public class </span><span class="element-name type-name-label">C1</span>
110                     <span class="extends-implements">extends java.lang.Object
111                     implements java.io.Serializable</span></div>""");
112         checkOutput("pkg1/C4.html", true,
113                 """
114                     <dl class="notes">
115                     <dt>Default:</dt>
116                     <dd>true</dd>
117                     </dl>""");
118 
119         // Test for valid HTML generation which should not comprise of empty
120         // definition list tags.
121         List<String> files= new ArrayList<>(Arrays.asList(
122             "pkg1/package-summary.html",
123             "pkg1/C1.html",
124             "pkg1/C1.ModalExclusionType.html",
125             "pkg1/C2.html",
126             "pkg1/C2.ModalType.html",
127             "pkg1/C3.html",
128             "pkg1/C4.html",
129             "overview-tree.html",
130             "serialized-form.html"
131         ));
132 
133         if (checkC5)
134             files.add("pkg1/C5.html");
135 
136         for (String f: files) {
137             checkOutput(f, false,
138                     "<dl></dl>",
139                     "<dl>\n</dl>",
140                     "<dl class=\"notes\"></dl>",
141                     "<dl class=\"notes\">\n</dl>");
142         }
143     }

381     void checkNoCommentNoDeprecated(boolean expectFound) {
382         // Test with -nocomment and -nodeprecated options. The ClassDocs whould
383         // not display definition lists for any member details.
384         checkOutput("pkg1/C1.html", expectFound,
385                 """
386                     <div class="member-signature"><span class="modifiers">public</span>&nbsp;<span c\
387                     lass="return-type">void</span>&nbsp;<span class="element-name">readObject</span>()
388                                     throws <span class="exceptions">java.io.IOException</span></div>
389                     </section>
390                     </li>""");
391 
392         checkOutput("pkg1/C2.html", expectFound,
393                 """
394                     <div class="member-signature"><span class="modifiers">public</span>&nbsp;<span class="element-name">C2</span>()</div>

396                     </li>""");

399                 """
400                     <div class="member-signature"><span class="modifiers">public static final</span>\
401                     &nbsp;<span class="return-type"><a href="C1.ModalExclusionType.html" title="enum\
402                      class in pkg1">C1.ModalExclusionType</a></span>&nbsp;<span class="element-name">APPLICATION_E\
403                     XCLUDE</span></div>
404                     </section>
405                     </li>""");

409                     <pre>boolean undecorated</pre>
