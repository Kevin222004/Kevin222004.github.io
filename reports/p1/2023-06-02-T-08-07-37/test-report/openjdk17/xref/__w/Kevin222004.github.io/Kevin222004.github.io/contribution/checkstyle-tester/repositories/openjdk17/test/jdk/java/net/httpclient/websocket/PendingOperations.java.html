24  import org.testng.annotations.AfterMethod;
25  import org.testng.annotations.DataProvider;
26  
27  import java.io.IOException;
28  import java.net.http.HttpClient;
29  import java.net.http.WebSocket;
30  import java.util.concurrent.Callable;
31  import java.util.concurrent.CompletableFuture;
32  import java.util.concurrent.CompletionStage;
33  import java.util.function.BooleanSupplier;
34  
35  import static java.net.http.HttpClient.Builder.NO_PROXY;
36  import static java.net.http.HttpClient.newBuilder;
37  
38  /* Common infrastructure for tests that check pending operations */
39  public class PendingOperations {
40  
41      static final Class<IllegalStateException> ISE = IllegalStateException.class;
42      static final Class<IOException> IOE = IOException.class;
43      // Time after which we deem that the local send buffer and remote
44      // receive buffer must be full. This has been heuristically determined.
45      // At the time of writing, using anything <= 5s on Mac will make the
46      // tests fail intermittently.
47      static final long MAX_WAIT_SEC = 10; // seconds.
48  
49      DummyWebSocketServer server;
50      WebSocket webSocket;
51  
52      protected HttpClient httpClient() {
53          return newBuilder().proxy(NO_PROXY).build();
54      }
55  
56      @AfterMethod
57      public void cleanup() {
58          // make sure we have a trace both on System.out and System.err
59          // to help with diagnosis.
60          System.out.println("cleanup: Closing server");
61          System.err.println("cleanup: Closing server");
62          server.close();
63          webSocket.abort();
64      }
65  
66      /* shortcut */
67      static void assertHangs(CompletionStage<?> stage) {
68          Support.assertHangs(stage);
69      }
70  
71      /* shortcut */
72      static void assertFails(Class<? extends Throwable> clazz,
73                              CompletionStage<?> stage) {
74          Support.assertCompletesExceptionally(clazz, stage);
75      }
76  
77      static void assertNotDone(CompletableFuture<?> future) {
78          Support.assertNotDone(future);
79      }
80  
81      @DataProvider(name = "booleans")
82      public Object[][] booleans() {
83          return new Object[][]{{Boolean.TRUE}, {Boolean.FALSE}};
84      }
85  
86      static boolean isMacOS() {
87          return System.getProperty("os.name").contains("OS X");
88      }
89      static boolean isWindows() {
90          return System.getProperty("os.name").toLowerCase().startsWith("win");
91      }
92  
93      private static final int ITERATIONS = 3;
94  
95      void repeatable(Callable<Void> callable,
96                      BooleanSupplier repeatCondition)
97          throws Exception
98      {
99          int iterations = 0;
100         do {
101             iterations++;
102             System.out.println("--- iteration " + iterations + " ---");
103             try {
104                 callable.call();
105                 break;
106             } catch (AssertionError e) {
107                 var isMac = isMacOS();
108                 var isWindows = isWindows();
109                 var repeat = repeatCondition.getAsBoolean();
110                 System.out.printf("repeatable: isMac=%s, isWindows=%s, repeat=%s, iterations=%d%n",
111                                   isMac, isWindows, repeat, iterations);
112                 if ((isMac || isWindows) && repeat) {
113                     // ## This is loathsome, but necessary because of observed
114                     // ## automagic socket buffer resizing on recent macOS platforms
115                     try { cleanup(); } catch (Throwable x) {}
116                     continue;
117                 } else {
118                     throw e;
119                 }
120             } finally {
121                 // gives some time to gc to cleanup any resource that might
122                 // be eligible for garbage collection
123                 System.gc();
124                 Thread.sleep(100);
125             }
126         } while (iterations <= ITERATIONS);
127     }
128 }
