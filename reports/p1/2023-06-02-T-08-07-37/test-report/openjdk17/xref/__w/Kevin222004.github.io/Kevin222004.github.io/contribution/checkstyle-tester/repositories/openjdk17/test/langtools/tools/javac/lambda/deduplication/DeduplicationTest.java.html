81      public static void main(String[] args) throws Exception {
82          JavacFileManager fileManager = new JavacFileManager(new Context(), false, UTF_8);
83          JavacTool javacTool = JavacTool.create();
84          Listener diagnosticListener = new Listener();
85          Path testSrc = Paths.get(System.getProperty("test.src"));
86          Path file = testSrc.resolve("Deduplication.java");
87          String sourceVersion = Integer.toString(Runtime.version().feature());
88          JavacTask task =
89                  javacTool.getTask(
90                          null,
91                          null,
92                          diagnosticListener,
93                          Arrays.asList(
94                                  "-d",
95                                  ".",
96                                  "-XDdebug.dumpLambdaToMethodDeduplication",
97                                  "-XDdebug.dumpLambdaToMethodStats"),
98                          null,
99                          fileManager.getJavaFileObjects(file));
100         Map<JCLambda, JCLambda> dedupedLambdas = new LinkedHashMap<>();
101         task.addTaskListener(new TreeDiffHashTaskListener(dedupedLambdas));
102         Iterable<? extends JavaFileObject> generated = task.generate();
103         if (!diagnosticListener.unexpected.isEmpty()) {
104             throw new AssertionError(
105                     diagnosticListener
106                             .unexpected
107                             .stream()
108                             .map(
109                                     d ->
110                                             String.format(
111                                                     "%s: %s",
112                                                     d.getCode(), d.getMessage(Locale.getDefault())))
113                             .collect(joining(", ", "unexpected diagnostics: ", "")));
114         }
115 
116         // Assert that each group of lambdas was deduplicated.
117         Map<JCLambda, JCLambda> actual = diagnosticListener.deduplicationTargets();
118         dedupedLambdas.forEach(
119                 (k, v) -> {
120                     if (!actual.containsKey(k)) {
121                         throw new AssertionError("expected " + k + " to be deduplicated");
122                     }
123                     if (!v.equals(actual.get(k))) {
124                         throw new AssertionError(
125                                 String.format(
126                                         "expected %s to be deduplicated to:\n  %s\nwas:  %s",
127                                         k, v, actual.get(v)));
128                     }
129                 });
130 
131         // Assert that the output contains only the canonical lambdas, and not the deduplicated
132         // lambdas.
133         Set<String> bootstrapMethodNames = new TreeSet<>();
134         for (JavaFileObject output : generated) {
135             ClassFile cf = ClassFile.read(output.openInputStream());
136             BootstrapMethods_attribute bsm =
137                     (BootstrapMethods_attribute) cf.getAttribute(Attribute.BootstrapMethods);
138             for (BootstrapMethodSpecifier b : bsm.bootstrap_method_specifiers) {
139                 bootstrapMethodNames.add(
140                         ((CONSTANT_MethodHandle_info)
141                                         cf.constant_pool.get(b.bootstrap_arguments[1]))
142                                 .getCPRefInfo()
143                                 .getNameAndTypeInfo()
144                                 .getName());
145             }
146         }
147         Set<String> deduplicatedNames =
148                 diagnosticListener
149                         .expectedLambdaMethods()
150                         .stream()
151                         .map(s -> s.getSimpleName().toString())
152                         .sorted()
153                         .collect(toSet());
154         if (!deduplicatedNames.equals(bootstrapMethodNames)) {
155             throw new AssertionError(
156                     String.format(
157                             "expected deduplicated methods: %s, but saw: %s",
158                             deduplicatedNames, bootstrapMethodNames));
159         }
160     }

162     /** Returns the parameter symbols of the given lambda. */
163     private static List<Symbol> paramSymbols(JCLambda lambda) {
164         return lambda.params.stream().map(x -> x.sym).collect(toList());
165     }
166 
167     /** A diagnostic listener that records debug messages related to lambda desugaring. */
168     @Trusted
169     static class Listener implements DiagnosticListener<JavaFileObject> {
170 
171         /** A map from method symbols to lambda trees for desugared lambdas. */
172         final Map<MethodSymbol, JCLambda> lambdaMethodSymbolsToTrees = new LinkedHashMap<>();
173 
174         /**
175          * A map from lambda trees that were deduplicated to the method symbol of the canonical
176          * lambda implementation method they were deduplicated to.
177          */
178         final Map<JCLambda, MethodSymbol> deduped = new LinkedHashMap<>();
179 
180         final List<Diagnostic<? extends JavaFileObject>> unexpected = new ArrayList<>();
181 
182         @Override
183         public void report(Diagnostic<? extends JavaFileObject> diagnostic) {
184             JCDiagnostic d = (JCDiagnostic) diagnostic;
185             switch (d.getCode()) {
186                 case "compiler.note.lambda.stat":
187                     lambdaMethodSymbolsToTrees.put(
188                             (MethodSymbol) d.getArgs()[1],
189                             (JCLambda) d.getDiagnosticPosition().getTree());
190                     break;
191                 case "compiler.note.verbose.l2m.deduplicate":
192                     deduped.put(
193                             (JCLambda) d.getDiagnosticPosition().getTree(),
194                             (MethodSymbol) d.getArgs()[0]);
195                     break;
196                 case "compiler.note.preview.filename":
197                 case "compiler.note.preview.recompile":
198                     break; //ignore
199                 default:
200                     unexpected.add(diagnostic);
201             }
202         }
203 
204         /** Returns expected lambda implementation method symbols. */
205         Set<MethodSymbol> expectedLambdaMethods() {
206             return lambdaMethodSymbolsToTrees
207                     .entrySet()
208                     .stream()
209                     .filter(e -> !deduped.containsKey(e.getValue()))
210                     .map(Map.Entry::getKey)
211                     .collect(toSet());
212         }
213 
214         /**
215          * Returns a mapping from deduplicated lambda trees to the tree of the canonical lambda they
216          * were deduplicated to.
217          */
218         Map<JCLambda, JCLambda> deduplicationTargets() {
219             return deduped.entrySet()
220                     .stream()
221                     .collect(
222                             toMap(
223                                     Map.Entry::getKey,
224                                     e -> lambdaMethodSymbolsToTrees.get(e.getValue()),
225                                     (a, b) -> {
226                                         throw new AssertionError();
227                                     },
228                                     LinkedHashMap::new));
229         }
230     }

232     /**
233      * A task listener that tests {@link TreeDiffer} and {@link TreeHasher} on all lambda trees in a
234      * compilation, post-analysis.
235      */
236     private static class TreeDiffHashTaskListener implements TaskListener {
237 
238         /**
239          * A map from deduplicated lambdas to the canonical lambda they are expected to be
240          * deduplicated to.
241          */
242         private final Map<JCLambda, JCLambda> dedupedLambdas;
243 
244         public TreeDiffHashTaskListener(Map<JCLambda, JCLambda> dedupedLambdas) {
245             this.dedupedLambdas = dedupedLambdas;
246         }
247 
248         @Override
249         public void finished(TaskEvent e) {
250             if (e.getKind() != Kind.ANALYZE) {
251                 return;
252             }
253             // Scan the compilation for calls to a varargs method named 'group', whose arguments
254             // are a group of lambdas that are equivalent to each other, but distinct from all
255             // lambdas in the compilation unit outside of that group.
256             List<List<JCLambda>> lambdaGroups = new ArrayList<>();
257             new TreeScanner() {
258                 @Override
259                 public void visitApply(JCMethodInvocation tree) {
260                     if (tree.getMethodSelect().getTag() == Tag.IDENT
261                             && ((JCIdent) tree.getMethodSelect())
262                                     .getName()
263                                     .contentEquals("group")) {
264                         List<JCLambda> xs = new ArrayList<>();
265                         for (JCExpression arg : tree.getArguments()) {
266                             if (arg instanceof JCTypeCast) {
267                                 arg = ((JCTypeCast) arg).getExpression();
268                             }
269                             xs.add((JCLambda) arg);
270                         }
271                         lambdaGroups.add(xs);
272                     }
273                     super.visitApply(tree);
274                 }
275             }.scan((JCCompilationUnit) e.getCompilationUnit());
276             for (int i = 0; i < lambdaGroups.size(); i++) {
277                 List<JCLambda> curr = lambdaGroups.get(i);
278                 JCLambda first = null;
279                 // Assert that all pairwise combinations of lambdas in the group are equal, and
280                 // hash to the same value.
281                 for (JCLambda lhs : curr) {
282                     if (first == null) {
283                         first = lhs;
284                     } else {
285                         dedupedLambdas.put(lhs, first);
286                     }
287                     for (JCLambda rhs : curr) {
288                         if (!new TreeDiffer(paramSymbols(lhs), paramSymbols(rhs))
289                                 .scan(lhs.body, rhs.body)) {
290                             throw new AssertionError(
291                                     String.format(
292                                             "expected lambdas to be equal\n%s\n%s", lhs, rhs));
293                         }
294                         if (TreeHasher.hash(lhs, paramSymbols(lhs))
295                                 != TreeHasher.hash(rhs, paramSymbols(rhs))) {
296                             throw new AssertionError(
297                                     String.format(
298                                             "expected lambdas to hash to the same value\n%s\n%s",
299                                             lhs, rhs));
300                         }
301                     }
302                 }
303                 // Assert that no lambdas in a group are equal to any lambdas outside that group,
304                 // or hash to the same value as lambda outside the group.
305                 // (Note that the hash collisions won't result in correctness problems but could
306                 // regress performs, and do not currently occurr for any of the test inputs.)
307                 for (int j = 0; j < lambdaGroups.size(); j++) {
308                     if (i == j) {
309                         continue;
310                     }
311                     for (JCLambda lhs : curr) {
312                         for (JCLambda rhs : lambdaGroups.get(j)) {
313                             if (new TreeDiffer(paramSymbols(lhs), paramSymbols(rhs))
314                                     .scan(lhs.body, rhs.body)) {
315                                 throw new AssertionError(
316                                         String.format(
317                                                 "expected lambdas to not be equal\n%s\n%s",
318                                                 lhs, rhs));
319                             }
320                             if (TreeHasher.hash(lhs, paramSymbols(lhs))
321                                     == TreeHasher.hash(rhs, paramSymbols(rhs))) {
322                                 throw new AssertionError(
323                                         String.format(
324                                                 "expected lambdas to hash to different values\n%s\n%s",
325                                                 lhs, rhs));
326                             }
327                         }
328                     }
329                 }
330             }
331             lambdaGroups.clear();
332         }
333     }
