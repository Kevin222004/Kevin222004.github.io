24  /*
25   * @test
26   * @summary Ensure that a package whose module has not been defined to the boot loader
27   *          is correctly located with -Xbootclasspath/a
28   * @requires !(os.family == "windows")
29   * @library /test/lib
30   * @modules java.base/jdk.internal.misc
31   *          java.management
32   * @run driver XbootcpVisibility
33   */
34  
35  import java.io.File;
36  import java.nio.file.Files;
37  import java.nio.file.Paths;
38  
39  import jdk.test.lib.compiler.InMemoryJavaCompiler;
40  import jdk.test.lib.process.OutputAnalyzer;
41  import jdk.test.lib.process.ProcessTools;
42  import jdk.test.lib.helpers.ClassFileInstaller;
43  
44  public class XbootcpVisibility {
45  
46      public static void main(String[] args) throws Throwable {
47  
48          String Vis1_B_src =
49                  "package p2;" +
50                  "public class Vis1_B { public void m() { System.out.println(\"In B's m()\"); } }";
51  
52          ClassFileInstaller.writeClassToDisk("p2/Vis1_B",
53              InMemoryJavaCompiler.compile("p2.Vis1_B", Vis1_B_src), System.getProperty("test.classes"));
54          ClassFileInstaller.writeClassToDisk("p2/Vis1_B", "mods1");
55  
56          String Vis1_C_src =
57                  "package p2;" +
58                  "public class Vis1_C { public void m() { System.out.println(\"In C's m()\"); } }";
59  
60          ClassFileInstaller.writeClassToDisk("p2/Vis1_C",
61              InMemoryJavaCompiler.compile("p2.Vis1_C", Vis1_C_src), System.getProperty("test.classes"));
62          ClassFileInstaller.writeClassToDisk("p2/Vis1_C", "mods1");
63  
64          String Vis1_A_src =
65              "public class Vis1_A {" +
66              "    public static void main(String args[]) throws Exception {" +
67              // Try loading a class within a named package
68              // in the unnamed module.
69              // Ensure the class can still be loaded successfully by the
70              // boot loader since it is located on -Xbootclasspath/a.
71              "        try {" +
72              "            p2.Vis1_B b = new p2.Vis1_B();" +
73              "            if (b.getClass().getClassLoader() != null) {" +
74              "              throw new RuntimeException(\"XbootcpVisibility FAILED - class B " +
75                                                        "should be loaded by boot class loader\\n\");" +
76              "            }" +
77              "            b.m();" +
78              // Now that the package p2 has been recorded in the
79              // unnamed module within the boot loader's PackageEntryTable,
80              // ensure that a different class within the same package
81              // can be located on -Xbootclasspath/a as well.
82              "            p2.Vis1_C c = new p2.Vis1_C();" +
83              "            if (c.getClass().getClassLoader() != null) {" +
84              "              throw new RuntimeException(\"XbootcpVisibility FAILED - class C " +
85                                                         "should be loaded by boot class loader\\n\");" +
86              "            }" +
87              "            c.m();" +
88              "        } catch (Exception e) {" +
89              "            System.out.println(e);" +
90              "            throw new RuntimeException(\"XbootcpVisibility FAILED - " +
91                                                       "test should not throw exception\\n\");" +
92              "        }" +
93              "        System.out.println(\"XbootcpVisibility PASSED\\n\");" +
94              "    }" +
95              "}";
96  
97          ClassFileInstaller.writeClassToDisk("Vis1_A",
98                  InMemoryJavaCompiler.compile("Vis1_A", Vis1_A_src), System.getProperty("test.classes"));
99  
100         // Make sure the classes are actually being loaded from mods1
101         Files.delete(Paths.get(System.getProperty("test.classes") +  File.separator +
102                                                                "p2" + File.separator + "Vis1_B.class"));
103         Files.delete(Paths.get(System.getProperty("test.classes") +  File.separator +
104                                                                "p2" + File.separator + "Vis1_C.class"));
105 
106         new OutputAnalyzer(ProcessTools.createJavaProcessBuilder(
107                 "-Xbootclasspath/a:nonexistent.jar",
108                 "-Xbootclasspath/a:mods1",
109                 "Vis1_A")
110             .start()).shouldHaveExitValue(0);
111     }
112 }
