24  /*
25   * @test
26   * @bug      4780441 4874845 4978816 8014017 8016328 8025633 8026567 8175200 8182765
27   * @summary  Make sure that when the -private flag is not used, members
28   *           inherited from package private class are documented in the child.
29   *
30   *           Make sure that when a method inherits documentation from a method
31   *           in a non-public class/interface, the non-public class/interface
32   *           is not mentioned anywhere (not even in the signature or tree).
33   *
34   *           Make sure that when a private interface method with generic parameters
35   *           is implemented, the comments can be inherited properly.
36   *
37   *           Make sure when no modifier appear in the class signature, the
38   *           signature is displayed correctly without extra space at the beginning.
39   * @library  ../../lib
40   * @modules jdk.javadoc/jdk.javadoc.internal.tool
41   * @build    javadoc.tester.*
42   * @run main TestPrivateClasses
43   */
44  import javadoc.tester.JavadocTester;
45  
46  public class TestPrivateClasses extends JavadocTester {
47  
48      public static void main(String... args) throws Exception {
49          TestPrivateClasses tester = new TestPrivateClasses();
50          tester.runTests();
51      }
52  
53      @Test
54      public void testDefault() {
55          javadoc("-d", "out-default",
56                  "-sourcepath", testSrc,
57                  "--no-platform-links",
58                  "pkg", "pkg2");
59          checkExit(Exit.OK);
60  
61          checkOutput("pkg/PublicChild.html", true,
62                  // Field inheritence from non-public superclass.
63                  """
64                      <a href="#fieldInheritedFromParent" class="member-name-link">fieldInheritedFromParent</a>""",
65                  // Method inheritance from non-public superclass.
66                  """
67                      <a href="#methodInheritedFromParent(int)" class="member-name-link">methodInheritedFromParent</a>""",
68                  // private class does not show up in tree
69                  """
70                      <div class="inheritance" title="Inheritance Tree">java.lang.Object
71                      <div class="inheritance">pkg.PublicChild</div>
72                      </div>""",
73                  // Method is documented as though it is declared in the inheriting method.
74                  """
75                      <div class="member-signature"><span class="modifiers">public</span>&nbsp;<span c\
76                      lass="return-type">void</span>&nbsp;<span class="element-name">methodInheritedFr\
77                      omParent</span><wbr><span class="parameters">(int&nbsp;p1)</span>
78                                                     throws <span class="exceptions">java.lang.Exception</span></div>""",
79                  """
80                      <dl class="notes">
81                      <dt>All Implemented Interfaces:</dt>
82                      <dd><code><a href="PublicInterface.html" title="interface in pkg">PublicInterface</a></code></dd>
83                      </dl>""");
84  
85          checkOutput("pkg/PublicChild.html", false,
86                  // Should not document that a method overrides method from private class.
87                  """
88                      <span class="overrideSpecifyLabel">Overrides:</span>""",
89                  // Should not document that a method specified by private interface.
90                  """
91                      <span class="overrideSpecifyLabel">Specified by:</span>""",
92                  // Should not mention that any documentation was copied.
93                  "Description copied from",
94                  // Don't extend private classes or interfaces
95                  "PrivateParent",
96                  "PrivateInterface");
97  
98          checkOutput("pkg/PublicChild.html", false,
99                  // Should not document comments from private inherited interfaces
100                 """
101                     <td class="col-last"><code><span class="member-name-link"><a href="#methodInterf\
102                     ace(int)">methodInterface</a></span><wbr>(int&nbsp;p1)</code>
103                     <div class="block">Comment from interface.</div>
104                     </td>""",
105                 // and similarly one more
106                 """
107                     <td class="col-last"><code><span class="member-name-link"><a href="#methodInterf\
108                     ace2(int)">methodInterface2</a></span><wbr>(int&nbsp;p1)</code>
109                     <div class="block">Comment from interface.</div>
110                     </td>"""
111         );
112 
113         checkOutput("pkg/PublicInterface.html", true,
114                 // Field inheritance from non-public superinterface.
115                 """
116                     <a href="#fieldInheritedFromInterface" class="member-name-link">fieldInheritedFromInterface</a>""",
117                 // Method inheritance from non-public superinterface.
118                 """
119                     <a href="#methodInterface(int)" class="member-name-link">methodInterface</a>""",
120                 //Make sure implemented interfaces from private superclass are inherited
121                 """
122                     <dl class="notes">
123                     <dt>All Known Implementing Classes:</dt>
124                     <dd><code><a href="PublicChild.html" title="class in pkg">PublicChild</a></code></dd>
125                     </dl>""");
126 
127         checkOutput("pkg/PublicInterface.html", false,
128                 """
129                     <span class="overrideSpecifyLabel">Specified by:</span>""",
130                 "Description copied from",
131                 "PrivateInterface",
132                 "All Superinterfaces");
133 
134         checkOutput("pkg2/C.html", false,
135                 //Generic interface method test.
136                 "This comment should get copied to the implementing class");
137 
138         checkOutput("pkg2/C.html", false,
139                 //Do not inherit private interface method with generic parameters.
140                 //This method has been implemented.
141                 """
142                     <span class="member-name-link"><a href="I.html#hello(T)">hello</a></span>""");
143 
144         checkOutput("constant-values.html", false,
145                 // Make inherited constant are documented correctly.
146                 "PrivateInterface");
147     }

149     @Test
150     public void testPrivate() {
151         javadoc("-d", "out-private",
152                 "-sourcepath", testSrc,
153                 "--no-platform-links",
154                 "-private",
155                 "pkg", "pkg2");
156         checkExit(Exit.OK);
157 
158         checkOutput("pkg/PublicChild.html", true,
159                 // Field inheritence from non-public superclass.
160                 """
161                     Fields inherited from class&nbsp;pkg.<a href="PrivateParent.html" title="class in pkg">PrivateParent</a>""",
162                 """
163                     <a href="PrivateParent.html#fieldInheritedFromParent">fieldInheritedFromParent</a>""",
164                 // Method inheritence from non-public superclass.
165                 """
166                     Methods inherited from class&nbsp;pkg.<a href="PrivateParent.html" title="class in pkg">PrivateParent</a>""",
167                 """
168                     <a href="PrivateParent.html#methodInheritedFromParent(int)">methodInheritedFromParent</a>""",
169                 // Should document that a method overrides method from private class.
170                 """
171                     <dt>Overrides:</dt>
172                     <dd><code><a href="PrivateParent.html#methodOverriddenFromParent(char%5B%5D,int,\
173                     T,V,java.util.List)">methodOverriddenFromParent</a></code>&nbsp;in class&nbsp;<c\
174                     ode><a href="PrivateParent.html" title="class in pkg">PrivateParent</a></code></\
175                     dd>""",
176                 // Should document that a method is specified by private interface.
177                 """
178                     <dt>Specified by:</dt>
179                     <dd><code><a href="PrivateInterface.html#methodInterface(int)">methodInterface</\
180                     a></code>&nbsp;in interface&nbsp;<code><a href="PrivateInterface.html" title="in\
181                     terface in pkg">PrivateInterface</a></code></dd>""",
182                 // Should mention that any documentation was copied.
183                 "Description copied from",
184                 // Extend documented private classes or interfaces
185                 "extends",
186                 """
187                     <dl class="notes">
188                     <dt>All Implemented Interfaces:</dt>
189                     <dd><code><a href="PrivateInterface.html" title="interface in pkg">PrivateInterf\
190                     ace</a></code>, <code><a href="PublicInterface.html" title="interface in pkg">Pu\
191                     blicInterface</a></code></dd>
192                     </dl>""",
193                 """
194                     <div class="type-signature"><span class="modifiers">public class </span><span cl\
195                     ass="element-name type-name-label">PublicChild</span>""");
196 
197         checkOutput("pkg/PublicInterface.html", true,
198                 // Field inheritence from non-public superinterface.
199                 """
200                     Fields inherited from interface&nbsp;pkg.<a href="PrivateInterface.html" title="interface in pkg">PrivateInterface</a>""",
201                 """
202                     <a href="PrivateInterface.html#fieldInheritedFromInterface">fieldInheritedFromInterface</a>""",
203                 // Method inheritance from non-public superinterface.
204                 """
205                     Methods inherited from interface&nbsp;pkg.<a href="PrivateInterface.html" title="interface in pkg">PrivateInterface</a>""",
206                 // Extend documented private classes or interfaces
207                 "extends",
208                 "All Superinterfaces",
209                 //Make sure implemented interfaces from private superclass are inherited
210                 """
211                     <dl class="notes">
212                     <dt>All Known Implementing Classes:</dt>
213                     <dd><code><a href="PrivateParent.html" title="class in pkg">PrivateParent</a></c\
214                     ode>, <code><a href="PublicChild.html" title="class in pkg">PublicChild</a></cod\
215                     e></dd>
216                     </dl>""");
217 
218         checkOutput("pkg/PrivateInterface.html", true,
219                 """
220                     <a href="#methodInterface(int)" class="member-name-link">methodInterface</a>"""
221         );
222 
223         checkOutput("pkg2/C.html", true,
224                 //Since private flag is used, we can document that private interface method
225                 //with generic parameters has been implemented.
226                 """
227                     <span class="descfrm-type-label">Description copied from interface:&nbsp;<code><\
228                     a href="I.html#hello(T)">I</a></code></span>""",
229                 """
230                     <dt>Specified by:</dt>
231                     <dd><code><a href="I.html#hello(T)">hello</a></code>&nbsp;in interface&nbsp;<cod\
232                     e><a href="I.html" title="interface in pkg2">I</a>&lt;java.lang.String&gt;</code\
233                     ></dd>""");
234 
235         checkOutput("pkg/PrivateParent.html", true,
236                 //Make sure when no modifier appear in the class signature, the
237                 //signature is displayed correctly without extra space at the beginning.
238                 """
239                     <div class="type-signature"><span class="modifiers">class </span><span class="el\
240                     ement-name type-name-label">PrivateParent</span>""");
241 
242         checkOutput("pkg/PrivateParent.html", false,
243                 """
244                     <div class="type-signature"><span class="modifiers"> class </span><span class="el\
245                     ement-name type-name-label">PrivateParent</span>""");
246     }
