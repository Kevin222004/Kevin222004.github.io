153     public static void main(String[] args) throws Exception {
154 
155         if (args.length >= 1 && "CRL".equalsIgnoreCase(args[0])) {
156             ValidatePathWithParams pathValidator = new ValidatePathWithParams(null);
157             pathValidator.enableCRLCheck();
158 
159             // Validate int, EE certs don't have CRLs
160             pathValidator.validate(new String[]{INT},
161                     ValidatePathWithParams.Status.GOOD, null, System.out);
162 
163             return;
164         }
165 
166         // OCSP check by default
167         // intermediate cert R3 doesn't specify OCSP responder
168         ValidatePathWithParams pathValidator = new ValidatePathWithParams(new String[]{INT});
169         pathValidator.enableOCSPCheck();
170 
171         // Perform backdate check as test artifacts expire in July 2021
172         pathValidator.setValidationDate("June 15, 2021");
173 
174         // Validate valid
175         pathValidator.validate(new String[]{VALID},
176                 ValidatePathWithParams.Status.GOOD, null, System.out);
177 
178         // Validate Revoked
179         pathValidator.validate(new String[]{REVOKED},
180                 ValidatePathWithParams.Status.REVOKED,
181                 "Thu Apr 08 17:05:26 PDT 2021", System.out);
182     }
