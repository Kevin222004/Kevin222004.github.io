12      }

14      interface NumberConverter<T extends Number> {
15          T map(Object o);
16      }
17  
18      static class MyMapper<A extends Object, B extends Number> {
19          void map(ObjectConverter<? extends A> m) { }
20          void map(NumberConverter<? extends B> m) { }
21      }
22  
23      public static void main(String[] args) {
24          MyMapper<Number, Double> mm = new MyMapper<Number, Double>();
25          mm.map(e->1.0); //ambiguous - implicit
