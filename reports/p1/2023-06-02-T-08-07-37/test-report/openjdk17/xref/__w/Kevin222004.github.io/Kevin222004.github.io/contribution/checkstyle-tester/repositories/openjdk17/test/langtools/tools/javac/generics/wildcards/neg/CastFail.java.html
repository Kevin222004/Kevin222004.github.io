14      // --- Directly transferring parameters ---
15  
16      private class AA<T> { }
17  
18      private class AB<T> extends AA<T> { }
19      private class AC<T> extends AA<Vector<T>> { }
20      private class AD<T> extends AA<Vector<? extends T>> { }
21      private class AE<T> extends AA<Vector<? super T>> { }
22      private class AF<T> extends AA<T[]> { }
23      private class AG<T> extends AA<String> { }
24  
25      private void parameterTransfer() {
26          Object o;
27  
28          o = (AB<String>) (AA<Number>) null; // <<fail 1>>
29          o = (AC<String>) (AA<Vector<Number>>) null; // <<fail 2>>
30          o = (AC<String>) (AA<Stack<String>>) null; // <<fail 3>>
31          o = (AD<String>) (AA<Vector<? extends Number>>) null; // <<fail 4>>
32          o = (AE<Number>) (AA<Vector<? super String>>) null; // <<fail 5>>
33          o = (AF<String>) (AA<Number[]>) null; // <<fail 6>>
34          o = (AG<?>) (AA<Number>) null; // <<fail 7>>
35      }
36  
37      // --- Inconsistent matches ---
38  
39      private class BA<T> { }
40      private class BB<T, S> { }
41  
42      private class BC<T> extends BA<Integer> { }
43      private class BD<T> extends BB<T, T> { }
44  
45      private void inconsistentMatches() {
46          Object o;
47  
48          o = (BC<?>) (BA<String>) null; // <<fail 8>>
49          o = (BD<String>) (BB<String, Number>) null; // <<fail 9>>
50          o = (BD<String>) (BB<Number, String>) null; // <<fail 10>>
51      }
52  
53      // --- Transferring parameters via supertypes ---
54  
55      private interface CA<T> { }
56      private interface CB<T> extends CA<T> { }
57      private interface CC<T> extends CA<T> { }
58  
59      private class CD<T> implements CB<T> { }
60      private interface CE<T> extends CC<T> { }
61  
62      private interface CF<S> { }
63      private interface CG<T> { }
64      private class CH<S, T> implements CF<S>, CG<T> { }
65      private interface CI<S> extends CF<S> { }
66      private interface CJ<T> extends CG<T> { }
67      private interface CK<S, T> extends CI<S>, CJ<T> { }
68  
69      private void supertypeParameterTransfer() {
70          Object o;
71          CD<?> cd = (CE<?>) null; // <<fail 11>>
72          CE<?> ce = (CD<?>) null; // <<fail 12>>

79      // --- Disjoint ---
80  
81      private interface DA<T> { }
82      private interface DB<T> extends DA<T> { }
83      private interface DC<T> extends DA<Integer> { }
84  
85      private <N extends Number, I extends Integer, R extends Runnable, S extends String> void disjointness() {
86          Object o;
87  
88          // Classes
89          o = (DA<Number>) (DA<Integer>) null; // <<fail 14>>
90          o = (DA<? extends Integer>) (DA<Number>) null; // <<fail 15>>
91          o = (DA<? super Number>) (DA<Integer>) null; // <<fail 16>>
92          o = (DA<? extends Runnable>) (DA<? extends String>) null; // <<fail 17>>
93          o = (DA<? super Number>) (DA<? extends Integer>) null; // <<fail 18>>
94  
95          // Typevars
96          o = (DA<? extends String>) (DA<I>) null; // <<fail 19>>
97          o = (DA<S>) (DA<R>) null; // <<fail 20>>
98  
99          // Raw (asymmetrical!)
100         o = (DC<?>) (DA<? super String>) null; // <<fail 21>>
101     }
