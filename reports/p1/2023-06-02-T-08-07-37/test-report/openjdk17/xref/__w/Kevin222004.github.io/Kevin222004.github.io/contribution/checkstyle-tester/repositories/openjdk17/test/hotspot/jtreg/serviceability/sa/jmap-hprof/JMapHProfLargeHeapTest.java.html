41  /**
42   * @test
43   * @bug 6313383
44   * @requires vm.hasSA
45   * @summary Regression test for hprof export issue due to large heaps (>2G)
46   * @library /test/lib
47   * @modules java.base/jdk.internal.misc
48   *          java.compiler
49   *          java.management/sun.management
50   *          jdk.internal.jvmstat/sun.jvmstat.monitor
51   * @build JMapHProfLargeHeapProc
52   * @run driver JMapHProfLargeHeapTest
53   */
54  
55  public class JMapHProfLargeHeapTest {
56      private static final String HEAP_DUMP_FILE_NAME = "heap.bin";
57      private static final String HPROF_HEADER_1_0_2 = "JAVA PROFILE 1.0.2";
58      private static final long M = 1024L;
59      private static final long G = 1024L * M;
60  
61      public static void main(String[] args) throws Exception {
62          SATestUtils.skipIfCannotAttach(); // throws SkippedException if attach not expected to work.
63  
64          // All heap dumps should create 1.0.2 file format
65          testHProfFileFormat("-Xmx1g", 22 * M, HPROF_HEADER_1_0_2);
66  
67          /**
68           * This test was deliberately commented out since the test system lacks
69           * support to handle the requirements for this kind of heap size in a
70           * good way. If or when it becomes possible to run this kind of tests in
71           * the test environment the test should be enabled again.
72           * */
73          // Large heap 2,2 gigabytes, should create 1.0.2 file format
74          // testHProfFileFormat("-Xmx4g", 2 * G + 2 * M, HPROF_HEADER_1_0_2);
75      }
76  
77      private static void testHProfFileFormat(String vmArgs, long heapSize,
78              String expectedFormat) throws Exception, IOException,
79              InterruptedException, FileNotFoundException {
80          ProcessBuilder procBuilder = ProcessTools.createJavaProcessBuilder(
81                  "--add-exports=java.management/sun.management=ALL-UNNAMED", vmArgs, "JMapHProfLargeHeapProc", String.valueOf(heapSize));
82          procBuilder.redirectError(ProcessBuilder.Redirect.INHERIT);
83          Process largeHeapProc = procBuilder.start();
84  
85          try (Scanner largeHeapScanner = new Scanner(
86                  largeHeapProc.getInputStream());) {
87              String pidstring = null;
88              if (!largeHeapScanner.hasNext()) {
89                  throw new RuntimeException ("Test failed: could not open largeHeapScanner.");
90              }
91              while ((pidstring = largeHeapScanner.findInLine("PID\\[[0-9].*\\]")) == null) {
92                  Thread.sleep(500);
93              }
94              int pid = Integer.parseInt(pidstring.substring(4,
95                      pidstring.length() - 1));
96              System.out.println("Extracted pid: " + pid);
97  
98              JDKToolLauncher jMapLauncher = JDKToolLauncher
99                      .createUsingTestJDK("jhsdb");
100             jMapLauncher.addVMArgs(Utils.getTestJavaOpts());
101             jMapLauncher.addToolArg("jmap");
102             jMapLauncher.addToolArg("--binaryheap");
103             jMapLauncher.addToolArg("--pid");
104             jMapLauncher.addToolArg(String.valueOf(pid));
105 
106             ProcessBuilder jMapProcessBuilder = SATestUtils.createProcessBuilder(jMapLauncher);
107             System.out.println("jmap command: "
108                     + Arrays.toString(jMapLauncher.getCommand()));
109 
110             Process jMapProcess = jMapProcessBuilder.start();
111             OutputAnalyzer analyzer = new OutputAnalyzer(jMapProcess);
112             analyzer.shouldHaveExitValue(0);
113             analyzer.shouldContain(HEAP_DUMP_FILE_NAME);
114 
115             largeHeapProc.getOutputStream().write('\n');
116 
117             File dumpFile = new File(HEAP_DUMP_FILE_NAME);
118             Asserts.assertTrue(dumpFile.exists(), "Heap dump file not found.");
119 
120             try (Reader reader = new BufferedReader(new FileReader(dumpFile))) {
121                 CharBuffer buf = CharBuffer.allocate(expectedFormat.length());
122                 reader.read(buf);
123                 buf.clear();
124                 Asserts.assertEQ(buf.toString(), expectedFormat,
125                         "Wrong file format. Expected '" + expectedFormat
126                                 + "', but found '" + buf.toString() + "'");
127             }
128 
129             System.out.println("Success!");
130 
131         } finally {
132             largeHeapProc.destroyForcibly();
133         }
134     }
