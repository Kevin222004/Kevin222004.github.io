137     /**
138      * Run the program. A base directory can be provided for file arguments.
139      * In jtreg mode, the -r option can be given to change the default base
140      * directory to the test root directory. For other options, see usage().
141      * @param baseDir base directory for any file arguments.
142      * @param args command line args
143      * @return true if successful or in gui mode
144      */
145     boolean run(File baseDir, String... args) throws IOException {
146         try {
147             if (args.length == 0) {
148                 usage(System.out);
149                 return true;
150             }
151 
152             List<File> files = new ArrayList<File>();
153             for (int i = 0; i < args.length; i++) {
154                 String arg = args[i];
155                 if (arg.equals("-encoding") && i + 1 < args.length)
156                     encoding = args[++i];
157                 else if (arg.equals("-gui"))
158                     gui = true;
159                 else if (arg.equals("-q"))
160                     quiet = true;
161                 else if (arg.equals("-v"))
162                     verbose = true;
163                 else if (arg.equals("-t") && i + 1 < args.length)
164                     tags.add(args[++i]);
165                 else if (arg.equals("-ef") && i + 1 < args.length)
166                     excludeFiles.add(new File(baseDir, args[++i]));
167                 else if (arg.equals("-et") && i + 1 < args.length)
168                     excludeTags.add(args[++i]);
169                 else if (arg.equals("-r")) {
170                     if (excludeFiles.size() > 0)
171                         throw new Error("-r must be used before -ef");
172                     File d = baseDir;
173                     while (!new File(d, "TEST.ROOT").exists()) {
174                         d = d.getParentFile();
175                         if (d == null)
176                             throw new Error("cannot find TEST.ROOT");
177                     }
178                     baseDir = d;
179                 }
180                 else if (arg.startsWith("-"))
181                     throw new Error("unknown option: " + arg);
182                 else {
183                     while (i < args.length)
184                         files.add(new File(baseDir, args[i++]));
185                 }
186             }
187 
188             for (File file: files) {
189                 if (file.exists())
190                     test(file);
191                 else
192                     error("File not found: " + file);
193             }
194 
195             if (fileCount != 1)
196                 System.err.println(fileCount + " files read");
197             if (errors > 0)
198                 System.err.println(errors + " errors");
199 
200             return (gui || errors == 0);
201         } finally {
202             fm.close();
203         }
204     }

229     /**
230      * Test a file. If the file is a directory, it will be recursively scanned
231      * for java files.
232      * @param file the file or directory to test
233      */
234     void test(File file) {
235         if (excludeFiles.contains(file)) {
236             if (!quiet)
237                 error("File " + file + " excluded");
238             return;
239         }
240 
241         if (file.isDirectory()) {
242             for (File f: file.listFiles()) {
243                 test(f);
244             }
245             return;
246         }
247 
248         if (file.isFile() && file.getName().endsWith(".java")) {
249             try {
250                 if (verbose)
251                     System.err.println(file);
252                 fileCount++;
253                 PosTester p = new PosTester();
254                 p.test(read(file));
255             } catch (ParseException e) {
256                 if (!quiet) {
257                     error("Error parsing " + file + "\n" + e.getMessage());
258                 }
259             } catch (IOException e) {
260                 error("Error reading " + file + ": " + e);
261             }
262             return;
263         }
264 
265         if (!quiet)
266             error("File " + file + " ignored");
267     }

276     /**
277      * Read a file.
278      * @param file the file to be read
279      * @return the tree for the content of the file
280      * @throws IOException if any IO errors occur
281      * @throws TreePosTest.ParseException if any errors occur while parsing the file
282      */
283     JCCompilationUnit read(File file) throws IOException, ParseException {
284         JavacTool tool = JavacTool.create();
285         r.errors = 0;
286         Iterable<? extends JavaFileObject> files = fm.getJavaFileObjects(file);
287         JavacTask task = tool.getTask(pw, fm, r, List.of("-proc:none"), null, files);
288         Iterable<? extends CompilationUnitTree> trees = task.parse();
289         pw.flush();
290         if (r.errors > 0)
291             throw new ParseException(sw.toString());
292         Iterator<? extends CompilationUnitTree> iter = trees.iterator();
293         if (!iter.hasNext())
294             throw new Error("no trees found");
295         JCCompilationUnit t = (JCCompilationUnit) iter.next();
296         if (iter.hasNext())
297             throw new Error("too many trees found");
298         return t;
299     }

341     /**
342      * Main class for testing assertions concerning tree positions for tree nodes.
343      */
344     private class PosTester extends TreeScanner {
345         void test(JCCompilationUnit tree) {
346             sourcefile = tree.sourcefile;
347             endPosTable = tree.endPositions;
348             encl = new Info();
349             tree.accept(this);
350         }
351 
352         @Override
353         public void scan(JCTree tree) {
354             if (tree == null)
355                 return;
356 
357             Info self = new Info(tree, endPosTable);
358             if (check(encl, self)) {
359                 // Modifiers nodes are present throughout the tree even where
360                 // there is no corresponding source text.
361                 // Redundant semicolons in a class definition can cause empty
362                 // initializer blocks with no positions.
363                 if ((self.tag == MODIFIERS || self.tag == BLOCK)
364                         && self.pos == NOPOS) {
365                     // If pos is NOPOS, so should be the start and end positions
366                     check("start == NOPOS", encl, self, self.start == NOPOS);
367                     check("end == NOPOS", encl, self, self.end == NOPOS);
368                 } else {
369                     // For this node, start , pos, and endpos should be all defined
370                     check("start != NOPOS", encl, self, self.start != NOPOS);
371                     check("pos != NOPOS", encl, self, self.pos != NOPOS);
372                     check("end != NOPOS", encl, self, self.end != NOPOS);
373                     // The following should normally be ordered
374                     // encl.start <= start <= pos <= end <= encl.end
375                     // In addition, the position of the enclosing node should be
376                     // within this node.
377                     // The primary exceptions are for array type nodes, because of the
378                     // need to support legacy syntax:
379                     //    e.g.    int a[];    int[] b[];    int f()[] { return null; }
380                     // and because of inconsistent nesting of left and right of
381                     // array declarations:
382                     //    e.g.    int[][] a = new int[2][];
383                     if (!(encl.tag == REFERENCE && self.tag == ANNOTATED_TYPE)) {
384                     check("encl.start <= start", encl, self, encl.start <= self.start);
385                     }
386                     check("start <= pos", encl, self, self.start <= self.pos);
387                     if (!( (self.tag == TYPEARRAY ||
388                             isAnnotatedArray(self.tree))
389                             && (encl.tag == VARDEF ||
390                                 encl.tag == METHODDEF ||
391                                 encl.tag == TYPEARRAY ||
392                                 isAnnotatedArray(encl.tree))
393                            ||
394                             encl.tag == ANNOTATED_TYPE && self.tag == SELECT
395                            ||
396                             encl.tag == REFERENCE && self.tag == ANNOTATED_TYPE
397                          )) {
398                         check("encl.pos <= start || end <= encl.pos",
399                                 encl, self, encl.pos <= self.start || self.end <= encl.pos);
400                     }
401                     check("pos <= end", encl, self, self.pos <= self.end);
402                     if (!( (self.tag == TYPEARRAY || isAnnotatedArray(self.tree)) &&
403                             (encl.tag == TYPEARRAY || isAnnotatedArray(encl.tree))
404                            ||
405                             encl.tag == MODIFIERS && self.tag == ANNOTATION
406                          ) ) {
407                         check("end <= encl.end", encl, self, self.end <= encl.end);
408                     }
409                 }
410             }
411 
412             Info prevEncl = encl;
413             encl = self;
414             tree.accept(this);
415             encl = prevEncl;
416         }
417 
418         private boolean isAnnotatedArray(JCTree tree) {
419             return tree.hasTag(ANNOTATED_TYPE) &&
420                             ((JCAnnotatedType)tree).underlyingType.hasTag(TYPEARRAY);
421         }
422 
423         @Override
424         public void visitMethodDef(JCMethodDecl tree) {
425             // ignore compact record constructors
426             if ((tree.mods.flags & Flags.COMPACT_RECORD_CONSTRUCTOR) == 0) {
427                 super.visitMethodDef(tree);
428             }
429         }
430 
431         @Override
432         public void visitVarDef(JCVariableDecl tree) {
433             // enum member declarations are desugared in the parser and have
434             // ill-defined semantics for tree positions, so for now, we
435             // skip the synthesized bits and just check parts which came from
436             // the original source text
437             if ((tree.mods.flags & Flags.ENUM) != 0) {
438                 scan(tree.mods);
439                 if (tree.init != null) {
440                     if (tree.init.hasTag(NEWCLASS)) {
441                         JCNewClass init = (JCNewClass) tree.init;
442                         if (init.args != null && init.args.nonEmpty()) {
443                             scan(init.args);
444                         }
445                         if (init.def != null && init.def.defs != null) {
446                             scan(init.def.defs);
447                         }
448                     }
449                 }
450             } else
451                 super.visitVarDef(tree);
452         }
453 
454         @Override
455         public void visitCase(JCCase tree) {
456             if (tree.getCaseKind() == CaseKind.RULE) {
457                 scan(tree.getBody());
458             } else {
459                 super.visitCase(tree);
460             }
461         }
462 
463         boolean check(Info encl, Info self) {
464             if (excludeTags.size() > 0) {
465                 if (encl != null && excludeTags.contains(getTagName(encl.tag))
466                         || excludeTags.contains(getTagName(self.tag)))
467                     return false;
468             }
469             return tags.size() == 0 || tags.contains(getTagName(self.tag));
470         }
471 
472         void check(String label, Info encl, Info self, boolean ok) {
473             if (!ok) {
474                 if (gui) {
475                     if (viewer == null)
476                         viewer = new Viewer();
477                     viewer.addEntry(sourcefile, label, encl, self);
478                 }
479 
480                 String s = "encl: " + encl.tree.toString() +
481                         "  this: " + self.tree.toString();
482                 String msg = sourcefile.getName() + ": " + label + ": " +
483                         "encl:" + encl + " this:" + self + "\n" +
484                         s.substring(0, Math.min(80, s.length())).replaceAll("[\r\n]+", " ");
485                 error(msg);
486             }
487         }
488 
489         JavaFileObject sourcefile;
490         EndPosTable endPosTable;
491         Info encl;
492 
493     }

495     /**
496      * Utility class providing easy access to position and other info for a tree node.
497      */
498     private class Info {
499         Info() {
500             tree = null;
501             tag = ERRONEOUS;
502             start = 0;
503             pos = 0;
504             end = Integer.MAX_VALUE;
505         }
506 
507         Info(JCTree tree, EndPosTable endPosTable) {
508             this.tree = tree;
509             tag = tree.getTag();
510             start = TreeInfo.getStartPos(tree);
511             pos = tree.pos;
512             end = TreeInfo.getEndPos(tree, endPosTable);
513         }
514 
515         @Override
516         public String toString() {
517             return getTagName(tree.getTag()) + "[start:" + start + ",pos:" + pos + ",end:" + end + "]";
518         }
519 
520         final JCTree tree;
521         final JCTree.Tag tag;
522         final int start;
523         final int pos;
524         final int end;
525     }

644                 setTitle(e.file.getName());
645                 checkField.setText(e.check);
646                 enclPanel.setInfo(e.encl);

668         /** Add a highlighted region based on the positions in an Info object. */
669         private void addHighlight(Highlighter h, Info info, Color c) {
670             int start = info.start;
671             int end = info.end;
672             if (start == -1 && end == -1)
673                 return;
674             if (start == -1)
675                 start = end;
676             if (end == -1)
677                 end = start;
678             try {
679                 h.addHighlight(info.start, info.end,
680                         new DefaultHighlighter.DefaultHighlightPainter(c));
681                 if (info.pos != -1) {
682                     Color c2 = new Color(c.getRed(), c.getGreen(), c.getBlue(), (int)(.4f * 255)); // 40%
683                     h.addHighlight(info.pos, info.pos + 1,
684                         new DefaultHighlighter.DefaultHighlightPainter(c2));
685                 }
686             } catch (BadLocationException e) {
687                 e.printStackTrace();
688             }
689         }

691         /** Get the minimum valid position in a set of info objects. */
692         private int getMinPos(Info... values) {
693             int i = Integer.MAX_VALUE;
694             for (Info info: values) {
695                 if (info.start >= 0) i = Math.min(i, info.start);
696                 if (info.pos   >= 0) i = Math.min(i, info.pos);
697                 if (info.end   >= 0) i = Math.min(i, info.end);
698             }
699             return (i == Integer.MAX_VALUE) ? 0 : i;
700         }

709         /** Scroll a text area to display a given position near the middle of the visible area. */
710         private void scroll(final JTextArea t, final int pos) {
711             // Using invokeLater appears to give text a chance to sort itself out
712             // before the scroll happens; otherwise scrollRectToVisible doesn't work.
713             // Maybe there's a better way to sync with the text...
714             EventQueue.invokeLater(new Runnable() {
715                 public void run() {
716                     try {
717                         Rectangle r = t.modelToView(pos);
718                         JScrollPane p = (JScrollPane) SwingUtilities.getAncestorOfClass(JScrollPane.class, t);
719                         r.y = Math.max(0, r.y - p.getHeight() * 2 / 5);
720                         r.height += p.getHeight() * 4 / 5;
721                         t.scrollRectToVisible(r);
722                     } catch (BadLocationException ignore) {
723                     }
724                 }
725             });
726         }
