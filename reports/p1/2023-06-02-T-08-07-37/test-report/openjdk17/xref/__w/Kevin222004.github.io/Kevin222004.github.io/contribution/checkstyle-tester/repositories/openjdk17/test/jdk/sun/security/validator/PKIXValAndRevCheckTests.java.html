431     static void validatePath(X509Certificate[] path, List<byte[]> responses,
432             PKIXBuilderParameters params, Exception expectedExc) {
433         try {
434             Validator val = Validator.getInstance(Validator.TYPE_PKIX,
435                     Validator.VAR_TLS_SERVER, params);
436             val.validate(path, null, responses, null, "RSA");
437             if (expectedExc != null) {
438                 // We expected to receive an exception here
439                 throw new RuntimeException("Did not receive expected " +
440                         expectedExc.getClass().getName());
441             }
442         } catch (CertificateException certExc) {
443             if (expectedExc == null) {
444                 // This test was supposed to pass, so wrap it in a Runtime
445                 throw new RuntimeException("Received unexpected exception: ",
446                         certExc);
447             } else {
448                 Throwable cause = certExc.getCause();
449                 if (cause == null) {
450                     throw new RuntimeException("Missing expected cause: " +
451                             expectedExc.getClass().getName(),
452                             certExc);
453                 } else {
454                     verifyCause(cause, expectedExc);
455                 }
456             }
457         }
458     }
459 
460     static void verifyCause(Throwable cause, Throwable expectedExc) {
461         if (cause.getClass() != expectedExc.getClass()) {
462             throw new RuntimeException("Exception class mismatch: expected = " +
463                     expectedExc.getClass().getName() + ", actual = " +
464                     cause.getClass().getName());
465         } else if (cause instanceof CertPathValidatorException) {
466             CertPathValidatorException actual =
467                     (CertPathValidatorException)cause;
468             CertPathValidatorException expected =
469                     (CertPathValidatorException)expectedExc;
470             // The failure index and reason should be the same
471             if (actual.getIndex() != expected.getIndex() ||
472                     actual.getReason() != expected.getReason()) {
473                 throw new RuntimeException("CertPathValidatorException " +
474                         "differs from expected.  Expected: index = " +
475                         expected.getIndex() + ", reason = " +
476                         expected.getReason() + ", Actual: index = " +
477                         actual.getIndex() + ", reason = " +
478                         actual.getReason(), actual);
479             }
480         }
481     }

483     static X509Certificate getCert(CertificateFactory fac, String pemCert) {
484         try {
485             ByteArrayInputStream bais =
486                     new ByteArrayInputStream(pemCert.getBytes("UTF-8"));
487             return (X509Certificate)fac.generateCertificate(bais);
488         } catch (UnsupportedEncodingException | CertificateException exc) {
489             throw new RuntimeException(exc);
490         }
491     }
