25  import java.io.File;
26  import java.nio.charset.Charset;
27  import java.nio.file.Files;
28  import java.nio.file.Path;
29  import java.nio.file.Paths;
30  import java.util.List;
31  import java.util.ArrayList;
32  import jdk.test.lib.cds.CDSTestUtils;
33  
34  /*
35   * @test
36   * @summary Try to archive lots of classes by searching for classes from the jrt:/ file system. With JDK 12
37   *          this will produce an archive with over 30,000 classes.
38   * @requires vm.cds
39   * @library /test/lib /test/hotspot/jtreg/runtime/cds/appcds /test/hotspot/jtreg/runtime/cds/appcds/dynamicArchive/test-classes
40   * @build LoadClasses
41   * @build sun.hotspot.WhiteBox
42   * @run driver jdk.test.lib.helpers.ClassFileInstaller -jar loadclasses.jar LoadClasses
43   * @run driver jdk.test.lib.helpers.ClassFileInstaller -jar whitebox.jar sun.hotspot.WhiteBox
44   * @run main/othervm/timeout=500 -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI -Xbootclasspath/a:./whitebox.jar DynamicLotsOfClasses
45   */
46  
47  import jdk.test.lib.helpers.ClassFileInstaller;
48  
49  public class DynamicLotsOfClasses extends DynamicArchiveTestBase {
50  
51      public static void main(String[] args) throws Exception {
52          runTest(DynamicLotsOfClasses::testDefaultBase);
53      }
54  
55      static void testDefaultBase() throws Exception {
56          String topArchiveName = getNewArchiveName("top");
57          doTest(topArchiveName);
58      }
59  
60      private static void doTest(String topArchiveName) throws Exception {
61          ArrayList<String> list = new ArrayList<>();
62          TestCommon.findAllClasses(list);
63  
64          String classList = CDSTestUtils.getOutputDir() + File.separator +
65                             "LotsOfClasses.list";
66          List<String> lines = list;
67          Path file = Paths.get(classList);
68          Files.write(file, lines, Charset.forName("UTF-8"));
69  
70          String appJar = ClassFileInstaller.getJarPath("loadclasses.jar");
71          String mainClass = "LoadClasses";
72  
73          String whiteBoxJar = ClassFileInstaller.getJarPath("whitebox.jar");
74          String bootClassPath = "-Xbootclasspath/a:" + whiteBoxJar;
75          dump(topArchiveName,
76               "--add-modules",
77               "ALL-SYSTEM",
78               "-Xlog:hashtables",
79               "-Xmx500m",
80               "-Xlog:cds=debug", // test detailed metadata info printing
81               "-Xlog:cds+dynamic=info",
82               bootClassPath,
83               "-XX:+UnlockDiagnosticVMOptions", "-XX:+WhiteBoxAPI",
84               "-cp", appJar, mainClass, classList)
85              .assertNormalExit("Written dynamic archive 0x",
86                                "Detailed metadata info");
87      }
88  }
