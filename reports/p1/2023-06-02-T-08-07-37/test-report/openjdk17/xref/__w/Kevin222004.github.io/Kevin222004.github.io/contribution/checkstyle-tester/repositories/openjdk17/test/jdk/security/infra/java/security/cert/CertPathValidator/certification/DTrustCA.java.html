35      public static void main(String[] args) throws Exception {
36  
37          ValidatePathWithParams pathValidator = new ValidatePathWithParams(null);
38  
39          boolean ocspEnabled = true;
40  
41          if (args.length >= 1 && "CRL".equalsIgnoreCase(args[0])) {
42              pathValidator.enableCRLCheck();
43              ocspEnabled = false;
44          } else {
45              // OCSP check by default
46              pathValidator.enableOCSPCheck();
47          }
48  
49          new RootClass3CA2().runTest(pathValidator, ocspEnabled);
50          new RootClass3CA2EV().runTest(pathValidator, ocspEnabled);
51      }
