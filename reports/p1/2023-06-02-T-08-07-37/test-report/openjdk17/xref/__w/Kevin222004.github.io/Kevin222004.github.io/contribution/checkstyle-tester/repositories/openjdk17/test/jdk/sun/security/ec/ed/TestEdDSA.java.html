324     private static void runSignTest(String algorithm,
325             AlgorithmParameterSpec params, String privateKey, String publicKey,
326             String message, String signature) throws Exception {
327 
328         byte[] privKeyBytes = HexFormat.of().parseHex(privateKey);
329         EdECPoint pubKeyPoint = Convert.hexStringToEdPoint(publicKey);
330         byte[] msgBytes = HexFormat.of().parseHex(message);
331         byte[] computedSig;
332 
333         NamedParameterSpec namedSpec = new NamedParameterSpec(algorithm);
334         EdECPrivateKeySpec privKeySpec =
335             new EdECPrivateKeySpec(namedSpec, privKeyBytes);
336         KeyFactory kf = KeyFactory.getInstance(algorithm);
337         PrivateKey privKey = kf.generatePrivate(privKeySpec);
338         Signature sig = Signature.getInstance(algorithm);
339         if (params != null) {
340             sig.setParameter(params);
341         }
342         sig.initSign(privKey);
343         sig.update(msgBytes);
344         computedSig = sig.sign();
345         if (!Arrays.equals(computedSig, HexFormat.of().parseHex(signature))) {
346             throw new RuntimeException("Incorrect signature");
347         }
348 
349         // test verification
350         sig = Signature.getInstance(algorithm);
351         if (params != null) {
352             sig.setParameter(params);
353         }
354         EdECPublicKeySpec pubKeySpec =
355             new EdECPublicKeySpec(namedSpec, pubKeyPoint);
356         PublicKey pubKey = kf.generatePublic(pubKeySpec);
357         sig.initVerify(pubKey);
358         sig.update(msgBytes);
359         if (!sig.verify(computedSig)) {
360             throw new RuntimeException("Signature did not verify");
361         }
362     }

381     private static void runBasicTest(String name, Object param)
382         throws Exception {
383 
384         KeyPairGenerator kpg = KeyPairGenerator.getInstance(name);
385         if (param instanceof Integer) {
386             kpg.initialize((Integer) param, random);
387         } else if (param instanceof String) {
388             kpg.initialize(new NamedParameterSpec((String) param), random);
389         }
390         KeyPair kp = kpg.generateKeyPair();
391 
392         Signature sig = Signature.getInstance(name);
393         sig.initSign(kp.getPrivate());
394         byte[] testMessage = new byte[1024];
395         random.nextBytes(testMessage);
396         sig.update(testMessage);
397         byte[] msgSig = sig.sign();
398 
399         // sign again, should return false
400         byte[] x = sig.sign();
401         if (Arrays.compare(msgSig, x) == 0) {
402             throw new RuntimeException("Second sign identical, didn't reset");
403         }
404 
405         // verify the signature
406         sig.initVerify(kp.getPublic());
407         sig.update(testMessage);
408         if (!sig.verify(msgSig)) {
409             throw new RuntimeException("Valid signature did not verify");
410         }
411 
412         // verify again, should return false
413         if (sig.verify(msgSig)) {
414             throw new RuntimeException("Second verify succeeded, didn't reset");
415         }
416 
417         // try verifying an incorrect signature
418         testMessage[0] ^= (byte) 0x01;
419         sig.update(testMessage);
420         if (sig.verify(msgSig)) {
421             throw new RuntimeException("Invalid signature verified");
422         }
423 
424         KeyFactory kf = KeyFactory.getInstance(name);
425         // Test with X509 and PKCS8 key specs
426         X509EncodedKeySpec pubSpec =
427             kf.getKeySpec(kp.getPublic(), X509EncodedKeySpec.class);
428         PKCS8EncodedKeySpec priSpec =
429             kf.getKeySpec(kp.getPrivate(), PKCS8EncodedKeySpec.class);
430 
431         PublicKey pubKey = kf.generatePublic(pubSpec);
432         PrivateKey priKey = kf.generatePrivate(priSpec);
433 
434         sig.initSign(priKey);
435         sig.update(testMessage);
436         msgSig = sig.sign();
437         sig.initVerify(pubKey);
438         sig.update(testMessage);
439         if (!sig.verify(msgSig)) {
440             throw new RuntimeException("Valid signature did not verify");
441         }
442 
443         // test with EdEC key specs
444         EdECPublicKeySpec edPublic =
445             kf.getKeySpec(kp.getPublic(), EdECPublicKeySpec.class);
446         EdECPrivateKeySpec edPrivate =
447             kf.getKeySpec(kp.getPrivate(), EdECPrivateKeySpec.class);
448         PublicKey pubKey2 = kf.generatePublic(edPublic);
449         PrivateKey priKey2 = kf.generatePrivate(edPrivate);
450         sig.initSign(priKey2);
451         sig.update(testMessage);
452         msgSig = sig.sign();
453         sig.initVerify(pubKey2);
454         sig.update(testMessage);
455         if (!sig.verify(msgSig)) {
456             throw new RuntimeException("Valid signature did not verify");
457         }
458     }

471     private static void runCurveMixTest(String providerName, String name,
472                                         Object param) throws Exception {
473 
474         KeyPairGenerator kpg = KeyPairGenerator.getInstance(name,
475             providerName);
476 
477         try {
478             if (param instanceof Integer) {
479                 kpg.initialize((Integer) param);
480             } else if (param instanceof String) {
481                 kpg.initialize(new NamedParameterSpec((String) param));
482             }
483             throw new RuntimeException(name + " KeyPairGenerator accepted "
484                     + param.toString() + " parameters");
485         } catch (InvalidParameterException ex) {
486             if (param instanceof String) {
487                 throw new RuntimeException(
488                         "InvalidAlgorithmParameterException expected", ex);
489             }
490             // expected
491 
492         } catch (InvalidAlgorithmParameterException ex) {
493             if (param instanceof Integer) {
494                 throw new RuntimeException("InvalidParameterException expected",
495                         ex);
496             }
497             // expected
498         }
499 
500         // the rest of the test uses the parameter as an algorithm name to
501         // produce keys
502         if (param instanceof Integer) {
503             return;
504         }
505         String otherName = (String) param;
506         KeyPairGenerator otherKpg = KeyPairGenerator.getInstance(otherName,
507             providerName);
508         KeyPair otherKp = otherKpg.generateKeyPair();
509 
510         // ensure the KeyFactory rejects incorrect keys
511         KeyFactory kf = KeyFactory.getInstance(name, providerName);
512         try {
513             kf.getKeySpec(otherKp.getPublic(), EdECPublicKeySpec.class);
514             throw new RuntimeException(name + " KeyFactory accepted "
515                 + param.toString() + " key");
516         } catch (InvalidKeySpecException ex) {
517             // expected
518         }
519         try {
520             kf.getKeySpec(otherKp.getPrivate(), EdECPrivateKeySpec.class);
521             throw new RuntimeException(name + " KeyFactory accepted "
522                 + param.toString() + " key");
523         } catch (InvalidKeySpecException ex) {
524             // expected
525         }
526 
527         try {
528             kf.translateKey(otherKp.getPublic());
529             throw new RuntimeException(name + " KeyFactory accepted "
530                 + param.toString() + " key");
531         } catch (InvalidKeyException ex) {
532             // expected
533         }
534         try {
535             kf.translateKey(otherKp.getPrivate());
536             throw new RuntimeException(name + " KeyFactory accepted "
537                 + param.toString() + " key");
538         } catch (InvalidKeyException ex) {
539             // expected
540         }
541 
542         KeyFactory otherKf = KeyFactory.getInstance(otherName, providerName);
543         EdECPublicKeySpec otherPubSpec = otherKf.getKeySpec(otherKp.getPublic(),
544             EdECPublicKeySpec.class);
545         try {
546             kf.generatePublic(otherPubSpec);
547             throw new RuntimeException(name + " KeyFactory accepted "
548                 + param.toString() + " key");
549         } catch (InvalidKeySpecException ex) {
550             // expected
551         }
552         EdECPrivateKeySpec otherPriSpec =
553             otherKf.getKeySpec(otherKp.getPrivate(), EdECPrivateKeySpec.class);
554         try {
555             kf.generatePrivate(otherPriSpec);
556             throw new RuntimeException(name + " KeyFactory accepted "
557                 + param.toString() + " key");
558         } catch (InvalidKeySpecException ex) {
559             // expected
560         }
561 
562         // ensure the Signature rejects incorrect keys
563         Signature sig = Signature.getInstance(name, providerName);
564         try {
565             sig.initSign(otherKp.getPrivate());
566             throw new RuntimeException(name + " Signature accepted "
567                 + param.toString() + " signing key");
568         } catch (InvalidKeyException ex) {
569             // expected
570         }
571 
572         try {
573             sig.initVerify(otherKp.getPublic());
574             throw new RuntimeException(name + " Signature accepted "
575                 + param.toString() + " verification key");
576         } catch (InvalidKeyException ex) {
577             // expected
578         }
