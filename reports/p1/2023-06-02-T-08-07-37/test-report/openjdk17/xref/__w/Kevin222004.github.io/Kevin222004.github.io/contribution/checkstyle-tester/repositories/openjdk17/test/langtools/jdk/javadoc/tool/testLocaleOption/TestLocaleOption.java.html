24  /*
25   * @test
26   * @bug     8222793 8238437
27   * @summary Javadoc tool ignores "-locale" param and uses default locale for
28   *          all messages and texts
29   * @library /tools/lib
30   * @modules jdk.javadoc/jdk.javadoc.internal.api
31   *          jdk.javadoc/jdk.javadoc.internal.tool
32   *          jdk.javadoc/jdk.javadoc.internal.tool.resources:open
33   *          jdk.javadoc/jdk.javadoc.internal.doclets.toolkit.resources:open
34   *          jdk.javadoc/jdk.javadoc.internal.doclets.formats.html.resources:open
35   * @build   toolbox.JavadocTask toolbox.ToolBox
36   * @run main TestLocaleOption
37   */

39  import java.io.File;
40  import java.io.Writer;
41  import java.util.Enumeration;
42  import java.util.ArrayList;
43  import java.util.List;
44  import java.util.Locale;
45  import java.util.Objects;
46  import java.util.Properties;
47  import java.util.ResourceBundle;
48  import java.nio.file.Files;
49  import java.nio.file.Path;
50  
51  import jdk.javadoc.internal.tool.Main;
52  
53  import toolbox.JavadocTask;
54  import toolbox.Task;
55  import toolbox.TestRunner;
56  import toolbox.ToolBox;

58  /**
59   * Tests the {@code -locale} option.
60   *
61   * The test generates a set of resources files for javadoc and the doclet
62   * that can be used to "simulate" a non-default locale. These resource files
63   * have to be patched into the {@code jdk.javadoc} module, which means that
64   * the tool must be run in a separate VM, meaning that we cannot use the
65   * standard {@code JavadocTester} framework. Instead, we fall back on ToolBox.
66   */
67  
68  public class TestLocaleOption extends TestRunner {
69      public static void main(String... args) throws Exception {
70          Locale.setDefault(Locale.US);
71          TestLocaleOption t = new TestLocaleOption();
72          t.run();
73      }
74  
75      private ToolBox tb = new ToolBox();
76      private Path patchDir;
77      private Path srcDir;
78  
79      /**
80       * Locale for the generated resource files with uppercase values.
81       */
82      private static final Locale ALLCAPS = Locale.forLanguageTag("en-GB-ALLCAPS");
83  
84      TestLocaleOption() {
85          super(System.err);
86      }
87  
88      public void run() throws Exception {
89          patchDir = Path.of("patch");
90          generateBundle(patchDir, "jdk.javadoc.internal.tool.resources.javadoc");
91          generateBundle(patchDir, "jdk.javadoc.internal.doclets.toolkit.resources.doclets");
92          generateBundle(patchDir, "jdk.javadoc.internal.doclets.formats.html.resources.standard");
93  
94          srcDir = Path.of("src");
95          tb.writeJavaFiles(srcDir,
96                  """
97                      package p;
98                      public class HelloWorld {
99                          public static void main(String... args) {
100                             System.out.println("Hello World!");
101                         }
102                     }
103                     """);
104 
105         runTests(m -> new Object[]{Path.of(m.getName())});
106     }
107 
108     @Test
109     public void testHelpDefault_US(Path base) {
110         testHelp(null, null);
111     }
112 
113     @Test
114     public void testHelpDefault_ALLCAPS(Path base) {
115         testHelp(ALLCAPS, null);
116     }
117 
118     @Test
119     public void testHelpLocale(Path base) {
120         testHelp(null, ALLCAPS);
121     }
122 
123     private void testHelp(Locale defaultLocale, Locale localeOption) {
124         String stdOut = javadoc(defaultLocale, localeOption, "-help")
125                 .writeAll()
126                 .getOutput(Task.OutputKind.STDOUT);
127 
128         if (Objects.equals(defaultLocale, ALLCAPS)) {
129             checkContains(stdOut,
130                     """
131                         USAGE:
132                             JAVADOC [OPTIONS] [PACKAGENAMES] [SOURCEFILES] [@FILES]""");
133         } else {
134             checkContains(stdOut,
135                     """
136                         Usage:
137                             javadoc [options] [packagenames] [sourcefiles] [@files]""");
138         }
139     }
140 
141     @Test
142     public void testHelloWorldDefault_US(Path base) throws Exception {
143         testHelloWorld(base, null, null);
144     }
145 
146     @Test
147     public void testHelloWorldDefault_ALLCAPS(Path base) throws Exception {
148         testHelloWorld(base, ALLCAPS, null);
149     }
150 
151     @Test
152     public void testHelloWorldLocale(Path base) throws Exception {
153         testHelloWorld(base, null, ALLCAPS);
154     }
155 
156     private void testHelloWorld(Path base, Locale defaultLocale, Locale localeOption) throws Exception {
157         Path apiDir = base.resolve("api");
158         String stdErr = javadoc(defaultLocale,
159                                 localeOption,
160                                 "-sourcepath", srcDir.toString(),
161                                 "-d", apiDir.toString(),
162                                 "p")
163                 .writeAll()
164                 .getOutput(Task.OutputKind.STDERR);
165 
166         // check console messages
167         if (Objects.equals(defaultLocale, ALLCAPS)) {
168             checkContains(stdErr,
169                     """
170                         LOADING SOURCE FILES FOR PACKAGE p...
171                         CONSTRUCTING JAVADOC INFORMATION...""");
172         } else {
173             checkContains(stdErr,
174                     """
175                         Loading source files for package p...
176                         Constructing Javadoc information...""");
177         }
178 
179         // check generated files
180         String hw = Files.readString(apiDir.resolve("p/HelloWorld.html"));
181         Locale docLocale = localeOption != null ? localeOption : defaultLocale;
182         if (Objects.equals(docLocale, ALLCAPS)) {
183             checkContains(hw,
184                     "<h2>METHOD SUMMARY</h2>",
185                     """
186                         <div class="table-header col-first">MODIFIER AND TYPE</div>""",
187                     """
188                         <div class="table-header col-second">METHOD</div>""",
189                     """
190                         <div class="table-header col-last">DESCRIPTION</div>""");
191         } else {
192             checkContains(hw,
193                     "<h2>Method Summary</h2>",
194                     """
195                         <div class="table-header col-first">Modifier and Type</div>""",
196                     """
197                         <div class="table-header col-second">Method</div>""",
198                     """
199                         <div class="table-header col-last">Description</div>""");
200         }
201     }
202 
203     /**
204      * Generates a copy of a resource bundle, with the values converted to uppercase.
205      *
206      * @param dir  the root directory in which to write the bundle
207      * @param name the name of the bundle
208      * @throws Exception if an error occurs
209      */
210     private void generateBundle(Path dir, String name) throws Exception {
211         Module m = Main.class.getModule();
212         ResourceBundle rb = ResourceBundle.getBundle(name, m);
213         Properties p = new Properties();
214         Enumeration<String> e = rb.getKeys();
215         while (e.hasMoreElements()) {
216             String key = e.nextElement();
217             String value = rb.getString(key);
218             p.put(key, value.toUpperCase(Locale.US));
219         }
220         String localeSuffix = ALLCAPS.toString().replace("-", "_");
221         Path outPath = dir.resolve(name.replace(".", File.separator) + "_" + localeSuffix + ".properties");
222         Files.createDirectories(outPath.getParent());
223         try (Writer out = Files.newBufferedWriter(outPath)) {
224             p.store(out, "Generated by TestLocaleOption");
225             System.err.println("wrote: " + outPath);
226         }
227     }
228 
229     /**
230      * Runs javadoc, with the specified arguments,
231      * optionally specifying the default locale and locale option
232      *
233      * @param defaultLocale the default locale for the VM, or null if not specified
234      * @param localeOption  the value for the locale option, or null if not specified
235      * @param args          additional command-line args
236      * @return the task result
237      */
238     private Task.Result javadoc(Locale defaultLocale, Locale localeOption, String... args) {
239         List<String> options = new ArrayList<>();
240         options.add("-J--patch-module=jdk.javadoc=" + patchDir);
241         if (defaultLocale != null) {
242             options.add("-J-Duser.language=" + defaultLocale.getLanguage());
243             options.add("-J-Duser.country=" + defaultLocale.getCountry());
244             options.add("-J-Duser.variant=" + defaultLocale.getVariant());
245         }
246         if (localeOption != null) {
247             options.addAll(List.of("-locale", localeOption.toString()));
248         }
249         options.addAll(List.of(args));
250         System.err.println("Options: " + options);
251 
252         return new JavadocTask(tb, Task.Mode.EXEC)
253                 .options(options)
254                 .run();
255     }
256 
257     private String NL = System.lineSeparator();
258     private void checkContains(String found, String... expect) {
259         for (String e : expect) {
260             String e2 = e.replace("\n", NL);
261             if (!found.contains(e2)) {
262                 error("expected string not found: '" + e2 + "'");
263             }
264         }
265     }
