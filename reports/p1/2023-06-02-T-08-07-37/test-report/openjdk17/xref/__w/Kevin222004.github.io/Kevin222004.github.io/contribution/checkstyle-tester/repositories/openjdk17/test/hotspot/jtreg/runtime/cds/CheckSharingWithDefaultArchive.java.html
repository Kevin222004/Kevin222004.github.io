24  /**
25   * @test Default CDS archive
26   * @summary Sharing should be enabled by default on supported platform/binaries.
27   *          No -Xshare:dump is needed. No -Xshare:auto or -Xshare:on in needed.
28   *          Verify a set of well-known shared classes.
29   * @requires vm.cds
30   * @library /test/lib
31   * @build sun.hotspot.WhiteBox
32   * @run driver jdk.test.lib.helpers.ClassFileInstaller sun.hotspot.WhiteBox
33   * @run main/othervm -Xbootclasspath/a:. -XX:+UnlockDiagnosticVMOptions
34   *      -XX:+WhiteBoxAPI CheckSharingWithDefaultArchive -showversion
35   */
36  import jdk.test.lib.Platform;
37  import jtreg.SkippedException;
38  import sun.hotspot.WhiteBox;
39  
40  public class CheckSharingWithDefaultArchive {
41      public static void main(String[] args) throws Exception {
42          if (!Platform.isDefaultCDSArchiveSupported()) {
43              throw new SkippedException("Supported platform");
44          }
45  
46          WhiteBox wb = WhiteBox.getWhiteBox();
47          String classes[] = {"java.lang.Object",
48                              "java.lang.String",
49                              "java.lang.Class"};
50          // If maping fails, sharing is disabled
51          if (wb.isSharingEnabled()) {
52              for (int i = 0; i < classes.length; i++) {
53                  Class c = Class.forName(classes[i]);
54                  if (wb.isSharedClass(c)) {
55                      System.out.println(classes[i] + " is shared.");
56                  } else {
57                      throw new RuntimeException(classes[i] + " is not shared");
58                  }
59              }
60          } else {
61             throw new SkippedException("Sharing is not enabled.");
62          }
63      }
64  }
