67   * Checks that the various prohibited Serialization magic methods, and
68   * Externalizable methods, are not invoked ( effectively ignored ) for
69   * record objects.

73      public interface ThrowingExternalizable extends Externalizable {
74          default void writeExternal(ObjectOutput out) {
75              fail("should not reach here");
76          }
77          default void readExternal(ObjectInput in) {
78              fail("should not reach here");
79          }
80      }

82      record Wibble () implements ThrowingExternalizable { }
83  
84      record Wobble (long l) implements ThrowingExternalizable { }
85  
86      record Wubble (Wobble wobble, Wibble wibble, String s) implements ThrowingExternalizable { }

90      /**
91       * Generates the serializable record classes used by the test. First creates
92       * the initial bytecode for the record classes using javac, then adds the
93       * prohibited magic serialization methods. For example, effectively generates:
94       *
95       *   public record Foo () implements Serializable {
96       *       private void writeObject(ObjectOutputStream out)             {
97       *           fail("writeObject should not be invoked");               }
98       *       private void readObject(ObjectInputStream in)                {
99       *           fail("readObject should not be invoked");                }
100      *       private void readObjectNoData()                              {
101      *           fail("readObjectNoData should not be invoked");          }
102      *   }
103      */
104     @BeforeTest
105     public void setup() {
106         {
107             byte[] byteCode = InMemoryJavaCompiler.compile("Foo",
108                     "public record Foo () implements java.io.Serializable { }");
109             byteCode = addWriteObject(byteCode);
110             byteCode = addReadObject(byteCode);
111             byteCode = addReadObjectNoData(byteCode);
112             serializableRecordLoader = new ByteCodeLoader("Foo", byteCode, ProhibitedMethods.class.getClassLoader());
113         }
114         {
115             byte[] byteCode = InMemoryJavaCompiler.compile("Bar",
116                     "public record Bar (int x, int y) implements java.io.Serializable { }");
117             byteCode = addWriteObject(byteCode);
118             byteCode = addReadObject(byteCode);
119             byteCode = addReadObjectNoData(byteCode);
120             serializableRecordLoader = new ByteCodeLoader("Bar", byteCode, serializableRecordLoader);
121         }
122         {
123             byte[] byteCode = InMemoryJavaCompiler.compile("Baz",
124                     "import java.io.Serializable;" +
125                     "public record Baz<U extends Serializable,V extends Serializable>(U u, V v) implements Serializable { }");
126             byteCode = addWriteObject(byteCode);
127             byteCode = addReadObject(byteCode);
128             byteCode = addReadObjectNoData(byteCode);
129             serializableRecordLoader = new ByteCodeLoader("Baz", byteCode, serializableRecordLoader);
130         }
131     }

181     @Test(dataProvider = "recordInstances")
182     public void roundTrip(Object objToSerialize) throws Exception {
183         out.println("\n---");
184         out.println("serializing : " + objToSerialize);
185         var objDeserialized = serializeDeserialize(objToSerialize);
186         out.println("deserialized: " + objDeserialized);
187         assertEquals(objToSerialize, objDeserialized);
188         assertEquals(objDeserialized, objToSerialize);
189     }

191     <T> byte[] serialize(T obj) throws IOException {
192         ByteArrayOutputStream baos = new ByteArrayOutputStream();
193         ObjectOutputStream oos = new ObjectOutputStream(baos);
194         oos.writeObject(obj);
195         oos.close();
196         return baos.toByteArray();
197     }
198 
199     @SuppressWarnings("unchecked")
200     <T> T deserialize(byte[] streamBytes)
201         throws IOException, ClassNotFoundException
202     {
203         ByteArrayInputStream bais = new ByteArrayInputStream(streamBytes);
204         ObjectInputStream ois  = new ObjectInputStream(bais) {
205             @Override
206             protected Class<?> resolveClass(ObjectStreamClass desc)
207                     throws ClassNotFoundException {
208                 return Class.forName(desc.getName(), false, serializableRecordLoader);
209             }
210         };
211         return (T) ois.readObject();
212     }
213 
214     <T> T serializeDeserialize(T obj)
215         throws IOException, ClassNotFoundException
216     {
217         return deserialize(serialize(obj));
218     }

222     static byte[] addWriteObject(byte[] classBytes) {
223         return addMethod(classBytes, cv -> new WriteObjectVisitor(cv));
224     }
225 
226     static byte[] addReadObject(byte[] classBytes) {
227         return addMethod(classBytes, cv -> new ReadObjectVisitor(cv));
228     }
229 
230     static byte[] addReadObjectNoData(byte[] classBytes) {
231         return addMethod(classBytes, cv -> new ReadObjectNoDataVisitor(cv));
232     }
233 
234     static byte[] addMethod(byte[] classBytes,
235                             Function<ClassVisitor,ClassVisitor> classVisitorCreator) {
236         ClassReader reader = new ClassReader(classBytes);
237         ClassWriter writer = new ClassWriter(reader, COMPUTE_MAXS | COMPUTE_FRAMES);
238         reader.accept(classVisitorCreator.apply(writer), 0);
239         return writer.toByteArray();
240     }

242     static abstract class AbstractVisitor extends ClassVisitor {
243         final String nameOfMethodToAdd;
244         AbstractVisitor(ClassVisitor cv, String nameOfMethodToAdd) {
245             super(ASM8, cv);
246             this.nameOfMethodToAdd = nameOfMethodToAdd;
247         }
248         @Override
249         public MethodVisitor visitMethod(final int access,
250                                          final String name,
251                                          final String descriptor,
252                                          final String signature,
253                                          final String[] exceptions) {
254             // the method-to-be-added should not already exist
255             assert !name.equals(nameOfMethodToAdd) : "Unexpected " + name + " method";
256             return cv.visitMethod(access, name, descriptor, signature, exceptions);
257         }
258         @Override
259         public void visitEnd() {
260             throw new UnsupportedOperationException("implement me");
261         }
262     }
263 
264     /** A visitor that generates and adds a writeObject method. */
265     static final class WriteObjectVisitor extends AbstractVisitor {
266         static final String WRITE_OBJECT_NAME = "writeObject";
267         static final String WRITE_OBJECT_DESC = "(Ljava/io/ObjectOutputStream;)V";
268         WriteObjectVisitor(ClassVisitor cv) { super(cv, WRITE_OBJECT_NAME); }
269         @Override
270         public void visitEnd() {
271             MethodVisitor mv = cv.visitMethod(ACC_PRIVATE, WRITE_OBJECT_NAME, WRITE_OBJECT_DESC, null, null);
272             mv.visitCode();
273             mv.visitLdcInsn(WRITE_OBJECT_NAME + " should not be invoked");
274             mv.visitMethodInsn(INVOKESTATIC, "org/testng/Assert", "fail", "(Ljava/lang/String;)V", false);
275             mv.visitInsn(RETURN);
276             mv.visitMaxs(0, 0);
277             mv.visitEnd();
278 
279             cv.visitEnd();
280         }
281     }
282

323     static final Class<ReflectiveOperationException> ROE = ReflectiveOperationException.class;
324 
325     /** Checks to ensure correct operation of the test's generation logic. */
326     @Test
327     public void wellFormedGeneratedClasses() throws Exception {
328         out.println("\n---");
329         for (Object obj : new Object[] { newFoo(), newBar(3, 4), newBaz(1,"s") }) {
330             out.println(obj);
331             {   // writeObject
332                 Method m = obj.getClass().getDeclaredMethod("writeObject", ObjectOutputStream.class);
333                 assertTrue((m.getModifiers() & Modifier.PRIVATE) != 0);
334                 m.setAccessible(true);
335                 ReflectiveOperationException t = expectThrows(ROE, () ->
336                         m.invoke(obj, new ObjectOutputStream(OutputStream.nullOutputStream())));
337                 Throwable assertionError = t.getCause();
338                 out.println("caught expected AssertionError: " + assertionError);
339                 assertTrue(assertionError instanceof AssertionError,
340                            "Expected AssertionError, got:" + assertionError);
341                 assertEquals(assertionError.getMessage(), "writeObject should not be invoked");
342             }
343             {   // readObject
344                 Method m = obj.getClass().getDeclaredMethod("readObject", ObjectInputStream.class);
345                 assertTrue((m.getModifiers() & Modifier.PRIVATE) != 0);
346                 m.setAccessible(true);
347                 ReflectiveOperationException t = expectThrows(ROE, () ->
348                         m.invoke(obj, new ObjectInputStream() {
349                         }));
350                 Throwable assertionError = t.getCause();
351                 out.println("caught expected AssertionError: " + assertionError);
352                 assertTrue(assertionError instanceof AssertionError,
353                            "Expected AssertionError, got:" + assertionError);
354                 assertEquals(assertionError.getMessage(), "readObject should not be invoked");
355             }
356             {   // readObjectNoData
357                 Method m = obj.getClass().getDeclaredMethod("readObjectNoData");
358                 assertTrue((m.getModifiers() & Modifier.PRIVATE) != 0);
359                 m.setAccessible(true);
360                 ReflectiveOperationException t = expectThrows(ROE, () -> m.invoke(obj));
361                 Throwable assertionError = t.getCause();
362                 out.println("caught expected AssertionError: " + assertionError);
363                 assertTrue(assertionError instanceof AssertionError,
364                            "Expected AssertionError, got:" + assertionError);
365                 assertEquals(assertionError.getMessage(), "readObjectNoData should not be invoked");
366             }
367         }
