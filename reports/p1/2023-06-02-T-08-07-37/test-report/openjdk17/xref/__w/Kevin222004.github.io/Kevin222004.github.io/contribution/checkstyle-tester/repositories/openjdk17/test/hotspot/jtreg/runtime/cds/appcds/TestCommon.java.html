109     // Attempt to clean old archives to preserve space
110     // Archives are large artifacts (20Mb or more), and much larger than
111     // most other artifacts created in jtreg testing.
112     // Therefore it is a good idea to clean the old archives when they are not needed.
113     // In most cases the deletion attempt will succeed; on rare occasion the
114     // delete operation will fail since the system or VM process still holds a handle
115     // to the file; in such cases the File.delete() operation will silently fail, w/o
116     // throwing an exception, thus allowing testing to continue.
117     public static void deletePriorArchives() {
118         File dir = CDSTestUtils.getOutputDirAsFile();
119         String files[] = dir.list();
120         for (String name : files) {
121             if (name.startsWith("appcds-") && name.endsWith(".jsa")) {
122                 if (!(new File(dir, name)).delete())
123                     System.out.println("deletePriorArchives(): delete failed for file " + name);
124             }
125         }
126     }

180     private static void captureVerifyOpts(ArrayList<String> opts, ArrayList<String> verifyOpts) {
181         boolean addedDiagnosticOpt = false;
182         for (String s : opts) {
183             if (s.startsWith("-XX:-BytecodeVerification")) {
184                 if (!addedDiagnosticOpt) {
185                     verifyOpts.add("-XX:+UnlockDiagnosticVMOptions");
186                     addedDiagnosticOpt = true;
187                 }
188                 verifyOpts.add(s);
189             }
190             if (s.startsWith("-Xverify")) {
191                 verifyOpts.add(s);
192             }
193         }
194     }

196     // Create AppCDS archive using appcds options
197     public static OutputAnalyzer createArchive(AppCDSOptions opts)
198         throws Exception {
199         ArrayList<String> cmd = new ArrayList<String>();
200         ArrayList<String> verifyOpts = new ArrayList<String>();
201         startNewArchiveName();
202 
203         for (String p : opts.prefix) cmd.add(p);
204 
205         if (opts.appJar != null) {
206             cmd.add("-cp");
207             cmd.add(opts.appJar);
208             File jf = new File(opts.appJar);
209             if (DYNAMIC_DUMP && !jf.isDirectory()) {
210                 patchJarForDynamicDump(opts.appJar);
211             }
212         } else {
213             cmd.add("-Djava.class.path=");
214         }
215 
216         if (opts.archiveName == null) {
217             opts.archiveName = getCurrentArchiveName();
218         }
219 
220         if (DYNAMIC_DUMP) {
221             File baseArchive = null;
222             captureVerifyOpts(opts.suffix, verifyOpts);
223             int size = verifyOpts.size();
224             if (tempBaseArchive == null || !(new File(tempBaseArchive)).isFile() || size > 0) {
225                 tempBaseArchive = getNewArchiveName("tempBaseArchive");
226                 if (size == 0) {
227                     dumpBaseArchive(tempBaseArchive);
228                 } else {
229                     dumpBaseArchive(tempBaseArchive, verifyOpts.toArray(new String[size]));
230                 }
231             }
232             cmd.add("-Xshare:on");
233             cmd.add("-XX:SharedArchiveFile=" + tempBaseArchive);
234             cmd.add("-XX:ArchiveClassesAtExit=" + opts.archiveName);
235             cmd.add("-Xlog:cds");
236             cmd.add("-Xlog:cds+dynamic");
237             boolean mainModuleSpecified = false;
238             boolean patchModuleSpecified = false;
239             for (String s : opts.suffix) {
240                 if (s.length() == 0) {
241                     continue;
242                 }
243                 if (s.equals("-m")) {
244                     mainModuleSpecified = true;
245                 }
246                 if (s.startsWith("--patch-module=")) {
247                     patchModuleSpecified = true;
248                 }
249                 cmd.add(s);
250             }
251 
252             if (opts.appJar != null) {
253                 // classlist is supported only when we have a Jar file to patch (to insert
254                 // cdsutils.DynamicDumpHelper)
255                 if (opts.classList == null) {
256                     throw new RuntimeException("test.dynamic.dump requires classList file");
257                 }
258 
259                 if (!mainModuleSpecified && !patchModuleSpecified) {
260                     cmd.add("cdsutils.DynamicDumpHelper");
261                     File classListFile = makeClassList(opts.classList);
262                     cmd.add(classListFile.getPath());
263                 }
264             } else {
265                 if (!mainModuleSpecified && !patchModuleSpecified) {
266                     // If you have an empty classpath, you cannot specify a classlist!
267                     if (opts.classList != null && opts.classList.length > 0) {
268                         throw new RuntimeException("test.dynamic.dump is not supported with an empty classpath while the classlist is not empty");
269                     }
270                     cmd.add("-version");
271                 }
272             }
273         } else {
274             // static dump
275             cmd.add("-Xshare:dump");
276             cmd.add("-Xlog:cds");
277             cmd.add("-XX:SharedArchiveFile=" + opts.archiveName);
278 
279             if (opts.classList != null) {
280                 File classListFile = makeClassList(opts.classList);
281                 cmd.add("-XX:ExtraSharedClassListFile=" + classListFile.getPath());
282             }
283             for (String s : opts.suffix) {
284                 cmd.add(s);
285             }
286         }
287 
288         ProcessBuilder pb = ProcessTools.createTestJvm(cmd);
289         if (opts.appJarDir != null) {
290             pb.directory(new File(opts.appJarDir));
291         }
292 
293         OutputAnalyzer output = executeAndLog(pb, "dump");
294         if (DYNAMIC_DUMP && isUnableToMap(output)) {
295             throw new SkippedException(UnableToMapMsg);
296         }
297         return output;
298     }

300     // This allows you to run the AppCDS tests with JFR enabled at runtime (though not at
301     // dump time, as that's uncommon for typical AppCDS users).
302     //
303     // To run in this special mode, add the following to your jtreg command-line
304     //    -Dtest.cds.run.with.jfr=true
305     //
306     // Some AppCDS tests are not compatible with this mode. See the group
307     // hotspot_appcds_with_jfr in ../../TEST.ROOT for details.
308     private static final boolean RUN_WITH_JFR = Boolean.getBoolean("test.cds.run.with.jfr");
309     // This method simulates -Xshare:dump with -XX:ArchiveClassesAtExit. This way, we
310     // can re-use many tests (outside of the ./dynamicArchive directory) for testing
311     // general features of JDK-8215311 (JEP 350: Dynamic CDS Archives).
312     //
313     // We insert the cdsutils/DynamicDumpHelper.class into the first Jar file in
314     // the classpath. We use this class to load all the classes specified in the classlist.
315     //
316     // There's no need to change the run-time command-line: in this special mode, two
317     // archives are involved. The command-line specifies only the top archive. However,
318     // the location of the base archive is recorded in the top archive, so it can be
319     // determined by the JVM at runtime start-up.
320     //
321     // To run in this special mode, specify the following in your jtreg command-line
322     //    -Dtest.dynamic.cds.archive=true
323     //
324     // Note that some tests are not compatible with this special mode, including
325     //    + Tests in ./dynamicArchive: these tests are specifically written for
326     //      dynamic archive, and do not use TestCommon.createArchive(), which works
327     //      together with patchJarForDynamicDump().
328     //    + Tests related to cached objects and shared strings: dynamic dumping
329     //      does not support these.
330     //    + Custom loader tests: DynamicDumpHelper doesn't support the required
331     //      classlist syntax. (FIXME).
332     //    + Extra symbols and extra strings.
333     // See the hotspot_appcds_dynamic in ../../TEST.ROOT for details.
334     //
335     // To run all tests that are compatible with this mode:
336     //    cd test/hotspot/jtreg
337     //    jtreg -Dtest.dynamic.cds.archive=true :hotspot_appcds_dynamic
338     //
339     private static void patchJarForDynamicDump(String cp) throws Exception {
340         System.out.println("patchJarForDynamicDump: classpath = " + cp);
341         String firstJar = cp;
342         int n = firstJar.indexOf(File.pathSeparator);
343         if (n > 0) {
344             firstJar = firstJar.substring(0, n);
345         }
346         String classDir = System.getProperty("test.classes");
347         String expected = getOutputDir() + File.separator;
348 
349         if (!firstJar.startsWith(expected)) {
350             throw new RuntimeException("FIXME: jar file not at a supported location ('"
351                                        + expected + "'): " + firstJar);
352         }
353 
354         String replaceJar = firstJar + ".tmp";
355         String patchClass = "cdsutils/DynamicDumpHelper.class";
356         ZipFile zipFile = new ZipFile(firstJar);
357         byte[] buf = new byte[1024];
358         int len;
359         if (zipFile.getEntry(patchClass) == null) {
360             FileOutputStream fout = new FileOutputStream(replaceJar);
361             final ZipOutputStream zos = new ZipOutputStream(fout);
362 
363             zos.putNextEntry(new ZipEntry(patchClass));
364             InputStream is = new FileInputStream(classDir + File.separator + patchClass);
365             while ((len = (is.read(buf))) > 0) {
366                 zos.write(buf, 0, len);
367             }
368             zos.closeEntry();
369             is.close();
370 
371             for (Enumeration e = zipFile.entries(); e.hasMoreElements(); ) {
372                 ZipEntry entryIn = (ZipEntry) e.nextElement();
373                 zos.putNextEntry(entryIn);
374                 is = zipFile.getInputStream(entryIn);
375                 while ((len = is.read(buf)) > 0) {
376                     zos.write(buf, 0, len);
377                 }
378                 zos.closeEntry();
379                 is.close();
380             }
381 
382             zos.close();
383             fout.close();
384             zipFile.close();
385 
386             File oldFile = new File(firstJar);
387             File newFile = new File(replaceJar);
388             oldFile.delete();
389             newFile.renameTo(oldFile);
390             System.out.println("firstJar = " + firstJar + " Modified");
391         } else {
392             zipFile.close();
393             System.out.println("firstJar = " + firstJar);
394         }
395     }

397     // Execute JVM using AppCDS archive with specified AppCDSOptions
398     public static OutputAnalyzer runWithArchive(AppCDSOptions opts)
399         throws Exception {
400 
401         ArrayList<String> cmd = new ArrayList<String>();
402 
403         for (String p : opts.prefix) cmd.add(p);
404 
405         cmd.add("-Xshare:" + opts.xShareMode);
406         cmd.add("-showversion");
407         cmd.add("-XX:SharedArchiveFile=" + getCurrentArchiveName());
408         cmd.add("-Dtest.timeout.factor=" + timeoutFactor);
409 
410         if (opts.appJar != null) {
411             cmd.add("-cp");
412             cmd.add(opts.appJar);
413         }
414 
415         for (String s : opts.suffix) cmd.add(s);
416 
417         if (RUN_WITH_JFR) {
418             boolean usesJFR = false;
419             for (String s : cmd) {
420                 if (s.startsWith("-XX:StartFlightRecording") || s.startsWith("-XX:FlightRecorderOptions")) {
421                     System.out.println("JFR option might have been specified. Don't interfere: " + s);
422                     usesJFR = true;
423                     break;
424                 }
425             }
426             if (!usesJFR) {
427                 System.out.println("JFR option not specified. Enabling JFR ...");
428                 cmd.add(0, "-XX:StartFlightRecording:dumponexit=true");
429                 System.out.println(cmd);
430             }
431         }
432 
433         ProcessBuilder pb = ProcessTools.createTestJvm(cmd);
434         if (opts.appJarDir != null) {
435             pb.directory(new File(opts.appJarDir));
436         }
437         return executeAndLog(pb, "exec");
438     }

646     public static boolean checkOutputStrings(String outputString1,
647                                              String outputString2,
648                                              String split_regex) {
649         String[] sa1 = outputString1.split(split_regex);
650         String[] sa2 = outputString2.split(split_regex);
651         Arrays.sort(sa1);
652         Arrays.sort(sa2);
653 
654         int i = 0;
655         for (String s : sa1) {
656             if (!s.equals(sa2[i])) {
657                 throw new RuntimeException(s + " is different from " + sa2[i]);
658             }
659             i ++;
660         }
661         return true;
662     }

667         // Find all the classes in the jrt file system
668         pattern = Pattern.compile("/modules/[a-z.]*[a-z]+/([^-]*)[.]class");
669         FileSystem fs = FileSystems.getFileSystem(URI.create("jrt:/"));
670         Path base = fs.getPath("/modules/");
671         findAllClassesAtPath(base, list);
672     }

674     private static void findAllClassesAtPath(Path p, ArrayList<String> list) throws Exception {
675         try (DirectoryStream<Path> stream = Files.newDirectoryStream(p)) {
676             for (Path entry: stream) {
677                 Matcher matcher = pattern.matcher(entry.toString());
678                 if (matcher.find()) {
679                     String className = matcher.group(1);
680                     list.add(className);
681                 }
682                 try {
683                     findAllClassesAtPath(entry, list);
684                 } catch (Exception ex) {}
685             }
686         }
687     }

689     public static String composeRelPath(String appJar) {
690          int idx = appJar.lastIndexOf(File.separator);
691          String jarName = appJar.substring(idx + 1);
692          String jarDir = appJar.substring(0, idx);
693          String lastDir = jarDir.substring(jarDir.lastIndexOf(File.separator));
694          String relPath = jarDir + File.separator + ".." + File.separator + lastDir;
695          String newJar = relPath + File.separator + jarName;
696          return newJar;
697     }

700     public static File createSymLink(String appJar) throws Exception {
701          int idx = appJar.lastIndexOf(File.separator);
702          String jarName = appJar.substring(idx + 1);
703          String jarDir = appJar.substring(0, idx);
704          File origJar = new File(jarDir, jarName);
705          String linkedJarName = "linked_" + jarName;
706          File linkedJar = null;
707          if (!Platform.isWindows()) {
708              linkedJar = new File(jarDir, linkedJarName);
709              if (linkedJar.exists()) {
710                  linkedJar.delete();
711              }
712              Files.createSymbolicLink(linkedJar.toPath(), origJar.toPath());
713          }
714          return linkedJar;
715     }

717     // Remove all UL log messages from a JVM's STDOUT (such as those printed by -Xlog:cds)
718     static Pattern logPattern = Pattern.compile("^\\[[0-9. ]*s\\].*");
719     public static String filterOutLogs(String stdout) {
720         StringBuilder sb = new StringBuilder();
721         String prefix = "";
722         for (String line : stdout.split("\n")) {
723             if (logPattern.matcher(line).matches()) {
724                 continue;
725             }
726             sb.append(prefix);
727             sb.append(line);
728             prefix = "\n";
729         }
730         if (stdout.endsWith("\n")) {
731             // String.split("A\n") returns {"A"}, not {"A", ""}.
732             sb.append("\n");
733         }
734         return sb.toString();
735     }
