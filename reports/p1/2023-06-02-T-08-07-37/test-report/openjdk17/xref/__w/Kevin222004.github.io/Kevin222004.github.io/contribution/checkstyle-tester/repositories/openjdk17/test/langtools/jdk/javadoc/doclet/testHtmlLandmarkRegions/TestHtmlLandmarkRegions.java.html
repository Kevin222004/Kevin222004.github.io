38  import java.nio.file.Files;
39  import java.nio.file.Path;
40  import java.nio.file.Paths;
41  import java.util.List;
42  
43  import builder.ClassBuilder;
44  import toolbox.ModuleBuilder;
45  import toolbox.ToolBox;
46  
47  import javadoc.tester.JavadocTester;
48  
49  public class TestHtmlLandmarkRegions extends JavadocTester {
50  
51      final ToolBox tb;
52  
53      public static void main(String... args) throws Exception {
54          TestHtmlLandmarkRegions tester = new TestHtmlLandmarkRegions();
55          tester.runTests(m -> new Object[]{Paths.get(m.getName())});
56      }
57  
58      TestHtmlLandmarkRegions() {
59          tb = new ToolBox();
60      }
61  
62      @Test
63      public void testModules(Path base) throws Exception {
64          Path srcDir = base.resolve("src");
65          createModules(srcDir);
66  
67          Path outDir = base.resolve("out");
68          javadoc("-d", outDir.toString(),
69                  "-doctitle", "Document Title",
70                  "-header", "Test Header",
71                  "-bottom", "bottom text",
72                  "--module-source-path", srcDir.toString(),
73                  "--module", "m1,m2");
74  
75          checkExit(Exit.OK);
76  
77          checkOrder("index.html",
78                  """
79                      <header role="banner" class="flex-header">
80                      <nav role="navigation">""",
81                  """
82                      <main role="main">
83                      <div class="header">
84                      <h1 class="title">Document Title</h1>""",
85                  """
86                      <footer role="contentinfo"> """,
87                  """
88                      bottom text"""
89          );
90      }
91  
92      @Test
93      public void testPackages(Path base) throws Exception {
94          Path srcDir = base.resolve("src");
95          createPackages(srcDir);
96  
97          Path outDir = base.resolve("out");
98          javadoc("-d", outDir.toString(),
99                  "-doctitle", "Document Title",
100                 "-header", "Test Header",
101                 "-bottom", "bottom text",
102                 "-sourcepath", srcDir.toString(),
103                 "pkg1", "pkg2");
104 
105         checkExit(Exit.OK);
106 
107         checkOrder("index.html",
108                 """
109                     <header role="banner" class="flex-header">
110                     <nav role="navigation">""",
111                 """
112                     <main role="main">
113                     <div class="header">
114                     <h1 class="title">Document Title</h1>""",
115                 """
116                     <footer role="contentinfo">""",
117                 """
118                     bottom text""");
119     }
120 
121     @Test
122     public void testDocFiles(Path base) throws Exception {
123         Path srcDir = base.resolve("src");
124         createPackages(srcDir);
125         Path docFiles = Files.createDirectory(srcDir.resolve("pkg1").resolve("doc-files"));
126         Files.write(docFiles.resolve("s.html"), List.of(
127                 """
128                     <html>
129                       <head>
130                         <title>"Hello World"</title>
131                       </head>
132                       <body>
133                          A sample doc file.
134                       </body>
135                     </html>"""));
136 
137         Path outDir = base.resolve("out");
138         javadoc("-d", outDir.toString(),
139                 "-bottom", "bottom text",
140                 "-sourcepath", srcDir.toString(),
141                 "pkg1", "pkg2");
142 
143         checkExit(Exit.OK);
144 
145         checkOrder("pkg1/doc-files/s.html",
146                 """
147                     <header role="banner" class="flex-header">
148                     <nav role="navigation">
149                     """,
150                 """
151                     <main role="main">A sample doc file""",
152                 """
153                     <footer role="contentinfo">""",
154                 """
155                     bottom text"""
156                 );
157     }
158 
159     void createModules(Path srcDir) throws Exception {
160         new ModuleBuilder(tb, "m1")
161                 .classes("package p1; public class a { }")
162                 .classes("package p2; public class b { }")
163                 .write(srcDir);
164         new ModuleBuilder(tb, "m2")
165                 .classes("package p3; public class c { }")
166                 .classes("package p4; public class d { }")
167                 .write(srcDir);
168     }
169 
170     void createPackages(Path srcDir) throws Exception {
171         new ClassBuilder(tb, "pkg1.A")
172                 .setModifiers("public", "class")
173                 .write(srcDir);
174         new ClassBuilder(tb, "pkg2.B")
175                 .setModifiers("public", "class")
176                 .write(srcDir);
177     }
