147     void assertBodyPublisherEqual(HttpRequest r1, HttpRequest r2) {
148         if (r1.bodyPublisher().isPresent()) {
149             assertTrue(r2.bodyPublisher().isPresent());
150             var bp1 = r1.bodyPublisher().get();
151             var bp2 = r2.bodyPublisher().get();
152 
153             assertEquals(bp1.getClass(), bp2.getClass());
154             assertEquals(bp1.contentLength(), bp2.contentLength());
155 
156             final class TestSubscriber implements Flow.Subscriber<ByteBuffer> {
157                 final BodySubscriber<String> s;
158                 TestSubscriber(BodySubscriber<String> s) { this.s = s; }
159                 @Override
160                 public void onSubscribe(Flow.Subscription subscription) { s.onSubscribe(subscription); }
161                 @Override
162                 public void onNext(ByteBuffer item) { s.onNext(List.of(item)); }
163                 @Override
164                 public void onError(Throwable throwable) { fail("TestSubscriber failed"); }
165                 @Override
166                 public void onComplete() { s.onComplete(); }
167             }
168             var bs1 = BodySubscribers.ofString(UTF_8);
169             bp1.subscribe(new TestSubscriber(bs1));
170             var b1 = bs1.getBody().toCompletableFuture().join().getBytes();
171 
172             var bs2 = BodySubscribers.ofString(UTF_8);
173             bp2.subscribe(new TestSubscriber(bs2));
174             var b2 = bs2.getBody().toCompletableFuture().join().getBytes();
175 
176             assertEquals(b1, b2);
177         } else {
178             assertFalse(r2.bodyPublisher().isPresent());
179         }
180     }

182     void assertAllOtherElementsEqual(HttpRequest r1, HttpRequest r2, String... except) {
183         var ignoreList = Arrays.asList(except);
184         REQUEST_ASSERTIONS.stream()
185                 .filter(a -> !ignoreList.contains(a.name()))
186                 .forEach(testCaseAssertion -> testCaseAssertion.test().accept(r1, r2));
187     }

189     void testBodyPublisher(String methodName, HttpRequest request) {
190         // method w/body
191         var r = HttpRequest.newBuilder(request, (n, v) -> true)
192                 .method(methodName, HttpRequest.BodyPublishers.ofString("testData"))
193                 .build();
194         assertEquals(r.method(), methodName);
195         assertTrue(r.bodyPublisher().isPresent());
196         assertEquals(r.bodyPublisher().get().contentLength(), 8);
197         assertAllOtherElementsEqual(r, request, "method");
198 
199         // method w/o body
200         var noBodyPublisher = HttpRequest.BodyPublishers.noBody();
201         var r1 = HttpRequest.newBuilder(request, (n, v) -> true)
202                 .method(methodName, noBodyPublisher)
203                 .build();
204         assertEquals(r1.method(), methodName);
205         assertTrue(r1.bodyPublisher().isPresent());
206         assertEquals(r1.bodyPublisher().get(), noBodyPublisher);
207         assertAllOtherElementsEqual(r1, request, "method");
208     }

315     @Test(dataProvider = "testRequests")
316     public void testRemoveHeader(HttpRequest request) {
317         if(!request.headers().map().isEmpty()) {
318             assertTrue(request.headers().map().containsKey("testName1"));
319         }
320         BiPredicate<String, String> filter = (n, v) -> !n.equalsIgnoreCase("testName1");
321 
322         var r = HttpRequest.newBuilder(request, filter).build();
323         assertFalse(r.headers().map().containsKey("testName1"));
324         assertEquals(r.headers().map(), HttpHeaders.of(request.headers().map(), filter).map());
325     }

340     @Test(dataProvider = "testRequests")
341     public void testRemoveMultipleHeaders(HttpRequest request) {
342         BiPredicate<String, String> isTestName1Value1 = (n ,v) ->
343                 n.equalsIgnoreCase("testName1") && v.equals("testValue1");
344         BiPredicate<String, String> isTestName2Value2 = (n ,v) ->
345                 n.equalsIgnoreCase("testName2") && v.equals("testValue2");
346         var filter = (isTestName1Value1.or(isTestName2Value2)).negate();
347 
348         var r = HttpRequest.newBuilder(request, filter).build();
349         assertEquals(r.headers().map(), HttpHeaders.of(request.headers().map(), filter).map());
350 
351         BiPredicate<String, String> filter1 = (n, v) ->
352                 !(n.equalsIgnoreCase("testName1") && (v.equals("testValue1") || v.equals("testValue2")));
353 
354         var r1 = HttpRequest.newBuilder(request, filter1).build();
355         assertEquals(r1.headers().map(), HttpHeaders.of(request.headers().map(), filter1).map());
356     }
