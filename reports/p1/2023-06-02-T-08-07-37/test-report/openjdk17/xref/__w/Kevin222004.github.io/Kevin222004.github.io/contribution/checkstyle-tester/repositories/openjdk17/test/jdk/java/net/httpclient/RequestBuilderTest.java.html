103     @Test
104     public void testURI() {
105         assertThrows(ISE, () -> newBuilder().build());
106         List<URI> uris = List.of(
107                 URI.create("ws://foo.com"),
108                 URI.create("wss://foo.com"),
109                 URI.create("ftp://foo.com"),

112                 URI.create("scheme:example.com"),
113                 URI.create("scheme:example.com/path"),
114                 URI.create("path"),
115                 URI.create("/path")
116         );
117         for (URI u : uris) {
118             assertThrows(IAE, () -> newBuilder(u));
119             assertThrows(IAE, () -> newBuilder().uri(u));

121 
122         assertEquals(newBuilder(uri).build().uri(), uri);

124         URI https = URI.create("https://foo.com");
125         assertEquals(newBuilder(https).build().uri(), https);
126         assertEquals(newBuilder().uri(https).build().uri(), https);

129     @Test
130     public void testMethod() {
131         HttpRequest request = newBuilder(uri).build();
132         assertEquals(request.method(), "GET");
133         assertTrue(!request.bodyPublisher().isPresent());
134 
135         request = newBuilder(uri).GET().build();
136         assertEquals(request.method(), "GET");
137         assertTrue(!request.bodyPublisher().isPresent());
138 
139         request = newBuilder(uri).POST(BodyPublishers.ofString("")).GET().build();
140         assertEquals(request.method(), "GET");
141         assertTrue(!request.bodyPublisher().isPresent());
142 
143         request = newBuilder(uri).PUT(BodyPublishers.ofString("")).GET().build();
144         assertEquals(request.method(), "GET");
145         assertTrue(!request.bodyPublisher().isPresent());
146 
147         request = newBuilder(uri).DELETE().GET().build();
148         assertEquals(request.method(), "GET");
149         assertTrue(!request.bodyPublisher().isPresent());
150 
151         request = newBuilder(uri).POST(BodyPublishers.ofString("")).build();
152         assertEquals(request.method(), "POST");
153         assertTrue(request.bodyPublisher().isPresent());
154 
155         request = newBuilder(uri).PUT(BodyPublishers.ofString("")).build();
156         assertEquals(request.method(), "PUT");
157         assertTrue(request.bodyPublisher().isPresent());
158 
159         request = newBuilder(uri).DELETE().build();
160         assertEquals(request.method(), "DELETE");
161         assertTrue(!request.bodyPublisher().isPresent());
162 
163         request = newBuilder(uri).GET().POST(BodyPublishers.ofString("")).build();
164         assertEquals(request.method(), "POST");
165         assertTrue(request.bodyPublisher().isPresent());
166 
167         request = newBuilder(uri).GET().PUT(BodyPublishers.ofString("")).build();
168         assertEquals(request.method(), "PUT");
169         assertTrue(request.bodyPublisher().isPresent());
170 
171         request = newBuilder(uri).GET().DELETE().build();
172         assertEquals(request.method(), "DELETE");
173         assertTrue(!request.bodyPublisher().isPresent());
174 
175         // CONNECT is disallowed in the implementation, since it is used for
176         // tunneling, and is handled separately for security checks.
177         assertThrows(IAE, () -> newBuilder(uri).method("CONNECT", BodyPublishers.noBody()).build());
178 
179         request = newBuilder(uri).method("GET", BodyPublishers.noBody()).build();
180         assertEquals(request.method(), "GET");
181         assertTrue(request.bodyPublisher().isPresent());
182 
183         request = newBuilder(uri).method("POST", BodyPublishers.ofString("")).build();
184         assertEquals(request.method(), "POST");
185         assertTrue(request.bodyPublisher().isPresent());
186     }

188     @Test
189     public void testHeaders() {
190         HttpRequest.Builder builder = newBuilder(uri);
191 
192         String[] empty = new String[0];
193         assertThrows(IAE, () -> builder.headers(empty).build());
194         assertThrows(IAE, () -> builder.headers("1").build());
195         assertThrows(IAE, () -> builder.headers("1", "2", "3").build());
196         assertThrows(IAE, () -> builder.headers("1", "2", "3", "4", "5").build());
197         assertEquals(builder.build().headers().map().size(),0);
198 
199         List<HttpRequest> requests = List.of(
200                 // same header built from different combinations of the API
201                 newBuilder(uri).header("A", "B").build(),
202                 newBuilder(uri).headers("A", "B").build(),
203                 newBuilder(uri).setHeader("A", "B").build(),
204                 newBuilder(uri).header("A", "F").setHeader("A", "B").build(),
205                 newBuilder(uri).headers("A", "F").setHeader("A", "B").build()
206         );
207 
208         for (HttpRequest r : requests) {
209             assertEquals(r.headers().map().size(), 1);
210             assertTrue(r.headers().firstValue("A").isPresent());
211             assertTrue(r.headers().firstValue("a").isPresent());
212             assertEquals(r.headers().firstValue("A").get(), "B");
213             assertEquals(r.headers().allValues("A"), List.of("B"));
214             assertEquals(r.headers().allValues("C").size(), 0);
215             assertEquals(r.headers().map().get("A"), List.of("B"));
216             assertThrows(NFE, () -> r.headers().firstValueAsLong("A"));
217             assertFalse(r.headers().firstValue("C").isPresent());
218             // a non-exhaustive list of mutators
219             assertThrows(UOE, () -> r.headers().map().put("Z", List.of("Z")));
220             assertThrows(UOE, () -> r.headers().map().remove("A"));
221             assertThrows(UOE, () -> r.headers().map().remove("A", "B"));
222             assertThrows(UOE, () -> r.headers().map().clear());
223             assertThrows(UOE, () -> r.headers().allValues("A").remove("B"));
224             assertThrows(UOE, () -> r.headers().allValues("A").remove(1));
225             assertThrows(UOE, () -> r.headers().allValues("A").clear());
226             assertThrows(UOE, () -> r.headers().allValues("A").add("Z"));
227             assertThrows(UOE, () -> r.headers().allValues("A").addAll(List.of("Z")));
228             assertThrows(UOE, () -> r.headers().allValues("A").add(1, "Z"));
229         }
230 
231         requests = List.of(
232                 // same headers built from different combinations of the API
233                 newBuilder(uri).header("A", "B")
234                                .header("C", "D").build(),
235                 newBuilder(uri).header("A", "B")
236                                .headers("C", "D").build(),
237                 newBuilder(uri).header("A", "B")
238                                .setHeader("C", "D").build(),
239                 newBuilder(uri).headers("A", "B")
240                                .headers("C", "D").build(),
241                 newBuilder(uri).headers("A", "B")
242                                .header("C", "D").build(),
243                 newBuilder(uri).headers("A", "B")
244                                .setHeader("C", "D").build(),
245                 newBuilder(uri).setHeader("A", "B")
246                                .setHeader("C", "D").build(),
247                 newBuilder(uri).setHeader("A", "B")
248                                .header("C", "D").build(),
249                 newBuilder(uri).setHeader("A", "B")
250                                .headers("C", "D").build(),
251                 newBuilder(uri).headers("A", "B", "C", "D").build()
252         );
253 
254         for (HttpRequest r : requests) {
255             assertEquals(r.headers().map().size(), 2);
256             assertTrue(r.headers().firstValue("A").isPresent());
257             assertEquals(r.headers().firstValue("A").get(), "B");
258             assertEquals(r.headers().allValues("A"), List.of("B"));
259             assertTrue(r.headers().firstValue("C").isPresent());
260             assertEquals(r.headers().firstValue("C").get(), "D");
261             assertEquals(r.headers().allValues("C"), List.of("D"));
262             assertEquals(r.headers().map().get("C"), List.of("D"));
263             assertThrows(NFE, () -> r.headers().firstValueAsLong("C"));
264             assertFalse(r.headers().firstValue("E").isPresent());
265             // a smaller non-exhaustive list of mutators
266             assertThrows(UOE, () -> r.headers().map().put("Z", List.of("Z")));
267             assertThrows(UOE, () -> r.headers().map().remove("C"));
268             assertThrows(UOE, () -> r.headers().allValues("A").remove("B"));
269             assertThrows(UOE, () -> r.headers().allValues("A").clear());
270             assertThrows(UOE, () -> r.headers().allValues("C").add("Z"));
271         }
272 
273         requests = List.of(
274                 // same multi-value headers built from different combinations of the API
275                 newBuilder(uri).header("A", "B")
276                                .header("A", "C").build(),
277                 newBuilder(uri).header("A", "B")
278                                .headers("A", "C").build(),
279                 newBuilder(uri).headers("A", "B")
280                                .headers("A", "C").build(),
281                 newBuilder(uri).headers("A", "B")
282                                .header("A", "C").build(),
283                 newBuilder(uri).setHeader("A", "B")
284                                .header("A", "C").build(),
285                 newBuilder(uri).setHeader("A", "B")
286                                .headers("A", "C").build(),
287                 newBuilder(uri).header("A", "D")
288                                .setHeader("A", "B")
289                                .headers("A", "C").build(),
290                 newBuilder(uri).headers("A", "B", "A", "C").build()
291         );
292 
293         for (HttpRequest r : requests) {
294             assertEquals(r.headers().map().size(), 1);
295             assertTrue(r.headers().firstValue("A").isPresent());
296             assertTrue(r.headers().allValues("A").containsAll(List.of("B", "C")));
297             assertEquals(r.headers().allValues("C").size(), 0);
298             assertEquals(r.headers().map().get("A"), List.of("B", "C"));
299             assertThrows(NFE, () -> r.headers().firstValueAsLong("A"));
300             assertFalse(r.headers().firstValue("C").isPresent());
301             // a non-exhaustive list of mutators
302             assertThrows(UOE, () -> r.headers().map().put("Z", List.of("Z")));
303             assertThrows(UOE, () -> r.headers().map().remove("A"));
304             assertThrows(UOE, () -> r.headers().map().remove("A", "B"));
305             assertThrows(UOE, () -> r.headers().map().clear());
306             assertThrows(UOE, () -> r.headers().allValues("A").remove("B"));
307             assertThrows(UOE, () -> r.headers().allValues("A").remove(1));
308             assertThrows(UOE, () -> r.headers().allValues("A").clear());
309             assertThrows(UOE, () -> r.headers().allValues("A").add("Z"));
310             assertThrows(UOE, () -> r.headers().allValues("A").addAll(List.of("Z")));
311             assertThrows(UOE, () -> r.headers().allValues("A").add(1, "Z"));
312         }
313 
314         // case-insensitivity
315         requests = List.of(
316                 newBuilder(uri)
317                         .header("Accept-Encoding", "gzip, deflate").build(),
318                 newBuilder(uri)
319                         .header("accept-encoding", "gzip, deflate").build(),
320                 newBuilder(uri)
321                         .header("AccePt-EncodINg", "gzip, deflate").build(),
322                 newBuilder(uri)
323                         .header("AcCEpt-EncoDIng", "gzip, deflate").build()
324         );
325         for (HttpRequest r : requests) {
326             for (String name : List.of("Accept-Encoding", "accept-encoding",
327                                        "aCCept-EnCODing", "accepT-encodinG")) {
328                 assertTrue(r.headers().firstValue(name).isPresent());
329                 assertTrue(r.headers().allValues(name).contains("gzip, deflate"));
330                 assertEquals(r.headers().firstValue(name).get(), "gzip, deflate");
331                 assertEquals(r.headers().allValues(name).size(), 1);
332                 assertEquals(r.headers().map().size(), 1);
333                 assertEquals(r.headers().map().get(name).size(), 1);
334                 assertEquals(r.headers().map().get(name).get(0), "gzip, deflate");
335             }
336         }

360     interface WithHeader {
361         HttpRequest.Builder withHeader(HttpRequest.Builder builder, String name, String value);
362     }
363 
364     @Test
365     public void testRestricted()  throws URISyntaxException {
366         URI uri = new URI("http://localhost:80/test/");
367         Map<String, WithHeader> lambdas = Map.of(
368                 "Builder::header",    HttpRequest.Builder::header,
369                 "Builder::headers",   (b, n, v) -> b.headers(n,v),
370                 "Builder::setHeader", HttpRequest.Builder::setHeader
371                 );
372         for (Map.Entry<String, WithHeader> e : lambdas.entrySet()) {
373             System.out.println("Testing restricted headers with " + e.getKey());
374             WithHeader f = e.getValue();
375             for (String name : RESTRICTED) {
376                 String value = name + "-value";
377                 HttpRequest req = f.withHeader(HttpRequest.newBuilder(uri)
378                         .GET(), "x-" + name, value).build();
379                 String v = req.headers().firstValue("x-" + name).orElseThrow(
380                         () -> new RuntimeException("header x-" + name + " not set"));
381                 assertEquals(v, value);
382                 try {
383                     f.withHeader(HttpRequest.newBuilder(uri)
384                             .GET(), name, value).build();
385                     throw new RuntimeException("Expected IAE not thrown for " + name);
386                 } catch (IllegalArgumentException x) {
387                     System.out.println("Got expected IAE for " + name + ": " + x);
388                 }
389             }
390         }
391     }

394     @Test
395     public void testCopy() {
396         HttpRequest.Builder builder = newBuilder(uri).expectContinue(true)
397                                                      .header("A", "B")
398                                                      .POST(BodyPublishers.ofString(""))
399                                                      .timeout(ofSeconds(30))
400                                                      .version(HTTP_1_1);
401         HttpRequest.Builder copy = builder.copy();
402         assertTrue(builder != copy);
403 
404         // modify the original builder before building from the copy
405         builder.GET().timeout(ofSeconds(5)).version(HTTP_2).setHeader("A", "C");
406 
407         HttpRequest copyRequest = copy.build();
408         assertEquals(copyRequest.uri(), uri);
409         assertEquals(copyRequest.expectContinue(), true);
410         assertEquals(copyRequest.headers().map().get("A"), List.of("B"));
411         assertEquals(copyRequest.method(), "POST");
412         assertEquals(copyRequest.bodyPublisher().isPresent(), true);
413         assertEquals(copyRequest.timeout().get(), ofSeconds(30));
414         assertTrue(copyRequest.version().isPresent());
415         assertEquals(copyRequest.version().get(), HTTP_1_1);
416 
417         // lazy set URI ( maybe builder as a template )
418         copyRequest = newBuilder().copy().uri(uri).build();
419         assertEquals(copyRequest.uri(), uri);
420 
421         builder = newBuilder().header("C", "D");
422         copy = builder.copy();
423         copy.uri(uri);
424         copyRequest = copy.build();
425         assertEquals(copyRequest.uri(), uri);
426         assertEquals(copyRequest.headers().firstValue("C").get(), "D");
427     }

450     @Test
451     public void testEquals() {
452         assertNotEquals(newBuilder(URI.create("http://foo.com")),
453                         newBuilder(URI.create("http://bar.com")));
454 
455         HttpRequest.Builder builder = newBuilder(uri);
456         assertEquals(builder.build(), builder.build());
457         assertEquals(builder.build(), newBuilder(uri).build());
458 
459         builder.POST(BodyPublishers.noBody());
460         assertEquals(builder.build(), builder.build());
461         assertEquals(builder.build(), newBuilder(uri).POST(BodyPublishers.noBody()).build());
462         assertEquals(builder.build(), newBuilder(uri).POST(BodyPublishers.ofString("")).build());
463         assertNotEquals(builder.build(), newBuilder(uri).build());
464         assertNotEquals(builder.build(), newBuilder(uri).GET().build());
465         assertNotEquals(builder.build(), newBuilder(uri).PUT(BodyPublishers.noBody()).build());
466 
467         builder = newBuilder(uri).header("x", "y");
468         assertEquals(builder.build(), builder.build());
469         assertEquals(builder.build(), newBuilder(uri).header("x", "y").build());
470         assertNotEquals(builder.build(), newBuilder(uri).header("x", "Z").build());
471         assertNotEquals(builder.build(), newBuilder(uri).header("z", "y").build());
