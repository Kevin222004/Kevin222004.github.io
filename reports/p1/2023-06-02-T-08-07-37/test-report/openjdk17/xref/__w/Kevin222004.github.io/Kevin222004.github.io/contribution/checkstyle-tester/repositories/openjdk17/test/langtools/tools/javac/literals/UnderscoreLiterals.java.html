210     void test(int value, int expect) {
211         count++;
212         if (value != expect)
213             error("test " + count + "\nexpected: 0x" + Integer.toHexString(expect) + "\n   found: 0x" + Integer.toHexString(value));
214     }
215 
216     void test(double value, double expect) {
217         count++;
218         if (value != expect)
219             error("test " + count + "\nexpected: 0x" + expect + "\n   found: 0x" + value);
220     }
221 
222     void test(long value, long expect) {
223         count++;
224         if (value != expect)
225             error("test " + count + "\nexpected: 0x" + Long.toHexString(expect) + "\n   found: 0x" + Long.toHexString(value));
226     }
227 
228     void error(String message) {
229         System.out.println(message);
230         errors++;
231     }
