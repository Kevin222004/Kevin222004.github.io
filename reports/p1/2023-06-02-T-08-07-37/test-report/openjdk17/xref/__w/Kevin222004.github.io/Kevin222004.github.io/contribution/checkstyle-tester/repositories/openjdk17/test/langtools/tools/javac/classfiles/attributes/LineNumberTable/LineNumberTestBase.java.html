39  /**
40   * Base class for line number table attribute tests.
41   * To add new tests cases(e.g. for new language constructions) you should modify TestData in LineNumberTest.
42   * If you plan to add new tests you should extends LineNumberTestBase and invoke one of two "test(...)" methods.
43   *
44   * @see #test(Container) test methods for more info.
45   */
46  public class LineNumberTestBase extends TestBase {
47      /**
48       * Generates test cases and passes to {@link #test(java.util.List)}
49       * Generation: Replaces placeholder in template by value of enum {@link Constructions}.
50       */
51      protected void test(Container container) throws Exception {
52          test(container.generate(Constructions.values()));
53      }
54  
55      /**
56       * Takes list of test cases. Compiles source of test case.
57       * Checks what expected lines are covered by line number table.
58       * Does general check of line number table for consistency.
59       *
60       * @param testCases list of test cases.
61       */
62      protected void test(List<TestCase> testCases) throws Exception {
63          boolean failed = false;
64          for (TestCase testCase : testCases) {
65              try {
66                  writeToFileIfEnabled(Paths.get(testCase.getName() + ".java"), testCase.src);
67                  Set<Integer> coveredLines = new HashSet<>();
68                  for (JavaFileObject file : compile(testCase.extraCompilerOptions, testCase.src).getClasses().values()) {
69                      ClassFile classFile = ClassFile.read(file.openInputStream());
70                      for (Method m : classFile.methods) {
71                          Code_attribute code_attribute = (Code_attribute) m.attributes.get(Code);
72  
73                          assertEquals(
74                                  countAttributes(LineNumberTable, code_attribute.attributes.attrs, classFile.constant_pool),
75                                  1,
76                                  "Can be more than one LNT attribute, but javac should generate only one.");
77  
78                          LineNumberTable_attribute tableAttribute =
79                                  (LineNumberTable_attribute) code_attribute.attributes.get(LineNumberTable);
80                          checkAttribute(testCase, tableAttribute, code_attribute.code_length);
81                          coveredLines.addAll(
82                                  Stream.of(tableAttribute.line_number_table)
83                                          .map(e -> e.line_number)
84                                          .collect(toList()));
85                      }
86                  }
87                  if (testCase.exactLines) {
88                      assertTrue(coveredLines.equals(testCase.expectedLines),
89                              format("Incorrect covered lines.%n" +
90                                      "Covered: %s%n" +
91                                      "Expected: %s%n", coveredLines, testCase.expectedLines));
92                  } else {
93                      assertTrue(coveredLines.containsAll(testCase.expectedLines),
94                              format("All significant lines are not covered.%n" +
95                                      "Covered: %s%n" +
96                                      "Expected: %s%n", coveredLines, testCase.expectedLines));
97                  }
98              } catch (AssertionFailedException | CompilationException ex) {
99                  System.err.printf("#       %-20s#%n", testCase.getName());
100                 int l = 0;
101                 for (String line : testCase.src.split("\n")) {
102                     System.err.println(++l + line);
103                 }
104                 System.err.println(ex);
105                 failed = true;
106                 continue;
107             }
108             System.err.printf("#       %-20s#%n", testCase.getName());
109             System.err.println("Passed");
110         }
111         if (failed) {
112             throw new RuntimeException("Test failed");
113         }
114     }
115 
116     private int countAttributes(String name, Attribute[] attrs, ConstantPool constant_pool) throws ConstantPoolException {
117         int i = 0;
118         for (Attribute attribute : attrs) {
119             if (name.equals(attribute.getName(constant_pool))) {
120                 i++;
121             }
122         }
123         return i;
124     }
125 
126     private void checkAttribute(TestCase testCase, LineNumberTable_attribute tableAttribute, int code_length) {
127         assertEquals(tableAttribute.line_number_table_length, tableAttribute.line_number_table.length,
128                 "Incorrect line number table length.");
129         //attribute length is offset(line_number_table_length) + element_size*element_count
130         assertEquals(tableAttribute.attribute_length, 2 + 4 * tableAttribute.line_number_table_length,
131                 "Incorrect attribute length");
132         testNonEmptyLine(testCase.src.split("\n"), tableAttribute);
133         assertEquals(
134                 Stream.of(tableAttribute.line_number_table)
135                         .filter(e -> e.start_pc >= code_length)
136                         .count()
137                 , 0L, "StartPC is out of bounds.");
138     }
139 
140     /**
141      * Expects line number table point to non empty lines.
142      * The method can't recognize commented lines as empty(insensible) in case of multiline comment.
143      */
144     private void testNonEmptyLine(String[] source, LineNumberTable_attribute attribute) {
145         for (LineNumberTable_attribute.Entry e : attribute.line_number_table) {
146             String line = source[e.line_number - 1].trim();
147             assertTrue(!("".equals(line) || line.startsWith("//") || line.startsWith("/*")),
148                     format("Expect that line #%d is not empty.%n", e.line_number));
149         }
150     }

152     protected static enum Constructions implements Container.Construction {
153         STORE("testField = 10;"),
154         LOAD("int p;\n" +
155                 "p = testField;", 2),
156         ASSERT("assert false: \"Assert error\";"),
157         ARRAY("double arr[] = new double[10];"),
158         ARRAY2("int arr2[][] = {{1,2},{}};"),
159         LAMBDA("Runnable runnable = () -> \n" +
160                 "   System.out.println();"),
161         LAMBDA_BODY("Runnable runnable = () -> {\n" +
162                 "   testField++;\n" +
163                 "};"),
164         METHOD_REFERENCE("Runnable run = System.out::println;\nrun.run();"),
165         INVOKE_STATIC_METHOD("System.out.println(\"\");"),
166         INVOKE_INTERFACE("Runnable runnable = new Runnable() {\n" +
167                 "    @Override\n" +
168                 "    public void run() {\n" +
169                 "        System.out.println(\"runnable\");\n" +
170                 "    }\n" +
171                 "};\n" +
172                 "runnable.run();", 1, 7),
173         INVOKE_VIRTUAL_METHOD("testMethod();"),
174         INVOKE_CONSTRUCTOR("new Integer(2);"),
175         INVOKE_LAMBDA(LAMBDA.getSource() + "\n" +
176                 "runnable.run();"),
177         DO_WHILE("do{\n" +
178                 "    testField++;\n" +
179                 "}while(testField == 1);", 2, 3),
180         WHILE("while(testField == 1);"),
181         FOR("for(int i = 0; i < 3 ; i++);"),
182         FOR_ENHANCEMENT("int[] ints = {1,2,3};\n" +
183                 "for(int i:  ints);"),
184         LABEL("int i=0;\n" +
185                 "label:{\n" +
186                 "    label2:\n" +
187                 "    for(;i<5;i++){\n" +
188                 "        if(i==3)\n" +
189                 "            break label;\n" +
190                 "        if(i==0){\n" +
191                 "            continue label2;\n" +
192                 "        }\n" +
193                 "        return;\n" +
194                 "    }\n" +
195                 "    i++;\n" +
196                 "}\n"
197                 , 1, 4, 5, 6, 7, 8, 10, 12),
198         CONDITION("int res = \n" +
199                 "testField == 2 ?\n" +
200                 "10\n" +
201                 ":9;", 2, 3, 4),
202         TRY("try{\n" +
203                 "    --testField;\n" +
204                 "}\n" +
205                 "catch(Exception e){\n" +
206                 "    --testField;\n" +
207                 "}\n" +
208                 "catch(Error e){\n" +
209                 "    System.out.print(e);\n" +
210                 "    throw e;\n " +
211                 "}\n" +
212                 "finally{\n" +
213                 "    ++testField;\n" +
214                 "}", 2, 4, 5, 7, 8, 9, 12),
215         TRY_WITH_RESOURCES("try (\n" +
216                 "    Writer writer = new StringWriter();\n" +
217                 "    Reader reader = new StringReader(\"\")) {\n" +
218                 "        writer.write(1);\n" +
219                 "        reader.read();\n" +
220                 "} catch (IOException e) {}\n"
221                 , 2, 3, 4, 5),
222         SYNCHRONIZE("" +
223                 "synchronized(this){\n" +
224                 "    testField++;\n" +
225                 "}"),
226         SWITCH("switch (testField){\n" +
227                 "case 1:\n" +
228                 "    break;\n" +
229                 "case 2:\n" +
230                 "    testField++;\n" +
231                 "default: \n" +
232                 "    testField+=2; \n" +
233                 "}", 1, 3, 5, 7),
234         SWITCH_STRING(
235                 "String str = String.valueOf(testField);\n" +
236                         "switch (str){\n" +
237                         "case \"1\":\n" +
238                         "    break;\n" +
239                         "case \"2\":\n" +
240                         "    testField++;\n" +
241                         "default: \n" +
242                         "    testField+=2; \n" +
243                         "}", 1, 2, 4, 6, 8);
244 
245         private final String source;
246         private int[] expectedLines;
247 
248         Constructions(String source) {
249             this.source = source;
250             expectedLines = IntStream.rangeClosed(1, source.split("\n").length).toArray();
251         }
252 
253         Constructions(String source, int... expectedLines) {
254             this.source = source;
255             this.expectedLines = expectedLines;
256         }
257 
258         @Override
259         public String getSource() {
260             return source;
261         }
262 
263         @Override
264         public int[] getExpectedLines() {
265             return expectedLines;
266         }
267     }
