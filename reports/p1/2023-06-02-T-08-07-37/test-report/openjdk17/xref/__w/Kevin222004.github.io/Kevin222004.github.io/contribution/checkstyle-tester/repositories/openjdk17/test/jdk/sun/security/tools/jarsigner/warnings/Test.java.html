167     protected void createAlias(String alias, String ... options)
168             throws Throwable {
169         List<String> cmd = new ArrayList<>();
170         cmd.addAll(List.of(
171                 "-genkeypair",
172                 "-alias", alias,
173                 "-keyalg", KEY_ALG,
174                 "-keysize", Integer.toString(KEY_SIZE),
175                 "-keystore", KEYSTORE,
176                 "-storepass", PASSWORD,
177                 "-keypass", PASSWORD,
178                 "-dname", "CN=" + alias));
179         cmd.addAll(Arrays.asList(options));
180 
181         keytool(cmd.toArray(new String[cmd.size()]))
182                 .shouldHaveExitValue(0);
183     }
184 
185     protected void issueCert(String alias, String ... options)
186             throws Throwable {
187         keytool("-certreq",
188                 "-alias", alias,
189                 "-keystore", KEYSTORE,
190                 "-storepass", PASSWORD,
191                 "-keypass", PASSWORD,
192                 "-file", alias + ".req")
193                     .shouldHaveExitValue(0);
194 
195         List<String> cmd = new ArrayList<>();
196         cmd.addAll(List.of(
197                 "-gencert",
198                 "-alias", CA_KEY_ALIAS,
199                 "-infile", alias + ".req",
200                 "-outfile", alias + ".cert",
201                 "-keystore", KEYSTORE,
202                 "-storepass", PASSWORD,
203                 "-keypass", PASSWORD,
204                 "-file", alias + ".req"));
205         cmd.addAll(Arrays.asList(options));
206 
207         keytool(cmd.toArray(new String[cmd.size()]))
208                 .shouldHaveExitValue(0);
209 
210         keytool("-importcert",
211                 "-alias", alias,
212                 "-keystore", KEYSTORE,
213                 "-storepass", PASSWORD,
214                 "-keypass", PASSWORD,
215                 "-file", alias + ".cert")
216                     .shouldHaveExitValue(0);
217     }

219     protected void checkVerifying(OutputAnalyzer analyzer, int expectedExitCode,
220             String... warnings) {
221         analyzer.shouldHaveExitValue(expectedExitCode);
222         int count = 0;
223         for (String warning : warnings) {
224             if (warning.startsWith("!")) {
225                 analyzer.shouldNotContain(warning.substring(1));
226             } else {
227                 count++;
228                 analyzer.shouldContain(warning);
229             }
230         }
231         if (count > 0) {
232             analyzer.shouldMatch(WARNING_OR_ERROR);
233         }
234         if (expectedExitCode == 0) {
235             analyzer.shouldContain(JAR_VERIFIED);
236         } else {
237             analyzer.shouldContain(JAR_VERIFIED_WITH_SIGNER_ERRORS);
238         }
239     }
240 
241     protected void checkSigning(OutputAnalyzer analyzer, String... warnings) {
242         analyzer.shouldHaveExitValue(0);
243         int count = 0;
244         for (String warning : warnings) {
245             if (warning.startsWith("!")) {
246                 analyzer.shouldNotContain(warning.substring(1));
247             } else {
248                 count++;
249                 analyzer.shouldContain(warning);
250             }
251         }
252         if (count > 0) {
253             analyzer.shouldMatch(WARNING_OR_ERROR);
254         }
255         analyzer.shouldContain(JAR_SIGNED);
256     }

258     protected OutputAnalyzer keytool(String... cmd) throws Throwable {
259         return tool(KEYTOOL, cmd);
260     }
261 
262     protected OutputAnalyzer jarsigner(String... cmd) throws Throwable {
263         return tool(JARSIGNER, cmd);
264     }

266     private OutputAnalyzer tool(String tool, String... args) throws Throwable {
267         List<String> cmd = new ArrayList<>();
268         cmd.add(tool);
269         cmd.add("-J-Duser.language=en");
270         cmd.add("-J-Duser.country=US");
271         cmd.addAll(Arrays.asList(args).stream().filter(arg -> {
272             return arg != null && !arg.isEmpty();
273         }).collect(Collectors.toList()));
274         return ProcessTools.executeCommand(cmd.toArray(new String[cmd.size()]));
275     }
