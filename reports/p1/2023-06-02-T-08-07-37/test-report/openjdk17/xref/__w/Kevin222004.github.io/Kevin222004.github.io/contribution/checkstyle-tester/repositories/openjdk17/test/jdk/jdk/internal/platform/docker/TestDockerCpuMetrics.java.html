24  import java.util.stream.Collectors;
25  import java.util.stream.IntStream;
26  import jdk.internal.platform.Metrics;
27  import jdk.test.lib.Utils;
28  import jdk.test.lib.containers.cgroup.CPUSetsReader;
29  import jdk.test.lib.containers.docker.Common;
30  import jdk.test.lib.containers.docker.DockerRunOptions;
31  import jdk.test.lib.containers.docker.DockerTestUtils;

33  /*
34   * @test
35   * @key cgroups
36   * @summary Test JDK Metrics class when running inside docker container
37   * @requires docker.support
38   * @library /test/lib
39   * @modules java.base/jdk.internal.platform
40   * @build MetricsCpuTester
41   * @run main/timeout=360 TestDockerCpuMetrics
42   */
43  
44  public class TestDockerCpuMetrics {
45      private static final String imageName = Common.imageName("metrics-cpu");
46  
47      public static void main(String[] args) throws Exception {
48          if (!DockerTestUtils.canTestDocker()) {
49              return;
50          }
51  
52          // These tests create a docker image and run this image with
53          // varying docker cpu options.  The arguments passed to the docker
54          // container include the Java test class to be run along with the
55          // resource to be examined and expected result.
56  
57          DockerTestUtils.buildJdkDockerImage(imageName, "Dockerfile-BasicTest", "jdk-docker");
58  
59          try {
60              int numCpus = CPUSetsReader.getNumCpus();
61              testCpuSet("0");
62              testCpuSet("0-" + (numCpus - 1));
63              if (numCpus > 2) {
64                  testCpuSet("0-" + ((numCpus - 1) / 2));
65                  testCpuSet((((numCpus - 1) / 2) + 1) + "-" + (numCpus - 1));
66              }
67              testCpuSet(IntStream.range(0, numCpus).mapToObj(a -> Integer.toString(a)).collect(Collectors.joining(",")));
68  
69              testCpuQuota(50 * 1000, 100 * 1000);
70              testCpuQuota(100 * 1000, 100 * 1000);
71              testCpuQuota(150 * 1000, 100 * 1000);
72              testCpuQuota(400 * 1000, 100 * 1000);
73  
74              testCpuShares(256);
75              testCpuShares(2048);
76              testCpuShares(4096);
77  
78              testCpuThrottling(0.5);// --cpus=<value>
79  
80              int[] cpuSetMems = Metrics.systemMetrics().getCpuSetMems();
81              String memNodes = null;
82              if (cpuSetMems != null && cpuSetMems.length > 1) {
83                  int endNode = (cpuSetMems[cpuSetMems.length - 1] - cpuSetMems[0]) / 2 + cpuSetMems[0];
84                  memNodes = cpuSetMems[0] + "-" + endNode;
85              } else if (cpuSetMems != null && cpuSetMems.length == 1) {
86                  memNodes = cpuSetMems[0] + "";
87              }
88  
89              if(memNodes != null)
90                  testCpuSetMems(memNodes);
91  
92              testComboOptions("0-" + (numCpus - 1), 200 * 1000, 100 * 1000, 4 * 1024);
93              testComboOptions("0", 200 * 1000, 100 * 1000, 1023);
94          } finally {
95              DockerTestUtils.removeDockerImage(imageName);
96          }
97      }
98  
99      private static void testCpuSetMems(String value) throws Exception {
100         Common.logNewTestCase("testCpuSetMems, mem nodes = " + value);
101         DockerRunOptions opts =
102                 new DockerRunOptions(imageName, "/jdk/bin/java", "MetricsCpuTester");
103         opts.addDockerOpts("--volume", Utils.TEST_CLASSES + ":/test-classes/");
104         opts.addDockerOpts("--cpuset-mems=" + value);
105         opts.addJavaOpts("-cp", "/test-classes/").addJavaOpts("--add-exports", "java.base/jdk.internal.platform=ALL-UNNAMED");
106         opts.addClassOptions("cpumems", value);
107         DockerTestUtils.dockerRunJava(opts).shouldHaveExitValue(0).shouldContain("TEST PASSED!!!");
108     }
109 
110     private static void testCpuSet(String value) throws Exception {
111         Common.logNewTestCase("testCpuSet, value = " + value);
112         DockerRunOptions opts =
113                 new DockerRunOptions(imageName, "/jdk/bin/java", "MetricsCpuTester");
114         opts.addDockerOpts("--volume", Utils.TEST_CLASSES + ":/test-classes/");
115         opts.addJavaOpts("-cp", "/test-classes/");
116         opts.addJavaOpts("--add-exports", "java.base/jdk.internal.platform=ALL-UNNAMED");
117         opts.addClassOptions("cpusets", value);
118         opts.addDockerOpts("--cpuset-cpus=" + value);
119         DockerTestUtils.dockerRunJava(opts).shouldHaveExitValue(0).shouldContain("TEST PASSED!!!");
120     }
121 
122     private static void testCpuQuota(long quota, long period) throws Exception {
123         Common.logNewTestCase("testCpuQuota, quota = " + quota + ", period = " + period);
124         DockerRunOptions opts =
125                 new DockerRunOptions(imageName, "/jdk/bin/java", "MetricsCpuTester");
126         opts.addDockerOpts("--volume", Utils.TEST_CLASSES + ":/test-classes/");
127         opts.addDockerOpts("--cpu-period=" + period).addDockerOpts("--cpu-quota=" + quota);
128         opts.addJavaOpts("-cp", "/test-classes/").addJavaOpts("--add-exports", "java.base/jdk.internal.platform=ALL-UNNAMED");
129         opts.addClassOptions("cpuquota", quota + "", period + "");
130         DockerTestUtils.dockerRunJava(opts).shouldHaveExitValue(0).shouldContain("TEST PASSED!!!");
131     }

155     private static void testComboOptions(String cpuset, int quota, int period, int shares) throws Exception {
156         Common.logNewTestCase("testComboOptions, shares = " + shares);
157         DockerRunOptions opts =
158                 new DockerRunOptions(imageName, "/jdk/bin/java", "MetricsCpuTester");
159         opts.addDockerOpts("--volume", Utils.TEST_CLASSES + ":/test-classes/");
160         opts.addDockerOpts("--cpuset-cpus", "" + cpuset)
161                 .addDockerOpts("--cpu-period=" + period)
162                 .addDockerOpts("--cpu-quota=" + quota)
163                 .addDockerOpts("--cpu-shares=" + shares);
164         opts.addJavaOpts("-cp", "/test-classes/").addJavaOpts("--add-exports", "java.base/jdk.internal.platform=ALL-UNNAMED");
165         opts.addClassOptions("combo", cpuset, quota + "", period + "", shares + "");
166         DockerTestUtils.dockerRunJava(opts).shouldHaveExitValue(0).shouldContain("TEST PASSED!!!");
167     }
