110     public static void main(String[] args) throws Exception
111     {
112         try {
113             SourceDebugExtension loader = new SourceDebugExtension();
114             /* The test program creates a class file from the header
115              * stored above and adding the content of a SourceDebugExtension
116              * attribute made of the character 0x02 repeated 68000 times.
117              * This attribute doesn't follow the syntax specified in JSR 45
118              * but it's fine because this test just checks that the JVM is
119              * able to load a class file with a SourceDebugExtension
120              * attribute bigger than 64KB. The JVM doesn't try to
121              * parse the content of the attribute, this work is performed
122              * by the SA or external tools.
123              */
124             byte[] buf = new byte[header.length + attrSize];
125             for(int i=0; i<header.length; i++) {
126                 buf[i] = header[i];
127             }
128             for(int i=0; i<attrSize; i++) {
129                 buf[header.length+i] = (byte)0x02;
130             }
131             Class c = loader.defineClass("TestProg", buf, 0, buf.length);
132             System.out.println("Test PASSES");
133         } catch(Exception e) {
134             System.out.println("Test FAILS");
135         }
136     }
