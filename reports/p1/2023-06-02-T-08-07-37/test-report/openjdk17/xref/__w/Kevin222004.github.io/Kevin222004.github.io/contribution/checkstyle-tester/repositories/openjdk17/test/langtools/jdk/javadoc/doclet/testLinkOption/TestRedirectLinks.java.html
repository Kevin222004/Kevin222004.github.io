81      /*
82       * This test requires access to a URL that is redirected
83       * from http: to https:.
84       * For now, we use the main JDK API on docs.oracle.com.
85       * The test is skipped if access to the server is not available.
86       * (A better solution is to use a local testing web server.)
87       */
88      @Test
89      public void testRedirects() throws Exception {
90          // This test relies on access to an external resource, which may or may not be
91          // reliably available, depending on the host system configuration and other
92          // networking issues. Therefore, it is disabled by default, unless the system
93          // property "javadoc.dev" is set "true".
94          String property = "javadoc.dev";
95          if (!Boolean.getBoolean(property)) {
96              out.println("Test case disabled by default; "
97                      + "set system property \"" + property + "\" to true to enable it.");
98              return;
99          }
100 
101         // test to see if access to external URLs is available, and that the URL uses a redirect
102 
103         URL testURL = new URL("http://docs.oracle.com/en/java/javase/11/docs/api/element-list");
104         String testURLHost = testURL.getHost();
105         try {
106             InetAddress testAddr = InetAddress.getByName(testURLHost);
107             out.println("Found " + testURLHost + ": " + testAddr);
108         } catch (UnknownHostException e) {
109             out.println("Setup failed (" + testURLHost + " not found); this test skipped");
110             return;
111         }
112 
113         boolean haveRedirectURL = false;
114         Instant start = Instant.now();
115         try {
116             URLConnection conn = testURL.openConnection();
117             conn.connect();
118             out.println("Opened connection to " + testURL);
119             if (conn instanceof HttpURLConnection) {
120                 HttpURLConnection httpConn = (HttpURLConnection) conn;
121                 int status = httpConn.getResponseCode();
122                 if (status / 100 == 3) {
123                     haveRedirectURL = true;
124                 }
125                 out.println("Status: " + status);
126                 int n = 0;
127                 while (httpConn.getHeaderField(n) != null) {
128                     out.println("Header: " + httpConn.getHeaderFieldKey(n) + ": " + httpConn.getHeaderField(n));
129                     n++;
130                 }
131                 httpConn.disconnect();
132             }
133         } catch (Exception e) {
134             out.println("Exception occurred: " + e);
135             Instant now = Instant.now();
136             out.println("Attempt took " + Duration.between(start, now).toSeconds() + " seconds");
137         }
138 
139         if (!haveRedirectURL) {
140             out.println("Setup failed (no redirect URL); this test skipped");
141             return;
142         }
143 
144         String apiURL = "http://docs.oracle.com/en/java/javase/11/docs/api";
145         String outRedirect = "outRedirect";
146         javadoc("-d", outRedirect,
147                 "-sourcepath", testSrc,
148                 "-link", apiURL,
149                 "-Xdoclint:none",
150                 "pkg");
151         checkExit(Exit.OK);
152         checkOutput("pkg/B.html", true,
153                 "<a href=\"" + apiURL + """
154                     /java.base/java/lang/String.html" title="class or interface in java.lang" class=\
155                     "external-link">Link-Plain to String Class</a>""");
156         checkOutput("pkg/C.html", true,
157                 "<a href=\"" + apiURL + """
158                     /java.base/java/lang/Object.html" title="class or interface in java.lang" class="external-link">Object</a>""");
159     }

165     /**
166      * This test verifies redirection using temporary localhost web servers,
167      * such that one server redirects to the other.
168      */
169     @Test
170     public void testWithServers() throws Exception {
171         // Set up a simple library
172         Path libSrc = Path.of("libSrc");
173         tb.writeJavaFiles(libSrc.resolve("mA"),
174                 "module mA { exports p1; exports p2; }",
175                 "package p1; public class C1 { }",
176                 "package p2; public class C2 { }");
177         tb.writeJavaFiles(libSrc.resolve("mB"),
178                 "module mB { exports p3; exports p4; }",
179                 "package p3; public class C3 { }",
180                 "package p4; public class C4 { }");
181 
182         Path libModules = Path.of("libModules");
183         Files.createDirectories(libModules);
184 
185         new JavacTask(tb)
186                 .outdir(libModules)
187                 .options("--module-source-path", libSrc.toString(),
188                         "--module", "mA,mB",
189                         "-Xdoclint:none")
190                 .run()
191                 .writeAll();
192 
193         javadoc("-d", libApi.toString(),
194                 "--module-source-path", libSrc.toString(),
195                 "--module", "mA,mB",
196                 "-Xdoclint:none" );
197 
198         // start web servers
199         // use loopback address to avoid any issues if proxy is in use
200         InetAddress localHost = InetAddress.getLoopbackAddress();
201         try {
202             oldServer = HttpServer.create(new InetSocketAddress(localHost, 0), 0);
203             String oldURL = "http:/" + oldServer.getAddress();
204             oldServer.createContext("/", this::handleOldRequest);
205             out.println("Starting old server (" + oldServer.getClass().getSimpleName() + ") on " + oldURL);
206             oldServer.start();
207 
208             SSLContext sslContext = new SimpleSSLContext().get();
209             newServer = HttpsServer.create(new InetSocketAddress(localHost, 0), 0);
210             String newURL = "https:/" + newServer.getAddress();
211             newServer.setHttpsConfigurator(new HttpsConfigurator(sslContext));
212             newServer.createContext("/", this::handleNewRequest);
213             out.println("Starting new server (" + newServer.getClass().getSimpleName() + ") on " + newURL);
214             newServer.start();
215 
216             // Set up API to use that library
217             Path src = Path.of("src");
218             tb.writeJavaFiles(src.resolve("mC"),
219                     "module mC { requires mA; requires mB; exports p5; exports p6; }",
220                     "package p5; public class C5 extends p1.C1 { }",
221                     "package p6; public class C6 { public p4.C4 c4; }");
222 
223             // Set defaults for HttpsURLConfiguration for the duration of this
224             // invocation of javadoc to use our testing sslContext
225             HostnameVerifier prevHostNameVerifier = HttpsURLConnection.getDefaultHostnameVerifier();
226             SSLSocketFactory prevSSLSocketFactory = HttpsURLConnection.getDefaultSSLSocketFactory();
227             try {
228                 HttpsURLConnection.setDefaultHostnameVerifier((hostName, session) -> true);
229                 HttpsURLConnection.setDefaultSSLSocketFactory(sslContext.getSocketFactory());
230 
231                 javadoc("-d", "api",
232                         "--module-source-path", src.toString(),
233                         "--module-path", libModules.toString(),
234                         "-link", "http:/" + oldServer.getAddress(),
235                         "--module", "mC",
236                         "-Xdoclint:none");
237 
238             } finally {
239                 HttpsURLConnection.setDefaultHostnameVerifier(prevHostNameVerifier);
240                 HttpsURLConnection.setDefaultSSLSocketFactory(prevSSLSocketFactory);
241             }
242 
243             // Verify the following:
244             // 1: A warning about the redirection is generated.
245             // 2: The contents of the redirected link were read successfully,
246             //    identifying the remote API
247             // 3: The original URL is still used in the generated docs, to avoid assuming
248             //    that all the other files at that link have been redirected as well.
249             checkOutput(Output.OUT, true,
250                     "warning: URL " + oldURL + "/element-list was redirected to " + newURL + "/element-list");
251             checkOutput("mC/p5/C5.html", true,
252                     "extends <a href=\"" + oldURL + """
253                         /mA/p1/C1.html" title="class or interface in p1" class="external-link">C1</a>""");
254             checkOutput("mC/p6/C6.html", true,
255                     "<a href=\"" + oldURL + """
256                         /mB/p4/C4.html" title="class or interface in p4" class="external-link">C4</a>""");
257         } finally {
258             if (oldServer != null) {
259                 out.println("Stopping old server on " + oldServer.getAddress());
260                 oldServer.stop(0);
261             }
262             if (newServer != null) {
263                 out.println("Stopping new server on " + newServer.getAddress());
264                 newServer.stop(0);
265             }
266         }
267     }

269     private void handleOldRequest(HttpExchange x) throws IOException {
270         out.println("old request: "
271                 + x.getProtocol() + " "
272                 + x.getRequestMethod() + " "
273                 + x.getRequestURI());
274         String newProtocol = (newServer instanceof HttpsServer) ? "https" : "http";
275         String redirectTo = newProtocol + ":/" + newServer.getAddress() + x.getRequestURI();
276         out.println("    redirect to: " + redirectTo);
277         x.getResponseHeaders().add("Location", redirectTo);
278         x.sendResponseHeaders(HttpURLConnection.HTTP_MOVED_PERM, 0);
279         x.getResponseBody().close();
280     }
281 
282     private void handleNewRequest(HttpExchange x) throws IOException {
283         out.println("new request: "
284                 + x.getProtocol() + " "
285                 + x.getRequestMethod() + " "
286                 + x.getRequestURI());
287         Path file = libApi.resolve(x.getRequestURI().getPath().substring(1).replace('/', File.separatorChar));
288         System.err.println(file);
289         if (Files.exists(file)) {
290             byte[] bytes = Files.readAllBytes(file);
291             // in the context of this test, the only request should be element-list,
292             // which we can say is text/plain.
293             x.getResponseHeaders().add("Content-type", "text/plain");
294             x.sendResponseHeaders(HttpURLConnection.HTTP_OK, bytes.length);
295             try (OutputStream responseStream = x.getResponseBody()) {
296                 responseStream.write(bytes);
297             }
298         } else {
299             x.sendResponseHeaders(HttpURLConnection.HTTP_NOT_FOUND, 0);
300             x.getResponseBody().close();
301         }
302     }

304     /**
305      * Creates a simple usable SSLContext for an HttpsServer using
306      * a default keystore in the test tree.
307      * <p>
308      * This class is based on
309      * test/jdk/java/net/httpclient/whitebox/java.net.http/jdk/internal/net/http/SimpleSSLContext.java
310      */
311     static class SimpleSSLContext {
312 
313         private final SSLContext ssl;
314 
315         /**
316          * Loads default keystore.
317          */
318         SimpleSSLContext() throws Exception {
319             Path p = Path.of(System.getProperty("test.src", ".")).toAbsolutePath();
320             while (!Files.exists(p.resolve("TEST.ROOT"))) {
321                 p = p.getParent();
322                 if (p == null) {
323                     throw new IOException("can't find TEST.ROOT");
324                 }
325             }
326 
327             System.err.println("Test suite root: " + p);
328             Path testKeys = p.resolve("../lib/jdk/test/lib/net/testkeys").normalize();
329             if (!Files.exists(testKeys)) {
330                 throw new IOException("can't find testkeys");
331             }
332             System.err.println("Test keys: " + testKeys);
333 
334             try (InputStream fis = Files.newInputStream(testKeys)) {
335                 ssl = init(fis);
336             }
337         }
338 
339         private SSLContext init(InputStream i) throws Exception {
340             char[] passphrase = "passphrase".toCharArray();
341             KeyStore ks = KeyStore.getInstance("PKCS12");
342             ks.load(i, passphrase);
343 
344             KeyManagerFactory kmf = KeyManagerFactory.getInstance("PKIX");
345             kmf.init(ks, passphrase);
346 
347             TrustManagerFactory tmf = TrustManagerFactory.getInstance("PKIX");
348             tmf.init(ks);
349 
350             SSLContext ssl = SSLContext.getInstance("TLS");
351             ssl.init(kmf.getKeyManagers(), tmf.getTrustManagers(), null);
352             return ssl;
353         }
354 
355         SSLContext get() {
356             return ssl;
357         }
358     }
