83      private static class Interceptor implements RespInterceptor {
84  
85          private final String path;
86  
87          Interceptor(String path) {
88              this.path = path;
89          }
90  
91          @Override
92          public X509Certificate[] getSignerCertChain(
93                  X509Certificate[] signerCertChain, boolean certReq)
94                  throws Exception {
95              if (path.equals("fullchain")) { // Only case 5 uses full chain
96                  return signerCertChain;
97              } else if (path.equals("nocert")) {
98                  return new X509Certificate[0];
99              } else {
100                 return new X509Certificate[] { signerCertChain[0] };
101             }
102         }
103 
104         @Override
105         public String getSigAlgo(String sigAlgo) throws Exception {
106             return "SHA256withRSA";
107         }
108 
109         @Override
110         public TsaParam getRespParam(TsaParam reqParam) {
111             TsaParam respParam = RespInterceptor.super.getRespParam(reqParam);
112 
113             String policyId
114                     = respParam.policyId() == null || path.equals("diffpolicy")
115                     ? TimestampCheck.defaultPolicyId
116                     : respParam.policyId();
117             respParam.policyId(policyId);
118 
119             String digestAlgo = respParam.digestAlgo();
120             if (path.equals("diffalg")) {
121                 digestAlgo = digestAlgo.contains("256")
122                         ? "SHA-1" : "SHA-256";
123             }
124             respParam.digestAlgo(digestAlgo);
125 
126             byte[] hashedMessage = respParam.hashedMessage();
127             if (path.equals("baddigest")) {
128                 hashedMessage[hashedMessage.length - 1] = (byte) 0x01;
129                 hashedMessage[hashedMessage.length - 2] = (byte) 0x02;
130                 hashedMessage[hashedMessage.length - 3] = (byte) 0x03;
131             }
132             respParam.hashedMessage(hashedMessage);
133 
134             Instant instant = Instant.now();
135             if (path.equals("tsold")) {
136                 instant = instant.minus(20, ChronoUnit.DAYS);
137             }
138             respParam.genTime(Date.from(instant));
139 
140             BigInteger nonce = respParam.nonce();
141             if (path.equals("diffnonce")) {
142                 nonce = BigInteger.valueOf(1234);
143             } else if (path.equals("nononce")) {
144                 nonce = null;
145             }
146             respParam.nonce(nonce);
147 
148             return respParam;
149         }
150     }

152     private static class Handler extends TsaHandler {
153 
154         Handler(String keystore) throws Exception {
155             super(KeyStoreUtils.loadKeyStore(keystore, PASSWORD), PASSWORD);
156         }
157 
158         public TsaSigner createSigner(HttpExchange exchange)
159                 throws Exception {
160             String path = exchange.getRequestURI().getPath().substring(1);
161 
162             SignerEntry signerEntry = createSignerEntry(
163                     path.startsWith("ts") ? path : "ts");
164             byte[] requestData = exchange.getRequestBody().readAllBytes();
165             RespInterceptor interceptor = new Interceptor(path);
166             return new TsaSigner(signerEntry, requestData, interceptor);
167         }
168     }

532     private static void checkInvalidTsaCertKeyUsage() throws Exception {
533 
534         // Hack: Rewrite the TSA cert inside normal.jar into ts2.jar.
535 
536         // Both the cert and the serial number must be rewritten.
537         byte[] tsCert = Files.readAllBytes(Paths.get("ts.cert"));
538         byte[] ts2Cert = Files.readAllBytes(Paths.get("ts2.cert"));
539         byte[] tsSerial = getCert(tsCert)
540                 .getSerialNumber().toByteArray();
541         byte[] ts2Serial = getCert(ts2Cert)
542                 .getSerialNumber().toByteArray();
543 
544         byte[] oldBlock;
545         try (JarFile normal = new JarFile("normal.jar")) {
546             oldBlock = normal.getInputStream(
547                     normal.getJarEntry("META-INF/SIGNER.RSA")).readAllBytes();
548         }
549 
550         JarUtils.updateJar("normal.jar", "ts2.jar",
551                 Map.of("META-INF/SIGNER.RSA",
552                         updateBytes(updateBytes(oldBlock, tsCert, ts2Cert),
553                                 tsSerial, ts2Serial)));
554 
555         verify("ts2.jar", "-verbose", "-certs")
556                 .shouldHaveExitValue(64)
557                 .shouldContain("jar verified")
558                 .shouldContain("Invalid TSA certificate chain: Extended key usage does not permit use for TSA server");
559     }

561     public static X509Certificate getCert(byte[] data)
562             throws CertificateException, IOException {
563         return (X509Certificate)
564                 CertificateFactory.getInstance("X.509")
565                         .generateCertificate(new ByteArrayInputStream(data));
566     }
567 
568     private static byte[] updateBytes(byte[] old, byte[] from, byte[] to) {
569         int pos = 0;
570         while (true) {
571             if (pos + from.length > old.length) {
572                 return null;
573             }
574             if (Arrays.equals(Arrays.copyOfRange(old, pos, pos+from.length), from)) {
575                 byte[] result = old.clone();
576                 System.arraycopy(to, 0, result, pos, from.length);
577                 return result;
578             }
579             pos++;
580         }
581     }

583     private static void checkMissingOrInvalidFiles(String s)
584             throws Throwable {
585

593                 .shouldHaveExitValue(16)
594                 .shouldContain("treated as unsigned")

599                 .shouldContain("treated as unsigned")
600                 .shouldContain("Unparsable signature-related file META-INF/SIGNER.SF");

602         verify("4.jar", "-verbose")
603                 .shouldHaveExitValue(16)
604                 .shouldContain("treated as unsigned")
605                 .shouldContain("Unparsable signature-related file META-INF/SIGNER.RSA");
606     }

616     static OutputAnalyzer verify(String file, String... extra)
617             throws Exception {
618         List<String> args = new ArrayList<>();
619         args.add("-verify");

768     static void checkTimestamp(String file, String policyId, String digestAlg)
769             throws Exception {
770         try (JarFile jf = new JarFile(file)) {
771             JarEntry je = jf.getJarEntry("META-INF/SIGNER.RSA");
772             try (InputStream is = jf.getInputStream(je)) {
773                 byte[] content = is.readAllBytes();
774                 PKCS7 p7 = new PKCS7(content);
775                 SignerInfo[] si = p7.getSignerInfos();
776                 if (si == null || si.length == 0) {
777                     throw new Exception("Not signed");
778                 }
779                 PKCS9Attribute p9 = si[0].getUnauthenticatedAttributes()
780                         .getAttribute(PKCS9Attribute.SIGNATURE_TIMESTAMP_TOKEN_OID);
781                 PKCS7 tsToken = new PKCS7((byte[]) p9.getValue());
782                 TimestampToken tt =
783                         new TimestampToken(tsToken.getContentInfo().getData());
784                 if (!tt.getHashAlgorithm().toString().equals(digestAlg)) {
785                     throw new Exception("Digest alg different");
786                 }
787                 if (!tt.getPolicyID().equals(policyId)) {
788                     throw new Exception("policyId different");
789                 }
790             }
791         }
792     }

911     static void gencert(String alias, String... extra) throws Exception {
912         keytool("-alias " + alias + " -certreq -file " + alias + ".req");
913         String genCmd = "-gencert -alias ca -infile " +
914                 alias + ".req -outfile " + alias + ".cert";
915         for (String s : extra) {
916             genCmd += " " + s;
917         }
918         keytool(genCmd);
919         keytool("-alias " + alias + " -importcert -file " + alias + ".cert");
920     }
