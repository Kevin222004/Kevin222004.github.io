10  import com.sun.source.util.JavacTask;
11  import java.io.File;
12  import java.net.URI;
13  import java.util.Arrays;
14  import java.util.Set;
15  import java.util.TreeSet;
16  import javax.tools.JavaCompiler;
17  import javax.tools.JavaFileObject;
18  import javax.tools.SimpleJavaFileObject;
19  import javax.tools.ToolProvider;
20  
21  public class IgnoreIgnorableCharactersInInput {
22  
23      public static void main(String... args) throws Exception {
24          new IgnoreIgnorableCharactersInInput().run();
25      }
26  
27      void run() throws Exception {
28          JavaCompiler comp = ToolProvider.getSystemJavaCompiler();
29          File classesDir = new File(System.getProperty("user.dir"), "classes");
30          classesDir.mkdirs();
31          JavaSource[] sources = new JavaSource[]{
32              new JavaSource("TestOneIgnorableChar", "AA\\u0000BB"),
33              new JavaSource("TestMultipleIgnorableChar", "AA\\u0000\\u0000\\u0000BB")};
34          JavacTask ct = (JavacTask)comp.getTask(null, null, null,
35                  Arrays.asList("-d", classesDir.getPath()),
36                  null, Arrays.asList(sources));
37          try {
38              if (!ct.call()) {
39                  throw new AssertionError("Error thrown when compiling test cases");
40              }
41          } catch (Throwable ex) {
42              throw new AssertionError("Error thrown when compiling test cases");
43          }
44          check(classesDir,
45                  "TestOneIgnorableChar.class",
46                  "TestOneIgnorableChar$AABB.class",
47                  "TestMultipleIgnorableChar.class",
48                  "TestMultipleIgnorableChar$AABB.class");
49          if (errors > 0)
50              throw new AssertionError("There are some errors in the test check the error output");
51      }
52  
53      /**
54       *  Check that a directory contains the expected files.
55       */
56      void check(File dir, String... paths) {
57          Set<String> found = new TreeSet<String>(Arrays.asList(dir.list()));
58          Set<String> expect = new TreeSet<String>(Arrays.asList(paths));
59          if (found.equals(expect))
60              return;
61          for (String f: found) {
62              if (!expect.contains(f))
63                  error("Unexpected file found: " + f);
64          }
65          for (String e: expect) {
66              if (!found.contains(e))
67                  error("Expected file not found: " + e);
68          }
69      }
70  
71      int errors;
72  
73      void error(String msg) {
74          System.err.println(msg);
75          errors++;
76      }
77  
78      class JavaSource extends SimpleJavaFileObject {
79  
80          String internalSource =
81              "public class #O {public class #I {} }";
82          public JavaSource(String outerClassName, String innerClassName) {
83              super(URI.create(outerClassName + ".java"), JavaFileObject.Kind.SOURCE);
84              internalSource =
85                      internalSource.replace("#O", outerClassName).replace("#I", innerClassName);
86          }
87  
88          @Override
89          public CharSequence getCharContent(boolean ignoreEncodingErrors) {
90              return internalSource;
91          }
92      }
93  }
