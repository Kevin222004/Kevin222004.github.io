35  import jdk.test.lib.Asserts;
36  import jdk.test.lib.process.OutputAnalyzer;
37  import lib.jdb.JdbCommand;
38  import lib.jdb.JdbTest;
39  
40  import java.util.regex.Matcher;
41  import java.util.regex.Pattern;
42  
43  class RedefineTTYLineNumberTarg {
44  
45      public void B() {
46          System.out.println("in B: @1 delete"); // delete 1 line before A method
47      }
48  
49      public void A() {
50          System.out.println("expected statement printed by jdb");
51      }
52  
53      public static void main(String[] args) {
54          RedefineTTYLineNumberTarg untitled41 = new RedefineTTYLineNumberTarg();
55          untitled41.A();
56          System.out.println("done");
57      }
58  }
59  
60  public class RedefineTTYLineNumber extends JdbTest {
61  
62      public static void main(String argv[]) {
63          new RedefineTTYLineNumber().run();
64      }
65  
66      private RedefineTTYLineNumber() {
67          super(DEBUGGEE_CLASS, SOURCE_FILE);
68      }
69  
70      private final static String DEBUGGEE_CLASS = RedefineTTYLineNumberTarg.class.getName();
71      private final static String SOURCE_FILE = "RedefineTTYLineNumber.java";
72  
73      // parses line number from the jdb "Breakpoint hit" message
74      private static int parseLineNum(String s) {
75          // Breakpoint hit: "thread=main", RedefineTTYLineNumberTarg.A(), line=49 bci=0
76          // 49            System.out.println("expected statement printed by jdb");
77          Matcher m = Pattern.compile("\\bline=(\\d+)\\b").matcher(s);
78          if (!m.find()) {
79              throw new RuntimeException("could not parse line number");
80          }
81          return Integer.parseInt(m.group(1));
82      }
83  
84      private void verifyBPSource(int n, String reply) {
85          if (!reply.contains("expected statement printed by jdb")) {
86              throw new RuntimeException("Breakpoint source (" + n + ") is not correct");
87          }
88      }
89  
90      @Override
91      protected void runCases() {
92          jdb.command(JdbCommand.stopIn(DEBUGGEE_CLASS, "A"));
93          String bp1Reply = execCommand(JdbCommand.run()).getStdout();
94          int bp1Line = parseLineNum(bp1Reply);
95          redefineClass(1, "-g");
96          jdb.command(JdbCommand.pop());
97          jdb.command(JdbCommand.stopIn(DEBUGGEE_CLASS, "A"));
98          String bp2Reply = execCommand(JdbCommand.cont()).getStdout();
99          int bp2Line = parseLineNum(bp2Reply);
100 
101         new OutputAnalyzer(getDebuggeeOutput())
102                 .shouldNotContain("Internal exception:");
103         // 1 line is deleted before RedefineTTYLineNumberTarg.A(),
104         // so bp2Line should be equals bp1Line-1
105         Asserts.assertEquals(bp2Line, bp1Line - 1, "BP line numbers");
106         verifyBPSource(1, bp1Reply);
107         // uncomment the following line to reproduce JDK-8210927
108         //verifyBPSource(2, bp2Reply);
109     }
110 }
