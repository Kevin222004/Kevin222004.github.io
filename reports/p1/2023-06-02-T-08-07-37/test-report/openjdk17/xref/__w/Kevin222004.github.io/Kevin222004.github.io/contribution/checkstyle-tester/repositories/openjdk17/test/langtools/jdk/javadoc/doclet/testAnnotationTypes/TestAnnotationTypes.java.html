38  public class TestAnnotationTypes extends JavadocTester {

45      @Test
46      public void test() {
47          javadoc("-d", "out-1",
48                  "-sourcepath", testSrc,
49                  "--no-platform-links",
50                  "pkg");
51          checkExit(Exit.OK);
52  
53          checkOutput("pkg/AnnotationTypeField.html", true,
54                  """
55                      <li>Summary:&nbsp;</li>
56                      <li><a href="#field-summary">Field</a>&nbsp;|&nbsp;</li>""",
57                  """
58                      <li>Detail:&nbsp;</li>
59                      <li><a href="#field-detail">Field</a>&nbsp;|&nbsp;</li>""",
60                  "<!-- =========== FIELD SUMMARY =========== -->",
61                  "<h2>Field Summary</h2>",
62                  """
63                      <div class="col-second even-row-color"><code><a href="#DEFAULT_NAME" class="memb\
64                      er-name-link">DEFAULT_NAME</a></code></div>""",
65                  "<!-- ============ FIELD DETAIL =========== -->",
66                  """
67                      <section class="detail" id="DEFAULT_NAME">
68                      <h3>DEFAULT_NAME</h3>
69                      <div class="member-signature"><span class="modifiers">static final</span>&nbsp;<\
70                      span class="return-type">java.lang.String</span>&nbsp;<span class="element-name">DEFAULT_NAME</span></div>
71                      """);
72  
73          checkOutput("pkg/AnnotationType.html", true,
74                  """
75                      <li>Summary:&nbsp;</li>
76                      <li>Field&nbsp;|&nbsp;</li>""",
77                  """
78                      <li>Detail:&nbsp;</li>
79                      <li>Field&nbsp;|&nbsp;</li>""");

81          checkOutput("pkg/AnnotationType.html", true,
82                      "<!-- ============ ANNOTATION INTERFACE MEMBER DETAIL =========== -->",
83                      "<ul class=\"member-list\">",
84                      "<li>",
85                      """
86                          <section class="details" id="annotation-interface-element-detail">""",
87                      "<h2>Element Details</h2>",
88                      "</a>",
89                      "<ul class=\"member-list\">",
90                      "<li>",
91                      """
92                          <section class="detail" id="value()">""",
93                      "<h3>value</h3>\n",
94                      """
95                          <div class="member-signature"><span class="return-type">int</span>&nbsp;<span class="element-name">value</span></div>""");
96  
97          checkOutput("pkg/AnnotationType.html", false,
98                  """
99                      <HR>
100 
101                     <P>
102 
103                     <P><!-- ========= END OF CLASS DATA ========= --><HR>""");
104     }

106     @Test
107     public void testLinkSource() {
108         javadoc("-d", "out-2",
109                 "-linksource",
110                 "--no-platform-links",
111                 "-sourcepath", testSrc,
112                 "pkg");
113         checkExit(Exit.OK);
114 
115         checkOutput("src-html/pkg/AnnotationType.html", true,
116                 "<title>Source code</title>",
117                 "@Documented public @interface AnnotationType {");
118 
119         checkOutput("src-html/pkg/AnnotationTypeField.html", true,
120                 "<title>Source code</title>",
121                 "@Documented public @interface AnnotationTypeField {");
122 
123         checkOutput("pkg/AnnotationType.html", true,
124                 """
125                     <span class="modifiers">public @interface </span><span class="element-name"><a href="../src-html/pkg/AnnotationType.html#line-31">AnnotationType</a></span></div>""");

128                 """
129                     <span class="modifiers">public @interface </span><span class="element-name"><a href="../src-html/pkg/AnnotationTypeField.html#line-31">AnnotationTypeField</a></span></div>""");
