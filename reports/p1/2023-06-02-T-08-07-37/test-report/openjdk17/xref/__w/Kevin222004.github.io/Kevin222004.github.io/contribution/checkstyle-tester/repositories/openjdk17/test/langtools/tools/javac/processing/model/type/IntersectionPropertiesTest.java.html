28   * @modules jdk.compiler/com.sun.tools.javac.util

34  import com.sun.source.util.*;
35  import com.sun.tools.javac.util.Assert;
36  import javax.annotation.processing.*;
37  import javax.lang.model.type.*;
38  import javax.lang.model.util.ElementFilter;
39  import javax.lang.model.element.*;
40  import javax.tools.*;
41  import java.util.*;
42  import java.io.*;
43  import javax.lang.model.util.Types;
44  
45  public class IntersectionPropertiesTest {
46  
47      private int errors = 0;
48      private static final String Intersection_name = "IntersectionTest.java";
49      private static final String Intersection_contents =
50          "import java.util.AbstractList;\n" +
51          "import java.util.List;\n" +
52          "import java.io.Serializable;\t" +
53          "public class IntersectionTest<S extends List & Serializable, One extends AbstractList & Runnable & Serializable, Two extends AbstractList & Serializable & Runnable> {\n" +
54          "  void method(S s, One o, Two t) { }\n" +
55          "  public static abstract class SubType extends AbstractList implements Runnable, Serializable { } \n" +
56          "}";
57  
58      private static final File classesdir = new File("intersectionproperties");
59      private static final JavaCompiler comp =
60          ToolProvider.getSystemJavaCompiler();
61      private static final StandardJavaFileManager fm =
62          comp.getStandardFileManager(null, null, null);
63  
64      public void runOne(final String Test_name, final String Test_contents)
65          throws IOException {
66          System.err.println("Testing " + Test_name);
67          final Iterable<? extends JavaFileObject> files =
68              fm.getJavaFileObjectsFromFiles(Collections.singleton(writeFile(classesdir, Test_name, Test_contents)));
69          final JavacTask ct =
70              (JavacTask)comp.getTask(null, fm, null, null, null, files);
71          ct.setProcessors(Collections.singleton(new TestProcessor()));
72  
73          if (!ct.call()) {
74              System.err.println("Compilation unexpectedly failed");
75              errors++;
76          }
77      }
78  
79      public void run() throws IOException {
80          try {
81              runOne(Intersection_name, Intersection_contents);
82  
83              if (0 != errors)
84                  throw new RuntimeException(errors + " errors occurred");
85          } finally {
86              fm.close();
87          }
88      }
89  
90      public static void main(String... args) throws IOException {
91          new IntersectionPropertiesTest().run();
92      }
93  
94      private static File writeFile(File dir, String path, String body)
95          throws IOException {
96          File f = new File(dir, path);
97          f.getParentFile().mkdirs();
98          try (FileWriter out = new FileWriter(f)) {
99              out.write(body);
100         }
101         return f;
102     }
103 
104     private class TestProcessor extends JavacTestingAbstractProcessor {
105 
106         private Set<? extends Element> rootElements;
107 
108         public boolean process(Set<? extends TypeElement> annotations, RoundEnvironment roundEnv) {
109             rootElements = roundEnv.getRootElements();
110             if (!rootElements.isEmpty()) {
111                 performCheck();
112             }
113             return true;
114         }
115 
116         private void performCheck() {
117             TypeElement typeElement = (TypeElement) rootElements.iterator().next();
118             ExecutableElement method = ElementFilter.methodsIn(typeElement.getEnclosedElements()).get(0);
119             TypeVariable typeVariable = (TypeVariable) method.getParameters().get(0).asType();
120 
121             final TypeMirror upperBound = typeVariable.getUpperBound();
122 
123             TypeParameterElement typeParameterElement = ((TypeParameterElement) typeVariable.asElement());
124             final List<? extends TypeMirror> bounds = typeParameterElement.getBounds();
125             Types types = processingEnv.getTypeUtils();
126             final HashSet<TypeMirror> actual = new HashSet<TypeMirror>(types.directSupertypes(upperBound));
127             final HashSet<TypeMirror> expected = new HashSet<TypeMirror>(bounds);
128             if (!expected.equals(actual)) {
129                 System.err.println("Mismatched expected and actual bounds.");
130                 System.err.println("Expected:");
131                 for(TypeMirror tm : expected)
132                     System.err.println("  " + tm);
133                 System.err.println("Actual:");
134                 for(TypeMirror tm : actual)
135                     System.err.println("  " + tm);
136                 errors++;
137             }
138 
139             TypeVariable oneTypeVariable = (TypeVariable) method.getParameters().get(1).asType();
140             TypeMirror oneUpperBound = oneTypeVariable.getUpperBound();
141             TypeVariable twoTypeVariable = (TypeVariable) method.getParameters().get(2).asType();
142             TypeMirror twoUpperBound = twoTypeVariable.getUpperBound();
143             TypeElement oneUpperBoundElement = (TypeElement) types.asElement(oneUpperBound);
144 
145             Assert.checkNonNull(oneUpperBoundElement);
146 
147             Assert.check("java.util.AbstractList".equals(oneUpperBoundElement.getSuperclass().toString()),
148                          oneUpperBoundElement.getSuperclass().toString());
149 
150             List<String> superInterfaces = new java.util.ArrayList<>();
151 
152             for (TypeMirror tm : oneUpperBoundElement.getInterfaces()) {
153                 superInterfaces.add(tm.toString());
154             }
155 
156             Assert.check(java.util.Arrays.asList("java.lang.Runnable",
157                                                  "java.io.Serializable").equals(superInterfaces),
158                          superInterfaces);
159 
160             Assert.check(types.isSameType(upperBound, types.capture(upperBound)));
161             Assert.check(types.isSameType(types.erasure(typeVariable), types.erasure(upperBound)));
162 
163             TypeElement subTypeClass = processingEnv.getElementUtils().getTypeElement("IntersectionTest.SubType");
164 
165             Assert.checkNonNull(subTypeClass);
166 
167             Assert.check(types.isAssignable(subTypeClass.asType(), oneUpperBound));
168             Assert.check(types.isSameType(oneUpperBound, twoUpperBound));
169             Assert.check(!types.isSameType(upperBound, twoUpperBound));
170             Assert.check(types.isSubtype(subTypeClass.asType(), oneUpperBound));
171             Assert.check(types.isSubtype(oneUpperBound, upperBound));
172         }
173 
174     }
