24  /**
25   * @test
26   * @bug 8003280
27   * @summary Add lambda tests
28   *   Test static method reference
29   * @compile MethodRef1.java
30   * @run main MethodRef1
31   */
32  
33  public class MethodRef1 {
34  
35      static interface A {void m();}
36  
37      static interface B {void m(int i);}
38  
39      static interface C {String m(String s);}
40  
41      private static void assertTrue(boolean cond) {
42          if (!cond)
43              throw new AssertionError();
44      }
45  
46      static void foo(int x) {
47          System.out.println("MethodRef1.foo(int) " + x);
48      }
49  
50      static void bar() {
51          System.out.println("MethodRef1.bar()");
52      }
53  
54      static void bar(int x) {
55          System.out.println("MethodRef1.bar(int) " + x);
56      }
57  
58      static String bar(String s) {
59          return "MethodRef1.bar(String) " + s;
60      }
61  
62      public static void main(String[] args) {
63  
64          A a = MethodRef1::bar; //static reference to bar()
65          a.m();
66  
67          B b = MethodRef1::foo; //static reference to foo(int), (int) omitted because method foo is not overloaded
68          b.m(1);
69  
70          b = MethodRef1::foo; //static reference to foo(int)
71          b.m(1);
72  
73          b = MethodRef1::bar; //static reference to bar(int)
74          b.m(2);
75  
76          C c = MethodRef1::bar; //static reference to bar(String)
77          assertTrue( c.m("hi").equals("MethodRef1.bar(String) hi") );
78      }
79  }
