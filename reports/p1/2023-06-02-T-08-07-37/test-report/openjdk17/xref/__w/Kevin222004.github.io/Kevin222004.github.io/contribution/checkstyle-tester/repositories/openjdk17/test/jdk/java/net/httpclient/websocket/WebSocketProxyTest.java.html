153     record bytes(byte[] bytes) {
154         @Override
155         public boolean equals(Object o) {
156             if (this == o) return true;
157             if (o instanceof bytes other) {
158                 return Arrays.equals(bytes(), other.bytes());
159             }
160             return false;
161         }
162         @Override
163         public int hashCode() { return Arrays.hashCode(bytes()); }
164         public String toString() {
165             return "0x" + HexFormat.of()
166                     .withUpperCase()
167                     .formatHex(bytes());
168         }
169     }

171     static List<bytes> ofBytes(List<byte[]> bytes) {
172         return bytes.stream().map(bytes::new).toList();
173     }
174 
175     static String diagnose(List<byte[]> a, List<byte[]> b) {
176         var actual = ofBytes(a);
177         var expected = ofBytes(b);
178         var message = actual.equals(expected) ? "match" : "differ";
179         return "%s and %s %s".formatted(actual, expected, message);
180     }

182     @Test(dataProvider = "servers")
183     public void simpleAggregatingBinaryMessages
184             (Function<int[],DummySecureWebSocketServer> serverSupplier,
185              Supplier<ProxyServer> proxyServerSupplier)
186         throws IOException
187     {
188         List<byte[]> expected = List.of("hello", "chegar")
189                 .stream()
190                 .map(s -> s.getBytes(StandardCharsets.US_ASCII))
191                 .collect(Collectors.toList());
192         int[] binary = new int[]{
193                 0x82, 0x05, 0x68, 0x65, 0x6C, 0x6C, 0x6F,       // hello
194                 0x82, 0x06, 0x63, 0x68, 0x65, 0x67, 0x61, 0x72, // chegar
195                 0x88, 0x00                                      // <CLOSE>
196         };
197         CompletableFuture<List<byte[]>> actual = new CompletableFuture<>();
198 
199         try (var proxyServer = proxyServerSupplier.get();
200              var server = serverSupplier.apply(binary)) {
201 
202             InetSocketAddress proxyAddress = new InetSocketAddress(
203                     InetAddress.getLoopbackAddress(), proxyServer.getPort());
204             server.open();
205             System.out.println("Server: " + server.getURI());
206             System.out.println("Proxy: " + proxyAddress);
207 
208             WebSocket.Listener listener = new WebSocket.Listener() {
209 
210                 List<byte[]> collectedBytes = new ArrayList<>();
211                 ByteBuffer buffer = ByteBuffer.allocate(1024);
212 
213                 @Override
214                 public CompletionStage<?> onBinary(WebSocket webSocket,
215                                                    ByteBuffer message,
216                                                    boolean last) {
217                     System.out.printf("onBinary(%s, %s)%n", message, last);
218                     webSocket.request(1);
219 
220                     append(message);
221                     if (last) {
222                         buffer.flip();
223                         byte[] bytes = new byte[buffer.remaining()];
224                         buffer.get(bytes);
225                         buffer.clear();
226                         processWholeBinary(bytes);
227                     }
228                     return null;
229                 }
230 
231                 private void append(ByteBuffer message) {
232                     if (buffer.remaining() < message.remaining()) {
233                         assert message.remaining() > 0;
234                         int cap = (buffer.capacity() + message.remaining()) * 2;
235                         ByteBuffer b = ByteBuffer.allocate(cap);
236                         b.put(buffer.flip());
237                         buffer = b;
238                     }
239                     buffer.put(message);
240                 }
241 
242                 private void processWholeBinary(byte[] bytes) {
243                     String stringBytes = new String(bytes, UTF_8);
244                     System.out.println("processWholeBinary: " + stringBytes);
245                     collectedBytes.add(bytes);
246                 }
247 
248                 @Override
249                 public CompletionStage<?> onClose(WebSocket webSocket,
250                                                   int statusCode,
251                                                   String reason) {
252                     actual.complete(collectedBytes);
253                     return null;
254                 }
255 
256                 @Override
257                 public void onError(WebSocket webSocket, Throwable error) {
258                     actual.completeExceptionally(error);
259                 }
260             };
261 
262             var webSocket = newBuilder()
263                     .proxy(ProxySelector.of(proxyAddress))
264                     .authenticator(new WSAuthenticator())
265                     .build().newWebSocketBuilder()
266                     .buildAsync(server.getURI(), listener)
267                     .join();
268 
269             List<byte[]> a = actual.join();
270             assertEquals(ofBytes(a), ofBytes(expected), diagnose(a, expected));
271         }
272     }

344     /*
345      * Ensures authentication does not succeed when no authenticator is present.
346      */
347     @Test
348     public void failNoAuthenticator() throws IOException  {
349         try (var proxyServer = AUTH_TUNNELING_PROXY_SERVER.get();
350              var server = new DummySecureWebSocketServer(USERNAME, PASSWORD)) {
351             server.open();
352             InetSocketAddress proxyAddress = new InetSocketAddress(
353                     InetAddress.getLoopbackAddress(), proxyServer.getPort());
354 
355             CompletableFuture<WebSocket> cf = newBuilder()
356                     .proxy(ProxySelector.of(proxyAddress)).build()
357                     .newWebSocketBuilder()
358                     .buildAsync(server.getURI(), new WebSocket.Listener() { });
359 
360             try {
361                 var webSocket = cf.join();
362                 fail("Expected exception not thrown");
363             } catch (CompletionException expected) {
364                 WebSocketHandshakeException e = (WebSocketHandshakeException)expected.getCause();
365                 HttpResponse<?> response = e.getResponse();
366                 assertEquals(response.statusCode(), 407);
367             }
368         }
369     }
