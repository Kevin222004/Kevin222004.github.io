144     private volatile static boolean clientRenegoReady = false;
145 
146     public static void main(String[] args) throws Exception {
147 
148         final String tlsProtocol = args[0];
149         final KeyType keyType = KeyType.valueOf(args[1]);
150         final String cipher = args[2];
151         Security.setProperty("jdk.tls.disabledAlgorithms", "");
152         CountDownLatch serverReady = new CountDownLatch(1);
153         Server server = new Server(tlsProtocol, keyType, cipher, serverReady);
154         server.start();
155 
156         // Wait till server is ready to accept connection.
157         serverReady.await();
158         new Client(tlsProtocol, keyType, cipher, server.port).doClientSide();
159         if (server.serverExc != null) {
160             throw new RuntimeException(server.serverExc);
161         }
162     }

164     public static class Server implements Runnable {
165 
166         private volatile int port = 0;
167         private final String tlsProtocol;
168         private final KeyType keyType;
169         private final String cipher;
170         private final CountDownLatch latch;
171         private volatile Exception serverExc;
172 
173         public Server(String tlsProtocol, KeyType keyType, String cipher,
174                 CountDownLatch latch) {
175             this.tlsProtocol = tlsProtocol;
176             this.keyType = keyType;
177             this.cipher = cipher;
178             this.latch = latch;
179         }
180 
181         public void start() {
182 
183             ExecutorService executor = null;
184             try {
185                 executor = Executors.newCachedThreadPool(new ThreadFactory() {
186                     @Override
187                     public Thread newThread(Runnable r) {
188                         Thread t = Executors.defaultThreadFactory()
189                                 .newThread(r);
190                         t.setDaemon(true);
191                         return t;
192                     }
193                 });
194                 executor.execute(this);
195             } finally {
196                 if (executor != null) {
197                     executor.shutdown();
198                 }
199             }
200         }
201 
202         /*
203          * Define the server side operation.
204          */
205         void doServerSide() throws Exception {
206 
207             SSLContext ctx = getSSLContext(tlsProtocol,
208                     keyType.getTrustedCert(), keyType.getEndCert(),
209                     keyType.getPrivateKey(), keyType.getKeyType());
210             SSLServerSocketFactory sslssf = ctx.getServerSocketFactory();
211             InetSocketAddress socketAddress =
212                     new InetSocketAddress(InetAddress.getLoopbackAddress(), port);
213             SSLServerSocket sslServerSocket
214                     = (SSLServerSocket) sslssf.createServerSocket();
215             sslServerSocket.bind(socketAddress);
216             port = sslServerSocket.getLocalPort();
217             System.out.println("Server listining on port: " + port);
218             // specify the enabled server cipher suites
219             sslServerSocket.setEnabledCipherSuites(new String[]{this.cipher});
220             sslServerSocket.setEnabledProtocols(new String[]{tlsProtocol});
221             sslServerSocket.setSoTimeout(25000);
222             /*
223              * Signal Client, the server is ready to accept client request.
224              */
225             latch.countDown();
226             try (SSLSocket sslSocket = (SSLSocket) sslServerSocket.accept()) {
227                 try (InputStream sslIS = sslSocket.getInputStream();
228                         OutputStream sslOS = sslSocket.getOutputStream();) {
229                     sslIS.read();
230                     sslOS.write(85);
231                     sslOS.flush();
232                     while (!clientRenegoReady) {
233                         System.out.println("Waiting for ClientHello");
234                         TimeUnit.MILLISECONDS.sleep(50);
235                     }
236                     for (int i = 0; i < 4; i++) {
237                         sslIS.read();
238                         sslOS.write(89);
239                         sslOS.flush();
240                         TimeUnit.MILLISECONDS.sleep(50);   // wait for a while
241                     }
242                 }
243             } finally {
244                 sslServerSocket.close();
245             }
246         }
247 
248         @Override
249         public void run() {
250             try {
251                 doServerSide();
252             } catch (Exception e) {
253                 // Print the exception for debug purpose.
254                 e.printStackTrace(System.out);
255                 serverExc = e;
256             }
257         }
258     }

263     public static class Client {
264 
265         private final int serverPort;
266         private final String tlsProtocol;
267         private final KeyType keyType;
268         private final String cipher;
269 
270         public Client(String tlsProtocol, KeyType keyType, String cipher,
271                 int serverPort) {
272             this.tlsProtocol = tlsProtocol;
273             this.keyType = keyType;
274             this.cipher = cipher;
275             this.serverPort = serverPort;
276         }
277 
278         void doClientSide() throws Exception {
279 
280             SSLContext ctx = getSSLContext(this.tlsProtocol,
281                     keyType.getTrustedCert(), null, null, keyType.getKeyType());
282             SSLSocketFactory sslsf = ctx.getSocketFactory();
283             try (SSLSocket sslSocket
284                     = (SSLSocket) sslsf.createSocket("localhost", serverPort)) {
285                 // Specify the client cipher suites
286                 sslSocket.setEnabledCipherSuites(new String[]{this.cipher});
287                 sslSocket.setEnabledProtocols(new String[]{this.tlsProtocol});
288                 InputStream sslIS = sslSocket.getInputStream();
289                 OutputStream sslOS = sslSocket.getOutputStream();
290                 try {
291                     sslOS.write(86);
292                     sslOS.flush();
293                     sslIS.read();
294                     // Re-handshake for key-update and session resumption.
295                     sslSocket.startHandshake();
296                     System.out.println("Client: Re-Handshake completed.");
297                 } finally {
298                     clientRenegoReady = true;
299                 }
300                 try {
301                     for (int i = 0; i < 4; i++) {
302                         sslOS.write(88);
303                         sslOS.flush();
304                         sslIS.read();
305                         TimeUnit.MILLISECONDS.sleep(50);  // wait for a while
306                     }
307                 } finally {
308                     sslIS.close();
309                     sslOS.close();
310                 }
311             }
312         }

315     // get the ssl context
316     protected static SSLContext getSSLContext(String tlsProtocol,
317             String trustedCertStr, String keyCertStr,
318             String privateKey, String keyType) throws Exception {
319 
320         // Generate certificate from cert string
321         CertificateFactory cf = CertificateFactory.getInstance("X.509");
322 
323         // Create a key store
324         KeyStore ts = KeyStore.getInstance("PKCS12");
325         KeyStore ks = KeyStore.getInstance("PKCS12");
326         ts.load(null, null);
327         ks.load(null, null);
328         char passphrase[] = "passphrase".toCharArray();
329 
330         // Import the trusted cert
331         ts.setCertificateEntry("trusted-cert-" + keyType,
332                 cf.generateCertificate(new ByteArrayInputStream(
333                         trustedCertStr.getBytes())));
334 
335         boolean hasKeyMaterials = keyCertStr != null && privateKey != null;
336         if (hasKeyMaterials) {
337 
338             // Generate the private key.
339             PKCS8EncodedKeySpec priKeySpec = new PKCS8EncodedKeySpec(
340                     Base64.getMimeDecoder().decode(privateKey));
341             KeyFactory kf = KeyFactory.getInstance(keyType);
342             PrivateKey priKey = kf.generatePrivate(priKeySpec);
343 
344             // Generate certificate chain
345             Certificate keyCert = cf.generateCertificate(
346                     new ByteArrayInputStream(keyCertStr.getBytes()));
347             Certificate[] chain = new Certificate[]{keyCert};
348 
349             // Import the key entry.
350             ks.setKeyEntry("cert-" + keyType, priKey, passphrase, chain);
351         }
352 
353         // Create SSL context
354         TrustManagerFactory tmf = TrustManagerFactory.getInstance("PKIX");
355         tmf.init(ts);
356 
357         SSLContext context = SSLContext.getInstance(tlsProtocol);
358         if (hasKeyMaterials) {
359             KeyManagerFactory kmf = KeyManagerFactory.getInstance("NewSunX509");
360             kmf.init(ks, passphrase);
361             context.init(kmf.getKeyManagers(), tmf.getTrustManagers(), null);
362         } else {
363             context.init(null, tmf.getTrustManagers(), null);
364         }
365         return context;
366     }
