24  import utils.*;
25  /*
26   * @test
27   * @summary Test checks output displayed with jstat -gcnew.
28   *          Test scenario:
29   *          test several times provokes garbage collection in the debuggee application and after each garbage
30   *          collection runs jstat. jstat should show that after garbage collection number of GC events and garbage
31   *          collection time increase.
32   * @requires vm.gc != "Z" & vm.gc != "Shenandoah"
33   * @modules java.base/jdk.internal.misc
34   * @library /test/lib
35   * @library ../share
36   * @run main/othervm -XX:+UsePerfData -Xmx128M GcNewTest
37   */
38  
39  public class GcNewTest {
40  
41      public static void main(String[] args) throws Exception {
42  
43          // We will be running "jstat -gc" tool
44          JstatGcNewTool jstatGcTool = new JstatGcNewTool(ProcessHandle.current().pid());
45  
46          // Run once and get the  results asserting that they are reasonable
47          JstatGcNewResults measurement1 = jstatGcTool.measure();
48          measurement1.assertConsistency();
49  
50          GcProvoker gcProvoker = new GcProvoker();
51  
52          // Provoke GC and run the tool again
53          gcProvoker.provokeGc();
54          JstatGcNewResults measurement2 = jstatGcTool.measure();
55          measurement2.assertConsistency();
56  
57          // Assert the increase in GC events and time between the measurements
58          assertThat(measurement2.getFloatValue("YGC") > measurement1.getFloatValue("YGC"), "YGC didn't increase between measurements 1 and 2");
59          assertThat(measurement2.getFloatValue("YGCT") > measurement1.getFloatValue("YGCT"), "YGCT time didn't increase between measurements 1 and 2");
60  
61          // Provoke GC and run the tool again
62          gcProvoker.provokeGc();
63          JstatGcNewResults measurement3 = jstatGcTool.measure();
64          measurement3.assertConsistency();
65  
66          // Assert the increase in GC events and time between the measurements
67          assertThat(measurement3.getFloatValue("YGC") > measurement2.getFloatValue("YGC"), "YGC didn't increase between measurements 1 and 2");
68          assertThat(measurement3.getFloatValue("YGCT") > measurement2.getFloatValue("YGCT"), "YGCT time didn't increase between measurements 1 and 2");
69  
70      }
71  
72      private static void assertThat(boolean result, String message) {
73          if (!result) {
74              throw new RuntimeException(message);
75          };
76      }
77  }
