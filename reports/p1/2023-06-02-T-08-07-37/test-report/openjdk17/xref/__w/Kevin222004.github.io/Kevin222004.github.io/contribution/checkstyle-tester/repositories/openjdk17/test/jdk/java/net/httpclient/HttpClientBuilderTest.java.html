69      @Test
70      public void testDefaults() throws Exception {
71          List<HttpClient> clients = List.of(HttpClient.newHttpClient(),
72                                             HttpClient.newBuilder().build());
73  
74          for (HttpClient client : clients) {
75              // Empty optionals and defaults
76              assertFalse(client.authenticator().isPresent());
77              assertFalse(client.cookieHandler().isPresent());
78              assertFalse(client.connectTimeout().isPresent());
79              assertFalse(client.executor().isPresent());
80              assertFalse(client.proxy().isPresent());
81              assertTrue(client.sslParameters() != null);
82              assertTrue(client.followRedirects().equals(HttpClient.Redirect.NEVER));
83              assertTrue(client.sslContext() == SSLContext.getDefault());
84              assertTrue(client.version().equals(HttpClient.Version.HTTP_2));
85          }
86      }

188     @Test
189     public void testSSLParameters() {
190         HttpClient.Builder builder = HttpClient.newBuilder();
191         SSLParameters a = new SSLParameters();
192         a.setCipherSuites(new String[] { "A" });
193         builder.sslParameters(a);
194         a.setCipherSuites(new String[] { "Z" });
195         assertTrue(builder.build().sslParameters() != (a));
196         assertTrue(builder.build().sslParameters().getCipherSuites()[0].equals("A"));
197         SSLParameters b = new SSLParameters();
198         b.setEnableRetransmissions(true);
199         builder.sslParameters(b);
200         assertTrue(builder.build().sslParameters() != b);
201         assertTrue(builder.build().sslParameters().getEnableRetransmissions());
202         assertThrows(NPE, () -> builder.sslParameters(null));
203         SSLParameters c = new SSLParameters();
204         c.setProtocols(new String[] { "C" });
205         builder.sslParameters(c);
206         c.setProtocols(new String[] { "D" });
207         assertTrue(builder.build().sslParameters().getProtocols()[0].equals("C"));
208     }

267     static final URI uri = URI.create("http://foo.com/");
268 
269     @Test
270     static void testHttpClientSendArgs() throws Exception {
271         HttpClient client = HttpClient.newHttpClient();
272         HttpRequest request = HttpRequest.newBuilder(uri).build();
273 
274         assertThrows(NPE, () -> client.send(null, BodyHandlers.discarding()));
275         assertThrows(NPE, () -> client.send(request, null));
276         assertThrows(NPE, () -> client.send(null, null));
277 
278         assertThrows(NPE, () -> client.sendAsync(null, BodyHandlers.discarding()));
279         assertThrows(NPE, () -> client.sendAsync(request, null));
280         assertThrows(NPE, () -> client.sendAsync(null, null));
281 
282         assertThrows(NPE, () -> client.sendAsync(null, BodyHandlers.discarding(), null));
283         assertThrows(NPE, () -> client.sendAsync(request, null, null));
284         assertThrows(NPE, () -> client.sendAsync(null, null, null));
285 
286         // CONNECT is disallowed in the implementation, since it is used for
287         // tunneling, and is handled separately for security checks.
288         HttpRequest connectRequest = new HttpConnectRequest();
289         assertThrows(IAE, () -> client.send(connectRequest, BodyHandlers.discarding()));
290         assertThrows(IAE, () -> client.sendAsync(connectRequest, BodyHandlers.discarding()));
291         assertThrows(IAE, () -> client.sendAsync(connectRequest, BodyHandlers.discarding(), null));
292     }
