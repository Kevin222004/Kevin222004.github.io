4    * @summary Annotation processors should be able to print multiple errors at the same location.

11  import java.util.*;
12  import javax.annotation.processing.*;
13  import javax.lang.model.element.*;
14  import javax.tools.Diagnostic.Kind;
15  import com.sun.source.util.TreePath;
16  import com.sun.source.util.Trees;
17  
18  public class TestMultipleErrors extends JavacTestingAbstractProcessor {
19      @Override
20      public boolean process(Set<? extends TypeElement> annotations, RoundEnvironment roundEnv) {
21          for (Element root : roundEnv.getRootElements()) {
22              processingEnv.getMessager().printMessage(Kind.ERROR, "error1", root);
23              processingEnv.getMessager().printMessage(Kind.ERROR, "error2", root);
24  
25              Trees trees = Trees.instance(processingEnv);
26              TreePath path = trees.getPath(root);
27  
28              trees.printMessage(Kind.ERROR, "error3", path.getLeaf(), path.getCompilationUnit());
29              trees.printMessage(Kind.ERROR, "error4", path.getLeaf(), path.getCompilationUnit());
30          }
31          return true;
32      }
33  }
