42          TestNewApiList test = new TestNewApiList();

46      @Test
47      public void testMultiRelease() throws Exception {
48          javadoc("-d", "out-multi",
49                  "--no-platform-links",
50                  "--module-source-path", testSrc,
51                  "--since", "0.9,v1.0,1.2,2.0b,3.2,5",
52                  "--since-label", "New API in recent releases",
53                  "--module", "mdl",
54                  "pkg");
55          checkExit(Exit.OK);
56          checkMultiReleaseContents();
57          checkMultiReleaseNewElements();
58          checkMultiReleaseDeprecatedElements();
59      }
60
