95      static void dumpArchive() throws Exception {
96          JarBuilder.build("classpathtests", "jdk/test/Main");
97          appJar = TestCommon.getTestJar("classpathtests.jar");
98  
99          JarBuilder.build("bootAppend",
100                          APP_MODULE_CLASS, BOOT_APPEND_MODULE_CLASS, BOOT_APPEND_CLASS);
101         bootAppendJar = TestCommon.getTestJar("bootAppend.jar");
102 
103         OutputAnalyzer output1  = TestCommon.dump(
104             appJar, TestCommon.list(ARCHIVE_CLASSES), "-Xbootclasspath/a:" + bootAppendJar);
105         TestCommon.checkDump(output1);
106 
107         if (!TestCommon.isUnableToMap(output1) &&
108             !CDSTestUtils.DYNAMIC_DUMP // these messages aren't available in dynamic dump model
109             ) {
110             // Make sure all the classes were successfully archived.
111             for (String archiveClass : ARCHIVE_CLASSES) {
112                 String msg = "Preload Warning: Cannot find " + archiveClass;
113                 if (archiveClass.equals(BOOT_APPEND_CLASS)) {
114                     // This class is in a package (sun/nio/cs/ext1) that doesn't belong to any
115                     // of the built-in modules. We can load it from -Xbootclasspath/a:
116                     output1.shouldNotContain(msg);
117                 } else {
118                     // This class belongs a package that belongs to a built-in module.
119                     // We shouldn't load it from -Xbootclasspath/a:
120                     output1.shouldContain(msg);
121                 }
122             }
123         }
124 
125         testArchiveName = TestCommon.getCurrentArchiveName();
126     }

128     // Test #1: A class in package defined in boot module
129     //    - should not be loaded from the -Xbootclasspath/a without AppCDS
130     public static void testBootAppendModuleClassWithoutAppCDS() throws Exception {
131         CDSOptions opts = (new CDSOptions())
132             .addPrefix("-Xbootclasspath/a:" + bootAppendJar, "-cp", appJar)
133             .setArchiveName(testArchiveName)
134             .addSuffix(MAIN_CLASS, "Test #1", BOOT_APPEND_MODULE_CLASS, "false");
135 
136         CDSTestUtils.runWithArchiveAndCheck(opts);
137     }

152     // Test #3: A class in excluded package defined in boot module
153     //     - should be loaded from the -Xbootclasspath/a by the boot classloader
154     public static void testBootAppendExcludedModuleClassWithoutAppCDS() throws Exception {
155         TestCommon.run(
156             "-Xbootclasspath/a:" + bootAppendJar, "-cp", appJar,
157             "-Xlog:class+load=info",
158             "--limit-modules", "java.base",
159             MAIN_CLASS, "Test #3", BOOT_APPEND_MODULE_CLASS, "true", "BOOT")
160             .assertSilentlyDisabledCDS(out -> {
161                 out.shouldHaveExitValue(0)
162                    .shouldMatch(".class.load. sun.nio.cs.ext.MyClass source:.*bootAppend.jar");
163             });
164     }
