48  /**
49   * Tests the order of the output of block tags in the generated output.
50   * There is a default order, embodied in the order of declaration of tags in
51   * {@code TagletManager}, but this can be overridden on the command line by
52   * specifying {@code -tag} options in the desired order.
53   */
54  public class TestTagOrder extends JavadocTester {
55      public static void main(String... args) throws Exception {
56          TestTagOrder tester = new TestTagOrder();
57          tester.runTests(m -> new Object[] { Path.of(m.getName()) });
58      }

60      ToolBox tb = new ToolBox();
61      Path src = Path.of("src");
62      Map<String, String> expectMethod = new LinkedHashMap<>();
63      Map<String, String> expectClass = new LinkedHashMap<>();
64  
65      TestTagOrder() throws IOException {
66          tb.writeJavaFiles(src,
67                  """
68                      package p;
69                      /** Class C1. */
70                      public class C1 {
71                          /**
72                           * This is method m.
73                           * @param p1 first parameter
74                           * @param p2 second parameter
75                           * @return zero
76                           * @throws IllegalArgumentException well, never
77                           * @since 1.0
78                           * @see <a href="http://example.com">example</a>
79                           */
80                          public int m(int p1, int p2) throws IllegalArgumentException {
81                              return 0;
82                          }
83                      }
84                      """, """
85                      package p;
86                      /**
87                       * Class C2.
88                       * @since 1.0
89                       * @author J. Duke.
90                       * @version 2.0
91                       * @see <a href="http://example.com">example</a>
92                       */
93                      public class C2 { }
94                      """);

96          // The following add map entries in the default order of appearance in the output.
97          // Note that the list is not otherwise ordered, such as alphabetically.
98  
99          expectMethod.put("@param", """

101                 <dd><code>p1</code> - first parameter</dd>

114 
115         expectMethod.put("@since", """

118                 """);
119 
120         expectMethod.put("@see", """

122                 <dd>
123                 <ul class="see-list">
124                 <li><a href="http://example.com">example</a></li>
125                 </ul>

131                 <dd>1.0</dd>

133

136                 <dd>2.0</dd>
137                 """);
138

140                 <dt>Author:</dt>

142                 """);

154     @Test
155     public void testDefault(Path base) {
156         test(base, null);
157     }
158 
159     @Test
160     public void testAlpha(Path base) {
161         test(base, Comparator.naturalOrder());
162     }
163 
164     @Test
165     public void testReverse(Path base) {
166         test(base, Comparator.reverseOrder());
167     }
168 
169     private void test(Path base, Comparator<String> c) {
170         List<String> args = new ArrayList<>();
171         args.addAll(List.of(
172                 "-d", base.resolve("out").toString(),
173                 "--source-path", src.toString(),
174                 "--no-platform-links",
175                 "-author",
176                 "-version"));
177         args.addAll(getTagArgs(c, expectMethod, expectClass));
178         args.add("p");
179 
180         javadoc(args.toArray(new String[0]));
181         checkExit(Exit.OK);
182 
183         checkOutput("p/C1.html", true,
184                 "<dl class=\"notes\">\n"
185                         + getExpectString(c, expectMethod)
186                         + "</dl>");
187 
188         checkOutput("p/C2.html", true,
189                 "<dl class=\"notes\">\n"
190                         + getExpectString(c, expectClass)
191                         + "</dl>");
192     }
193 
194     /**
195      * Returns a series of {@code -tag} options derived from the keys of a series of maps,
196      * sorted according to the given comparator, or an empty list if the comparator is {@code null}.
197      *
198      * @param c      the comparator, or {@code null}
199      * @param expect the maps from which to infer the options
200      *
201      * @return the list of options
202      */
203     @SafeVarargs
204     private List<String> getTagArgs(Comparator<String> c, Map<String, String>... expect) {
205         if (c == null) {
206             return List.of();
207         }
208 
209         SortedSet<String> allTags = new TreeSet<>(c);
210         for (Map<String, String> e : expect) {
211             allTags.addAll(e.keySet());
212         }
213         List<String> args = new ArrayList<>();
214         allTags.forEach(t -> { args.add("-tag"); args.add(t.substring(1)); });
215         return args;
216     }
217 
218     /**
219      * Returns the "expected string" derived from the values of a map, sorted according
220      * to the keys of the map with a given comparator if the comparator is not {@code null}.
221      *
222      * @param c      the comparator, or {@code null}
223      * @param expect the map
224      *
225      * @return the "expected string"
226      */
227     private String getExpectString(Comparator<String> c, Map<String, String> expect) {
228         Map<String, String> e;
229         if (c == null) {
230             e = expect;
231         } else {
232             e = new TreeMap<>(c);
233             e.putAll(expect);
234         }
235         return String.join("", e.values());
236     }
