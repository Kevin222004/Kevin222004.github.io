56      public static void main(String[] args) {
57  
58          boolean result = true;
59          for (String algo : new String[]{ALGO, OID}) {
60              System.out.println("With : " + algo);
61              result &= validateCert(algo, PROVIDER, PUBLIC_256);
62              result &= validateCert(algo, PROVIDER, PUBLIC_384);
63              result &= validateCert(algo, PROVIDER, PUBLIC_512);
64  
65              result &= validatePrivate(algo, PROVIDER, PRIVATE);
66          }
67          if (!result) {
68              throw new RuntimeException("Some test cases failed");
69          }
70      }

72      private static boolean validatePrivate(String algorithm, String provider,
73              String type) {
74  
75          try {
76              KeyFactory kf = KeyFactory.getInstance(algorithm, provider);
77              PKCS8EncodedKeySpec privSpec = new PKCS8EncodedKeySpec(
78                      Base64.getMimeDecoder().decode(type));
79              PrivateKey priv = kf.generatePrivate(privSpec);
80  
81              RSAPrivateCrtKey crtKey = (RSAPrivateCrtKey) priv;
82              PrivateKey priv1 = kf.generatePrivate(new RSAPrivateCrtKeySpec(
83                      crtKey.getModulus(),
84                      crtKey.getPublicExponent(),
85                      crtKey.getPrivateExponent(),
86                      crtKey.getPrimeP(),
87                      crtKey.getPrimeQ(),
88                      crtKey.getPrimeExponentP(),
89                      crtKey.getPrimeExponentQ(),
90                      crtKey.getCrtCoefficient(),
91                      crtKey.getParams()
92              ));
93              equals(priv, priv1);
94          } catch (NoSuchAlgorithmException | InvalidKeySpecException
95                  | NoSuchProviderException e) {
96              e.printStackTrace(System.out);
97              return false;
98          }
99          System.out.println("PASSED - validatePrivate");
100         return true;
101     }

103     private static boolean validateCert(String algorithm, String provider,
104             String type) {
105 
106         try {
107             CertificateFactory cf = CertificateFactory.getInstance("X.509");
108             Certificate cert = cf.generateCertificate(
109                     new ByteArrayInputStream(type.getBytes()));
110             System.out.println(cert);
111             KeyFactory kf = KeyFactory.getInstance(algorithm, provider);
112             X509EncodedKeySpec pubSpec = kf.getKeySpec(
113                     cert.getPublicKey(), X509EncodedKeySpec.class);
114             PublicKey pub = kf.generatePublic(pubSpec);
115             PublicKey pub1 = kf.generatePublic(new RSAPublicKeySpec(
116                     ((RSAPublicKey) pub).getModulus(),
117                     ((RSAPublicKey) pub).getPublicExponent(),
118                     ((RSAPublicKey) pub).getParams()));
119             equals(cert.getPublicKey(), pub);
120             equals(pub, pub1);
121         } catch (CertificateException | NoSuchAlgorithmException
122                 | InvalidKeySpecException | NoSuchProviderException e) {
123             e.printStackTrace(System.out);
124             return false;
125         }
126         System.out.println("PASSED - validateCert");
127         return true;
128     }

130     private static void equals(Key orig, Key gen) {
131         if (!orig.equals(gen) && orig.hashCode() != gen.hashCode()
132                 && !Arrays.equals(orig.getEncoded(), gen.getEncoded())) {
133             throw new RuntimeException("Key mismatch found");
134         }
135     }
