24  /*
25   * @test
26   * @summary Test writeable VM Options with ranges.
27   * @library /test/lib /runtime/CommandLine/OptionsValidation/common
28   * @modules java.base/jdk.internal.misc
29   *          jdk.attach/sun.tools.attach
30   *          java.management
31   * @run main/othervm -XX:MinHeapFreeRatio=0 -XX:MaxHeapFreeRatio=100 -Djdk.attach.allowAttachSelf TestOptionsWithRangesDynamic
32   */
33  
34  import java.util.List;
35  import jdk.test.lib.Asserts;
36  import optionsvalidation.JVMOption;
37  import optionsvalidation.JVMOptionsUtils;
38  
39  public class TestOptionsWithRangesDynamic {
40  
41      private static List<JVMOption> allWriteableOptions;
42  
43      private static void excludeTestRange(String optionName) {
44          for (JVMOption option: allWriteableOptions) {
45              if (option.getName().equals(optionName)) {
46                  option.excludeTestMinRange();
47                  option.excludeTestMaxRange();
48                  break;
49              }
50          }
51      }
52  
53      public static void main(String[] args) throws Exception {
54          int failedTests;
55  
56          /* Get only writeable options */
57          allWriteableOptions = JVMOptionsUtils.getOptionsWithRange(origin -> (origin.contains("manageable") || origin.contains("rw")));
58  
59          /*
60           * Exclude SoftMaxHeapSize as its valid range is only known at runtime.
61           */
62          excludeTestRange("SoftMaxHeapSize");
63  
64          Asserts.assertGT(allWriteableOptions.size(), 0, "Options with ranges not found!");
65  
66          System.out.println("Test " + allWriteableOptions.size() + " writeable options with ranges. Start test!");
67  
68          failedTests = JVMOptionsUtils.runDynamicTests(allWriteableOptions);
69  
70          failedTests += JVMOptionsUtils.runJcmdTests(allWriteableOptions);
71  
72          failedTests += JVMOptionsUtils.runAttachTests(allWriteableOptions);
73  
74          Asserts.assertEQ(failedTests, 0,
75                  String.format("%d tests failed! %s", failedTests, JVMOptionsUtils.getMessageWithFailures()));
76      }
77  }
