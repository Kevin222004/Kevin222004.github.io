51      /*
52       * Tests for dynamic archives should be written using this pattern:
53       *
54       * public class HelloDynamic extends DynamicArchiveTestBase {
55       *     public static void main(String[] args) throws Exception {
56       *        runTest(HelloDynamic::testDefaultBase); // launch one test case
57       *     }
58       *
59       *     // the body of a test case
60       *     static void testDefaultBase() throws Exception {
61       *         String topArchiveName = getNewArchiveName("top");
62       *         doTest(null, topArchiveName);
63       *     }
64       * }
65       *
66       * The reason for this is so that we can clean up the archive files
67       * created by prior test cases. Otherwise tests with lots of
68       * test cases may fill up the scratch directory.
69       */
70      public static void runTest(DynamicArchiveTest t) throws Exception {
71          executedIn_run = true;
72          try {
73              TestCommon.deletePriorArchives();
74              t.run();
75          } finally {
76              executedIn_run = false;
77          }
78      }
79  
80      public static void runTest(DynamicArchiveTestWithArgs t, String... args) throws Exception {
81          executedIn_run = true;
82          try {
83              TestCommon.deletePriorArchives();
84              t.run(args);
85          } finally {
86              executedIn_run = false;
87          }
88      }

97      /**
98       * Execute a JVM using the base archive (given by baseArchiveName) with the command line
99       * (given by cmdLineSuffix). At JVM exit, dump all eligible classes into the top archive
100      * (give by topArchiveName).
101      *
102      * If baseArchiveName is null, use the JDK's default archive as the base archive.
103      */
104     public static Result dump2(String baseArchiveName, String topArchiveName, String ... cmdLineSuffix)
105         throws Exception
106     {
107         String[] cmdLine = TestCommon.concat(
108             "-XX:ArchiveClassesAtExit=" + topArchiveName);
109         // to allow dynamic archive tests to be run in the "rt-non-cds-mode"
110         cmdLine = TestCommon.concat(cmdLine, "-Xshare:auto");
111         if (baseArchiveName == null && isUseSharedSpacesDisabled()) {
112             baseArchiveName = getTempBaseArchive();
113         }
114         if (baseArchiveName != null) {
115             cmdLine = TestCommon.concat(cmdLine, "-XX:SharedArchiveFile=" + baseArchiveName);
116         }
117         cmdLine = TestCommon.concat(cmdLine, cmdLineSuffix);
118         return execProcess("dump", null, cmdLine);
119     }

121     public static Result dump2_WB(String baseArchiveName, String topArchiveName, String ... cmdLineSuffix)
122         throws Exception
123     {
124         if (baseArchiveName == null && isUseSharedSpacesDisabled()) {
125             baseArchiveName = getTempBaseArchive();
126         }
127         return dump2(baseArchiveName, topArchiveName,
128                      TestCommon.concat(wbRuntimeArgs(), cmdLineSuffix));
129     }

153     private static String getWhiteBoxJar() {
154         String wbJar = ClassFileInstaller.getJarPath("WhiteBox.jar");
155         if (!(new File(wbJar)).exists()) {
156             throw new RuntimeException("Test error: your test must have " +
157                                        "'@run driver jdk.test.lib.helpers.ClassFileInstaller -jar WhiteBox.jar sun.hotspot.WhiteBox'");
158         }
159         return wbJar;
160     }

162     private static String[] wbRuntimeArgs() {
163         return TestCommon.concat("-Xbootclasspath/a:" + getWhiteBoxJar(),
164                                  "-XX:+UnlockDiagnosticVMOptions",
165                                  "-XX:+WhiteBoxAPI");

168     /**
169      * Execute a JVM using the base archive (given by baseArchiveName) and the top archive
170      * (give by topArchiveName), using the command line (given by cmdLineSuffix).
171      *
172      * If baseArchiveName is null, use the JDK's default archive as the base archive.
173      */
174     public static Result run2(String baseArchiveName, String topArchiveName, String ... cmdLineSuffix)
175         throws Exception {
176         if (baseArchiveName == null && topArchiveName == null) {
177             throw new RuntimeException("Both baseArchiveName and topArchiveName cannot be null at the same time.");
178         }
179         if (baseArchiveName == null && isUseSharedSpacesDisabled()) {
180             baseArchiveName = getTempBaseArchive();
181         }
182         String archiveFiles = (baseArchiveName == null) ? topArchiveName :
183             (topArchiveName == null) ? baseArchiveName :
184             baseArchiveName + File.pathSeparator + topArchiveName;
185         String[] cmdLine = TestCommon.concat(
186             "-Xshare:on",
187             "-XX:SharedArchiveFile=" + archiveFiles);
188         cmdLine = TestCommon.concat(cmdLine, cmdLineSuffix);
189         return execProcess("exec", null, cmdLine);
190     }

194         throws Exception {
195         if (baseArchiveName == null && topArchiveName == null) {

197         }
198         if (baseArchiveName == null && isUseSharedSpacesDisabled()) {
199             baseArchiveName = getTempBaseArchive();
200         }
201         String archiveFiles = (baseArchiveName == null) ? topArchiveName :
202             (topArchiveName == null) ? baseArchiveName :
203             baseArchiveName + File.pathSeparator + topArchiveName;

205             "-Xshare:on",
206             "-XX:SharedArchiveFile=" + archiveFiles);
207         cmdLine = TestCommon.concat(cmdLine, cmdLineSuffix);
208         return execProcess("exec", jarDir, cmdLine);
209     }

230     private static String getXshareMode(String[] cmdLine) {
231         for (int i = 0; i <= cmdLine.length - 1; i++) {
232             int j = cmdLine[i].indexOf("-Xshare:");
233             if (j != -1) {
234                 return (cmdLine[i].substring(j));
235             }
236         }
237         return null;
238    }

241     private static Result execProcess(String mode, String jarDir, String[] cmdLine) throws Exception {
242         if (!executedIn_run) {
243             throw new Exception("Test error: dynamic archive tests must be executed via DynamicArchiveTestBase.run()");
244         }
245         ProcessBuilder pb = ProcessTools.createTestJvm(cmdLine);
246         if (jarDir != null) {
247             pb.directory(new File(jarDir));
248         }
249         OutputAnalyzer output = TestCommon.executeAndLog(pb, mode);
250         CDSOptions opts = new CDSOptions();
251         String xShareMode = getXshareMode(cmdLine);
252         if (xShareMode != null) {
253             opts.setXShareMode(xShareMode);
254         }
255         return new Result(opts, output);
256     }

258     /**
259      * A convenience method for dumping and running, using the default CDS archive from the
260      * JDK. Both dumping and running should exit normally.
261      */
262     public static void dumpAndRun(String topArchiveName, String ... cmdLineSuffix) throws Exception {
263         dump(topArchiveName, cmdLineSuffix).assertNormalExit();
264         run(topArchiveName,  cmdLineSuffix).assertNormalExit();
265     }

267     private static String tempBaseArchive;
268     /**
269      * Return the name of a base archive.
270      * It will generate one if one doesn't exist.
271      */
272     private static String getTempBaseArchive() throws Exception {
273         if (tempBaseArchive == null) {
274             tempBaseArchive = getNewArchiveName("tempBaseArchive");
275             TestCommon.dumpBaseArchive(tempBaseArchive);
276         }
277         return tempBaseArchive;
278     }

280     /**
281      * Return true if the UseSharedSpaces flag has been disabled.
282      * By default, the VM will be started with -Xshare:auto.
283      * The UseSharedSpaces flag will be disabled by the VM if there's some
284      * problem in using the default CDS archive. It could happen under some
285      * situations such as follows:
286      * - the default CDS archive wasn't generated during build time because
287      *   the JDK was built via cross-compilation on a different platform;
288      * - the VM under test was started with a different options than the ones
289      *   when the default CDS archive was built. E.g. the VM was started with
290      *   -XX:+UseZGC which implicitly disabled the UseCompressedOoops and the
291      *   UseCompressedClassPointers options. Those "compressed" options were
292      *   enabled when the default CDS archive was built.
293      */
294     private static boolean isUseSharedSpacesDisabled() {
295         return (WB.getBooleanVMFlag("UseSharedSpaces") == false);
296     }
