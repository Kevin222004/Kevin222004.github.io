24  /**
25   * @test SharedBaseAddress
26   * @bug 8265705 8267351
27   * @summary Test variety of values for SharedBaseAddress, making sure
28   *          VM handles normal values as well as edge values w/o a crash.
29   * @requires vm.cds
30   * @library /test/lib
31   * @run driver SharedBaseAddress
32   */
33  
34  import jdk.test.lib.cds.CDSTestUtils;
35  import jdk.test.lib.cds.CDSOptions;
36  import jdk.test.lib.Platform;
37  import jdk.test.lib.process.OutputAnalyzer;
38  
39  public class SharedBaseAddress {
40  
41      // shared base address test table
42      private static final String[] testTable = {
43          "1g", "8g", "64g","512g", "4t",
44          "32t", "128t", "0",
45          "1", "64k", "64M",
46          "0x800001000",        // Default base address + 1 page - probably valid but unaligned to metaspace alignment, see JDK 8247522
47          "0xfffffffffff00000", // archive top wraps around 64-bit address space
48          "0xfff80000",         // archive top wraps around 32-bit address space
49          "0xffffffffffffffff", // archive bottom wraps around 64-bit address space -- due to align_up()
50          "0xffffffff",         // archive bottom wraps around 32-bit address space -- due to align_up()
51          "0x00007ffffff00000", // end of archive will go past the end of user space on linux/x64
52          "0x500000000",        // (20g) below 32g at a 4g aligned address, but cannot be expressed with a logical
53                                //    immediate on aarch64 (0x5_0000_0000) (see JDK-8265705)
54          "0",                  // always let OS pick the base address at runtime (ASLR for CDS archive)
55      };
56  
57      // failed pattern
58      private static String failedPattern = "os::release_memory\\(0x[0-9a-fA-F]*,\\s[0-9]*\\)\\sfailed";
59  
60      public static void main(String[] args) throws Exception {
61  
62          for (int run = 0; run < 2; run ++) {
63              // We run twice:
64              // Once, where we want to increase the chance that mapping the generated archive at the designated base
65              // succeeds, to test Klass pointer encoding at that weird location. We do this by sizing heap + class space
66              // small, and by switching off compressed oops.
67              // On the second run, we don't do this but instead go with default parameters. This is more of a test of
68              // CDS' ability to recover if mapping at runtime fails.
69              for (String testEntry : testTable) {
70                  String filename = "SharedBaseAddress-base" + testEntry + "-run" + run + ".jsa";
71                  System.out.println("sharedBaseAddress = " + testEntry);
72                  CDSOptions opts = (new CDSOptions())
73                          .setArchiveName(filename)
74                          .addPrefix("-XX:SharedBaseAddress=" + testEntry)
75                          .addPrefix("-Xlog:cds=debug")
76                          .addPrefix("-Xlog:cds+reloc=debug")
77                          .addPrefix("-Xlog:nmt=debug")
78                          .addPrefix("-Xlog:os=debug")
79                          .addPrefix("-Xlog:gc+metaspace")
80                          .addPrefix("-XX:NativeMemoryTracking=detail");
81  
82                  if (run == 0 && Platform.is64bit()) {
83                      opts.addPrefix("-Xmx128m")
84                          .addPrefix("-XX:CompressedClassSpaceSize=32m")
85                          .addPrefix("-XX:-UseCompressedOops");
86                  }
87                  CDSTestUtils.createArchiveAndCheck(opts);
88                  OutputAnalyzer out = CDSTestUtils.runWithArchiveAndCheck(opts);
89                  if (testEntry.equals("0")) {
90                      out.shouldContain("Archive(s) were created with -XX:SharedBaseAddress=0. Always map at os-selected address.")
91                              .shouldContain("Try to map archive(s) at an alternative address")
92                              .shouldNotMatch(failedPattern);
93                  }
94              }
95          }
96      }
97  }
