24  /**
25   * @test
26   * @bug 4361783
27   * @key intermittent
28   * @summary Test to see if ICMP Port Unreachable on non-connected
29   *          DatagramSocket causes a SocketException "socket closed"
30   *          exception on Windows 2000.
31   * @run main/othervm PortUnreachable
32   */
33  
34  import java.net.BindException;
35  import java.net.DatagramPacket;
36  import java.net.DatagramSocket;
37  import java.net.InetAddress;
38  import java.net.InetSocketAddress;
39  
40  public class PortUnreachable {
41  
42      DatagramSocket clientSock;
43      int serverPort;
44      int clientPort;
45  
46      public void serverSend() throws Exception {
47          InetAddress addr = InetAddress.getLocalHost();
48          Thread.sleep(1000);
49          // send a delayed packet which should mean a delayed icmp
50          // port unreachable
51          byte b[] = "A late msg".getBytes();
52          DatagramPacket packet = new DatagramPacket(b, b.length, addr,
53                  serverPort);
54          clientSock.send(packet);
55  
56          DatagramSocket sock = recreateServerSocket(serverPort);
57          b = "Greetings from the server".getBytes();
58          packet = new DatagramPacket(b, b.length, addr, clientPort);
59          sock.send(packet);
60          Thread.sleep(500);  // give time to the kernel to send packet
61          sock.close();
62      }
63  
64      DatagramSocket recreateServerSocket (int serverPort) throws Exception {
65          DatagramSocket serverSocket = null;
66          int retryCount = 0;
67          long sleeptime = 0;
68          System.out.println("Attempting to recreate server socket with port: " +
69                  serverPort);
70          // it's possible that this method intermittently fails, if some other
71          // process running on the machine grabs the port we want before us,
72          // and doesn't release it before the 10 * 500 ms are elapsed...
73          while (serverSocket == null) {
74              try {
75                  serverSocket = new DatagramSocket(serverPort, InetAddress.getLocalHost());
76              } catch (BindException bEx) {
77                  if (retryCount++ < 10) {
78                      sleeptime += sleepAtLeast(500);
79                  } else {
80                      System.out.println("Give up after 10 retries and " + sleeptime(sleeptime));
81                      System.out.println("Has some other process grabbed port " + serverPort + "?");
82                      throw bEx;
83                  }
84              }
85          }
86  
87          System.out.println("PortUnreachableTest.recreateServerSocket: returning socket == "
88                  + serverSocket.getLocalAddress() + ":" + serverSocket.getLocalPort()
89                  + " obtained at " + attempt(retryCount) + " attempt with " + sleeptime(sleeptime));
90          return serverSocket;
91      }
92  
93      long sleepAtLeast(long millis) throws Exception {
94          long start = System.nanoTime();
95          long ms = millis;
96          while (ms > 0) {
97              assert ms < Long.MAX_VALUE/1000_000L;
98              Thread.sleep(ms);
99              long elapsedms = (System.nanoTime() - start)/1000_000L;
100             ms = millis - elapsedms;
101         }
102         return millis - ms;
103     }
104 
105     String attempt(int retry) {
106         switch (retry) {
107             case 0: return "first";
108             case 1: return "second";
109             case 2: return "third";
110             default: return retry + "th";
111         }
112     }
113 
114     String sleeptime(long millis) {
115         if (millis == 0) return "no sleep";
116         long sec = millis / 1000L;
117         long ms =  millis % 1000L;
118         String sleeptime = "";
119         if (millis > 0) {
120            if (sec > 0) {
121                sleeptime = "" + sec + " s" +
122                    (ms > 0 ? " " : "");
123             }
124             if (ms > 0 ) {
125                 sleeptime += ms + " ms";
126             }
127         } else sleeptime = millis + " ms"; // should not happen
128         return sleeptime + " of sleep time";
129     }
130 
131     PortUnreachable() throws Exception {
132         clientSock = new DatagramSocket(new InetSocketAddress(InetAddress.getLocalHost(), 0));
133         clientPort = clientSock.getLocalPort();
134 
135     }
136 
137     void execute () throws Exception{
138 
139         // pick a port for the server
140         DatagramSocket sock2 = new DatagramSocket(new InetSocketAddress(InetAddress.getLocalHost(), 0));
141         serverPort = sock2.getLocalPort();
142 
143         // send a burst of packets to the unbound port - we should get back
144         // icmp port unreachable messages
145         //
146         InetAddress addr = InetAddress.getLocalHost();
147         byte b[] = "Hello me".getBytes();
148         DatagramPacket packet = new DatagramPacket(b, b.length, addr,
149                                                    serverPort);
150         //close just before sending
151         sock2.close();
152         for (int i=0; i<100; i++)
153             clientSock.send(packet);
154 
155         serverSend();
156 
157         // try to receive
158         b = new byte[25];
159         packet = new DatagramPacket(b, b.length, addr, serverPort);
160         clientSock.setSoTimeout(10000);
161         clientSock.receive(packet);
162         System.out.println("client received data packet " + new String(packet.getData()));
163 
164         // done
165         clientSock.close();
166     }
167 
168     public static void main(String[] args) throws Exception {
169         // A BindException might be thrown intermittently. In that case retry
170         // 3 times before propagating the exception to finish execution.
171         int catchCount = 0;
172 
173         while (true) {
174             try {
175                 PortUnreachable test = new PortUnreachable();
176                 test.execute();
177                 return;
178             } catch (BindException bEx) {
179                 System.out.println("Failed to bind server: " + bEx);
180                 if (++catchCount > 3) {
181                     System.out.printf("Max retry count exceeded (%d)%n", catchCount);
182                     throw bEx;
183                 }
184                 System.out.printf("Retrying; retry count: %d%n", catchCount);
185             }
186         }
187     }
188 }
