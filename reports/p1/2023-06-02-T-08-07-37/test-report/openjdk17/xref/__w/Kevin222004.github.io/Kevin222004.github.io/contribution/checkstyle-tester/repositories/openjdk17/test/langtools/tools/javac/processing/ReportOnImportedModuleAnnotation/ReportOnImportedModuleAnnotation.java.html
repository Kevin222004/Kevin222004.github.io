24  /*
25   * @test
26   * @bug 8235457
27   *      8235458
28   * @summary javac shouldn't fail when an annotation processor report a message about an annotation on a module
29   *          javac should process annotated module when imports statement are present
30   * @modules jdk.compiler
31   */
32  
33  import java.io.PrintWriter;
34  import java.io.StringWriter;
35  import java.nio.file.Files;
36  import java.nio.file.Path;
37  import java.util.List;
38  
39  import javax.tools.JavaCompiler;
40  import javax.tools.StandardJavaFileManager;
41  import javax.tools.StandardLocation;
42  import javax.tools.ToolProvider;
43  
44  public class ReportOnImportedModuleAnnotation {
45  
46      public static void main(String[] args) throws Exception {
47          final Path testBasePath = Path.of(System.getProperty("test.src"));
48          final Path testOutputPath = Path.of(System.getProperty("test.classes"));
49  
50          final JavaCompiler compiler = ToolProvider.getSystemJavaCompiler();
51  
52          // Compile annotation and processor modules
53          StandardJavaFileManager fileManager = compiler.getStandardFileManager(null, null, null);
54          fileManager.setLocationFromPaths(StandardLocation.MODULE_SOURCE_PATH, List.of(testBasePath.resolve("mods-src1/")));
55          fileManager.setLocationFromPaths(StandardLocation.CLASS_OUTPUT, List.of(testOutputPath));
56          compiler.getTask(new PrintWriter(System.out), fileManager, null, List.of("--module", "annotation,processor"), null, null).call();
57  
58          // Compile mod modules
59          fileManager = compiler.getStandardFileManager(null, null, null);
60          fileManager.setLocationFromPaths(StandardLocation.MODULE_SOURCE_PATH, List.of(testBasePath.resolve("mods-src2/")));
61          fileManager.setLocationFromPaths(StandardLocation.MODULE_PATH, List.of(testOutputPath.resolve("annotation")));
62          fileManager.setLocationFromPaths(StandardLocation.ANNOTATION_PROCESSOR_MODULE_PATH, List.of(testOutputPath.resolve("processor")));
63          fileManager.setLocationFromPaths(StandardLocation.CLASS_OUTPUT, List.of(testOutputPath));
64  
65          final StringWriter outputWriter = new StringWriter();
66          compiler.getTask(outputWriter, fileManager, null, List.of("-XDrawDiagnostics", "--module", "mod"), null, null).call();
67  
68          String actualOutput = outputWriter.toString();
69          String expectedOutput = Files.readString(testBasePath.resolve("ReportOnImportedModuleAnnotation.out"));
70  
71          String lineSep = System.getProperty("line.separator");
72          if(!actualOutput.replace(lineSep, "\n").equals(expectedOutput.replace(lineSep, "\n"))) {
73              System.err.println("Expected: [" + expectedOutput + "]");
74              System.err.println("Received: [" + actualOutput + "]");
75              throw new Exception("Invalid output");
76          }
77      }
78  }
