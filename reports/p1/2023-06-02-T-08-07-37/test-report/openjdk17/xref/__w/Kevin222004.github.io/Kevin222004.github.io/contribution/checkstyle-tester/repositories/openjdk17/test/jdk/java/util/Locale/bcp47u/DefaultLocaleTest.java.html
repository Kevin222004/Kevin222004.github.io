24  import java.util.Locale;
25  
26  /*
27   * Test application that verifies default locales. Invoked from
28   * SystemPropertyTests
29   */
30  public class DefaultLocaleTest {
31      public static void main(String... args) {
32          String defLoc = Locale.getDefault().toString();
33          String defFmtLoc = Locale.getDefault(Locale.Category.FORMAT).toString();
34          String defDspLoc = Locale.getDefault(Locale.Category.DISPLAY).toString();
35  
36          if (!defLoc.equals(args[0]) ||
37              !defFmtLoc.equals(args[1]) ||
38              !defDspLoc.equals(args[2])) {
39              System.err.println("Some default locale(s) don't match.\n" +
40                  "Default Locale expected: " + args[0] + ", result: " + defLoc + "\n" +
41                  "Default Format Locale expected: " + args[1] + ", result: " + defFmtLoc + "\n" +
42                  "Default Display Locale expected: " + args[2] + ", result: " + defDspLoc);
43              System.exit(-1);
44          }
45      }
46  }
