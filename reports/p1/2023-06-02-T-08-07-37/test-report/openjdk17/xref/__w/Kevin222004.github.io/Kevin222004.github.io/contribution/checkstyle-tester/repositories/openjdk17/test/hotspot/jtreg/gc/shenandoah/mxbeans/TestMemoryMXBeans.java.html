25  /**
26   * @test TestMemoryMXBeans
27   * @summary Test JMX memory beans
28   * @requires vm.gc.Shenandoah
29   * @modules java.base/jdk.internal.misc
30   *          java.management
31   * @run main/othervm -XX:+UnlockExperimentalVMOptions -XX:+UseShenandoahGC          -Xmx1g TestMemoryMXBeans   -1 1024
32   * @run main/othervm -XX:+UnlockExperimentalVMOptions -XX:+UseShenandoahGC -Xms1g   -Xmx1g TestMemoryMXBeans 1024 1024
33   * @run main/othervm -XX:+UnlockExperimentalVMOptions -XX:+UseShenandoahGC -Xms128m -Xmx1g TestMemoryMXBeans  128 1024
34   * @run main/othervm -XX:+UnlockExperimentalVMOptions -XX:+UseShenandoahGC -Xms1g   -Xmx1g -XX:ShenandoahUncommitDelay=0 TestMemoryMXBeans 1024 1024
35   * @run main/othervm -XX:+UnlockExperimentalVMOptions -XX:+UseShenandoahGC -Xms128m -Xmx1g -XX:ShenandoahUncommitDelay=0 TestMemoryMXBeans  128 1024
36   */
37  
38  import java.lang.management.*;
39  import java.util.*;
40  
41  public class TestMemoryMXBeans {
42  
43      public static void main(String[] args) throws Exception {
44          if (args.length < 2) {
45              throw new IllegalStateException("Should provide expected heap sizes");
46          }
47  
48          long initSize = 1L * Integer.parseInt(args[0]) * 1024 * 1024;
49          long maxSize  = 1L * Integer.parseInt(args[1]) * 1024 * 1024;
50  
51          // wait for GC to uncommit
52          Thread.sleep(1000);
53  
54          testMemoryBean(initSize, maxSize);
55      }
56  
57      public static void testMemoryBean(long initSize, long maxSize) {
58          MemoryMXBean memoryMXBean = ManagementFactory.getMemoryMXBean();
59          long heapInit = memoryMXBean.getHeapMemoryUsage().getInit();
60          long heapCommitted = memoryMXBean.getHeapMemoryUsage().getCommitted();
61          long heapMax = memoryMXBean.getHeapMemoryUsage().getMax();
62          long nonHeapInit = memoryMXBean.getNonHeapMemoryUsage().getInit();
63          long nonHeapCommitted = memoryMXBean.getNonHeapMemoryUsage().getCommitted();
64          long nonHeapMax = memoryMXBean.getNonHeapMemoryUsage().getMax();
65  
66          if (initSize > 0 && heapInit != initSize) {
67              throw new IllegalStateException("Init heap size is wrong: " + heapInit + " vs " + initSize);
68          }
69          if (maxSize > 0 && heapMax != maxSize) {
70              throw new IllegalStateException("Max heap size is wrong: " + heapMax + " vs " + maxSize);
71          }
72          if (initSize > 0 && maxSize > 0 && initSize != maxSize && heapCommitted == heapMax) {
73              throw new IllegalStateException("Committed heap size is max: " + heapCommitted +
74                                              " (init: " + initSize + ", max: " + maxSize + ")");
75          }
76          if (initSize > 0 && maxSize > 0 && initSize == maxSize && heapCommitted != heapMax) {
77              throw new IllegalStateException("Committed heap size is not max: " + heapCommitted +
78                                              " (init: " + initSize + ", max: " + maxSize + ")");
79          }
80          if (initSize > 0 && heapCommitted < initSize) {
81              throw new IllegalStateException("Committed heap size is below min: " + heapCommitted +
82                                              " (init: " + initSize + ", max: " + maxSize + ")");
83          }
84      }
85  }
