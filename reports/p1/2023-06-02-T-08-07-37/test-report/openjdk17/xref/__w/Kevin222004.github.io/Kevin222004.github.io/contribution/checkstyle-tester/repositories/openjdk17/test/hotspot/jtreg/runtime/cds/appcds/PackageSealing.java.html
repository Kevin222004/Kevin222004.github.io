25  /*
26   * @test
27   * @summary AppCDS handling of package.
28   * @requires vm.cds
29   * @library /test/lib
30   * @compile test-classes/C1.java test-classes/C2.java test-classes/C3.java
31   *          test-classes/PackageSealingTest.java test-classes/Hello.java
32   * @run driver PackageSealing
33   */
34  
35  import java.io.File;
36  import jdk.test.lib.process.OutputAnalyzer;
37  import jdk.test.lib.helpers.ClassFileInstaller;
38  
39  public class PackageSealing {
40      public static void main(String args[]) throws Exception {
41          String[] classList = {"foo/C1", "pkg/C2", "PackageSealingTest"};
42          String appJar = ClassFileInstaller.writeJar("foo-sealed.jar",
43              ClassFileInstaller.Manifest.fromSourceFile("test-classes/package_seal.mf"),
44              "PackageSealingTest", "foo/C1", "pkg/C2");
45  
46          String helloJar = JarBuilder.getOrCreateHelloJar();
47          String jars = helloJar + File.pathSeparator + appJar;
48  
49          // test shared package from -cp path
50          TestCommon.testDump(jars, TestCommon.list(classList));
51          OutputAnalyzer output;
52          output = TestCommon.exec(jars, "PackageSealingTest",
53                                   "foo/C1", "sealed", "pkg/C2", "notSealed");
54          TestCommon.checkExec(output, "OK");
55  
56          // test shared package from -Xbootclasspath/a
57          TestCommon.dump(helloJar, TestCommon.list(classList),
58                          "-Xbootclasspath/a:" + appJar);
59          output = TestCommon.exec(helloJar, "-Xbootclasspath/a:" + appJar,
60                                   "PackageSealingTest",
61                                   "foo/C1", "sealed", "pkg/C2", "notSealed");
62          TestCommon.checkExec(output, "OK");
63  
64          // Test loading of two classes from the same package from different jars.
65          // First loaded class is from a non-sealed package, the second loaded
66          // class is from the same package but sealed.
67          // The expected result is a SecurityException with a "sealing violation"
68          // for the second class.
69          classList[1] = "foo/C3"; // C3 is from a non-sealed package
70          String[] classList2 = {"foo/C3", "foo/C1", "PackageSealingTest"};
71          String nonSealedJar = ClassFileInstaller.writeJar("foo-unsealed.jar", "foo/C3");
72          jars = helloJar + File.pathSeparator + nonSealedJar;
73          TestCommon.testDump(jars, TestCommon.list(classList2));
74          jars += File.pathSeparator + appJar;
75          output = TestCommon.exec(jars, "-Xlog:class+load", "PackageSealingTest",
76                                   "foo/C3", "notSealed", "foo/C1", "sealed");
77          TestCommon.checkExec(output,
78                               "foo.C3 source: shared objects file",
79                               "sealing violation: can't seal package foo: already defined");
80  
81          // Use the jar with the sealed package during dump time.
82          // Reverse the class loading order during runtime: load the class in the
83          // sealed package following by another class in the same package but unsealed.
84          // Same "sealing violation should result in loading the second class.
85          jars = helloJar + File.pathSeparator + appJar;
86          TestCommon.testDump(jars, TestCommon.list(classList2));
87          jars += File.pathSeparator + nonSealedJar;
88          output = TestCommon.exec(jars, "-Xlog:class+load", "PackageSealingTest",
89                                   "foo/C1", "sealed", "foo/C3", "notSealed");
90          TestCommon.checkExec(output,
91                               "foo.C1 source: shared objects file",
92                               "sealing violation: package foo is sealed");
93      }
94  }
