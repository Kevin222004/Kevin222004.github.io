24  /*
25   * @test
26   * @bug 7020499
27   * @summary Verify that the close resource code works properly in all cases
28   * @library /tools/lib
29   * @modules jdk.compiler/com.sun.tools.javac.api
30   *          jdk.compiler/com.sun.tools.javac.main
31   * @build toolbox.ToolBox TwrClose
32   * @run main TwrClose
33   */
34  
35  import javax.tools.JavaFileManager;
36  import javax.tools.StandardLocation;
37  import javax.tools.ToolProvider;
38  
39  import toolbox.JavacTask;
40  import toolbox.ToolBox;
41  
42  public class TwrClose {
43  
44      private static final int MAX_RESOURCES = 5;
45      public static void main(String... args) throws Exception {
46          for (int i = 1; i < MAX_RESOURCES * 2; i++) {
47              new TwrClose().compile(i);
48          }
49      }
50  
51      ToolBox tb = new ToolBox();
52      JavaFileManager fm = ToolProvider.getSystemJavaCompiler()
53                                       .getStandardFileManager(null, null, null);
54  
55      void compile(int trysCount) throws Exception {
56          StringBuilder testInvocations = new StringBuilder();
57          StringBuilder testMethods = new StringBuilder();
58  
59          for (int i = 0; i < trysCount; i++) {
60              testInvocations.append(TEST_INVOCATIONS_TEMPLATE.replace("#N", Integer.toString(i)));
61              testMethods.append(TEST_METHOD_TEMPLATE.replace("#N", Integer.toString(i)));
62          }
63  
64          String sourceCode = FILE_TEMPLATE.replace("#TEST_INVOCATIONS", testInvocations.toString())
65                                           .replace("#TEST_METHODS", testMethods.toString());
66  
67          System.err.println("analyzing:");
68          System.err.println(sourceCode);
69  
70          try (ToolBox.MemoryFileManager mfm = new ToolBox.MemoryFileManager()) {
71              new JavacTask(tb).fileManager(mfm)
72                               .sources(sourceCode)
73                               .run()
74                               .writeAll();
75              ClassLoader cl = new ClassLoader(TwrClose.class.getClassLoader()) {
76                  @Override
77                  protected Class<?> findClass(String name) throws ClassNotFoundException {
78                      byte[] data = mfm.getFileBytes(StandardLocation.CLASS_OUTPUT, name);
79                      if (data != null) {
80                          return defineClass(name, data, 0, data.length);
81                      }
82                      return super.findClass(name);
83                  }
84              };
85  
86              ((Runnable) cl.loadClass("Test").newInstance()).run();
87          }
88      }
89  
90      final String TEST_INVOCATIONS_TEMPLATE =
91          "        test#N(false, false, Arrays.asList(\"close\"));\n" +
92          "        test#N(false, true, Arrays.asList(\"close\", \"close-exception\"));\n" +
93          "        test#N(true, false, Arrays.asList(\"close\", \"inTwr\"));\n" +
94          "        test#N(true, true, Arrays.asList(\"close\", \"inTwr\", \"close-exception\"));\n";
95  
96      final String TEST_METHOD_TEMPLATE =
97          "    private void test#N(boolean failInTwr, boolean failOnClose,\n" +
98          "                        List<String> expectedMessages) {\n" +
99          "        List<String> messages = new ArrayList<>();\n" +
100         "        try {\n" +
101         "            try (CloseableImpl c = new CloseableImpl(messages, failOnClose)) {\n" +
102         "                if (failInTwr)\n" +
103         "                    throw new IllegalStateException(\"inTwr\");\n" +
104         "            }\n" +
105         "        } catch (IllegalStateException ex) {\n" +
106         "            messages.add(ex.getMessage());\n" +
107         "            for (Throwable t : ex.getSuppressed()) {\n" +
108         "                messages.add(t.getMessage());\n" +
109         "            }\n" +
110         "        }\n" +
111         "        if (!expectedMessages.equals(messages))\n" +
112         "            throw new AssertionError(\"Expected and actual messages differ; expectedMessages=\" +\n" +
113         "                                     expectedMessages + \"; actual=\" + messages);\n" +
114         "    }\n";
115 
116     final String FILE_TEMPLATE =
117         "import java.util.*;\n" +
118         "public class Test implements Runnable {\n" +
119         "    public void run() {\n" +
120         "#TEST_INVOCATIONS" +
121         "    }\n" +
122         "#TEST_METHODS" +
123         "    static class CloseableImpl implements AutoCloseable {\n" +
124         "        private final List<String> messages;\n" +
125         "        private final boolean failOnClose;\n" +
126         "        public CloseableImpl(List<String> messages, boolean failOnClose) {\n" +
127         "            this.messages = messages;\n" +
128         "            this.failOnClose = failOnClose;\n" +
129         "        }\n" +
130         "        @Override\n" +
131         "        public void close() {\n" +
132         "            messages.add(\"close\");\n" +
133         "            if (failOnClose)\n" +
134         "                throw new IllegalStateException(\"close-exception\");\n" +
135         "        }\n" +
136         "    }\n" +
137         "}\n";
138 }
