234     @Test(dataProvider = "all")
235     void test(Schemes scheme, HttpClient.Version version, String uri, boolean async)
236             throws Throwable
237     {
238         checkSkip();
239         var name = String.format("test(%s, %s, %s)", scheme, version, async);
240         out.printf("%n---- starting %s ----%n", name);
241 
242         for (int i=0; i<ITERATIONS; i++) {
243             if (ITERATIONS > 1) out.printf("---- ITERATION %d%n",i);
244             try {
245                 doTest(scheme, version, uri, async);
246                 long count = sleepCount.incrementAndGet();
247                 System.err.println(now() + " Sleeping: " + count);
248                 Thread.sleep(SLEEP_AFTER_TEST);
249                 System.err.println(now() + " Waking up: " + count);
250             } catch (Throwable x) {
251                 FAILURES.putIfAbsent(name, x);
252                 throw x;
253             }
254         }
255     }

257     private <T> HttpResponse<T> send(HttpClient client,
258                                      URI uri,
259                                      HttpResponse.BodyHandler<T> handler,
260                                      boolean async) throws Throwable {
261         HttpRequest.Builder requestBuilder = HttpRequest
262                 .newBuilder(uri)
263                 .GET();
264 
265         HttpRequest request = requestBuilder.build();
266         out.println("Sending request: " + request.uri());
267 
268         HttpResponse<T> response = null;
269         if (async) {
270             response = client.send(request, handler);
271         } else {
272             try {
273                 response = client.sendAsync(request, handler).get();
274             } catch (ExecutionException ex) {
275                 throw ex.getCause();
276             }
277         }
278         return response;
279     }

281     private void doTest(Schemes scheme,
282                         HttpClient.Version version,
283                         String uriString,
284                         boolean async) throws Throwable {
285 
286         URI uri1 = URI.create(uriString + "/server/ProxySelectorTest");
287         URI uri2 = URI.create(uriString + "/proxy/noauth/ProxySelectorTest");
288         URI uri3 = URI.create(uriString + "/proxy/auth/ProxySelectorTest");
289 
290         HttpResponse<String> response;
291 
292         // First request should go with a direct connection.
293         // A plain server or https server should serve it, and we should get 200 OK
294         response = send(client, uri1, BodyHandlers.ofString(), async);
295         out.println("Got response from plain server: " + response);
296         assertEquals(response.statusCode(), HTTP_OK);
297         assertEquals(response.headers().firstValue("X-value"),
298                 scheme == Schemes.HTTPS ? Optional.of("https-server") : Optional.of("plain-server"));
299 
300         // Second request should go through a non authenticating proxy.
301         // For a clear connection - a proxy-server should serve it, and we should get 200 OK
302         // For an https connection - a tunnel should be established through the non
303         // authenticating proxy - and we should receive 200 OK from an https-server
304         response = send(client, uri2, BodyHandlers.ofString(), async);
305         out.println("Got response through noauth proxy: " + response);
306         assertEquals(response.statusCode(), HTTP_OK);
307         assertEquals(response.headers().firstValue("X-value"),
308                 scheme == Schemes.HTTPS ? Optional.of("https-server") : Optional.of("proxy-server"));
309 
310         // Third request should go through an authenticating proxy.
311         // For a clear connection - an auth-proxy-server should serve it, and we
312         // should get 407
313         // For an https connection - a tunnel should be established through an
314         // authenticating proxy - and we should receive 407 directly from the
315         // proxy - so the X-value header will be absent
316         response = send(client, uri3, BodyHandlers.ofString(), async);
317         out.println("Got response through auth proxy: " + response);
318         assertEquals(response.statusCode(), PROXY_UNAUTHORIZED);
319         assertEquals(response.headers().firstValue("X-value"),
320                 scheme == Schemes.HTTPS ? Optional.empty() : Optional.of("auth-proxy-server"));
321 
322     }

326     @BeforeTest
327     public void setup() throws Exception {
328         sslContext = new SimpleSSLContext().get();
329         if (sslContext == null)
330             throw new AssertionError("Unexpected null sslContext");
331 
332         InetSocketAddress sa = new InetSocketAddress(InetAddress.getLoopbackAddress(), 0);
333 
334         httpTestServer = HttpTestServer.of(HttpServer.create(sa, 0));
335         httpTestServer.addHandler(new PlainServerHandler("plain-server"), "/http1/");
336         httpURI = "http://" + httpTestServer.serverAuthority() + "/http1";
337         proxyHttpTestServer = HttpTestServer.of(HttpServer.create(sa, 0));
338         proxyHttpTestServer.addHandler(new PlainServerHandler("proxy-server"), "/http1/proxy/");
339         proxyHttpTestServer.addHandler(new PlainServerHandler("proxy-server"), "/http2/proxy/");
340         proxyHttpURI = "http://" + httpTestServer.serverAuthority() + "/http1";
341         authProxyHttpTestServer = HttpTestServer.of(HttpServer.create(sa, 0));
342         authProxyHttpTestServer.addHandler(new UnauthorizedHandler("auth-proxy-server"), "/http1/proxy/");
343         authProxyHttpTestServer.addHandler(new UnauthorizedHandler("auth-proxy-server"), "/http2/proxy/");
344         proxyHttpURI = "http://" + httpTestServer.serverAuthority() + "/http1";
345 
346         HttpsServer httpsServer = HttpsServer.create(sa, 0);
347         httpsServer.setHttpsConfigurator(new HttpsConfigurator(sslContext));
348         httpsTestServer = HttpTestServer.of(httpsServer);
349         httpsTestServer.addHandler(new PlainServerHandler("https-server"),"/https1/");
350         httpsURI = "https://" + httpsTestServer.serverAuthority() + "/https1";
351 
352         http2TestServer = HttpTestServer.of(new Http2TestServer("localhost", false, 0));
353         http2TestServer.addHandler(new PlainServerHandler("plain-server"), "/http2/");
354         http2URI = "http://" + http2TestServer.serverAuthority() + "/http2";
355         https2TestServer = HttpTestServer.of(new Http2TestServer("localhost", true, sslContext));
356         https2TestServer.addHandler(new PlainServerHandler("https-server"), "/https2/");
357         https2URI = "https://" + https2TestServer.serverAuthority() + "/https2";
358 
359         proxy = DigestEchoServer.createHttpsProxyTunnel(DigestEchoServer.HttpAuthSchemeType.NONE);
360         authproxy = DigestEchoServer.createHttpsProxyTunnel(DigestEchoServer.HttpAuthSchemeType.BASIC);
361 
362         client = TRACKER.track(HttpClient.newBuilder()
363                 .proxy(new TestProxySelector())
364                 .sslContext(sslContext)
365                 .executor(executor)
366                 .build());
367         clientCount.incrementAndGet();
368 
369 
370         httpTestServer.start();
371         serverCount.incrementAndGet();
372         proxyHttpTestServer.start();
373         serverCount.incrementAndGet();
374         authProxyHttpTestServer.start();
375         serverCount.incrementAndGet();
376         httpsTestServer.start();
377         serverCount.incrementAndGet();
378         http2TestServer.start();
379         serverCount.incrementAndGet();
380         https2TestServer.start();
381         serverCount.incrementAndGet();
382     }

400     class TestProxySelector extends ProxySelector {
401         @Override
402         public List<Proxy> select(URI uri) {
403             String path = uri.getPath();
404             out.println("Selecting proxy for: " + uri);
405             if (path.contains("/proxy/")) {
406                 if (path.contains("/http1/") || path.contains("/http2/")) {
407                     // Simple proxying
408                     var p = path.contains("/auth/") ? authProxyHttpTestServer : proxyHttpTestServer;
409                     return List.of(new Proxy(Proxy.Type.HTTP, p.getAddress()));
410                 } else {
411                     // Both HTTPS or HTTPS/2 require tunnelling
412                     var p = path.contains("/auth/") ? authproxy : proxy;
413                     return List.of(new Proxy(Proxy.Type.HTTP, p.getProxyAddress()));
414                 }
415             }
416             System.out.print("NO_PROXY for " + uri);
417             return List.of(Proxy.NO_PROXY);
418         }
419         @Override
420         public void connectFailed(URI uri, SocketAddress sa, IOException ioe) {
421             System.err.printf("Connect failed for: uri=\"%s\", sa=\"%s\", ioe=%s%n", uri, sa, ioe);
422         }
423     }

425     static class PlainServerHandler implements HttpTestHandler {
426 
427         final String serverType;
428         PlainServerHandler(String serverType) {
429             this.serverType = serverType;
430         }
431 
432         @Override
433         public void handle(HttpTestExchange t) throws IOException {
434             readAllRequestData(t); // shouldn't be any
435             String method = t.getRequestMethod();
436             String path = t.getRequestURI().getPath();
437             HttpTestRequestHeaders  reqh = t.getRequestHeaders();
438             HttpTestResponseHeaders rsph = t.getResponseHeaders();
439 
440             String xValue = serverType;
441             rsph.addHeader("X-value", serverType);
442 
443             t.getResponseHeaders().addHeader("X-value", xValue);
444             byte[] body = "RESPONSE".getBytes(UTF_8);
445             t.sendResponseHeaders(HTTP_OK, body.length);
446             try (var out = t.getResponseBody()) {
447                 out.write(body);
448             }
449         }
450     }

452     static class UnauthorizedHandler implements HttpTestHandler {
453 
454         final String serverType;
455         UnauthorizedHandler(String serverType) {
456             this.serverType = serverType;
457         }
458 
459         @Override
460         public void handle(HttpTestExchange t) throws IOException {
461             readAllRequestData(t); // shouldn't be any
462             String method = t.getRequestMethod();
463             String path = t.getRequestURI().getPath();
464             HttpTestRequestHeaders  reqh = t.getRequestHeaders();
465             HttpTestResponseHeaders rsph = t.getResponseHeaders();
466 
467             String xValue = serverType;
468             String srv = path.contains("/proxy/") ? "proxy" : "server";
469             String prefix = path.contains("/proxy/") ? "Proxy-" : "WWW-";
470             int code = path.contains("/proxy/") ? PROXY_UNAUTHORIZED : UNAUTHORIZED;
471             String resp = prefix + "Unauthorized";
472             rsph.addHeader(prefix + "Authenticate", "Basic realm=\"earth\", charset=\"UTF-8\"");
473 
474             byte[] body = resp.getBytes(UTF_8);
475             t.getResponseHeaders().addHeader("X-value", xValue);
476             t.sendResponseHeaders(code, body.length);
477             try (var out = t.getResponseBody()) {
478                 out.write(body);
479             }
480         }
481     }
