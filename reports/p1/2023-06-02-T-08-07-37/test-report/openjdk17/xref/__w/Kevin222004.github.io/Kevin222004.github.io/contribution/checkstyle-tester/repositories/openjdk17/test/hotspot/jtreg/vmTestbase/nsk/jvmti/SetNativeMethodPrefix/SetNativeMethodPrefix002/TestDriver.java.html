52  import jdk.test.lib.Platform;
53  import jdk.test.lib.process.ProcessTools;
54  
55  import java.io.File;
56  import java.util.Arrays;
57  
58  public class TestDriver {
59      public static void main(String[] args) throws Exception {
60          ProcessBuilder pb = ProcessTools.createTestJvm(
61                  "-agentlib:SetNativeMethodPrefix001=trace=all",
62                  "-agentlib:SetNativeMethodPrefix002-01=trace=all prefix=wa_",
63                  "-agentlib:SetNativeMethodPrefix002-02=trace=all prefix=wb_",
64                  "-agentlib:SetNativeMethodPrefix002-03=trace=all prefix=wc_",
65                  nsk.jvmti.SetNativeMethodPrefix.SetNativeMethodPrefix002.class.getName()
66          );
67  
68          String envName = Platform.sharedLibraryPathVariableName();
69          pb.environment()
70            .merge(envName, ".", (x, y) -> y + File.pathSeparator + x);
71  
72          String command = Arrays.toString(args);
73          System.out.println("exec " + command);
74          pb.inheritIO();
75  
76          int exitCode = pb.start().waitFor();
77  
78          if (exitCode != 95 && exitCode !=0 ) {
79              throw new AssertionError(command + " exit code is " + exitCode);
80          }
81      }
82  }
