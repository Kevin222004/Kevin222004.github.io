24  import jdk.test.lib.apps.LingeredApp;
25  
26  import java.lang.ref.Reference;
27  
28  interface Language {
29      static final long nbrOfWords = 99999;
30      public abstract long getNbrOfWords();
31  }
32  
33  class ParselTongue implements Language {
34      public long getNbrOfWords() {
35          return nbrOfWords * 4;
36      }
37  }
38  
39  public class LingeredAppWithInterface extends LingeredApp {
40  
41      public static void main(String args[]) {
42          ParselTongue lang = new ParselTongue();
43          Language muggleSpeak = new Language() {
44              public long getNbrOfWords() {
45                  return nbrOfWords * 8;
46              }
47          };
48  
49          // Not tested at this point. The test needs to be enhanced
50          // later to test for the sizes of the Lambda MetaFactory
51          // generated anonymous classes too. (After JDK-8160228 gets
52          // fixed.)
53          Runnable r2 = () -> System.out.println("Hello world!");
54          r2.run();
55  
56          System.out.println(lang.getNbrOfWords() + muggleSpeak.getNbrOfWords());
57  
58          LingeredApp.main(args);
59          Reference.reachabilityFence(lang);
60      }
61  }
