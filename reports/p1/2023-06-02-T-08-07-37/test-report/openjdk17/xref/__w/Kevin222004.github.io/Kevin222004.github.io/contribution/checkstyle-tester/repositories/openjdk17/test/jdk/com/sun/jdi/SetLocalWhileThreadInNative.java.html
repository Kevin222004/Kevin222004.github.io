38  import java.io.FileDescriptor;
39  import java.io.FileInputStream;
40  import java.io.IOException;
41  import java.io.OutputStream;
42  import java.util.List;

44  import com.sun.jdi.*;

47  import jdk.test.lib.Asserts;

49  /********** target program **********/
50  
51  class SetLocalWhileThreadInNativeTarget {
52  
53      public static final String name = SetLocalWhileThreadInNativeTarget.class.getName();
54      public static FileInputStream fis;
55      public static int count;
56      public static int bytesRead;
57  
58  
59      // Let D (debugger) be a JVMTI agent that updates a local of a compiled frame F owned by
60      // thread T. In this test F corresponds to dontinline_testMethod.
61      //
62      // Issue: The VM thread executes an VM op on behalf of D to set the local in F. Doing so it
63      // requests the deoptimization of F and then calls compiledVFrame::update_deferred_value(),
64      // where frame::is_deoptimized_frame() returns false causing an assertion failure on SPARC where
65      // NeedsDeoptSuspend is true.
66      //
67      // Analysis: The deoptimization of F is requested, while T is in the native method
68      // java.io.FileInputStream::read0() and F is the direct caller of read0(). This is a special
69      // case with NeedsDeoptSuspend in frame::deoptimize(). Effectively the deoptimization is not
70      // done synchronously, instead T deoptimizes F at a later point upon return from the native
71      // method.
72      public static int dontinline_testMethod() {
73          int zero = 0;
74          int val = 0;
75          try {
76              val = fis.read(); // Will be inlined. Calls native method java.io.FileInputStream::read0()
77              count++;
78          } catch (IOException e) { /* ignored */ }
79          return val + zero;
80      }
81  
82      public static void main(String[] args) {
83          System.out.println(name + " is up and running.");
84          fis = new FileInputStream(FileDescriptor.in);
85          bytesRead=0;
86          while (true) {
87              int val = dontinline_testMethod();
88              if (val == SetLocalWhileThreadInNative.STOP) {
89                  System.out.println("Debuggee: received STOP message");
90                  System.exit(0);
91              }
92              bytesRead++;
93              if ((bytesRead & ((1L << 14)-1)) == 0) {
94                  System.out.println("Called test method " + bytesRead + " times");
95              }
96          }
97      }
98  }

100  /********** test program **********/
101 
102 public class SetLocalWhileThreadInNative extends TestScaffold {
103     public static final int MESSAGE_COUNT = 10000;
104     public static final String MESSAGE    = "0123456789";
105     public static final int MESSAGE_SIZE  = MESSAGE.length();
106     public static final int TOTAL_BYTES   = MESSAGE_COUNT * MESSAGE_SIZE;
107     public static final int STOP = 255;
108 
109     ReferenceType mainClass;
110     ThreadReference mainThread;
111 
112     SetLocalWhileThreadInNative (String args[]) {
113         super(args);
114     }
115 
116     public static void main(String[] args)
117         throws Exception
118     {
119         new SetLocalWhileThreadInNative (args).startTests();
120     }
121 
122     /********** test core **********/
123 
124     protected void runTests()
125         throws Exception
126     {
127         String targetProgName = SetLocalWhileThreadInNativeTarget.class.getName();
128         String testName = getClass().getSimpleName();
129 
130         // Start debuggee and obtain reference to main thread an main class
131         BreakpointEvent bpe = startToMain(targetProgName);
132         mainClass = bpe.location().declaringType();
133         mainThread = bpe.thread();
134 
135         // Resume debuggee send some bytes
136         vm().resume();
137         OutputStream os = vm().process().getOutputStream();
138         byte[] ba = MESSAGE.getBytes();
139         for (int i = 0; i < MESSAGE_COUNT; i++) {
140             os.write(ba);
141         }
142         os.flush();
143 
144         // Wait for the debugee to read all the bytes.
145         int bytesRead = 0;
146         Field bytesReadField = mainClass.fieldByName("bytesRead");
147         do {
148             bytesRead = ((PrimitiveValue)mainClass.getValue(bytesReadField)).intValue();
149             System.out.println("debugee has read " + bytesRead + " of " + TOTAL_BYTES);
150             Thread.sleep(500);
151         } while (bytesRead < TOTAL_BYTES);
152 
153         // By now  dontinline_testMethod() will be compiled. The debugee will be blocked in java.io.FileInputStream::read0().
154         // Now set local variable in dontinline_testMethod().
155         vm().suspend();
156         System.out.println("Debuggee Stack:");
157         List<StackFrame> stack_frames = mainThread.frames();
158         int i = 0;
159         for (StackFrame ff : stack_frames) {
160             System.out.println("frame[" + i++ +"]: " + ff.location().method());
161         }
162         StackFrame frame = mainThread.frame(2);
163         Asserts.assertEQ(frame.location().method().toString(), "SetLocalWhileThreadInNativeTarget.dontinline_testMethod()");
164         List<LocalVariable> localVars = frame.visibleVariables();
165         boolean changedLocal = false;
166         for (LocalVariable lv : localVars) {
167             if (lv.name().equals("zero")) {
168                 frame.setValue(lv, vm().mirrorOf(0)); // triggers deoptimization!
169                 changedLocal = true;
170             }
171         }
172         Asserts.assertTrue(changedLocal);
173 
174         // signal stop
175         os.write(STOP);
176         os.flush();
177 
178 
179         // resume the target listening for events
180         listenUntilVMDisconnect();
181 
182 
183         // deal with results of test if anything has called failure("foo")
184         // testFailed will be true
185         if (!testFailed) {
186             println(testName + ": passed");
187         } else {
188             throw new Exception(testName + ": failed");
189         }
190     }
191 }
