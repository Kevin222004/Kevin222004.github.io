25  import java.io.File;
26  import java.lang.module.Configuration;
27  import java.lang.module.ModuleDescriptor;
28  import java.util.List;
29  import java.util.Set;
30  import sun.hotspot.WhiteBox;
31  
32  //
33  // Test archived system module graph when open archive heap objects are mapped:
34  //
35  public class CheckArchivedModuleApp {
36      static WhiteBox wb;
37      public static void main(String args[]) throws Exception {
38          wb = WhiteBox.getWhiteBox();
39  
40          if (!wb.areOpenArchiveHeapObjectsMapped()) {
41              System.out.println("Archived open_archive_heap objects are not mapped.");
42              System.out.println("This may happen during normal operation. Test Skipped.");
43              return;
44          }
45  
46          if (args.length != 2) {
47             throw new RuntimeException(
48                 "FAILED. Incorrect argument length: " + args.length);
49          }
50  
51          boolean expectArchivedDescriptors = "yes".equals(args[0]);
52          boolean expectArchivedConfiguration = "yes".equals(args[1]);
53          // -XX:+EnableJVMCI adds extra system modules, in which case the system
54          // module objects are not archived.
55          Boolean enableJVMCI = wb.getBooleanVMFlag("EnableJVMCI");
56          if (enableJVMCI != null && enableJVMCI) {
57              expectArchivedDescriptors = false;
58              expectArchivedConfiguration = false;
59          }
60  
61          checkModuleDescriptors(expectArchivedDescriptors);
62          checkConfiguration(expectArchivedConfiguration);
63          checkEmptyConfiguration(expectArchivedConfiguration);
64          checkEmptyLayer();
65      }
66  
67      private static void checkModuleDescriptors(boolean expectArchivedDescriptors) {
68          Set<Module> modules = ModuleLayer.boot().modules();
69          for (Module m : modules) {
70              ModuleDescriptor md = m.getDescriptor();
71              String name = md.name();
72              if (expectArchivedDescriptors) {
73                  if (wb.isShared(md)) {
74                      System.out.println(name + " is archived. Expected.");
75                  } else {
76                      throw new RuntimeException(
77                          "FAILED. " + name + " is not archived. Expect archived.");
78                  }
79              } else {
80                  if (!wb.isShared(md)) {
81                      System.out.println(name + " is not archived. Expected.");
82                  } else {
83                      throw new RuntimeException(
84                          "FAILED. " + name + " is archived. Expect not archived.");
85                  }
86              }
87          }
88      }
89  
90      private static void checkEmptyConfiguration(boolean expectArchivedConfiguration) {
91          // Configuration.EMPTY_CONFIGURATION uses the singletons,
92          // ListN.EMPTY_LIST, SetN.EMPTY_SET and MapN.EMPTY_MAP in
93          // ImmutableCollections for the 'parents', 'modules' and
94          // 'graph' fields. The ImmutableCollections singletons
95          // can be accessed via List.of(), Set.of() and Map.of() APIs.
96          // Configuration public APIs also allow access to the
97          // EMPTY_CONFIGURATION's 'parents' and 'modules'. When the
98          // archived java heap data is enabled at runtime, make sure
99          // the EMPTY_CONFIGURATION.parents and EMPTY_CONFIGURATION.modules
100         // are the archived ImmutableCollections singletons.
101         Configuration emptyCf = Configuration.empty();
102         List emptyCfParents = emptyCf.parents();
103         Set emptyCfModules = emptyCf.modules();
104         if (expectArchivedConfiguration) {
105             if (emptyCfParents == List.of() &&
106                 wb.isShared(emptyCfParents)) {
107                 System.out.println("Empty Configuration has expected parents.");
108             } else {
109                 throw new RuntimeException(
110                     "FAILED. Unexpected parents for empty Configuration.");
111             }
112             if (emptyCfModules == Set.of() &&
113                 wb.isShared(emptyCfModules)) {
114                 System.out.println("Empty Configuration has expected module set.");
115             } else {
116                 throw new RuntimeException(
117                     "FAILED. Unexpected module set for empty Configuration.");
118             }
119         }
120     }
121 
122 
123 
124     private static void checkConfiguration(boolean expectArchivedConfiguration) {
125         Configuration cf = ModuleLayer.boot().configuration();
126 
127         if (expectArchivedConfiguration) {
128             if (wb.isShared(cf)) {
129                 System.out.println("Boot layer configuration is archived. Expected.");
130             } else {
131                 throw new RuntimeException(
132                     "FAILED. Boot layer configuration is not archived.");
133             }
134         } else {
135             if (!wb.isShared(cf)) {
136                 System.out.println("Boot layer configuration is not archived. Expected.");
137             } else {
138                 throw new RuntimeException(
139                     "FAILED. Boot layer configuration is archived.");
140             }
141         }
142     }
143 
144     private static void checkEmptyLayer() {
145         // ModuleLayer.EMPTY_FIELD returned by empty() method is singleton.
146         // Check that with CDS there is still a single instance of EMPTY_LAYER
147         // and boot() layer parent is THE empty layer.
148         if (ModuleLayer.empty() != ModuleLayer.boot().parents().get(0)) {
149             throw new RuntimeException("FAILED. Empty module layer is not singleton");
150         }
151     }
