66      void fromHttpClientClasslevelDescription() throws Exception {
67          //Synchronous Example
68          HttpClient client = HttpClient.newBuilder()
69                  .version(Version.HTTP_1_1)
70                  .followRedirects(Redirect.NORMAL)
71                  .connectTimeout(Duration.ofSeconds(20))
72                  .proxy(ProxySelector.of(new InetSocketAddress("proxy.example.com", 80)))
73                  .authenticator(Authenticator.getDefault())
74                  .build();
75          HttpResponse<String> response = client.send(request, BodyHandlers.ofString());
76          System.out.println(response.statusCode());
77          System.out.println(response.body());
78  
79          //Asynchronous Example
80          HttpRequest request = HttpRequest.newBuilder()
81                  .uri(URI.create("https://foo.com/"))
82                  .timeout(Duration.ofMinutes(2))
83                  .header("Content-Type", "application/json")
84                  .POST(BodyPublishers.ofFile(Paths.get("file.json")))
85                  .build();
86          client.sendAsync(request, BodyHandlers.ofString())
87                  .thenApply(HttpResponse::body)
88                  .thenAccept(System.out::println);
89      }

91      void fromHttpRequest() throws Exception {
92          // HttpRequest class-level description
93          HttpClient client = HttpClient.newHttpClient();
94          HttpRequest request = HttpRequest.newBuilder()
95                  .uri(URI.create("http://foo.com/"))
96                  .build();
97          client.sendAsync(request, BodyHandlers.ofString())
98                  .thenApply(HttpResponse::body)
99                  .thenAccept(System.out::println)
100                 .join();
101 
102         // HttpRequest.BodyPublishers class-level description
103         // Request body from a String
104         HttpRequest request1 = HttpRequest.newBuilder()
105                 .uri(URI.create("https://foo.com/"))
106                 .header("Content-Type", "text/plain; charset=UTF-8")
107                 .POST(BodyPublishers.ofString("some body text"))
108                 .build();
109 
110         // Request body from a File
111         HttpRequest request2 = HttpRequest.newBuilder()
112                 .uri(URI.create("https://foo.com/"))
113                 .header("Content-Type", "application/json")
114                 .POST(BodyPublishers.ofFile(Paths.get("file.json")))
115                 .build();
116 
117         // Request body from a byte array
118         HttpRequest request3 = HttpRequest.newBuilder()
119                 .uri(URI.create("https://foo.com/"))
120                 .POST(BodyPublishers.ofByteArray(new byte[] { /*...*/ }))
121                 .build();
122 
123         // HttpRequest.Builder
124         // API note - newBuilder(HttpRequest, BiPredicate<String, String>)
125         // Retain all headers:
126         HttpRequest.newBuilder(request, (n, v) -> true);
127 
128         //Remove all headers:
129         HttpRequest.newBuilder(request, (n, v) -> false);
130 
131         // Remove a particular header (e.g. Foo-Bar):
132         HttpRequest.newBuilder(request, (name, value) ->
133                 !name.equalsIgnoreCase("Foo-Bar"));
134     }

136     void fromHttpResponse() throws Exception {
137         // HttpResponse class-level description
138         HttpResponse<String> response = client
139                 .send(request, BodyHandlers.ofString());
140 
141         // HttpResponse.BodyHandler class-level description
142         HttpRequest request = HttpRequest.newBuilder()
143                 .uri(URI.create("http://www.foo.com/"))
144                 .build();
145         client.sendAsync(request, BodyHandlers.ofFile(Paths.get("/tmp/f")))
146                 .thenApply(HttpResponse::body)
147                 .thenAccept(System.out::println);
148 
149         HttpRequest request1 = HttpRequest.newBuilder()
150                 .uri(URI.create("http://www.foo.com/"))
151                 .build();
152         BodyHandler<Path> bodyHandler = (info) -> info.statusCode() == 200
153                 ? BodySubscribers.ofFile(Paths.get("/tmp/f"))
154                 : BodySubscribers.replacing(Paths.get("/NULL"));
155         client.sendAsync(request, bodyHandler)
156                 .thenApply(HttpResponse::body)
157                 .thenAccept(System.out::println);
158 
159 
160         // HttpResponse.BodyHandlers class-level description
161         // Receives the response body as a String
162         HttpResponse<String> response1 = client
163                 .send(request, BodyHandlers.ofString());
164 
165         // Receives the response body as a file
166         HttpResponse<Path> response2 = client
167                 .send(request, BodyHandlers.ofFile(Paths.get("example.html")));
168 
169         // Receives the response body as an InputStream
170         HttpResponse<InputStream> respons3 = client
171                 .send(request, BodyHandlers.ofInputStream());
172 
173         // Discards the response body
174         HttpResponse<Void> respons4 = client
175                 .send(request, BodyHandlers.discarding());
176 
177 
178         // HttpResponse.BodySubscribers class-level description
179         // Streams the response body to a File
180         HttpResponse<Path> response5 = client
181                 .send(request, responseInfo -> BodySubscribers.ofFile(Paths.get("example.html")));
182 
183         // Accumulates the response body and returns it as a byte[]
184         HttpResponse<byte[]> response6 = client
185                 .send(request, responseInfo -> BodySubscribers.ofByteArray());
186 
187         // Discards the response body
188         HttpResponse<Void> response7 = client
189                 .send(request, responseInfo -> BodySubscribers.discarding());
190 
191         // Accumulates the response body as a String then maps it to its bytes
192         HttpResponse<byte[]> response8 = client
193                 .send(request, responseInfo ->
194                         BodySubscribers.mapping(BodySubscribers.ofString(UTF_8), String::getBytes));
195 
196         // Maps an InputStream to a Foo object.
197         HttpResponse<Supplier<Foo>> response9 = client.send(request,
198                 (resp) -> FromMappingSubscriber.asJSON(Foo.class));
199         String resp = response9.body().get().asString();
200

203     /**
204      * @apiNote This method can be used as an adapter between a {@code
205      * BodySubscriber} and a text based {@code Flow.Subscriber} that parses
206      * text line by line.
207      *
208      * <p> For example:
209      * <pre> {@code  // A PrintSubscriber that implements Flow.Subscriber<String>
210      *  // and print lines received by onNext() on System.out
211      *  PrintSubscriber subscriber = new PrintSubscriber(System.out);
212      *  client.sendAsync(request, BodyHandlers.fromLineSubscriber(subscriber))
213      *      .thenApply(HttpResponse::statusCode)
214      *      .thenAccept((status) -> {
215      *          if (status != 200) {
216      *              System.err.printf("ERROR: %d status received%n", status);
217      *          }
218      *      }); }</pre>
219      */
220     void fromLineSubscriber1() {
221          // A PrintSubscriber that implements Flow.Subscriber<String>
222          // and print lines received by onNext() on System.out
223          PrintSubscriber subscriber = new PrintSubscriber(System.out);
224          client.sendAsync(request, BodyHandlers.fromLineSubscriber(subscriber))
225                  .thenApply(HttpResponse::statusCode)
226                  .thenAccept((status) -> {
227                      if (status != 200) {
228                          System.err.printf("ERROR: %d status received%n", status);
229                      }
230                  });
231     }

233     /**
234      * @apiNote This method can be used as an adapter between a {@code
235      * BodySubscriber} and a text based {@code Flow.Subscriber} that parses
236      * text line by line.
237      *
238      * <p> For example:
239      * <pre> {@code  // A LineParserSubscriber that implements Flow.Subscriber<String>
240      *  // and accumulates lines that match a particular pattern
241      *  Pattern pattern = ...;
242      *  LineParserSubscriber subscriber = new LineParserSubscriber(pattern);
243      *  HttpResponse<List<String>> response = client.send(request,
244      *      BodyHandlers.fromLineSubscriber(subscriber, s -> s.getMatchingLines(), "\n"));
245      *  if (response.statusCode() != 200) {
246      *      System.err.printf("ERROR: %d status received%n", response.statusCode());
247      *  } }</pre>
248      *
249      */
250     void fromLineSubscriber2() throws IOException, InterruptedException {
251         // A LineParserSubscriber that implements Flow.Subscriber<String>
252         // and accumulates lines that match a particular pattern
253         Pattern pattern = p;
254         LineParserSubscriber subscriber = new LineParserSubscriber(pattern);
255         HttpResponse<List<String>> response = client.send(request,
256                 BodyHandlers.fromLineSubscriber(subscriber, s -> s.getMatchingLines(), "\n"));
257         if (response.statusCode() != 200) {
258             System.err.printf("ERROR: %d status received%n", response.statusCode());
259         }

262     static final class PrintSubscriber implements Flow.Subscriber<String> {
263         final PrintStream out;
264         PrintSubscriber(PrintStream out) {
265             this.out = out;
266         }
267         @Override
268         public void onSubscribe(Flow.Subscription subscription) {
269             subscription.request(Long.MAX_VALUE);
270         }
271         @Override
272         public void onNext(String item) {
273             out.println(item);
274         }
275 
276         @Override
277         public void onError(Throwable throwable) {
278             throwable.printStackTrace();
279         }
280         @Override
281         public void onComplete() {
282         }
283     }

285     static final class LineParserSubscriber implements Flow.Subscriber<String> {
286         final Pattern pattern;
287         final CopyOnWriteArrayList<String> matches = new CopyOnWriteArrayList<>();
288         LineParserSubscriber(Pattern pattern) {
289             this.pattern = pattern;
290         }
291         @Override
292         public void onSubscribe(Flow.Subscription subscription) {
293             subscription.request(Long.MAX_VALUE);
294         }
295         @Override
296         public void onNext(String item) {
297             if (pattern.matcher(item).matches()) {
298                 matches.add(item);
299             }
300         }
301         @Override
302         public void onError(Throwable throwable) {
303             throwable.printStackTrace();
304         }
305         @Override
306         public void onComplete() {
307         }
308 
309         public List<String> getMatchingLines() {
310             return Collections.unmodifiableList(matches);
311         }
312     }

324     static class ObjectMapper {
325         <W> W readValue(InputStream is, Class<W> targetType)
326             throws IOException
327         {
328                 byte[] bytes = is.readAllBytes();
329                 return map(bytes, targetType);
330         }
331 
332         static <W> W map(byte[] bytes, Class<W> targetType) {
333             try {
334                 return targetType.getConstructor(byte[].class).newInstance(bytes);
335             } catch (RuntimeException | Error x) {
336                 throw x;
337             } catch (Exception x) {
338                 throw new UndeclaredThrowableException(x);
339             }
340         }
341     }

343     static class FromMappingSubscriber {
344         public static <W> BodySubscriber<Supplier<W>> asJSON(Class<W> targetType) {
345             BodySubscriber<InputStream> upstream = BodySubscribers.ofInputStream();
346 
347             BodySubscriber<Supplier<W>> downstream = BodySubscribers.mapping(
348                     upstream, (InputStream is) -> () -> {
349                         try (InputStream stream = is) {
350                             ObjectMapper objectMapper = new ObjectMapper();
351                             return objectMapper.readValue(stream, targetType);
352                         } catch (IOException e) {
353                             throw new UncheckedIOException(e);
354                         }
355                     });
356             return downstream;
357         }
