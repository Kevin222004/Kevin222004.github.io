24  import java.io.File;
25  import java.net.URL;
26  import java.net.URLClassLoader;
27  import java.nio.file.Path;
28  import java.nio.file.Paths;
29  import java.util.jar.Attributes;
30  import java.util.jar.Manifest;
31  import jdk.test.lib.util.JarUtils;
32  import jdk.test.lib.compiler.InMemoryJavaCompiler;
33  import jdk.test.lib.helpers.ClassFileInstaller;
34  
35  /*
36   * @test
37   * @bug 8216401 8235361
38   * @summary Test classloading via JAR Class-Path entries
39   * @library /test/lib
40   *
41   * @run main/othervm JarClassPathFileEntry
42   * @run main/othervm -Djdk.net.URLClassPath.disableClassPathURLCheck=true JarClassPathFileEntry
43   * @run main/othervm -Djdk.net.URLClassPath.disableClassPathURLCheck=false JarClassPathFileEntry
44   */
45  
46  public class JarClassPathFileEntry {
47      private final static boolean IS_WINDOWS = System.getProperty("os.name").startsWith("Windows");
48  
49      private final static String TEST_CLASSES = System.getProperty("test.classes");
50      private final static String OTHER_DIR = TEST_CLASSES + "/OTHER/";
51  
52      private final static Path OTHER_JAR_PATH = Paths.get(OTHER_DIR, "Other.jar");
53      private final static Path CONTEXT_JAR_PATH = Paths.get(TEST_CLASSES, "Context.jar");
54  
55      public static void main(String[] args) throws Throwable {
56          String fileScheme = "file:" + (IS_WINDOWS ? toUnixPath(OTHER_JAR_PATH.toString())
57                                                        :        OTHER_JAR_PATH.toString());
58          doTest(fileScheme);
59  
60          if (IS_WINDOWS) {
61              // Relative URL encoding of absolute path, e.g. /C:\\path\\to\\file.jar
62              String driveLetter = "/" + OTHER_JAR_PATH;
63              doTest(driveLetter);
64          }
65      }
66  
67      /* Load a class from Other.jar via the given Class-Path entry */
68      private static void doTest(String classPathEntry) throws Throwable {
69          // Create Other.class in OTHER_DIR, off the default classpath
70          byte klassbuf[] = InMemoryJavaCompiler.compile("Other",
71                                                         "public class Other {}");
72          ClassFileInstaller.writeClassToDisk("Other", klassbuf, OTHER_DIR);
73  
74          // Create Other.jar in OTHER_DIR
75          JarUtils.createJarFile(OTHER_JAR_PATH,
76                                 Paths.get(OTHER_DIR),
77                                 Paths.get(OTHER_DIR, "Other.class"));
78  
79          // Create Context.class
80          klassbuf = InMemoryJavaCompiler.compile("Context",
81                                                  "public class Context {}");
82          ClassFileInstaller.writeClassToDisk("Context", klassbuf, TEST_CLASSES);
83  
84          // Create Context.jar w/ "file:" entry for Other.jar
85          Manifest mf = new Manifest();
86          Attributes attrs = mf.getMainAttributes();
87          attrs.put(Attributes.Name.MANIFEST_VERSION, "1.0");
88  
89          attrs.put(Attributes.Name.CLASS_PATH, classPathEntry);
90  
91          System.out.println("Creating Context.jar with Class-Path: " + classPathEntry);
92          JarUtils.createJarFile(CONTEXT_JAR_PATH, mf,
93                                 Paths.get(TEST_CLASSES),
94                                 Paths.get(TEST_CLASSES, "Context.class"));
95  
96          // Use URLClassLoader w/ Context.jar to load Other.class, which will
97          // load via the Class-Path entry
98          URL url = CONTEXT_JAR_PATH.toUri().toURL();
99          URLClassLoader ucl = new URLClassLoader("TestURLClassLoader",
100                                                 new URL[]{ url },
101                                                 null); // don't delegate to App CL
102         Class<?> otherClass = Class.forName("Other", true, ucl); // ClassNotFoundException -> fail
103         System.out.println("Loaded: " + otherClass);
104     }
105 
106     /* Convert a Windows path to a unix-style path, and remove any drive letter */
107     private static String toUnixPath(String orig) {
108         String retVal = new File(orig).toURI().getPath();
109         int colonAt = retVal.indexOf(':');
110 
111         if (colonAt != -1 && colonAt < 3) {
112             retVal = retVal.substring(colonAt + 1); // Start after the drive letter
113         }
114         return retVal;
115     }
116 }
