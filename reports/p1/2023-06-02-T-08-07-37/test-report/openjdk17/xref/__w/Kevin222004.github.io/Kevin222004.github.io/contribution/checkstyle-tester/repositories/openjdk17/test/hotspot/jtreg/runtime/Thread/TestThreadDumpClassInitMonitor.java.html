24  /*
25   * @test
26   * @bug     8213397 8217337
27   * @summary Check that the thread dump shows when a thread is blocked
28   *          on a class initialization monitor
29   *
30   * @library /test/lib
31   * @run main/othervm TestThreadDumpClassInitMonitor
32   */
33  
34  import jdk.test.lib.process.OutputAnalyzer;
35  import jdk.test.lib.JDKToolFinder;
36  import jdk.test.lib.Platform;
37  
38  import java.io.IOException;
39  import java.util.List;
40  
41  public class TestThreadDumpClassInitMonitor {
42      // jstack tends to be closely bound to the VM that we are running
43      // so use getTestJDKTool() instead of getCompileJDKTool() or even
44      // getJDKTool() which can fall back to "compile.jdk".
45      final static String JSTACK = JDKToolFinder.getTestJDKTool("jstack");
46      final static String PID = "" + ProcessHandle.current().pid();
47  
48      final static Thread current = Thread.currentThread();
49  
50      /*
51       * This is the output we're looking for:
52       *
53       * "TestThread" #22 prio=5 os_prio=0 cpu=1.19ms elapsed=0.80s tid=0x00007f8f9405d800 nid=0x568b in Object.wait()  [0x00007f8fd94d0000]
54       *   java.lang.Thread.State: RUNNABLE
55       * Thread: 0x00007f8f9405d800  [0x568b] State: _at_safepoint _has_called_back 0 _at_poll_safepoint 0  // DEBUG ONLY
56       *   JavaThread state: _thread_blocked                                                                // DEBUG ONLY
57       *         at TestThreadDumpClassInitMonitor$Target$1.run(TestThreadDumpClassInitMonitor.java:69)
58       *         - waiting on the Class initialization monitor for TestThreadDumpClassInitMonitor$Target
59       *
60       */
61      final static String TEST_THREAD = "TestThread";
62      final static String TEST_THREAD_ENTRY = "\"" + TEST_THREAD;
63      final static String IN_OBJECT_WAIT = "in Object.wait()";
64      final static String THREAD_STATE = "java.lang.Thread.State: RUNNABLE";
65      final static String THREAD_INFO = "Thread:"; // the details are not important
66      final static String JAVATHREAD_STATE = "JavaThread state: _thread_blocked";
67      final static String CURRENT_METHOD = "at TestThreadDumpClassInitMonitor$Target$1.run";
68      final static String WAIT_INFO = "- waiting on the Class initialization monitor for TestThreadDumpClassInitMonitor$Target";
69  
70      volatile static boolean ready = false;
71  
72      static List<String> stackDump;  // jstack output as lines
73  
74      static class Target {
75  
76          static int field;
77  
78          // The main thread will initialize this class and so
79          // execute the actual test logic here.
80          static {
81              if (Thread.currentThread() != current) {
82                  throw new Error("Initialization logic error");
83              }
84              System.out.println("Initializing Target class in main thread");
85  
86              Thread t  = new Thread() {
87                      public void run() {
88                          System.out.println("Test thread about to access Target");
89                          ready = true; // tell main thread we're close
90                          // This will block until the main thread completes
91                          // static initialization of target
92                          Target.field = 42;
93                          System.out.println("Test thread done");
94                      }
95                  };
96              t.setName(TEST_THREAD);
97              t.start();
98  
99              // We want to run jstack once the test thread is blocked but
100             // there's no programmatic way to detect that. So we check the flag
101             // that will be set just before it should block, then by the time
102             // we can exec jstack it should be ready.
103             try {
104                 while (!ready) {
105                     Thread.sleep(200);
106                 }
107             }
108             catch (InterruptedException ie) {
109                 throw new Error("Shouldn't happen");
110             }
111 
112             // Now run jstack
113             try {
114                 ProcessBuilder pb = new ProcessBuilder(JSTACK, PID);
115                 OutputAnalyzer output = new OutputAnalyzer(pb.start());
116                 output.shouldHaveExitValue(0);
117                 stackDump = output.asLines();
118             }
119             catch (IOException ioe) {
120                 throw new Error("Launching jstack failed", ioe);
121             }
122         }
123     }
124 
125 
126     public static void main(String[] args) throws Throwable {
127         // Implicitly run the main test logic
128         Target.field = 21;
129 
130         // Now check the output of jstack
131         try {
132             // product builds miss 2 lines of information in the stack
133             boolean isProduct = !Platform.isDebugBuild();
134             int foundLines = 0;
135             parseStack: for (String line : stackDump) {
136                 switch(foundLines) {
137                 case 0: {
138                     if (!line.startsWith(TEST_THREAD_ENTRY)) {
139                         continue;
140                     }
141                     foundLines++;
142                     if (!line.contains(IN_OBJECT_WAIT)) {
143                         throw new Error("Unexpected initial stack line: " + line);
144                     }
145                     continue;
146                 }
147                 case 1: {
148                     if (!line.trim().equals(THREAD_STATE)) {
149                         throw new Error("Unexpected thread state line: " + line);
150                     }
151                     if (isProduct) {
152                         foundLines += 3;
153                     } else {
154                         foundLines++;
155                     }
156                     continue;
157                 }
158                 case 2: { // Debug build
159                     if (!line.startsWith(THREAD_INFO)) {
160                         throw new Error("Unexpected thread info line: " + line);
161                     }
162                     foundLines++;
163                     continue;
164                 }
165                 case 3: { // Debug build
166                     if (!line.trim().equals(JAVATHREAD_STATE)) {
167                         throw new Error("Unexpected JavaThread state line: " + line);
168                     }
169                     foundLines++;
170                     continue;
171                 }
172                 case 4: {
173                     if (!line.trim().startsWith(CURRENT_METHOD)) {
174                         throw new Error("Unexpected current method line: " + line);
175                     }
176                     foundLines++;
177                     continue;
178                 }
179                 case 5: {
180                     if (!line.trim().equals(WAIT_INFO)) {
181                         throw new Error("Unexpected monitor information line: " + line);
182                     }
183                     break parseStack;
184                 }
185                 default: throw new Error("Logic error in case statement");
186                 }
187             }
188 
189             if (foundLines == 0) {
190                 throw new Error("Unexpected stack content - did not find line starting with "
191                                 + TEST_THREAD_ENTRY);
192             }
193         }
194         catch (Error e) {
195             // Dump the full stack trace on error so we can check the content
196             for (String line : stackDump) {
197                 System.out.println(line);
198             }
199             throw e;
200         }
201     }
