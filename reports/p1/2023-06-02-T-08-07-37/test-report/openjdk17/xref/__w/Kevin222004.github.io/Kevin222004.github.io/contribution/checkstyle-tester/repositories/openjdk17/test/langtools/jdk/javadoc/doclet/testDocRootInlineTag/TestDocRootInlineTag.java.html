24  /*
25   * @test
26   * @bug 4369014 4851991 8164407 8205593
27   * @summary Determine if the docRoot inline tag works properly.
28   * If docRoot performs as documented, the test passes.
29   * Make sure that the docRoot tag works with the -bottom option.
30   * @library ../../lib
31   * @modules jdk.javadoc/jdk.javadoc.internal.tool
32   * @build javadoc.tester.*
33   * @run main TestDocRootInlineTag
34   */
35  
36  import javadoc.tester.JavadocTester;
37  
38  public class TestDocRootInlineTag extends JavadocTester {
39  
40      public static void main(String... args) throws Exception {
41          TestDocRootInlineTag tester = new TestDocRootInlineTag();
42          tester.runTests();
43      }
44  
45      @Test
46      public void test() {
47          String uri = "http://www.java.sun.com/j2se/1.4/docs/api";
48  
49          javadoc("-bottom", """
50              The value of @docRoot is "{@docRoot}\"""",
51                  "-d", "out",
52                  "-source", "8",
53                  "-sourcepath", testSrc,
54                  "-linkoffline", uri, testSrc,
55                  testSrc("TestDocRootTag.java"), "pkg");
56          checkExit(Exit.OK);
57  
58          checkOutput("TestDocRootTag.html", true,
59                  "<a href=\"" + uri + """
60                      /java/io/File.html" title="class or interface in java.io" class="external-link"><code>File</code></a>""",
61                  """
62                      <a href="./index-all.html">index</a>""",
63                  "<a href=\"" + uri + """
64                      /java/io/File.html" title="class or interface in java.io" class="external-link"><code>Second File Link</code></a>""",
65                  "The value of @docRoot is \"./\"");
66  
67          checkOutput("index-all.html", true,
68                  """
69                      My package page is <a href="./pkg/package-summary.html">here</a>""");
70      }
71  }
