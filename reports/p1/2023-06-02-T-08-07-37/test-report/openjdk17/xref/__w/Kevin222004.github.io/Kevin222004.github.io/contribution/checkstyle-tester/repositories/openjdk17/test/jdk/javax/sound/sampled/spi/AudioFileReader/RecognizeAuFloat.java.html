648      * Tests that audio data is the same format as a command lines above.

650     private static void checkAudioStream(final InputStream is,
651                                          final byte[] data, final int bits)
652             throws Exception {
653         final AudioFileFormat aff = AudioSystem.getAudioFileFormat(is);
654         final int byteLength = aff.getByteLength();
655         if (byteLength != data.length) {
656             throw new RuntimeException("Wrong byteLength: " + byteLength);
657         }
658         final AudioFileFormat.Type type = aff.getType();
659         if (!type.equals(AudioFileFormat.Type.AU)) {
660             throw new RuntimeException("Wrong type: " + type);
661         }
662         final int frameLength = aff.getFrameLength();
663         if (frameLength != 0.01f * aff.getFormat().getFrameRate()) {
664             throw new RuntimeException("Wrong frameLength: " + frameLength);
665         }
666         checkAudioFormat(aff.getFormat(), bits);
667         try (AudioInputStream ais = AudioSystem.getAudioInputStream(is)) {
668             final AudioFormat format = ais.getFormat();
669             if (ais.getFrameLength() != 0.01f * format.getFrameRate()) {
670                 throw new RuntimeException(
671                         "Wrong frameLength: " + ais.getFrameLength());
672             }
673             checkAudioFormat(format, bits);
674         }

680     private static void checkAudioFormat(final AudioFormat audioFormat,
681                                          final int bits) {
682         final int channels = audioFormat.getChannels();
683         if (channels != 2) {
684             throw new RuntimeException("Wrong channels: " + channels);
685         }

692             throw new RuntimeException("Wrong frameSize: " + frameSize);
693         }
694         final AudioFormat.Encoding encoding = audioFormat.getEncoding();
695         if (!encoding.equals(AudioFormat.Encoding.PCM_FLOAT)) {
696             throw new RuntimeException("Wrong encoding: " + encoding);
697         }
698         final float frameRate = audioFormat.getFrameRate();
699         if (frameRate != 32000) {

701         }
702         final float sampleRate = audioFormat.getSampleRate();
703         if (sampleRate != 32000) {
704             throw new RuntimeException("Wrong sampleRate: " + sampleRate);
