276     // Ninety days in milliseconds
277     private static final long NINETY_DAYS = 7776000000L;
278 
279     private static boolean atLeastOneFailed = false;
280 
281     private static MessageDigest md;
282 
283     public static void main(String[] args) throws Exception {
284         System.out.println("cacerts file: " + CACERTS);
285         md = MessageDigest.getInstance("SHA-256");
286 
287         byte[] data = Files.readAllBytes(Path.of(CACERTS));
288         String checksum = HEX.formatHex(md.digest(data));
289         if (!checksum.equals(CHECKSUM)) {
290             atLeastOneFailed = true;
291             System.err.println("ERROR: wrong checksum\n" + checksum);
292             System.err.println("Expected checksum\n" + CHECKSUM);
293         }
294 
295         KeyStore ks = KeyStore.getInstance("JKS");
296         ks.load(new ByteArrayInputStream(data), "changeit".toCharArray());
297 
298         // check the count of certs inside
299         if (ks.size() != COUNT) {
300             atLeastOneFailed = true;
301             System.err.println("ERROR: " + ks.size() + " entries, should be "
302                     + COUNT);
303         }
304 
305         // check that all entries in the map are in the keystore
306         for (String alias : FINGERPRINT_MAP.keySet()) {
307             if (!ks.isCertificateEntry(alias)) {
308                 atLeastOneFailed = true;
309                 System.err.println("ERROR: " + alias + " is not in cacerts");
310             }
311         }
312 
313         // pull all the trusted self-signed CA certs out of the cacerts file
314         // and verify their signatures
315         Enumeration<String> aliases = ks.aliases();
316         while (aliases.hasMoreElements()) {
317             String alias = aliases.nextElement();
318             System.out.println("\nVerifying " + alias);
319             if (!ks.isCertificateEntry(alias)) {
320                 atLeastOneFailed = true;
321                 System.err.println("ERROR: " + alias
322                         + " is not a trusted cert entry");
323             }
324             X509Certificate cert = (X509Certificate) ks.getCertificate(alias);
325             if (!checkFingerprint(alias, cert)) {
326                 atLeastOneFailed = true;
327                 System.err.println("ERROR: " + alias + " SHA-256 fingerprint is incorrect");
328             }
329             // Make sure cert can be self-verified
330             try {
331                 cert.verify(cert.getPublicKey());
332             } catch (Exception e) {
333                 atLeastOneFailed = true;
334                 System.err.println("ERROR: cert cannot be verified:"
335                         + e.getMessage());
336             }
337 
338             // Make sure cert is not expired or not yet valid
339             try {
340                 cert.checkValidity();
341             } catch (CertificateExpiredException cee) {
342                 if (!EXPIRY_EXC_ENTRIES.contains(alias)) {
343                     atLeastOneFailed = true;
344                     System.err.println("ERROR: cert is expired");
345                 }
346             } catch (CertificateNotYetValidException cne) {
347                 atLeastOneFailed = true;
348                 System.err.println("ERROR: cert is not yet valid");
349             }
350 
351             // If cert is within 90 days of expiring, mark as failure so
352             // that cert can be scheduled to be removed/renewed.
353             Date notAfter = cert.getNotAfter();
354             if (notAfter.getTime() - System.currentTimeMillis() < NINETY_DAYS) {
355                 if (!EXPIRY_EXC_ENTRIES.contains(alias)) {
356                     atLeastOneFailed = true;
357                     System.err.println("ERROR: cert \"" + alias + "\" expiry \""
358                             + notAfter.toString() + "\" will expire within 90 days");
359                 }
360             }
361         }
362 
363         if (atLeastOneFailed) {
364             throw new Exception("At least one cacert test failed");
365         }
366     }

368     private static boolean checkFingerprint(String alias, Certificate cert)
369             throws Exception {
370         String fingerprint = FINGERPRINT_MAP.get(alias);
371         if (fingerprint == null) {
372             // no entry for alias
373             return true;
374         }
375         System.out.println("Checking fingerprint of " + alias);
376         byte[] digest = md.digest(cert.getEncoded());
377         return fingerprint.equals(HEX.formatHex(digest));
378     }
