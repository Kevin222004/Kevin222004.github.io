29  import java.lang.reflect.Field;
30  import java.lang.reflect.Modifier;
31  import java.util.Arrays;
32  import java.util.HashSet;
33  import java.util.Set;
34  import java.util.stream.Collectors;
35  
36  import static jdk.test.lib.Asserts.*;
37  
38  public class LambdaEagerInitTest {
39  
40      interface H {Object m(String s);}
41  
42      private static Set<String> allowedStaticFields(boolean nonCapturing) {
43          Set<String> s = new HashSet<>();
44          if (Boolean.getBoolean("jdk.internal.lambda.disableEagerInitialization")) {
45              if (nonCapturing) s.add("LAMBDA_INSTANCE$");
46          }
47          return s;
48      }
49  
50      private void nonCapturingLambda() {
51          H la = s -> s;
52          assertEquals("hi", la.m("hi"));
53          Class<? extends H> c1 = la.getClass();
54          verifyLambdaClass(la.getClass(), true);
55      }
56  
57      private void capturingLambda() {
58          H la = s -> concat(s, "foo");
59          assertEquals("hi foo", la.m("hi"));
60          verifyLambdaClass(la.getClass(), false);
61      }
62  
63      private void verifyLambdaClass(Class<?> c, boolean nonCapturing) {
64          Set<String> staticFields = new HashSet<>();
65          Set<String> instanceFields = new HashSet<>();
66          for (Field f : c.getDeclaredFields()) {
67              if (Modifier.isStatic(f.getModifiers())) {
68                  staticFields.add(f.getName());
69              } else {
70                  instanceFields.add(f.getName());
71              }
72          }
73          assertEquals(instanceFields.size(), nonCapturing ? 0 : 1, "Unexpected instance fields");
74          assertEquals(staticFields, allowedStaticFields(nonCapturing), "Unexpected static fields");
75      }
76  
77      private String concat(String... ss) {
78          return Arrays.stream(ss).collect(Collectors.joining(" "));
79      }
80  
81      public static void main(String[] args) {
82          LambdaEagerInitTest test = new LambdaEagerInitTest();
83          test.nonCapturingLambda();
84          test.capturingLambda();
85      }
86  }
