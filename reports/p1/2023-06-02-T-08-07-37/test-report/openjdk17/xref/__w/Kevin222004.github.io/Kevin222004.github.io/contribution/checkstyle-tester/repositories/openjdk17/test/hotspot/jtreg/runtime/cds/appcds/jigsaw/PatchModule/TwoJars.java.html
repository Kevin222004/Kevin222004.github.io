25  /*
26   * @test
27   * @requires vm.cds
28   * @summary a patched class found in --patch-module should be used at runtime
29   * @library ../..
30   * @library /test/hotspot/jtreg/testlibrary
31   * @library /test/lib
32   * @build PatchMain
33   * @run driver TwoJars
34   */
35  
36  import java.io.File;
37  import jdk.test.lib.compiler.InMemoryJavaCompiler;
38  import jdk.test.lib.process.OutputAnalyzer;
39  import jdk.test.lib.helpers.ClassFileInstaller;
40  
41  public class TwoJars {
42      private static String moduleJar;
43      private static String moduleJar2;
44  
45      public static void main(String args[]) throws Throwable {
46  
47          // Create a class file in the module java.naming. This class file
48          // will be put in the javanaming.jar file.
49          String source = "package javax.naming.spi; "                +
50                          "public class NamingManager { "             +
51                          "    static { "                             +
52                          "        System.out.println(\"I pass!\"); " +
53                          "    } "                                    +
54                          "}";
55  
56          // Create a class file in the module java.naming. This class file
57          // will be put in the javanaming2.jar file.
58          String source2 = "package javax.naming.spi; "                +
59                          "public class DirectoryManager { "             +
60                          "    static { "                             +
61                          "        System.out.println(\"I fail!\"); " +
62                          "    } "                                    +
63                          "}";
64  
65          ClassFileInstaller.writeClassToDisk("javax/naming/spi/NamingManager",
66               InMemoryJavaCompiler.compile("javax.naming.spi.NamingManager", source, "--patch-module=java.naming"),
67               System.getProperty("test.classes"));
68  
69          // Build the jar file that will be used for the module "java.naming".
70          JarBuilder.build("javanaming", "javax/naming/spi/NamingManager");
71          moduleJar = TestCommon.getTestJar("javanaming.jar");
72  
73          ClassFileInstaller.writeClassToDisk("javax/naming/spi/DirectoryManager",
74               InMemoryJavaCompiler.compile("javax.naming.spi.DirectoryManager", source2, "--patch-module=java.naming"),
75               System.getProperty("test.classes"));
76  
77          // Build the jar file that will be used for the module "java.naming".
78          JarBuilder.build("javanaming2", "javax/naming/spi/DirectoryManager");
79          moduleJar2 = TestCommon.getTestJar("javanaming2.jar");
80  
81          System.out.println("Test dumping with --patch-module");
82          OutputAnalyzer output =
83              TestCommon.dump(null,
84                  TestCommon.list("javax/naming/spi/NamingManager"),
85                  "--patch-module=java.naming=" + moduleJar2 + File.pathSeparator + moduleJar,
86                  "-Xlog:class+load",
87                  "-Xlog:class+path=info",
88                  "PatchMain", "javax.naming.spi.NamingManager");
89          output.shouldHaveExitValue(1)
90                .shouldContain("Cannot use the following option when dumping the shared archive: --patch-module");
91  
92          TestCommon.run(
93              "-XX:+UnlockDiagnosticVMOptions",
94              "--patch-module=java.naming=" + moduleJar2 + File.pathSeparator + moduleJar,
95              "-Xlog:class+load",
96              "-Xlog:class+path=info",
97              "PatchMain", "javax.naming.spi.NamingManager")
98              .assertSilentlyDisabledCDS(0, "I pass!");
99      }
100 }
