595     @SupportedAnnotationTypes("*")
596     @SupportedOptions("expectedEnclosedElements")
597     public static final class ListRequires extends AbstractProcessor {
598 
599         private int round;
600 
601         @Override
602         public boolean process(Set<? extends TypeElement> annotations, RoundEnvironment roundEnv) {
603             if (round++ == 0) {
604                 ModuleElement compileE = processingEnv.getElementUtils().getModuleElement("compile");
605                 ModuleElement testE = ElementFilter.exportsIn(compileE.getDirectives()).get(0).getTargetModules().get(0);
606 
607                 System.out.println("from directives:");
608                 for (RequiresDirective rd : ElementFilter.requiresIn(testE.getDirectives())) {
609                     System.out.println(rd.getDependency().getQualifiedName());
610                 }
611 
612                 System.out.println("from requires:");
613                 for (RequiresDirective rd : ((ModuleSymbol) testE).requires) {
614                     System.out.println(rd.getDependency().getQualifiedName());
615                 }
616             }
617 
618             return false;

795     @SupportedAnnotationTypes("*")
796     public static final class UnnamedPackageProcessor extends AbstractProcessor {
797 
798         int round = 0;
799 
800         @Override
801         public boolean process(Set<? extends TypeElement> annotations, RoundEnvironment roundEnv) {
802             if (round++ != 0)
803                 return false;
804 
805             Elements elements = processingEnv.getElementUtils();
806             PackageElement pe = elements.getPackageElement("");
807 
808             if (pe == null) {
809                 System.out.println("null");
810             } else {
811                 ModuleElement mod = (ModuleElement) pe.getEnclosingElement();
812                 if (mod == null) {
813                     System.out.println("noModule");
814                 } else if (mod.isUnnamed()) {
815                     System.out.println("unnamedModule");
816                 } else {
817                     System.out.println(mod);
818                 }
819             }
820 
821             ModuleElement m1x = elements.getModuleElement("m1x");
822             ModuleElement m2x = elements.getModuleElement("m2x");
823 
824             if (m1x != null && m2x != null) {
825                 System.out.println("m1x: " + (elements.getPackageElement(m1x, "") != null));
826                 System.out.println("m2x: " + (elements.getPackageElement(m2x, "") != null));
827             }
828 
829             return false;
830         }
831 
832     }
