124     private static void test() throws Exception {
125         byte[] p12 = new byte[P12_FILE.length() / 2];
126         for (int i = 0; i < p12.length; i++) {
127             p12[i] = Integer.valueOf(P12_FILE.substring(2 * i, 2 * i + 2), 16)
128                     .byteValue();
129         }
130         KeyStore ks = KeyStore.getInstance("pkcs12");
131         ks.load(new ByteArrayInputStream(p12), PASS);
132         System.out.println(ks.getKey("a", PASS));
133     }
134 
135     private static void reEncodeWithPBES2() throws Exception {
136         KeyStore keyStore = KeyStore.getInstance(new File("ks"), PASS);
137         KeyStore.PrivateKeyEntry privateKeyEntry = new KeyStore.PrivateKeyEntry(
138                 (PrivateKey)keyStore.getKey("a", PASS),
139                 new Certificate[] { keyStore.getCertificate("a") });
140         keyStore.setEntry("a", privateKeyEntry, new KeyStore.PasswordProtection(
141                 PASS, "PBEWithHmacSHA512AndAES_256",
142                 new PBEParameterSpec(new byte[20], 100_000)));
143         ByteArrayOutputStream bout = new ByteArrayOutputStream();
144         keyStore.store(bout, PASS);
145         byte[] p12 = bout.toByteArray();
146         StringBuilder sb = new StringBuilder();
147         for (int i = 0; i < p12.length; i++) {
148             if (i % 32 == 0) {
149                 if (i != 0) {
150                     sb.append("\" +\n");
151                 }
152                 sb.append("\"");
153             }
154             sb.append(String.format("%02x", p12[i] & 0xff));
155         }
156         sb.append("\";\n");
157         System.out.println(sb.toString());
158     }
