25  /*
26   * @test
27   * @summary Regression test for JDK-8098821
28   * @bug 8098821
29   * @requires vm.cds.archived.java.heap
30   * @library /test/lib /test/hotspot/jtreg/runtime/cds/appcds
31   * @run driver SysDictCrash
32   */
33  
34  import jdk.test.lib.cds.CDSOptions;
35  import jdk.test.lib.cds.CDSTestUtils;
36  import jdk.test.lib.process.OutputAnalyzer;
37  
38  public class SysDictCrash {
39      public static void main(String[] args) throws Exception {
40          SharedStringsUtils.run(args, SysDictCrash::test);
41      }
42  
43      public static void test(String[] args) throws Exception {
44          String vmOptionsPrefix[] = SharedStringsUtils.getChildVMOptionsPrefix();
45  
46          // SharedBaseAddress=0 puts the archive at a very high address, which provokes the crash.
47          boolean continueTest = true;
48  
49          CDSOptions opts = (new CDSOptions())
50              .addPrefix(vmOptionsPrefix,
51                         "-XX:+UseG1GC", "-XX:MaxRAMPercentage=12.5",
52                         "-cp", ".",
53                         "-XX:SharedBaseAddress=0",
54                         "-showversion", "-Xlog:cds,cds+hashtables")
55              .setArchiveName("./SysDictCrash.jsa");
56          OutputAnalyzer output = CDSTestUtils.createArchive(opts);
57          try {
58              TestCommon.checkDump(output);
59          } catch (java.lang.RuntimeException re) {
60              if (!output.getStdout().contains("UseCompressedOops and UseCompressedClassPointers have been disabled due to")) {
61                  throw re;
62              } else {
63                  System.out.println("Shared archive was not created due to UseCompressedOops and UseCompressedClassPointers have been disabled.");
64                  continueTest = false;
65              }
66          }
67  
68          if (!continueTest) {
69              return;
70          }
71  
72          opts = (new CDSOptions())
73              .setArchiveName("./SysDictCrash.jsa") // prevents the assignment of a default archive name
74              .setUseVersion(false) // the -version must be the last arg for this test to work
75              .addSuffix(vmOptionsPrefix,
76                         "-Xlog:cds",
77                         "-XX:+UseG1GC", "-XX:MaxRAMPercentage=12.5",
78                         "-XX:SharedArchiveFile=./SysDictCrash.jsa",
79                         "-version");
80          CDSTestUtils.run(opts)
81                      .assertNormalExit();
82      }
83  }
