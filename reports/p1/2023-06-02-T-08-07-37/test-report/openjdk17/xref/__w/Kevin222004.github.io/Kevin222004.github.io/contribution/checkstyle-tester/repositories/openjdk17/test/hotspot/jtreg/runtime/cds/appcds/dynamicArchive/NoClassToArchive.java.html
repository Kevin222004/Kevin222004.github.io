25  /*
26   * @test
27   * @summary A few edge cases where there's no class to be included in the dynamic archive.
28   * @requires vm.cds & !vm.graal.enabled
29   * @comment The test assumes that when "java -version" is executed, only a very limited number
30   *          of classes are loaded, and all of those are loaded from the default shared archive.
31   *
32   *          However, when graal is used as the JIT, many extra classes are loaded during VM start-up.
33   *          Some of those are loaded dynamically from jrt:/. Some classes are also defined by
34   *          LambdaMetafactory. This causes complexity that cannot be easily handled by this test.
35   *
36   *          The VM code covered by this test can be sufficiently tested with C1/C2. So there's no need
37   *          to bend over backwards to run this test with graal.
38   *
39   * @library /test/lib /test/hotspot/jtreg/runtime/cds/appcds /test/hotspot/jtreg/runtime/cds/appcds/dynamicArchive/test-classes
40   * @build StrConcatApp
41   * @build sun.hotspot.WhiteBox
42   * @run driver jdk.test.lib.helpers.ClassFileInstaller -jar strConcatApp.jar StrConcatApp
43   * @run driver jdk.test.lib.helpers.ClassFileInstaller sun.hotspot.WhiteBox
44   * @run main/othervm -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI -Xbootclasspath/a:. NoClassToArchive
45   */
46  
47  import java.io.File;
48  import jdk.test.lib.cds.CDSOptions;
49  import jdk.test.lib.cds.CDSTestUtils;
50  import jdk.test.lib.process.OutputAnalyzer;
51  import jdk.test.lib.helpers.ClassFileInstaller;
52  
53  public class NoClassToArchive extends DynamicArchiveTestBase {
54      static final String warningMessage =
55          "There is no class to be included in the dynamic archive";
56      static final String classList = CDSTestUtils.getOutputFileName("classlist");
57      static final String appClass = "StrConcatApp";
58  
59      public static void main(String[] args) throws Exception {
60          runTest(NoClassToArchive::testDefaultBase);
61          runTest(NoClassToArchive::testCustomBase);
62      }
63  
64      // (1) Test with default base archive + top archive
65      static void testDefaultBase() throws Exception {
66          String topArchiveName = getNewArchiveName("top");
67          doTest(topArchiveName);
68      }
69  
70      // (2) Test with custom base archive + top archive
71      static void testCustomBase() throws Exception {
72          String topArchiveName = getNewArchiveName("top2");
73          String baseArchiveName = getNewArchiveName("base");
74          doTestCustomBase(baseArchiveName, topArchiveName);
75      }
76  
77      private static void checkWarning(OutputAnalyzer output) throws Exception {
78          if (output.firstMatch("bytes: [0-9]+ checksum: [0-9a-f]+") != null) {
79              // Patterns like this indicate that a class was not loaded from CDS archive:
80              // [info ][class,load] jdk.internal.module.DefaultRoots$$Lambda$1/0x00007f80c4512048 source: jdk.internal.module.DefaultRoots
81              // [debug][class,load]  klass: 0x0000000800b77cf8 super: 0x0000000800007450 interfaces: 0x0000000800162538
82              //                      loader: [loader data: 0x00007f80f416a5b0 of 'bootstrap'] bytes: 403 checksum: 753e58aa
83              System.out.println("test skipped: this platform uses non-archived classes when running -version");
84          } else {
85              output.shouldContain(warningMessage);
86          }
87      }
88  
89      private static void doTest(String topArchiveName) throws Exception {
90          dump(topArchiveName,
91               "-Xlog:cds",
92               "-Xlog:cds+dynamic=debug",
93               "-Xlog:class+load=trace",
94               "-version")
95              .assertNormalExit(output -> checkWarning(output));
96  
97          dump(topArchiveName,
98               "-Xlog:cds",
99               "-Xlog:cds+dynamic=debug",
100              "-Xlog:class+load=trace",
101              "-help")
102             .assertNormalExit(output -> {
103                     // some classes will be loaded from the java.base module
104                     output.shouldContain("java.text.MessageFormat source: jrt:/java.base");
105                 });
106     }
107 
108     private static void doTestCustomBase(String baseArchiveName, String topArchiveName) throws Exception {
109         String appJar = ClassFileInstaller.getJarPath("strConcatApp.jar");
110         // dump class list by running the StrConcatApp
111         CDSTestUtils.dumpClassList(classList, "-cp", appJar, appClass)
112             .assertNormalExit(output -> {
113                 output.shouldContain("length = 0");
114             });
115 
116         // create a custom base archive based on the class list
117         TestCommon.dumpBaseArchive(baseArchiveName, "-XX:SharedClassListFile=" + classList);
118 
119         // create a dynamic archive with the custom base archive
120         // no class should be included in the dynamic archive
121         dump2(baseArchiveName, topArchiveName,
122               "-Xlog:cds",
123               "-Xlog:cds+dynamic=debug",
124               "-Xlog:class+load=trace",
125               "-version")
126             .assertNormalExit(out -> checkWarning(out));
127     }
128 }
