24  /* @test
25   * @bug 8081800 8165827
26   * @summary Add JNI invocation tests for private interface methods
27   * @run main/native PrivateInterfaceMethods
28   */
29  
30  public class PrivateInterfaceMethods {
31  
32      static {
33          System.loadLibrary("PrivateInterfaceMethods");
34      }
35  
36      static native int callIntVoid(Object target, String definingClassName, String methodName, boolean virtual);
37      static native void lookupIntVoid(String definingClassName, String methodName);
38  
39      static interface A {
40          static final int AmResult = 1;
41          private int m() { return AmResult; }
42          private int onlyA() { return 0; }
43      }
44  
45      static interface B extends A {
46          // No m() here
47          private int onlyB() { return 0; }
48      }
49  
50      static interface C extends B {
51          static final int CmResult = 2;
52          private int m() { return CmResult; }  // unrelated to A.m
53          private int onlyC() { return 0; }
54      }
55  
56      public static class Impl implements C {
57          static final int ImplmResult = 3;
58          private int m() { return ImplmResult; } // unrelated to A.m or C.m
59      }
60  
61      // We found that itable/vtable construction was affected by whether or not the
62      // implementation class declared a method with the same signature as the
63      // private interface method, so we test both variants.
64  
65      public static class Impl2 implements C {
66      }
67  
68      public static void main(String[] args) {
69  
70          // JNI getMethodID only works for methods declared in or inherited by a type.
71          // Private interface methods are not inherited and so should only be found
72          // in the declaring interface.
73  
74          lookup(A.class.getName(), "onlyA", null); // should succeed
75          lookup(B.class.getName(), "onlyA", NoSuchMethodError.class); // should fail
76          lookup(C.class.getName(), "onlyA", NoSuchMethodError.class); // should fail
77          lookup(Impl.class.getName(), "onlyA", NoSuchMethodError.class); // should fail
78          lookup(Impl2.class.getName(), "onlyA", NoSuchMethodError.class); // should fail
79  
80          lookup(B.class.getName(), "onlyB", null); // should succeed
81          lookup(A.class.getName(), "onlyB", NoSuchMethodError.class); // should fail
82          lookup(C.class.getName(), "onlyB", NoSuchMethodError.class); // should fail
83          lookup(Impl.class.getName(), "onlyB", NoSuchMethodError.class); // should fail
84          lookup(Impl2.class.getName(), "onlyB", NoSuchMethodError.class); // should fail
85  
86          lookup(C.class.getName(), "onlyC", null); // should succeed
87          lookup(A.class.getName(), "onlyC", NoSuchMethodError.class); // should fail
88          lookup(B.class.getName(), "onlyC", NoSuchMethodError.class); // should fail
89          lookup(Impl.class.getName(), "onlyC", NoSuchMethodError.class); // should fail
90          lookup(Impl2.class.getName(), "onlyC", NoSuchMethodError.class); // should fail
91  
92          Impl impl = new Impl();
93  
94          // Note: JNI doesn't enforce access control so we can make
95          //       private calls not possible in Java code.
96          //       Also it doesn't check that the receiver is a type that
97          //       defines the method!
98  
99          // test: ((A)impl).m() - should succeed
100         test(impl, A.class.getName(), "m", A.AmResult, true, null);
101         test(impl, A.class.getName(), "m", A.AmResult, false, null);
102 
103         // test: ((B)impl).m() - should fail: NoSuchMethodError
104         test(impl, B.class.getName(), "m", -1, true, NoSuchMethodError.class);
105         test(impl, B.class.getName(), "m", -1, false, NoSuchMethodError.class);
106 
107         // test: ((C)impl).m() - should succeed
108         test(impl, C.class.getName(), "m", C.CmResult, true, null);
109         test(impl, C.class.getName(), "m", C.CmResult, false, null);
110 
111         // test: impl.m() - should succeed
112         test(impl, Impl.class.getName(), "m", Impl.ImplmResult, true, null);
113         test(impl, Impl.class.getName(), "m", Impl.ImplmResult, false, null);
114 
115         // ---
116 
117         Impl2 impl2 = new Impl2();
118 
119         // test: ((A)impl2).m() - should succeed
120         test(impl2, A.class.getName(), "m", A.AmResult, true, null);
121         test(impl2, A.class.getName(), "m", A.AmResult, false, null);
122 
123         // test: ((B)impl2).m() - should fail: NoSuchMethodError
124         test(impl2, B.class.getName(), "m", -1, true, NoSuchMethodError.class);
125         test(impl2, B.class.getName(), "m", -1, false, NoSuchMethodError.class);
126 
127         // test: ((C)impl2).m() - should succeed
128         test(impl2, C.class.getName(), "m", C.CmResult, true, null);
129         test(impl2, C.class.getName(), "m", C.CmResult, false, null);
130 
131         // test: impl2.m() - should fail: NoSuchMethodError
132         test(impl2, Impl2.class.getName(), "m", -1, true, NoSuchMethodError.class);
133         test(impl2, Impl2.class.getName(), "m", -1, false, NoSuchMethodError.class);
134     }
135 
136     static void lookup(String definingClass, String method, Class<?> expectedException) {
137 
138         String desc = "Lookup of " + definingClass + "." + method;
139         try {
140             lookupIntVoid(definingClass, method);
141             if (expectedException != null)
142                 throw new Error(desc + " succeeded - but expected exception "
143                                 + expectedException.getSimpleName());
144             System.out.println(desc + " - passed");
145         }
146         catch (Throwable t) {
147            if (t.getClass() != expectedException)
148                throw new Error(desc + " failed: got exception " + t + " but expected exception "
149                                + expectedException.getSimpleName());
150            else
151               System.out.println(desc + " threw " + expectedException.getSimpleName() + " as expected");
152         }
153     }
154 
155     static void test(Object target, String definingClass, String method,
156                      int expected, boolean virtual, Class<?> expectedException) {
157 
158         String desc = (virtual ? "Virtual" : "Nonvirtual") + " Invocation of " +
159                        definingClass + "." + method + " on instance of class " +
160                        target.getClass().getName();
161         try {
162             int res = callIntVoid(target, definingClass, method, virtual);
163             if (expectedException != null)
164                 throw new Error(desc + " succeeded - but expected exception "
165                                 + expectedException.getSimpleName());
166             if (res != expected)
167                 throw new Error(desc + " got wrong result: " + res + " instead of " + expected);
168             System.out.println(desc + " - passed");
169         }
170         catch (Throwable t) {
171            if (t.getClass() != expectedException)
172                throw new Error(desc + " failed: got exception " + t + " but expected exception "
173                                + expectedException.getSimpleName());
174            else
175               System.out.println(desc + " threw " + expectedException.getSimpleName() + " as expected");
176         }
177     }
