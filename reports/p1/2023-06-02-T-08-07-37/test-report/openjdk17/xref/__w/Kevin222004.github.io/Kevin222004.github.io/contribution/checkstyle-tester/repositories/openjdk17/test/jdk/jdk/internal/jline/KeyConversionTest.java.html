24  /**
25   * @test
26   * @bug 8080679
27   * @summary Verify the conversion from key events to escape sequences works properly.
28   * @modules jdk.internal.le/jdk.internal.org.jline.terminal
29   *          jdk.internal.le/jdk.internal.org.jline.terminal.impl
30   */
31  
32  import java.io.IOException;
33  import java.io.StringWriter;
34  import java.nio.charset.Charset;
35  
36  import jdk.internal.org.jline.terminal.Size;
37  import jdk.internal.org.jline.terminal.impl.AbstractWindowsTerminal;
38  
39  public class KeyConversionTest {
40      public static void main(String... args) throws Exception {
41          new KeyConversionTest().run();
42      }
43  
44      void run() throws Exception {
45          checkKeyConversion(new KeyEvent(true, (short) 37, '\0', 256), "\033OD"); //LEFT
46          checkKeyConversion(new KeyEvent(true, (short) 37, '\0', 264), "\033[1;5D"); //Ctrl-LEFT
47          checkKeyConversion(new KeyEvent(true, (short) 37, '\0', 258), "\033[1;3D"); //Alt-LEFT
48          checkKeyConversion(new KeyEvent(true, (short) 112, '\0', 256), "\033OP"); //F1
49          checkKeyConversion(new KeyEvent(true, (short) 112, '\0', 264), "\033[1;5P"); //Ctrl-F1
50          checkKeyConversion(new KeyEvent(true, (short) 112, '\0', 258), "\033[1;3P"); //Alt-F1
51          checkKeyConversion(new KeyEvent(true, (short) 112, '\0', 272), "\033[1;2P"); //Shift-F1
52          checkKeyConversion(new KeyEvent(true, (short) 112, '\0', 280), "\033[1;6P"); //Ctrl-Shift-F1
53          checkKeyConversion(new KeyEvent(true, (short) 112, '\0', 274), "\033[1;4P"); //Alt-Shift-F1
54          checkKeyConversion(new KeyEvent(true, (short) 112, '\0', 282), "\033[1;8P"); //Ctrl-Alt-Shift-F1
55          checkKeyConversion(new KeyEvent(true, (short) 67, '\003', 8), "\003"); //Ctrl-C
56      }
57  
58      void checkKeyConversion(KeyEvent event, String expected) throws IOException {
59          StringBuilder result = new StringBuilder();
60          new AbstractWindowsTerminal(new StringWriter(), "", "windows", Charset.forName("UTF-8"),
61                                      0, true, null, in -> in) {
62              @Override
63              protected int getConsoleOutputCP() {
64                  throw new UnsupportedOperationException("Not supported yet.");
65              }
66              @Override
67              protected int getConsoleMode() {
68                  return 0;
69              }
70              @Override
71              protected void setConsoleMode(int mode) {
72                  throw new UnsupportedOperationException("Not supported yet.");
73              }
74              @Override
75              protected boolean processConsoleInput() throws IOException {
76                  throw new UnsupportedOperationException("Not supported yet.");
77              }
78              @Override
79              public Size getSize() {
80                  throw new UnsupportedOperationException("Not supported yet.");
81              }
82              @Override
83              public void processInputChar(char c) throws IOException {
84                  result.append(c);
85              }
86              @Override
87              public void processKeyEvent(boolean isKeyDown, short virtualKeyCode,
88                                          char ch, int controlKeyState) throws IOException {
89                  super.processKeyEvent(isKeyDown, virtualKeyCode, ch, controlKeyState);
90              }
91          }.processKeyEvent(event.isKeyDown, event.virtualKeyCode, event.ch, event.controlKeyState);
92          String actual = result.toString();
93  
94          if (!expected.equals(actual)) {
95              throw new AssertionError("Expected: " + expected + "; actual: " + actual);
96          }
97      }
98  
99      public static class KeyEvent {
100         public final boolean isKeyDown;
101         public final short virtualKeyCode;
102         public final char ch;
103         public final int controlKeyState;
104 
105         public KeyEvent(boolean isKeyDown, short virtualKeyCode, char ch, int controlKeyState) {
106             this.isKeyDown = isKeyDown;
107             this.virtualKeyCode = virtualKeyCode;
108             this.ch = ch;
109             this.controlKeyState = controlKeyState;
110         }
111 
112     }
113 }
