90      private static class ServiceItem {
91          private enum State {
92              REMOVED, UNCHANGED, ADDED
93          }
94  
95          final String name;
96          State state;
97  
98          private ServiceItem(final String name) {
99              this.name = name;
100             state = State.UNCHANGED;
101         }
102 
103         @Override
104         public String toString() {
105             return name;
106         }
107 
108         @Override
109         public boolean equals(Object obj) {
110             return (obj instanceof ServiceItem)
111                     && ((ServiceItem) obj).name.equals(name);
112         }
113 
114         @Override
115         public int hashCode() {
116             return name.hashCode();
117         }
118     }

120     private static class ServiceItemListModel extends AbstractListModel<ServiceItem> {
121         private final List<ServiceItem> list;
122 
123         private ServiceItemListModel(List<ServiceItem> list) {
124             this.list = list;
125         }
126 
127         @Override
128         public int getSize() {
129             return list.size();
130         }
131 
132         @Override
133         public ServiceItem getElementAt(int index) {
134             return list.get(index);
135         }
136 
137         private void refreshList(List<ServiceItem> newList) {
138             list.clear();
139             list.addAll(newList);
140             fireChanged();
141         }
142 
143         private void fireChanged() {
144             fireContentsChanged(this, 0, list.size() - 1);
145         }
146     }

148     private static class ServiceItemListRenderer extends DefaultListCellRenderer {
149         @Override
150         public Component getListCellRendererComponent(JList<?> list,
151                                                       Object value,
152                                                       int index,
153                                                       boolean isSelected,
154                                                       boolean cellHasFocus) {
155             Component component =
156                     super.getListCellRendererComponent(list, value, index,
157                                                        isSelected, cellHasFocus);
158             switch (((ServiceItem) value).state) {
159                 case REMOVED:
160                     component.setBackground(Color.RED);
161                     component.setForeground(Color.WHITE);
162                     break;
163                 case ADDED:
164                     component.setBackground(Color.GREEN);
165                     component.setForeground(Color.BLACK);
166                     break;
167                 case UNCHANGED:
168                 default:
169                     break;
170             }
171             return component;
172         }
173     }

213     private RemotePrinterStatusRefresh() {
214         frame = new JFrame("RemotePrinterStatusRefresh");
215         frame.addWindowListener(this);
216 
217 
218         JPanel northPanel = new JPanel(new BorderLayout());
219         northPanel.add(createInfoPanel(), BorderLayout.NORTH);
220         northPanel.add(createInstructionsPanel(), BorderLayout.SOUTH);
221 
222 
223         beforeList = new ServiceItemListModel(
224                 Collections.unmodifiableList(collectPrinterList()));
225         afterList = new ServiceItemListModel(new ArrayList<>());
226         logList("Before:", beforeList.list);
227 
228         JPanel listPanel = new JPanel(new GridLayout(1, 2));
229         listPanel.setBorder(createTitledBorder("Print Services"));
230         listPanel.add(createListPanel(beforeList, "Before:", 'b'));
231         listPanel.add(createListPanel(afterList, "After:", 'a'));
232 
233 
234         JPanel mainPanel = new JPanel(new BorderLayout());
235         mainPanel.add(northPanel, BorderLayout.NORTH);
236         mainPanel.add(listPanel, BorderLayout.CENTER);
237         mainPanel.add(createButtonPanel(), BorderLayout.SOUTH);
238 
239 
240         frame.add(mainPanel);
241         frame.pack();
242         refreshButton.requestFocusInWindow();
243         frame.setVisible(true);
244 
245 
246         timer = new Timer(1000, this::updateTimeLeft);
247         timer.start();
248         startTime = System.currentTimeMillis();
249         updateTimeLeft(null);
250     }

303                                    final String title,

305         JPanel panel = new JPanel();
306         panel.setLayout(new BoxLayout(panel, BoxLayout.Y_AXIS));
307         JList<ServiceItem> list = new JList<>(model);
308         list.setCellRenderer(new ServiceItemListRenderer());
309 
310         JLabel label = new JLabel(title);
311         label.setLabelFor(list);
312         label.setDisplayedMnemonic(mnemonic);
313         JPanel labelPanel = new JPanel();
314         labelPanel.setLayout(new BoxLayout(labelPanel, BoxLayout.X_AXIS));
315         labelPanel.add(label, BorderLayout.EAST);
316         labelPanel.add(Box.createHorizontalGlue());
317 
318         panel.add(labelPanel);
319         panel.add(new JScrollPane(list));
320         return panel;
321     }

345     private static List<ServiceItem> collectPrinterList() {
346         PrintService[] printServices = PrintServiceLookup.lookupPrintServices(null, null);
347         List<ServiceItem> list = new ArrayList<>(printServices.length);
348         for (PrintService service : printServices) {
349             list.add(new ServiceItem(service.getName()));
350         }
351         return list;
352     }

354     private static void logList(final String title, final List<ServiceItem> list) {
355         System.out.println(title);
356         for (ServiceItem item : list) {
357             System.out.println(item.name);
358         }
359         System.out.println();
360     }

362     private static void compareLists(final ServiceItemListModel before, final ServiceItemListModel after) {
363         boolean beforeUpdated = false;
364         boolean afterUpdated = false;
365 
366         for (ServiceItem item : before.list) {
367             if (!after.list.contains(item)) {
368                 item.state = ServiceItem.State.REMOVED;
369                 beforeUpdated = true;
370             } else if (item.state != ServiceItem.State.UNCHANGED) {
371                 item.state = ServiceItem.State.UNCHANGED;
372                 beforeUpdated = true;
373             }
374         }
375 
376         for (ServiceItem item : after.list) {
377             if (!before.list.contains(item)) {
378                 item.state = ServiceItem.State.ADDED;
379                 afterUpdated = true;
380             } else if (item.state != ServiceItem.State.UNCHANGED) {
381                 item.state = ServiceItem.State.UNCHANGED;
382                 afterUpdated = true;
383             }
384         }
385 
386         if (beforeUpdated) {
387             before.fireChanged();
388         }
389         if (afterUpdated) {
390             after.fireChanged();
391         }
392     }

406     @SuppressWarnings("unused")
407     private void refresh(ActionEvent e) {
408         System.out.println("Refresh button pressed");
409         afterList.refreshList(collectPrinterList());
410         compareLists(beforeList, afterList);
411         passButton.setEnabled(true);
412         failButton.setEnabled(true);
413         logList("After:", afterList.list);
414     }

430     @SuppressWarnings("unused")
431     private void updateTimeLeft(ActionEvent e) {
432         long elapsed = (System.currentTimeMillis() - startTime) / 1000;
433         long left = TIMEOUT - elapsed;
434         if (left < 0) {
435             testTimedOut = true;
436             disposeUI();
437         }
438         timeLeft.setText(formatTime(left));
439     }
440 
441     private static String formatTime(final long seconds) {
442         long minutes = seconds / 60;
443         return String.format("%d:%02d", minutes, seconds - minutes * 60);
444     }
