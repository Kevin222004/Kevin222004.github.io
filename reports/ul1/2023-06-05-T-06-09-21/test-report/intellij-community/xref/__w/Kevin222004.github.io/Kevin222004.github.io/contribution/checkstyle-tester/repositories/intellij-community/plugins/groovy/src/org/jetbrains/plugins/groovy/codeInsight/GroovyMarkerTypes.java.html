51    static final MarkerType OVERRIDING_PROPERTY_TYPE = new MarkerType("OVERRIDING_PROPERTY_TYPE", psiElement -> {
52      final PsiElement parent = psiElement.getParent();
53      if (!(parent instanceof GrField field)) return null;
54  
55      final List<GrAccessorMethod> accessors = GroovyPropertyUtils.getFieldAccessors(field);
56      StringBuilder builder = new StringBuilder();
57      builder.append("<html><body>");
58      int count = 0;
59      String sep = "";
60      for (GrAccessorMethod method : accessors) {
61        PsiMethod[] superMethods = method.findSuperMethods(false);
62        count += superMethods.length;
63        if (superMethods.length == 0) continue;
64        PsiMethod superMethod = superMethods[0];
65        boolean isAbstract = method.hasModifierProperty(PsiModifier.ABSTRACT);
66        boolean isSuperAbstract = superMethod.hasModifierProperty(PsiModifier.ABSTRACT);
67  
68        @NonNls final String key;
69        if (isSuperAbstract && !isAbstract) {
70          key = "method.implements.in";
71        }
72        else {
73          key = "method.overrides.in";
74        }
75        builder.append(sep);
76        sep = "<br>";
77        //noinspection UnresolvedPropertyKey
78        composeText(superMethods, DaemonBundle.message(key), builder);
79      }
80      if (count == 0) return null;
81      builder.append("</html></body>");
82      return builder.toString();
83    }, new LineMarkerNavigator() {
84      @Override
85      public void browse(MouseEvent e, PsiElement element) {
86        PsiElement parent = element.getParent();
87        if (!(parent instanceof GrField field)) return;
88        final List<GrAccessorMethod> accessors = GroovyPropertyUtils.getFieldAccessors(field);
89        final ArrayList<PsiMethod> superMethods = new ArrayList<>();
90        for (GrAccessorMethod method : accessors) {
91          Collections.addAll(superMethods, method.findSuperMethods(false));
92        }
93        if (superMethods.isEmpty()) return;
94        final PsiMethod[] supers = superMethods.toArray(PsiMethod.EMPTY_ARRAY);
95        boolean showMethodNames = !PsiUtil.allMethodsHaveSameSignature(supers);
96        PsiElementListNavigator.openTargets(e, supers,
97                                            DaemonBundle.message("navigation.title.super.method", field.getName()),
98                                            DaemonBundle.message("navigation.findUsages.title.super.method", field.getName()),
99                                            new MethodCellRenderer(showMethodNames));
100     }
101   });
102   static final MarkerType OVERRIDEN_PROPERTY_TYPE = new MarkerType("OVERRIDEN_PROPERTY_TYPE", element -> {
103     PsiElement parent = element.getParent();
104     if (!(parent instanceof GrField)) return null;
105     final List<GrAccessorMethod> accessors = GroovyPropertyUtils.getFieldAccessors((GrField)parent);
106 
107     PsiElementProcessor.CollectElementsWithLimit<PsiMethod> processor = new PsiElementProcessor.CollectElementsWithLimit<>(5);
108 
109     for (GrAccessorMethod method : accessors) {
110       OverridingMethodsSearch.search(method).forEach(new PsiElementProcessorAdapter<>(processor));
111     }
112     if (processor.isOverflow()) {
113       return DaemonBundle.message("method.is.overridden.too.many");
114     }
115 
116     PsiMethod[] overridings = processor.toArray(new PsiMethod[processor.getCollection().size()]);
117     if (overridings.length == 0) return null;
118 
119     Comparator<PsiMethod> comparator = PsiElementRenderingInfo.getComparator(new PsiMethodRenderingInfo(false));
120     Arrays.sort(overridings, comparator);
121 
122     String start = DaemonBundle.message("method.is.overriden.header");
123     @NonNls String pattern = "&nbsp;&nbsp;&nbsp;&nbsp;{1}";
124     return GutterIconTooltipHelper.composeText(overridings, start, pattern);
125   }, new LineMarkerNavigator() {
126     @Override
127     public void browse(MouseEvent e, PsiElement element) {
128       PsiElement parent = element.getParent();
129       if (!(parent instanceof GrField field)) return;
130       if (DumbService.isDumb(element.getProject())) {
131         DumbService.getInstance(element.getProject()).showDumbModeNotification(
132           GroovyBundle.message("popup.content.navigation.to.overriding.classes.unavailable")
133         );
134         return;
135       }
136 
137 
138       Set<PsiMethod> result = new HashSet<>();
139       Processor<PsiMethod> collectProcessor = Processors.cancelableCollectProcessor(result);
140       if (!ProgressManager.getInstance().runProcessWithProgressSynchronously(() -> ApplicationManager.getApplication().runReadAction(() -> {
141         for (GrAccessorMethod method : GroovyPropertyUtils.getFieldAccessors(field)) {
142           OverridingMethodsSearch.search(method).forEach(collectProcessor);
143         }
144       }), JavaAnalysisBundle.message("searching.for.overriding.methods"), true, field.getProject(), (JComponent)e.getComponent())) {
145         return;
146       }
147 
148       PsiMethod[] overridings = result.toArray(PsiMethod.EMPTY_ARRAY);
149       if (overridings.length == 0) return;
150       String title = DaemonBundle.message("navigation.title.overrider.method", field.getName(), overridings.length);
151       boolean showMethodNames = !PsiUtil.allMethodsHaveSameSignature(overridings);
152       MethodCellRenderer renderer = new MethodCellRenderer(showMethodNames);
153       Arrays.sort(overridings, renderer.getComparator());
154       PsiElementListNavigator.openTargets(e, overridings, title, GroovyBundle.message("overriding.methods.of.0",field.getName()) , renderer);
155     }
156   }

195   public static final MarkerType GR_OVERRIDDEN_METHOD =
196     new MarkerType("GR_OVERRIDEN_METHOD", (NullableFunction<PsiElement, String>)element -> {
197       PsiElement parent = element.getParent();
198       if (!(parent instanceof GrMethod method)) return null;
199 
200       final PsiElementProcessor.CollectElementsWithLimit<PsiMethod> processor =
201         new PsiElementProcessor.CollectElementsWithLimit<>(5);
202 
203       for (GrMethod m : PsiImplUtil.getMethodOrReflectedMethods(method)) {
204         OverridingMethodsSearch.search(m).forEach(
205           new ReadActionProcessor<>() {
206             @Override
207             public boolean processInReadAction(PsiMethod method) {
208               if (method instanceof GrTraitMethod) {
209                 return true;
210               }
211               return processor.execute(method);
212             }
213           });
214       }
215 
216       boolean isAbstract = method.hasModifierProperty(PsiModifier.ABSTRACT);
217 
218       if (processor.isOverflow()) {
219         return isAbstract ? DaemonBundle.message("method.is.implemented.too.many") : DaemonBundle.message("method.is.overridden.too.many");
220       }
221 
222       PsiMethod[] overridings = processor.toArray(new PsiMethod[processor.getCollection().size()]);
223       if (overridings.length == 0) return null;
224 
225       Comparator<PsiMethod> comparator = PsiElementRenderingInfo.getComparator(new PsiMethodRenderingInfo(false));
226       Arrays.sort(overridings, comparator);
227 
228       String start = isAbstract ? DaemonBundle.message("method.is.implemented.header") : DaemonBundle.message("method.is.overriden.header");
229       @NonNls String pattern = "&nbsp;&nbsp;&nbsp;&nbsp;{1}";
230       return GutterIconTooltipHelper.composeText(overridings, start, pattern);
231     }, new LineMarkerNavigator() {
232       @Override
233       public void browse(MouseEvent e, PsiElement element) {
234         PsiElement parent = element.getParent();
235         if (!(parent instanceof GrMethod method)) return;
236         if (DumbService.isDumb(element.getProject())) {
237           DumbService.getInstance(element.getProject()).showDumbModeNotification(
238             GroovyBundle.message("popup.content.navigation.to.overriding.classes.unavailable"));
239           return;
240         }
241 
242         final OverridingMethodsUpdater methodsUpdater = new OverridingMethodsUpdater(method);
243         new PsiTargetNavigator<>(() -> getOverridingMethods(method, (JComponent)e.getComponent()))
244           .tabTitle(GroovyBundle.message("overriding.methods.of.0", method.getName()))
245           .updater(methodsUpdater)
246           .navigate(e, null, element.getProject());
247       }
248     });

250   private static @NotNull Collection<PsiMethod> getOverridingMethods(GrMethod method, JComponent component) {
251     //collect all overriding methods (including fields with implicit accessors and method with default parameters)
252     final PsiElementProcessor.CollectElementsWithLimit<PsiMethod> collectProcessor =
253       new PsiElementProcessor.CollectElementsWithLimit<>(2, new HashSet<>());
254     for (GrMethod m : PsiImplUtil.getMethodOrReflectedMethods(method)) {
255       OverridingMethodsSearch.search(m).forEach(new ReadActionProcessor<>() {
256         @Override
257         public boolean processInReadAction(PsiMethod psiMethod) {
258           if (psiMethod instanceof GrReflectedMethod) {
259             psiMethod = ((GrReflectedMethod)psiMethod).getBaseMethod();
260           }
261           return collectProcessor.execute(psiMethod);
262         }
263       });
264     }
265     return collectProcessor.getCollection();
266   }

271   private static Set<PsiMethod> collectSuperMethods(GrMethod method) {
272     Set<PsiMethod> superMethods = new HashSet<>();
273     for (GrMethod m : PsiImplUtil.getMethodOrReflectedMethods(method)) {
274       for (PsiMethod superMethod : m.findSuperMethods(false)) {
275         if (superMethod instanceof GrReflectedMethod) {
276           superMethod = ((GrReflectedMethod)superMethod).getBaseMethod();
277         }
278 
279         superMethods.add(superMethod);
280       }
281     }
282     return superMethods;
283   }

285   private static StringBuilder composeText(PsiElement @NotNull [] elements, final String pattern, StringBuilder result) {
286     Set<String> names = new LinkedHashSet<>();
287     for (PsiElement element : elements) {
288       String methodName = ((PsiMethod)element).getName();
289       PsiClass aClass = ((PsiMethod)element).getContainingClass();
290       String className = aClass == null ? "" : ClassPresentationUtil.getNameForClass(aClass, true);
291       names.add(MessageFormat.format(pattern, methodName, className));
292     }
293 
294     @NonNls String sep = "";
295     for (String name : names) {
296       result.append(sep);
297       sep = "<br>";
298       result.append(name);
299     }
300     return result;
301   }

303   private static class OverridingMethodsUpdater extends TargetUpdaterTask {
304     private final GrMethod myMethod;
305 
306     OverridingMethodsUpdater(@NotNull GrMethod method) {
307       super(method.getProject(), JavaAnalysisBundle.message("searching.for.overriding.methods"));
308       myMethod = method;
309     }
310 
311     @Override
312     public @Nls @NotNull String getCaption(int size) {
313       return myMethod.hasModifierProperty(PsiModifier.ABSTRACT) ?
314              DaemonBundle.message("navigation.title.implementation.method", myMethod.getName(), size) :
315              DaemonBundle.message("navigation.title.overrider.method", myMethod.getName(), size);
316     }
317 
318     @Override
319     public void run(@NotNull final ProgressIndicator indicator) {
320       super.run(indicator);
321       for (PsiMethod method : PsiImplUtil.getMethodOrReflectedMethods(myMethod)) {
322         OverridingMethodsSearch.search(method).forEach(
323           new CommonProcessors.CollectProcessor<>() {
324             @Override
325             public boolean process(PsiMethod psiMethod) {
326               if (!updateComponent(com.intellij.psi.impl.PsiImplUtil.handleMirror(psiMethod))) {
327                 indicator.cancel();
328               }
329               indicator.checkCanceled();
330               return true;
331             }
332           });
333       }
334     }
