6    * @see <a href="https://openjdk.java.net/jeps/406">JEP 406: Pattern Matching for switch (Preview)</a>
7    */
8   public class GuardedAndParenthesizedPatterns {
9   
10  
11      static void test(Object o) {
12          switch (o) {
13              case String s && (s.length() == 1)    -> System.out.println("single char string");
14              case String s                         -> System.out.println("string");
15              case (Integer i && i.intValue() == 1) -> System.out.println("integer 1");
16              case (((Long l && l.longValue() == 1L))) -> System.out.println("long 1 with parens");
17              case (((Double d)))                   -> System.out.println("double with parens");
18              default                               -> System.out.println("default case");
19          }
20      }
21  
22      static void instanceOfPattern(Object o) {
23          if (o instanceof String s && s.length() > 2) {
24              System.out.println("A string containing at least two characters");
25          }
26          if (o != null && (o instanceof String s && s.length() > 3)) {
27              System.out.println("A string containing at least three characters");
28          }
29          if (o instanceof (String s && s.length() > 4)) {
30              System.out.println("A string containing at least four characters");
31          }
32      }
33  
34      public static void main(String[] args) {
35          test("a");
36          test("fooo");
37          test(1);
38          test(1L);
39          instanceOfPattern("abcde");
40          test(null);
41      }
42  }
